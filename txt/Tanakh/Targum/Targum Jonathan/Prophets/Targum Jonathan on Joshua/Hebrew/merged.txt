תרגום יונתן על יהושע



Chapter 1

וַהֲוָה בָּתַר דְמִית מֹשֶׁה עַבְדָא דַייָ וַאֲמַר יְיָ לִיהוֹשֻׁעַ בַּר נוּן מְשַׁמְשׁוּנֵיהּ דְמֹשֶׁה לְמֵימָר:
מֹשֶׁה עַבְדִי מִית וּכְעַן קוּם עִיבַּר יַת יַרְדְנָא הָדֵין אַתְּ וְכָל עַמָא הָדֵין לְאַרְעָא דַאֲנָא יְהִיב לְהוֹן לִבְנֵי יִשְׂרָאֵל:
כָּל אַתְרָא דְתִדְרַךְ פַּרְסַת רַגְלְכוֹן בֵּיהּ לְכוֹן יְהַבִתָהּ כְּמָא דְמַלֵילִית עִם מֹשֶׁה:
מִמַדְבְּרָא וּלְבָנוֹנָא הָדֵין וְעַד נַהֲרָא רַבָּא נַהֲרָא דִפְרָת כֹּל אַרַע חִתָּאֵי וְעַד יַמָא רַבָּא מַעֲלָנֵי שִׁמְשָׁא יְהֵא תְחוּמְכוֹן:
לָא יִתְעַתַּד אֱנָשׁ קֳדָמָךְ כָּל יוֹמֵי חַיָיךְ כְּמָא דַהֲוָא מֵימְרִי בְּסַעֲדֵיהּ דְמֹשֶׁה כֵּן אֶהֱוֵי עִמָךְ לָא אַשְׁבְּקִינָךְ וְלָא אַרְחְקִינָךְ:
תְּקֵיף וַעְלַם אֲרֵי אַתְּ תַּחְסֵין יַת עַמָא הָדֵין יַת אַרְעָא דְקַיֵימִית לַאֲבָהַתְהוֹן לְמִתַּן לְהוֹן:
לְחוֹד תְּקֵיף וֶעְלֵם לַחֲדָא לְמִטַר לְמֶעְבַּד כְּכָל אוֹרַיְתָא דְפַקְדָךְ מֹשֶׁה עַבְדִי לָא תִסְטֵי מִינֵיהּ לְיַמִינָא וְלִשְׂמָאלָא בְּדִיל דְתִצְלַח בְּכָל אֲתַר דִתְהָךְ:
לָא יַעְדֵי סִיפְרָא דְאוֹרַיְתָא הָדֵין מִפּוּמָךְ וּתְהֵי הֲגֵי בֵּיהּ יֵימָם וְלֵילְיָא בְּדִיל דְתִטַר לְמֶעְבַּד כְּכָל דִכְתִיב בֵּיהּ אֲרֵי בְּכֵן תַּצְלַח יַת דַרְכָּךְ וּבְכֵן תַּכְשָׁט:
הֲלָא פַקֵידְתָּךְ תְּקֵף וֶעְלֵם לָא תִדְחַל וְלָא תִתְּבַר אֲרֵי בְּסַעֲדָךְ מֵימְרָא דַייָ אֱלָהָךְ בְּכָל אֲתַר דִתְהָךְ:
וּפַקֵיד יְהוֹשֻׁעַ יַת סַרְכֵי עַמָא לְמֵימָר:
עִבָרוּ בְּגוֹ מַשְׁרִיתָא וּפַקִידוּ יַת עַמָא לְמֵימָר אַתְקִינוּ לְכוֹן זְוָדִין אֲרֵי בְּסוֹף תְּלָתָא יוֹמִין אַתּוּן עָבְרִין יַת יַרְדְנָא הָדֵין לְמֵיעַל לְמֵירַת יַת אַרְעָא דַייָ אֱלָהָכוֹן יָהֵיב לְכוֹן לְמִירְתָהּ:
וּלְשֵׁבֶט רְאוּבֵן וּלְשֵׁבֶט גָד וּלְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה אֲמַר יְהוֹשֻׁעַ לְמֵימָר:
הֲווֹ דְכִירִין יַת פִּתְגָמָא דִפַקֵיד יַתְכוֹן מֹשֶׁה עַבְדָא דַייָ לְמֵימָר יְיָ אֱלָהָכוֹן מַנִיחַ לְכוֹן וִיהַב לְכוֹן יַת אַרְעָא הָדָא:
נְשֵׁיכוֹן טַפְלֵיכוֹן וּבְעִירְכוֹן יֵתְבוּן בְּאַרְעָא דִיהַב לְכוֹן מֹשֶׁה בְּעִבְרָא דְיַרְדְנָא וְאַתּוּן תְעָבְרוּן מְזָרְזִין קֳדָם אֲחוּכוֹן כֹּל גִבָּרֵי חֵילָא וּתְסַעֲדוּן יַתְהוֹן:
עַד דִינִיחַ יְיָ לַאֲחוּכוֹן כְּוַתְכוֹן וְיֵירְתוּן אַף אִינוּן יַת אַרְעָא דַייָ אֱלָהָכוֹן יָהֵיב לְכוֹן וּתְתוּבוּן לְאַרַע יְרוּתַתְכוֹן וְתֵירְתוּן יָתָהּ דִיהַב לְכוֹן מֹשֶׁה עַבְדָא דַייָ בְּעִבְרָא דְיַרְדְנָא מַדְנַח שִׁמְשָׁא:
וַאֲתִיבוּ יַת יְהוֹשֻׁעַ לְמֵימָר כֹּל דְפַקֵדְתָּנָא נַעֲבַד וּלְכָל אֲתַר דְתִשְׁלַחְנָנָא נְהַךְ:
כְּכֹל דְקַבֵּלְנָא מִן מֹשֶׁה כֵּן נְקַבֵּל מִנָךְ לְחוֹד יְהֵי מִימְרָא דַייָ אֱלָהָךְ בְּסַעֲדָךְ כְּמָא דַהֲוָה בְּסַעֲדֵיהּ דְמֹשֶׁה:
כָּל גְבַר דִיסַרֵב עַל מֵימְרָךְ וְלָא יְקַבֵל יַת פִּתְגָמָךְ לְכֹל דְתַפְקְדִנֵיהּ יִתְקְטֵיל לְחוֹד תְּקֵף וֶעְלָם:

Chapter 2

וּשְׁלַח יְהוֹשֻׁעַ בַּר נוּן מִן שִׁטִין תְּרֵין גוּבְרִין מְאַלְלִין בְּרָז לְמֵימָר אֲזִילוּ חֲזוּ יַת אַרְעָא וְיַת יְרִיחוֹ וַאֲזַלוּ וְעַלוּ לְבֵית אִתְּתָא פּוּנְדְקִיתָא וּשְׁמָה רָחָב וּשְׁכִיבוּ תַמָן:
וְאִתְאַמַר לְמַלְכָּא דִירִיחוֹ לְמֵימָר הָא גוּבְרַיָא אֲתוֹ הָכָא בְּלֵילְיָא מִבְּנֵי יִשְׂרָאֵל לְאַלָלָא יַת אַרְעָא:
וּשְׁלַח מַלְכָּא דִירִיחוֹ לְוַת רָחָב לְמֵימָר אַפִיקִי גֻבְרַיָא דַאֲתוֹ לְוָתִיךְ דַעֲלוּ לְבֵיתֵיךְ אֲרֵי לְאַלָלָא יַת כָּל אַרְעָא אֲתוֹ:
וּדְבִירַת אִתְּתָא יַת תְּרֵין גֻבְרַיָא וְאַטְמַרַתְנוּן וַאֲמָרַת בְּקוּשְׁטָא אֲתוֹ לְוָתִי גֻבְרַיָא וְלָא יִדְעֵית אֵי מִנָן אִינוּן:
וַהֲוָה עִדָן לְמֵיחַד תַּרְעָא בְּקִבְלָה וְגֻבְרַיָא נְפַקוּ לָא יַדְעֵית לְאָן אֲזַלוּ גֻבְרַיָא רִדְפוּ בִּפְרִיעַ בַּתְרֵיהוֹן אֲרֵי תַדְבְּקוּנוּן:
וְהִיא אַסְקַתְנוּן לְאִגְרָא וְאַטְמַרַתְנוּן בִּטְעוּנֵי כִתָּנָא דִסְדִירִין לָהּ עַל אִגְרָא:
וְגֻבְרַיָא רְדָפוּ בַּתְרֵיהוֹן אוֹרַח יַרְדֵנָא עַד מְגִזָתָא וְתַרְעָא אֲחָדוּ בָּתַר דִנְפַקוּ רַדְפַיָא בַּתְרֵיהוֹן:
וְאִינוּן עַד לָא שְׁכִיבוּ וְהִיא סְלֵיקַת לְוַתְהוֹן לְאִגְרָא:
וַאֲמַרַת לְגֻבְרַיָא יְדַעְנָא אֲרֵי יְהַב יְיָ לְכוֹן יַת אַרְעָא וַאֲרֵי נְפָלַת אֵמַתְכוֹן עֲלָנָא וַאֲרֵי אִתְּבָּרוּ כָּל יַתְבֵי אַרְעָא מִן קֳדָמֵיכוֹן:
אֲרֵי שְׁמַעְנָא יַת דְיַבֵּשׁ יְיָ יַת מֵי יַמָא דְסוּף מִן קֳדָמֵיכוֹן בְּמִפַּקְכוֹן מִמִצְרַיִם וְדִי עֲבַדְתּוּן לִתְרֵין מַלְכֵי אֱמוֹרָאָה דִי בְּעַבְרָא דְיַרְדְנָא לְסִיחוֹן וּלְעוֹג דִי גְמַרְתּוּן יַתְהוֹן:
וּשְׁמַעְנָא וְאִתְמְסֵי לִבָּנָא וְלָא אִשְׁתְּאָרַת עוֹד רוּחָא בֶּאֱנַשׁ מִן קֳדָמֵיכוֹן אֲרֵי יְיָ אֱלָהָכוֹן הוּא אֱלָהָא דִשְׁכִינְתֵּיהּ בִּשְׁמַיָא מִלְעֵילָא וְשַׁלִיט עַל אַרְעָא מִלְרָע:
וּכְעַן קַיְימוּ כְעַן לִי בְּמֵימְרָא דַייָ אֲרֵי עֲבָדִית עִמְכוֹן טִיבוּ וְתַעְבְּדוּן אַף אַתּוּן עִם בֵּית אַבָּא טִיבוּ וְתִתְּנוּן לִי אָת דִקְשׁוֹט:
וְתֵיחוּן יַת אַבָּא וְיַת אִמָא וְיַת אֲחַי וְיַת אֲחָתֵי וְיַת כָּל דִילְהוֹן וּתְשֵׁזְבוּן יַת נַפְשָׁתָנָא מִמוֹתָא:
וַאֲמַרוּ לָהּ גֻבְרַיָא נַפְשָׁנָא מְסִירָן חֲלַף נַפְשָׁתֵיכוֹן לְמֵמָת אִם לָא תְחַווּן יַת פִּתְגָמָנָא דֵין וִיהֵי כַּד יִתֵּן יְיָ לָנָא יַת אַרְעָא וְנַעֲבֵד עִמָךְ טֵיבוּ וּקְשׁוֹט:
וְשַׁלְשְׁלַתְנוּן בְּאַטוּנָא מִן חֲרַכָּא אֲרֵי בֵיתָהּ בְּכוֹתֵל שׁוּרָא וּבְשׁוּרָא הִיא יָתְבָה:
וַאֲמַרַת לְהוֹן לְטוּרָא אֱזִילוּ דִלְמָא יְעַרְעוּן בְּכוֹן רָדְפַיָא וְתִטַמְרוּן תַּמָן תְּלָתָא יוֹמִין עַד דִיתוּבוּן רָדְפַיָא וּבָתַר כֵּן תְּהָכוּן לְאוֹרְחַתְכוֹן:
וַאֲמַרוּ לָהּ גֻבְרַיָא זַכָּאִין אֲנַחְנָא מִמוֹמָתֵךְ הָדָא דְקַיֵמְתָּ עֲלָנָא:
הָא אֲנַחְנָא עָלִין בְּאַרְעָא יַת תּוֹרָא דְחוּט זְהוֹרִיתָא הָדֵין תִקְטְרִי בַּחֲרַכָּא דְשַׁלְשַלְתָּנָא בֵּיהּ וְיַת אֲבוּךְ וְיַת אִמֵךְ וְיַת אֲחָיְכִי וְיַת כָּל בֵּית אָבוּךְ תִּכְנְשִׁין לְוָתֵיךְ לְבֵיתָא:
וִיהֵי כָּל דְיִפּוֹק בַּר מִדָשֵׁי בֵּיתֵךְ לְבָרָא חוֹבַת קְטוֹלֵיהּ בְּרֵישֵׁיהּ וַאֲנַחְנָא זַכָּאִין וְכֹל דִי יְהֵי עִמָךְ בְּבֵיתָא חוֹבַת קְטוֹלֵיהּ בְּרֵישָׁנָא אִם יַד אֱנָשׁ תְהֵי בֵיהּ:
וְאִם תְּחַוִין יַת פִּתְגָמָנָא דֵין וּנְהֵי זַכָּאִין מִמוֹמָתִיךְ דְקַיֵימְתְּ עֲלָנָא:
וַאֲמָרַת כְּפִתְגָמֵיכוֹן כָּן הוּא וּשְׁלַחַתְנוּן וַאֲזָלוּ וְקִטְרַת יַת תּוֹרָא דִזְהוֹרִיתָא בַּחֲרַכָּא:
וַאֲזַלוּ וְעָלוּ לְטוֹרָא וִיתָבוּ תַמָן תְּלָתָא יוֹמִין עַד דְתָבוּ רָדְפַיָא וּבְעוֹ רָדְפַיָא בְּכָל אָרְחָא וְלָא אַשְׁכָּחוּ:
וְתָבוּ תְּרֵין גֻבְרָין וּנְחִיתוּ מִן טוּרָא וַעֲבָרוּ וַאֲתוֹ לְוַת יְהוֹשֻׁעַ בַּר נוּן וְאִשְׁתְּעִיאוּ לֵיהּ יַת כָּל דְעַרָעָא יַתְהוֹן:
וַאֲמַרוּ לִיהוֹשֻׁעַ אֲרֵי מְסַר יְיָ בִּידָנָא יַת כָּל אַרְעָא וְאַף אִתְּבַּרוּ כָּל יָתְבֵי אַרְעָא מִן קֳדָמָנָא:

Chapter 3

וְאַקְדֵים יְהוֹשֻׁעַ בְּצַפְרָא וּנְטַלוּ מִשִׁטִין וַאֲתוֹ עַד יַרְדְנָא הוּא וְכָל בְּנֵי יִשְׂרָאֵל וּבָתוּ תַמָן קֳדָם יַעַבְּרוּן:
וַהֲוָה מִסוֹף תְּלָתָא יוֹמִין וַעֲבָרוּ סַרְכַיָא בְּגוֹ מַשְׁרִיתָא:
וּפַקֵידוּ יַת עַמָא לְמֵימָר בְּמֶחְזֵיכוֹן יַת אֲרוֹן קְיָמָא דַייָ אֱלָהָכוֹן וְכֹהֲנַיָא לְוָאֵי נַטְלִין יָתֵיהּ וְאַתּוּן תִּטְלוּן מֵאַתְרֵיכוֹן וּתְהָכוּן בַּתְרוֹהִי:
בְּרַם רָחִיק יְהֵי בְּינֵיכוֹן וּבֵינוֹהִי כִּתְרֵין אַלְפִין אַמִין בְּמִשְׁחָתָא לָא תְקַרְבוּן לְוָתֵיהּ בְּדִיל דְתִידְעוּן יַת אוֹרְחָא דִי תְהָכוּן בָּהּ אֲרֵי לָא עֲבַרְתּוּן בְּאוֹרְחָא מֵאִתְמָלִי וּמִדְקָמוֹהִי:
וַאֲמַר יְהוֹשֻׁעַ לְעַמָא אִזְדַמְנוּ אֲרֵי מָחָר יְעַבֵּד יְיָ בֵּינֵיכוֹן פְּרִישָׁן:
וַאֲמַר יְהוֹשֻׁעַ לְכַהֲנַיָא לְמֵימָר טוּלוּ יַת אֲרוֹן קְיָמָא וַעֲבַרוּ קֳדָם עַמָא וּנְטָלוּ יַת אֲרוֹנָא דִקְיָמָא וַאֲזַלוּ קֳדָם עַמָא:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ יוֹמָא הָדֵין אַשְׁרֵי לְרַבָּיוּתָךְ בְּעֵינֵי כָּל יִשְׂרָאֵל דְיֵדְעִין אֲרֵי בְּמָא דַהֲוָה מֵימְרִי בְּסַעְדֵיהּ דְמֹשֶׁה כֵּן יְהֵי מֵימְרִי בְּסַעְדָךְ:
וְאַתְּ תְּפַקֵד יַת כַּהֲנַיָא נַטְלֵי אֲרוֹנָא דִקְיָמָא לְמֵימָר כְּמִמְטֵיכוֹן עַד קְצַת מֵי יַרְדְנָא בְּיַרְדְנָא תְּקוּמוּן:
וַאֲמַר יְהוֹשֻׁעַ לִבְנֵי יִשְׂרָאֵל אִתְקְרִיבוּ הַלְכָא וּשְׁמָעוּ יַת פִּתְגָמַיָא דַייָ אֱלָהָכוֹן:
וַאֲמַר יְהוֹשֻׁעַ בְּדָא תֵדְעוּן אֲרֵי אֱלָהּ קַיָם אִתְרְעֵי לְאַשְׁרָאָה שְׁכִנְתֵּיהּ בְּינֵיכוֹן וּתְרָכָא יְתָרֵךְ מִן קָדָמֵיכוֹן יַת כְּנַעֲנָאֵי וְיַת חִתָּאֵי וְיַת חִוָאֵי וְיַת פְּרִיזָאֵי וְיַת גִרְגְשָׁאֵי וֶאֱמוֹרָאֵי וִיבוּסָאֵי:
הָא אֲרוֹן קְיָמָא דְרִבּוֹן כָּל אַרְעָא עֲבַר קֳדָמֵיכוֹן בְּיַרְדְנָא:
וּכְעַן דַבָּרוּ לְכוֹן תְּרֵי עֲסַר גַבְרִין מִשִׁבְטַיָא דְיִשְׂרָאֵל גַבְרָא חָד גַבְרָא חָד לְשִׁבְטָא:
וִיהֵי כִּמְנָח פַּרְסַת רִגְלֵי כַהֲנַיָא נַטְלֵי אֲרוֹנָא דַייָ רִבּוֹן כָּל אַרְעָא בְּמֵי יַרְדְנָא מֵי יַרְדְנָא יִפְסְקוּן מַיָא דְנַחֲתִין מִלְעֵילָא וִיקוּמוּן רוּקְבָא חָד:
וַהֲוָה כַּד נְטַלוּ עַמָא מְקִרְוֵיהוֹן לְמֶעֱבַּר יַת יַרְדְנָא וְכַהֲנַיָא נַטְלֵי אֲרוֹנָא דִקְיָמָא קֳדָם עַמָא:
וּכְמִמְטֵי נַטְלֵי אֲרוֹנָא עַד יַרְדְנָא וְרַגְלֵי כַהֲנַיָא נַטְלֵי אֲרוֹנָא טְבִילָא בִּקְצַת מַיָא וְיַרְדְנָא מְלִי עַל כָּל כַּפּוֹהִי כֹּל יוֹמֵי חֲצָדָא:
וְקָמוּ מַיָא דְנַחְתִין מִלְעֵילָא קָמוּ רוּקְבָא חָד רְחִיקִין לַחֲדָא מֵאָדָם קַרְתָּא דְבִסְטַר צָרְתָן וּדְנַחֲתִין לְיַמָא דְמֵישְׁרָא לְיַמָא דְמִלְחָא שְׁלִימוּ פְּסָקוּ וְעַמָא עֲבָרוּ לָקֳבֵיל יְרֵיחוֹ:
וְקָמוּ כַהֲנַיָא נַטְלֵי אֲרוֹן קַיָמָא דַייָ בְּיַבֶּשְׁתָּא בְּגוֹ יַרְדְנָא מְתַקְנַיָא וְכָל יִשְׂרָאֵל עָבְרִין בְּיַבֶּשְׁתָּא עַד דִשְׁלִימוּ כָּל עַמָא לְמֶעְבַּר יַת יַרְדְנָא:

Chapter 4

וַהֲוָה כַּד שְׁלִימוּ כָל עַמָא לְמֶעְבַּר יַת יַרְדְנָא וַאֲמַר יְיָ לִיהוֹשֻׁעַ לְמֵימָר:
דַבָּרוּ לְכוֹן מִן עַמָא תְּרֵי עֲסַר גַבְרִין גַבְרָא חָד גַבְרָא חָד מִשִׁבְטָא:
וּפַקֵידוּ יַתְהוֹן לְמֵימָר טוּלוּ לְכוֹן מִכָּא מִגוֹ יַרְדְנָא מֵאֲתַר מֵיקַם רִגְלֵי כַהֲנַיָא מְתַקְנַיָא תַּרְתֵּי עַסְרֵי אַבְנַיָא וּתְעַבְרִנוּן יַתְהוֹן עִמְכוֹן וְתַחֲתוּן יַתְהוֹן בְּבֵית מַבְּתָא דִתְבִיתוּן בֵּיהּ בְּלֵילְיָא:
וּקְרָא יְהוֹשֻׁעַ לִתְרֵי עֲסַר גַבְרָא דְאַתְקִין מִבְּנֵי יִשְׂרָאֵל גַבְרָא חָד גַבְרָא חָד מִשִׁבְטָא:
וַאֲמַר לְהוֹן יְהוֹשֻׁעַ עֲבָרוּ קֳדָם אֲרוֹנָא דַייָ אֱלָהָכוֹן לְגוֹ יַרְדְנָא וַאֲרִימוּ לְכוֹן גְבַר אַבְנָא חָדָה עַל כִּתְפֵיהּ לְמִנְיַן שִׁבְטַיָא דִבְנֵי יִשְׂרָאֵל:
בְּדִיל דִתְהֵי דָא אָת בֵּינֵכוֹן אֲרֵי יִשְׁאֲלִין בְּנֵיכוֹן מְחָר לְמֵימָר מָא אַבְנַיָא הָאִלֵין לְכוֹן:
וְתֵימְרוּן לְהוֹן דִי פְּסַקוּ מֵי יַרְדְנָא מִן קֳדָם אֲרוֹן קְיָמָא דַייָ בְּמֶעְבְּרֵיהּ בְּיַרְדְנָא פְּסַקוּ מֵי יַרְדְנָא וִיהֶוְיָן אַבְנַיָא הָאִלֵין לְדָכְרָנָא לִבְנֵי יִשְׂרָאֵל עַד עָלְמָא:
וַעֲבָדוּ כֵן בְּנֵי יִשְׂרָאֵל כְּמָא דִי פַּקֵד יְהוֹשֻׁעַ וּנְטַלוּ תַּרְתָּא עַשְׂרֵי אַבְנַיָא מִגוֹ יַרְדְנָא כְּמָא דְמַלֵיל יְיָ עִם יְהוֹשֻׁעַ לְמִנְיַן שִׁבְטַיָא דִבְנֵי יִשְׂרָאֵל וְאַעַבְּרוּנוּן עִמְהוֹן לְבֵית מַבָּתָא וְאַצְנְעוּנוּן תַּמָן:
וְתַרְתָּא עַשְׂרֵי אַבְנַיָא אֲקֵים יְהוֹשֻׁעַ בְּגוֹ יַרְדְנָא בַּאֲתַר מֵיקַם רִגְלֵי כַהֲנַיָא נַטְלֵי אֲרוֹנָא דִקְיָמָא וַהֲווֹ תַּמָן עַד יוֹמָא הָדֵין:
וְכַהֲנַיָא נַטְלֵי אֲרוֹנָא קַיְמִין בְּגוֹ יַרְדְנָא עַד דִשְׁלִים כָּל פִּתְגָמָא דְפַקֵיד יְיָ יַת יְהוֹשֻׁעַ לְמַלָלָא עִם עַמָא כְּכֹל דְפַקֵיד מֹשֶׁה יַת יְהוֹשֻׁעַ וְאוֹחִיאוּ עַמָא וַעֲבָרוּ:
וַהֲוָה כַּד שְׁלִים כָּל עַמָא לְמֶעְבַּר וַעֲבַר אֲרוֹנָא דַייָ וְכַהֲנַיָא קֳדָם עַמָא:
וַעֲבָרוּ בְּנֵי רְאוּבֵן וּבְנֵי גָד וּפַּלְגוּת שִׁבְטָא דִמְנַשֶׁה מְזָרְזִין קֳדָם בְּנֵי יִשְׂרָאֵל כְּמָא דְמַלֵיל עִמְהוֹן מֹשֶׁה:
כְּאַרְבְּעִין אַלְפִין מְזָרְזֵי חֵילָא עֲבָרוּ קֳדָם עַמָא דַייָ לִקְרָבָא לְמֵשְׁרֵי יְרֵיחוֹ:
בְּיוֹמָא הַהוּא רַבֵּי יְיָ יַת יְהוֹשֻׁעַ בְּעֵינֵי כָּל יִשְׂרָאֵל וּדְחִילוּ מִנֵיהּ כְּמָא דַהֲווֹ דָחֲלִין מִן מֹשֶׁה כָּל יוֹמֵי חַיוֹהִי:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ לְמֵימָר:
פַּקֵיד יַת כַּהֲנַיָא נַטְלֵי אֲרוֹנָא דְסַהֲדוּתָא וְיִסְקוּן מִן יַרְדְנָא:
וּפַקֵיד יְהוֹשֻעַ יַת כַּהֲנַיָה לְמֵימָר סַקוּ מִן יַרְדְנָא:
וַהֲוָה כַּד בְּעוֹ לְמִסַק כַּהֲנַיָא נַטְלֵי אֲרוֹן קְיָמָא דַייָ מִגוֹ יַרְדְנָא אִתְנְגִידָא פַּרְסַת רִגְלֵי כַּהֲנַיָא וּנְחָא עַל יַבֶּשְׁתָּא וְתָבוּ מֵי יַרְדְנָא לְאַתְרֵיהוֹן וַאֲזַלוּ כְמֵאִתְמָלֵי וּמִדְקֳדָמוֹהִי עַל כָּל כַּפּוֹהִי:
וְעַמָא סְלִיקוּ מִן יַרְדְנָא בְּעַשְׂרָא לְיַרְחָא קַדְמָאָה וּשְׁרוֹ בְגִלְגָלָא בִּסְיָפֵי מַדְנַח יְרִיחוֹ:
וְיַת תַּרְתָּא עַשְׂרֵי אַבְנַיָא הָאִלֵין דִנְסִיבוּ מִן יַרְדְנָא אֲקֵים יְהוֹשֻׁעַ בְּגִלְגָלָא:
וַאֲמַר לִבְנֵי יִשְׂרָאֵל לְמֵימָר דְיִשְׁאֲלוּן בְּנֵיכוֹן מְחָר יַת אֲבָהַתְהוֹן לְמֵימָר מָה אַבְנַיָא הָאִלֵין:
וּתְהוֹדְעוּן יַת בְּנֵיכוֹן לְמֵימָר בְּיַבֶּשְׁתָּא עֲבַר יִשְׂרָאֵל יַת יַרְדְנָא הָדֵין:
דְיַבֵּשׁ יְיָ אֱלָהָכוֹן יַת מֵי יַרְדְנָא מִן קֳדָמֵיכוֹן עַד דַעֲבַרְתּוּן כְּמָא דַעֲבַד יְיָ אֱלָהָכוֹן לְיַמָא דְסוּף דִי יַבֵּשׁ מִן קֲדָמָנָא עַד דַעֲבָרְנָא:
בְּדִיל דְיֵדְעוּן כָּל עַמְמֵי אַרְעָא יַת גְבוּרְתָּא דַייָ אֲרֵי תַקִיפָה הִיא בְּדִיל דְתִדְחֲלוּן יַת יְיָ אֱלָהָכוֹן כָּל יוֹמַיָא:

Chapter 5

וַהֲוָה כַד שְׁמָעוּ כָּל מַלְכֵי אֱמוֹרָאֵי דִי בְּעִבְרָא דְיַרְדְנָא מַעֲרָבָא וְכָל מַלְכֵי כְנַעֲנָאָה דִי עַל יַמָא יַת דְיַבֵּשׁ יְיָ יַת מֵי יַרְדְנָא מִן קֳדָם בְּנֵי יִשְׂרָאֵל עַד דַעֲבָרוּ וְאִתְמְסֵי לִבְּהוֹן וְלָא אִשְׁתַּאֲרַת בְּהוֹן עוֹד רוּחַ מִן קֳדָם בְּנֵי יִשְׂרָאֵל:
בְּעִידָנָא הַהִיא אֲמַר יְיָ לִיהוֹשֻׁעַ עֲבַד לָךְ אִזְמְלָוָן חֲרִיפִין וְתוּב גְזַר יַת בְּנֵי יִשְׂרָאֵל תִּנְיָנוּת:
וַעֲבַד לֵיהּ יְהוֹשֻׁעַ אִזְמְלָוָן חֲרִיפִין וּגְזַר יַת בְּנֵי יִשְׂרָאֵל בְּגִבְעַתָא וּקְרָא לֵיהּ גִבְעַת עַרְלָתָא:
וְדֵין פִּתְגָמָא דִי גְזַר יְהוֹשֻׁעַ כָּל עַמָא דִנְפַק מִמִצְרַיִם דָכְרַיָא כָּל גַבְרֵי מְגִיחֵי קְרָבָא מִיתוּ בְמַדְבְּרָא בְּאוֹרְחָא בְּמִפַּקְהוֹן מִמִצְרָיִם:
אֲרֵי גְזִירִין הֲווֹ כָּל עַמָא דִי נְפַקוּ וְכָל עַמָא דְאִתְיְלִידוּ בְּמַדְבְּרָא בְּאָרְחָא בְּמִפַּקְהוֹן מִמִצְרַיִם לָא גְזָרוּ:
אֲרֵי אַרְבְּעִין שְׁנִין הֲלִיכוּ בְנֵי יִשְׂרָאֵל בְּמַדְבְּרָא עַד דְסָף כָּל עַמָא גַבְרֵי מְגִיחֵי קְרָבָא דִנְפַקוּ מִמִצְרַיִם דְלָא קַבִּילוּ לְמֵימְרָא דַייָ דְקַיֵם יְיָ לְהוֹן בְּדִיל דְלָא לְאַחֲזָיוּתְהוֹן יַת אַרְעָא דְקַיֵם יְיָ לַאֲבָהַתְהוֹן לְמִתַן לָנָא אַרְעָא עַבְדָא חֲלַב וּדְבָשׁ:
וְיַת בְּנֵיהוֹן דְקָמוּ בַתְרֵיהוֹן יַתְהוֹן גְזַר יְהוֹשֻעַ אֲרֵי עֲרֵלִין הֲווֹ אֲרֵי לָא גְזָרוּ יַתְהוֹן בְּאָרְחָא:
וַהֲוָה כַּד שְׁלִימוּ כָל עַמָא לְמִגְזַר וִיתֵבוּ בְּאַתְרֵיהוֹן בְּמַשְׁרִיתָא עַד דְאִתַּסִיאוּ:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ יוֹמָא דֵין אַעֲדֵית יַת חִיסוּדֵי מִצְרָאֵי מִן קֳדָמֵיכוֹן וּקְרָא שְׁמָא דְאַתְרָא הַהוּא גִלְגְלָא עַד יוֹמָא הָדֵין:
וּשְׁרוֹ בְנֵי יִשְׂרָאֵל בְּגִלְגָלָא וַעֲבַדוּ יַת פִּסְחָא בְּאַרְבַּעַת עַשְׂרָא יוֹמָא לְיַרְחָא בְּרַמְשָׁא בְּמֵישְׁרֵי יְרִיחוֹ:
וַאֲכַלוּ מֵעִבּוּרָא דְאַרְעָא מִבָּתַר פִּסְחָא פַּטִיר וּקְלִי בִּכְרַן יוֹמָא הָדֵין:
וּפְסַק מַנָא מִיוֹמָא דְבַתְרוֹהִי בְּמֵיכַלְהוֹן מֵעִיבּוּרָא דְאַרְעָא וְלָא הֲוָה עוֹד לִבְנֵי יִשְׂרָאֵל מַנָא וַאֲכַלוּ מֵעֲלְלַת אַרְעָא דִכְנַעַן בְּשַׁתָּא הַהִיא:
וַהֲוָה כַּד הֲוָה יְהוֹשֻׁעַ בִּירִיחוֹ וּזְקַף עֵינוֹהִי וַחֲזָא וְהָא גַבְרָא קָאִים לְקִבְלֵיהּ וְחַרְבֵּיהּ שְׁלִיפָא בִּידֵיהּ וַאֲזַל יְהוֹשֻׁעַ לְוָתֵיהּ וַאֲמַר לֵיהּ הַלְמִסְעַדָנָא אַתֵיתָא אִם לְבַעֲלֵי דְבָבָנָא:
וַאֲמַר לָא אֲרֵי אֲנָא מַלְאַךְ שְׁלִיחַ מִן קֳדָם יְיָ כְּעַן אֲתֵיתִי וּנְפַל יְהוֹשֻׁעַ עַל אַפּוֹהִי עַל אַרְעָא וּסְגִיד וַאֲמַר לֵיהּ מַה רִבּוֹנִי מְמַלֵיל עִם עַבְדֵיהּ:
וַאֲמַר מַלְאָכָה דִשְׁלִיחַ מִן קֳדָם יְיָ לִיהוֹשֻׁעַ שְׁרֵי סִינָךְ מֵעַל רִגְלָךְ אֲרֵי אַתְרָא דִי אַתְּ קָאִים עֲלוֹהִי קַדִישָׁא הוּא וַעֲבַד יְהוֹשֻׁעַ כֵּן:

Chapter 6

וִירִיחוֹ אֲחִידָא הֲוַת בְּדָשִׁין דְפַרְזְלָא וּמְתַקְפָא בְּעַבְּרִין דִנְחָשׁ מִן קֳדָם בְּנֵי יִשְׂרָאֵל לֵית דִנְפַק מִנָהּ לְמֶעְבַּד קְרָבָא וְלֵית דְעָלֵל בְּגַוָהּ לְמִשְׁאַל בִּשְׁלָמָא:
וַאֲמַר יְיָ לִיהוֹשֻׁיעַ חֲזֵי דִמְסָרִית בִּידָךְ יַת יְרִיחוֹ וְיַת מַלְכָּהּ גִבָּרֵי חֵילָא:
וְתִסְחֲרוּן יַת קַרְתָּא כָּל גַבְרֵי עָבְדֵי קְרָבָא אַקֵיף יַת קַרְתָּא זִמְנָא חָדָא כְּדֵין תַּעְבֵּד שִׁיתָּא יוֹמִין:
וְשִׁבְעָא כַהֲנַיָא יִסְבוּן שִׁבְעָא שׁוֹפָרַיָא דְקֶרֶן דִכְרַיָא קֳדָם אֲרוֹנָא וּבְיוֹמָא שְׁבִיעָאָה תִּסְחֲרוּן יַת קַרְתָּא שְׁבַע זִמְנִין וְכַהֲנַיָא יִתְקְעוּן בְּשׁוֹפָרַיָא:
וִיהֵי בְּמֵיגַד בְּקְרַן דִכְרַיָא בְּמִשְׁמַעֲכוֹן יַת קַל שׁוֹפָרָא יְיַבְּבוּן כָּל עַמָא יַבָּבָא רַבָּא וְיִפּוֹל שׁוּרָא דְקַרְתָּא וְיִתְבְּלַע תְּחוֹתָהּ וְיִסְקוּן עַמָא גְבַר לְקִבְלֵיהּ:
וּקְרָא יְהוֹשֻׁעַ בַּר נוּן לְכַהֲנַיָא וַאֲמַר לְהוֹן טוּלוּ יַת אֲרוֹן קְיָמָא דַייָ וְשִׁבְעָא כַהֲנַיָא יִטְלוּן שַׁבְעָא שׁוֹפָרַיָא דְקֶרֶן דִכְרַיָא קֳדָם אֲרוֹנָא דַיָי:
וַאֲמַר לְעַמָא עֲבָרוּ וְאַקִיפוּ יַת קַרְתָּא וּמְזַרְזֵי חֵילָא יַעַבְּרוּן קֳדָם אֲרוֹנָא דַיָי:
וַהֲוָה כַּד אֲמַר יְהוֹשֻׁעַ לְעַמָא וְשַׁבְעָא כַהֲנַיָא נָסְבִין שַׁבְעָא שׁוֹפָרַיָא דְקָרֶן דִכְרַיָא קֳדָם אֲרוֹנָא דַייָ עָבְרִין וְתָקְעִין בְשׁוֹפְרַיָא וַאֲרוֹן קְיָמָא דַייָ אָזֵיל בַּתְרֵיהוֹן:
וּמְזַרְזֵי חֵילָא אָזְלִן קֳדָם כַּהֲנַיָא תָּקְעֵי שׁוֹפָרַיָא וְשִׁבְטָא דְבֵית דָן אָזַל בָּתַר אֲרוֹנָא וְכַהֲנַיָא אָזְלִין וְתָקְעוּן בְּשׁוֹפָרַיָא:
וְיַת עַמָא פַּקֵיד יְהוֹשֻעַ לְמֵימָר לָא תְיַבְבוּן וְלָא תַשְׁמְעוּן יַת קָלְכוֹן וְלָא יִפּוֹק מְפּוּמְכוֹן פִּתְגָמָא עַד יוֹמָא דְאֵימַר לְכוֹן יַבִּיבוּ וּתְיַבְּבוּן:
וְאַסְחַר אֲרוֹנָא דַייָ יַת קַרְתָּא אַקֵיף זִמְנָא חָדָא וְעָלוּ בְּמַשְׁרִיתָא וּבָתוּ בְמַשְׁרִיתָא:
וְאַקְדֵים יְהוֹשֻׁעַ בְּצַפְרָא וּנְטָלוּ כַהֲנַיָא יַת אֲרוֹנָא דַיָי:
וְשִׁבְעָא כַהֲנַיָא נְסִיבוּ שִׁבְעָא שׁוֹפָרַיָא דְקֶרֶן דִכְרַיָא קֳדָם אֲרוֹנָא דַייָ אָזְלִין מֵיזַל וְתָקְעִין בְּשׁוֹפָרַיָא וּמְזָרְזֵי חֵילָא אָזְלִין קֳדָמֵיהוֹן וְשִׁבְטָא דְבֵית דָן אָזֵיל בָּתַר אֲרוֹנָא דַייָ וְכַהֲנַיָא אָזְלִין וְתָקְעִין בְּשׁוֹפָרַיָא:
וְאַסְחָרוּ יַת קַרְתָּא בְּיוֹמָא תִנְיָנָא זִמְנָא חָדָא וְתָבוּ לְמַשְׁרִיתָא כְּדֵין עֲבָדוּ שִׁתָּא יוֹמִין:
וַהֲוָה בְּיוֹמָא שְׁבִיעָאָה וְאַקְדִימוּ כְּמִיסַק צַפְרָא וְאַסְחֲרוּ יַת קַרְתָּא כְּהִלְכָתָא הָדֵין שְׁבַע זִמְנִין לְחוֹד בְּיוֹמָא הַהוּא אַקִיפוּ יַת קַרְתָּא שְׁבַע זִמְנִין:
וַהֲוָה בְּזִמְנָא שְׁבִיעִיתָא תְּקָעוּ כַהֲנַיָא בְּשׁוֹפָרַיָא וַאֲמַר יְהוֹשֻׁעַ לְעַמָא יַבִּיבוּ אֲרֵי יְהַב יְיָ לְכוֹן יַת קַרְתָּא:
וּתְהֵי קַרְתָּא חֶרְמָא הִיא וְכָל דְבָהּ קֳדָם יְיָ לְחוֹד רָחָב פּוּנְדְקִיתָא תְחֵי הִיא וְכָל דְעִמָהּ בְּבֵיתָא אֲרֵי אַטְמַרַת יַת אִזְגַדַיָא דִי שְׁלָחָנָא:
וּלְחוֹד אַתּוּן אִסְתַּמָרוּ מִן חֶרְמָא דִלְמָא תְחַרְמוּן וְתִסְבוּן מִן חֶרְמָא וּתְשַׁווּן יַת מַשְׁרִיתָא דְיִשְׂרָאֵל לְחֶרְמָא וּתְעַכְרוּן יָתֵיהּ:
וְכָל כַּסְפָּא וְדַהֲבָא וּמָנֵי נְחָשָׁא וּבַרְזְלָא קוּדְשָׁא אִינוּן קֳדָם יְיָ לְאוֹצַר בֵּית מַקְדְשָׁא דַייָ יִתְעָלוּן:
וְיַבִּיבוּ עַמָא וּתְקָעוּ בְּשׁוֹפְרַיָא וַהֲוָה כַּד שְׁמַע עַמָא יַת קַל שֹׁפָרָא וְיַבִּיבוּ עַמָא יַבָּבָא רַבָּא וּנְפַל שׁוּרָא דְקַרְתָּא וְאִתְבְּלַע תְּחוֹתוֹהִי וּסְלִיק עַמָא לְקַרְתָּא גְבַר לְקִבְלֵיהּ וּכְבָשׁוּ יַת קַרְתָּא:
וּגְמָרוּ יַת כָּל דִבְקַרְתָּא מִגְבַר וְעַד אִתָּא מֵעוּלֵימָא וְעַד סָבָא וְעַד תּוֹר וְאִמַר וַחֲמָר לְפִתְגַם דְחֶרֶב:
וְלִתְרֵין גַבְרַיָא דְאַלִילוּ יַת אַרְעָא אֲמַר יְהוֹשֻׁעַ עֲלוּ לְבֵית אִתְּתָא פּוּנְדְקִיתָא וְאַפִּיקוּ מִתַּמָן יַת אִתְּתָא וְיַת כָּל דִי לָהּ כְמָא דְקַיֵמְתּוּן לָהּ:
וְעָלוּ עוּלֵמַיָא מְאַלְלַיָא וְאַפִּיקוּ יַת רָחָב וְיַת אֲבוּהָא וְיַת אִמָהּ וְיַת אֲחוּהָא וְיַת כָּל דִי לָהּ וְיַת כָּל זַרְעֲיָתָהָא אַפִּיקוּ וְאַשְׁרוּנוּן מִבָּרָא לְמַשְׁרִיתָא דְיִשְׂרָאֵל:
וְקַרְתָּא אוֹקִידוּ בְנוּרָא וְכָל דִי בָהּ לְחוֹד כַּסְפָּא וְדַהֲבָא וּמָאנֵי נְחָשָׁא וּבַרְזְלָא יְהַבוּ בְּאוֹצַר בֵּית מַקְדְשָׁא דַיָי:
וְיַת רָחָב פּוּנְדְקִיתָא וְיַת בֵּית אֲבוּהָא וְיַת כָּל דִי לָהּ קַיֵם יְהוֹשֻׁעַ וִיתִיבַת בְּגוֹ יִשְׂרָאֵל עַד יוֹמָא הָדֵין אֲרֵי אִטְמַרַת יַת אִזְגַדַיָא דִשְׁלַח יְהוֹשֻׁעַ לַאֲלָלָא יַת אַרְעָא:
וְאוֹמֵי יְהוֹשֻׁעַ בְּעִידָנָא הַהִיא לְמֵימָר לִיט גַבְרָא קֳדָם יְיָ דִיקוּם וְיִבְנֵי יַת קַרְתָּא הָדָא יַת יְרִיחוֹ בִּבְכוֹרֵיהּ יְשַׁכְלְלִינָהּ וּבִזְעֵיר בְּנוֹהִי יְקִים דְשָׁהָא:
וַהֲוָה מֵימְרָא דַייָ בְּסַעֲדֵיהּ דִיהוֹשֻׁעַ וַהֲוָה שִׁמְעֵיהּ סַגִי בְּכָל אַרְעָא:

Chapter 7

וְשַׁקָרוּ בְּנֵי יִשְׂרָאֵל שְׁקַר בְּחֶרְמָא וּנְסִיב עָכָן בַּר כַּרְמִי בַר זַבְדִי בַר זֶרַח לְשִׁבְטָא דִיהוּדָה מִן חֶרְמָא וּתְקֵיף רוּגְזָא דַייָ בִּבְנֵי יִשְׂרָאֵל:
וּשְׁלַח יְהוֹשֻׁעַ גֻבְרִין מִירִיחוֹ לְעַי דְעִם בֵּית אָוֶן מִמַדְנַח לְבֵית אֵל וַאֲמַר לְהוֹן לְמֵימַר סַקוּ וְאַלִילוּ יַת אַרְעָא וּסְלִיקוּ גַבְרַיָא וְאַלִילוּ יַת עָי:
וְתָבוּ לְוַת יְהוֹשֻׁעַ וַאֲמַרוּ לֵיהּ לָא יִסַק כָּל עַמָא כִּתְרֵין אַלְפִין גַבְרָא אוֹ כִּתְלָתָא אַלְפִין גַבְרָא יִסְקוּן וְיִמְחוּן יַת עָי לָא תַרְגִישׁ לְתַמָן יַת כָּל עַמָא אֲרֵי זְעִירִין אִינוּן:
וּסְלִיקוּ מִן עַמָא לְתַמָן כִּתְלָתָא אַלְפִין גַבְרָא וְאַפִּיכוּ קֳדָם אֱנָשֵׁי עָי:
וּקְטַלוּ מִנְהוֹן אֱנָשֵׁי עַי כִּתְלָתִין וְשִׁתָּא גַבְרָא וּרְדָפוּנוּן קֳדָם תַּרְעָא עַד דְתַבְּרוּנוּן וּמְחוּנוּן בְּמַחֲתָנָא וְאִתְמְסֵי לִבָּא דְעַמָא וַהֲוָה לְמַיָא:
וּבְזַע יְהוֹשֻׁעַ לְבוּשׁוֹהִי וּנְפַל עַל אַפּוֹהִי עַל אַרְעָא קֳדָם אֲרוֹנָא דַייָ עַד רַמְשָׁא הוּא וְסָבֵי יִשְׂרָאֵל וְאַסִיקוּ עַפְרָא עַל רֵישֵׁיהוֹן:
וַאֲמַר יְהוֹשֻׁעַ קַבֵּיל בָּעוּתִי יְיָ אֱלֹהִים לְמָא אַעֲבַרְתָּא אַעֲבָרָא יַת עַמָא הָדֵין יַת יַרְדְנָא לְמִמְסַר יָתָנָא בִּידָא דֶאֱמוֹרָאָה לְאוֹבָדוּתָנָא וּלְוַי פוֹן דִשְׁרֵינָא וִיתֵיבְנָא בְּעִבְרָא דְיַרְדֵנָא:
בְּבָעוּ אֲדֹנָי מָה אֵימַר בָּתַר דְאַחֲזָרוּ יִשְׂרָאֵל קְדַלְהוֹן לְמֶהְפַךְ קֳדָם בַּעֲלֵי דְבָבֵיהוֹן:
וְיִשְׁמְעוּן כְּנַעֲנָאֵי וְכָל יַתְבֵי אַרְעָא וְיִסְתַּחֲרוּן עֲלָנָא וִישֵׁצוּן יַת שְׁמָנָא מִן אַרְעָא וּמָה תַּעֲבֵּד בְּדִיל שְׁמָךְ רַבָּא:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ קוּם לָךְ לָמָה דְנָן אַתְּ רָמֵי עַל אַפָּךְ:
חֲטָא יִשְׂרָאֵל וְאַף עֲבָרוּ עַל קְיָמִי דְפַקְדֵית יַתְהוֹן וְאַף נְסִיבוּ מִן חֶרְמָא וְאַף גְנָבוּ וְאַף כַּדִיבוּ וְאַף שַׁוִיאוּ בְּמָנֵיהוֹן:
וְלָא יִכְלוּן בְּנֵי יִשְׂרָאֵל לְמֵקַם קֳדָם בַּעֲלֵי דְבָבֵיהוֹן קְדַלְהוֹן יַחְזְרוּן לְמֵיפַּךְ קֳדָם בַּעֲלֵי דְבָבֵיהוֹן אֲרֵי הֲווֹ לְחֶרְמָא לָא אוֹסִיף לְמֶהֱוַי מֵימְרִי בְּסַעְדְכוֹן אִם לָא תְשֵׁיצוּן חֶרְמָא מִבֵּנֵיכוֹן:
קוּם זַמֵן יַת עַמָא וְתֵימַר אִזְדַמְנוּ לִמְחַר אֲרֵי כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל חֶרְמָא בֵינָךְ יִשְׂרָאֵל לָא תִכּוּל לְמֵיקַם קֳדָם בַּעֲלֵי דְבָבָךְ עַד דְתַעֲדוּן חֶרְמָא מִבֵּינֵיכוֹן:
וְתִתְקַרְבוּן בְּצַפְרָא לְשִׁבְטֵיכוֹן וִיהֵי שִׁבְטָא דְיִתְאַחַד מִן קֳדָם יָיָ יִקְרַב לְזַרְעֲיָן וְזַרְעִיתָא דְיִתְאַחַד מִן קֳדָם יְיָ תִּקְרַב לְבָתִּין וּבֵיתָא דְיִתְאַחַד מִן קֳדָם יָיָ יִתְקַרַב לְגַבְרַיָא:
וִיהֵי דְיִתְאַחַד בְּחֶרְמָא יִתּוֹקַד בְּנוּרָא יָתֵיהּ וְכָל דִי לֵיהּ אֲרֵי עֲבַר עַל קְיָמָא דַייָ וַאֲרַי עֲבַד דְלָא כָשַׁר בְּיִשְׂרָאֵל:
וְאַקְדֵם יְהוֹשֻׁעַ בְּצַפְרָא וְקָרֵיב יַת יִשְׂרָאֵל לְשִׁבְטוֹהִי וְאִתְאַחַד שִׁבְטָא דְבֵית יְהוּדָה:
וְקָרֵיב יַת זַרְעֲיַת יְהוּדָה וְאַחַד יַת זַרְעֲיַת זָרַח וְקָרֵיב יַת זַרְעֲיַת זָרַח לְגַבְרַיָא וְאִתְאַחַד זַבְדִי:
וְקָרֵיב יַת בֵּיתֵיהּ לְגַבְרַיָא וְאִתְאַחַד עָכָן בַּר כַּרְמִי בַּר זַבְדִי בַּר זֶרַח לְשִׁבְטָא דִיהוּדָא:
וַאֲמַר יְהוֹשֻׁעַ לְעָכָן בְּרִי שַׁוִי כְעַן יְקָרָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל וְהַב קֳדָמוֹהִי הוֹדָאָה וְחַוֵי כְּעַן לִי מַה עֲבַדְתָּא לָא תְכַסֵי מִינִי:
וַאֲתֵיב עָכָן יַת יְהוֹשֻׁעַ וַאֲמַר בְּקוּשְׁטָא אֲנָא חָבִית קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל וּכְדֵין וּכְדֵין עֲבָדִית:
וַחֲזִית בְּבִזְתָא אִצְטְלִי בַּבְלִי חָד שַׁפִּיר וּמָאתָן סַלְעִין דִכְסַף וְלִישְׁנָא דְדַהֲבָא חָד חַמְשִׁין סַלְעִין מַתְקְלֵיהּ וְרִגַגְתִינוּן וּנְסֵבְתִּנוּן וְהָא אִנוּן טְמִירִין בְּאַרְעָא בְּגוֹ מַשְׁכְּנִי וְכַסְפָּא תְּחוֹתָא:
וּשְׁלַח יְהוֹשֻׁעַ אִזְגַדוּן וּרְהַטוּ לְמַשְׁכְּנָא וְהָא טְמִירִין בְּמַשְׁכְּנֵיהּ וְכַסְפָּא תְּחוֹתָא:
וּנְסִיבוּנוּן מִגוֹ מַשְׁכְּנָא וְאַיְתִיאוּנוּן לְוַת יְהוֹשֻׁעַ וּלְוַת כָּל בְּנֵי יִשְׂרָאֵל וְאַתִּיכוּנוּן קֳדָם יְיָ:
וּדְבַר יְהוֹשֻׁעַ יַת עָכָן בַּר זֶרַח וְיַת כַּסְפָּא וְיַת אִצְטְלָא וְיַת לִישָׁנָא דְדַהֲבָא וּדְבַר יַת בְּנוֹהִי וְיַת בְּנָתֵיהּ וְיַת תּוֹרֵיהּ וְיַת חֲמָרֵיהּ וְיַת עָנֵיהּ וְיַת מַשְׁכְּנֵיהּ וְיַת כָּל דִילֵיהּ וְכָל יִשְׂרָאֵל עִמֵיהּ וְאַסִיקוּ יַתְהוֹן לְמֵישַׁר עָכוֹר:
וַאֲמַר יְהוֹשֻׁעַ מָה עַכַרְתָּנָא יַעְכְּרִינָךְ יְיָ בְּיוֹמָא הָדֵין וּרְגָמוּ יָתֵיהּ כָּל יִשְׂרָאֵל בְּאַבְנַיָא וְאוֹקִידוּ יַתְהוֹן בְּנוּרָא בָּתַר דִרְגִימוּ יַתְהוֹן בְּאַבְנַיָא:
וְאַקִימוּ עֲלוֹהִי דְגוֹר אַבְנִין רַב עַד יוֹמָא הָדֵין וְתַב יְיָ מִתְקוֹף רוּגְזֵיהּ עַל כֵּן קְרָא שְׁמָא דְאַתְרָא הַהוּא מֵישַׁר עָכוֹר עַד יוֹמָא הָדֵין:

Chapter 8

וַאֲמַר יְיָ לִיהוֹשֻׁעַ לָא תִדְחַל וְלָא תִתְּבַּר דְבַר עִמָךְ יַת כָּל עַמָא עָבְדֵי קְרָבָא וְקוּם סַק לְעָי חֲזֵי דִמְסָרִית בִּידָךְ יַת מַלְכָּא דְעַי וְיַת עַמֵיהּ יַת קַרְתֵּיהּ וְיַת אַרְעֵיהּ:
וְתַעְבֵּיד לְעַי וּלְמַלְכָּה כְּמָא דַעֲבַדְתָּ לִירִיחוֹ וּלְמַלְכָּהּ לְחוֹד עֲדָאָהּ וּבְעִירָהּ תִּבְזוֹן לְכוֹן אַתְקַן לָךְ כְּמָנָא לְקַרְתָּא מֵאֲחוֹרָאָה:
וְקָם יְהוֹשֻׁעַ וְכָל עַמָא עָבְדֵי קְרָבָא לְמִיסַק לְעָי וּבְחַר יְהוֹשֻׁעַ תְּלָתִין אַלְפִין גַבְרָא גִבָּרֵי חֵילָא וְשַׁלְחִינוּן בְּלֵילְיָא:
וּפַקֵיד יַתְהוֹן לְמֵימָר חֲזוּ דְאַתּוּן כָּמְנִין לְקַרְתָּא מֵאֲחוֹרֵי קַרְתָּא לָא תַרְחִיקוּן מִן קַרְתָּא לַחֲדָא וּתְהוֹן כּוּלְכוֹן מְתַקְנַיָא:
וַאֲנָא וְכָל עַמָא דְעִמִי נִתְקָרֵב לְקַרְתָּא וִיהֵי אֲרֵי יִפְקוּן לִקְדָמֻתָנָא כִּדְבְּקַדְמֵיתָא וְנִפּוֹךְ קֳדָמֵיהוֹן:
וְיִפְקוּן בַּתְרָנָא עַד דְנִגוֹד יַתְהוֹן מִן קַרְתָּא אֲרֵי יֵימְרוּן תְּבִירִין קֳדָמָנָא כִּדְבְּקַדְמֵיתָא וְנִפּוֹךְ קֳדָמֵיהוֹן:
וְאַתּוּן תְּקוּמוּן מִכְּמָנָא וּתְתַרְכוּן יַת קַרְתָּא וְיִמְסְרְנָהּ יְיָ אֱלָהָכוֹן בְּיֶדְכוֹן:
וִיהֵי כְּמֵיחַדְכוֹן יַת קַרְתָּא תַּדְלְקוּן יַת קַרְתָּא בְּנוּרָא כְּפִתְגָמָא דַייָ תַּעַבְדוּן חֲזוֹ דְפַקְדִית יַתְכוֹן:
וּשְׁלָחִינוּן יְהוֹשֻׁעַ וַאֲזַלוּ לִכְמָנָא וִיתַבוּ בֵּין בֵּית אֵל וּבֵין עָי מַעֲרַב לְעָי וּבָת יְהוֹשֻׁעַ בְּלֵילְיָא הַהוּא בְּגוֹ עַמָא:
וְאַקְדִים יְהוֹשֻׁעַ בְּצַפְרָא וּמְנָא יַת עַמָא וּסְלִיק הוּא וְסָבֵי יִשְׂרָאֵל קֳדָם עַמָא לָעָי:
וְכָל עַמָא עָבְדֵי קְרָבָא דְעִמֵיהּ סְלִיקוּ וְאִתְקְרָבוּ וַאֲתוֹ לָקֳבֵל קַרְתָּא וּשְׁרוֹ מִצִפּוּנָא לָעַי וְחֵילְתָא בֵּינוֹהִי וּבֵין עָי:
וּדְבַר כְּחַמְשָׁא אַלְפִין גַבְרָא וְשַׁוִי יַתְהוֹן כְּמָנָא בֵּין בֵּית אֵל וּבֵין עָי מִמַעֲרָב לְעָי:
וְשַׁוִיאוּ עַמָא יַת כָּל מַשְׁרִיתָא דְמִצִפוּנָא לְקַרְתָּא וְיַת כְּמָנֵיהּ מַעֲרַב לְקַרְתָּא וַאֲזַל יְהוֹשֻׁעַ בְּלֵילְיָא הַהוּא בְּגוֹ מֵישְׁרָא:
וַהֲוָה כַּד חֲזָא מַלְכָּא דְעַי וְאוֹחִיאוּ וְאַקְדִימוּ וּנְפַקוּ אֱנָשֵׁי קַרְתָּא לִקְדָמוּת יִשְׂרָאֵל לְאַגָחָא קְרָבָא הוּא וְכָל עַמֵיהּ לְזִמְנָא דִמְתַקֵן לֵיהּ לִקְדָם מֵישְׁרָא וְאִינוּן לָא יָדְעִין אֲרֵי כְּמָנָא לְהוֹן מֵאֲחוֹרֵי קַרְתָּא:
וְאִתְּבָרוּ יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל קֳדָמֵיהוֹן וְאַפָּכוּ לְאוֹרַח מַדְבְּרָא:
וְאִתְכְּנִישׁוּ כָּל עַמָא דִי בְקַרְתָּא בָּעַי לְמִרְדַף בַּתְרֵיהוֹן וּרְדַפוּ בָּתַר יְהוֹשֻׁעַ וְאִתְנְגִידוּ מִן קַרְתָּא:
וְלָא אִשְׁתְּאַר אֱנַשׁ בְּעַי וּבֵית אֵל דְלָא נְפַקוּ בָּתַר יִשְׂרָאֵל וּשְׁבַקוּ יַת קַרְתָּא כַּד פְּתִיחָא וּרְדַפוּ בָּתַר יִשְׂרָאֵל:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ אֲרֵים בְּרוּמְחָא דִי בִידָךְ עַל עַי אֲרֵי בִּידָךְ אִמְסְרִינָה וַאֲרֵים יְהוֹשֻׁעַ בְּרוּמְחָא דְבִידֵיהּ עַל קַרְתָּא:
וּכְמָנָא קָם בִּפְרִיעַ מֵאַתְרֵיהּ וּרְהָטוּ כִּדְאָרֵים יְדֵיהּ וְעָלוּ לְקַרְתָּא וְכַבְּשׁוּהָ וְאוֹחִיאוּ וְאַדְלִיקוּ יַת קַרְתָּא בְּנוּרָא:
וְאִתְפְּנִיאוּ אֱנָשֵׁי עַי לַאֲחוֹרֵיהֶן וַחֲזוֹ וְהָא סְלִיק תְּנָנָא דְקַרְתָּא לְצֵית שְׁמַיָא וְלָא הֲוָה בְהוֹן חֵילָא לְמַעֲרָק לְכָא וּלְכָא וְעַמָא דְאַפֵיךְ לְמַדְבְּרָא אִתְחֲזַר עַל רָדְפָא:
וִיהוֹשֻׁעַ וְכָל יִשְׂרָאֵל חֲזוֹ אֲרֵי כְבַשׁ כְּמָנָא יַת קַרְתָּא וַאֲרֵי סְלֵיק תְּנָנָא דְקַרְתָּא וְתָבוּ וּמְחוֹ יַת אֱנָשֵׁי עָי:
וְאִלֵין נְפַקוּ מִן קַרְתָּא לִקְדָמוּתְהוּן וַהֲווֹ לְיִשְׂרָאֵל בִּמְצִיעָא אִלֵין מִכָּא וְאִלֵין מִכָּא וּמִחוֹ יַתְהוֹן עַד דְלָא אִשְׁתָּאַר לְהוֹן שְׁאֵר וּמְשֵׁזֵיב:
וְיַת מַלְכָּא דְעַי אֲחָדוּ כַּד חָי וְקָרִיבוּ יָתֵיהּ לְוַת יְהוֹשֻׁעַ:
וַהֲוָה כַּד שֵׁיצֵי יִשְׂרָאֵל לְקַטָלָא יַת כָּל יַתְבֵי עַי בְּחַקְלָא בְּמַדְבְּרָא דִי רְדָפוּנוּן בֵּיהּ וְאִתְקְטָלוּ כּוּלְהוֹן לְפִתְגַם דְחָרֶב עַד דִשְׁלִימוּ וְתָבוּ כָל יִשְׂרָאֵל לְעַי וּמְחוֹ יָתָהּ לְפִתְגַם דְחָרֶב:
וַהֲווֹ כָּל דְאִתְקְטָלוּ בְּיוֹמָא הַהוּא מִגְבַר וְעַד אִתְּתָא תְּרֵי עֲשַׂר אַלְפִין כֹּל אֱנָשֵׁי עָי:
וִיהוֹשֻׁעַ לָא אָתֵיב יְדֵיהּ דַאֲרִים בְּרוּמְחָא עַד דְגַמַר יַת כָּל יַתְבֵי עָי:
לְחוֹד בְּעִירָא וַעֲדִי קַרְתָּא הַהִיא בְּזוֹ לְהוֹן יִשְׂרָאֵל בְּפִתְגָמָא דַייָ דְפַקֵיד יַת יְהוֹשֻׁעַ:
וְאוֹקִיד יְהוֹשֻׁעַ יַת עָי וְשַׁוְיָהּ תֵּל חֲרִיב לְעָלַם לְצָדוּ עַד יוֹמָא הָדֵין:
וְיַת מַלְכָּא דְעַי צְלַב עַל צְלִיבָא עַד עִידַן רַמְשָׁא וּכְמֵיעַל שִׁמְשָׁא פַּקֵד יְהוֹשֻׁעַ וַאֲחִיתוּ יַת נִבְלָתֵיהּ מִן צְלִיבָא וּרְמוֹ יָתֵיהּ בְּמַעֲלָנָא תְּרַע קַרְתָּא וְאַקִימוּ עֲלוֹהִי דְגוֹר אַבְנִין רַב עַד יוֹמָא הָדֵין:
בְכֵן בְּנָא יְהוֹשֻׁעַ מַדְבְּחָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל בְּטוּרָא דְעֵיבָל:
כְּמָא דְפַקֵיד מֹשֶׁה עַבְדָא דַייָ יַת בְּנֵי יִשְׂרָאֵל כְּמָא דִכְתִיב בִּסְפַר אוֹרַיְיתָא דְמֹשֶׁה מַדְבַּח אַבְנִין שַׁלְמִין דְלָא אִתְּרַם עֲלֵיהוֹן בַּרְזְלָא וְאַסִיקוּ עֲלוֹהִי עֲלָוָן קֳדָם יְיָ וּנְכִיסוּ נִיכְסַת קוּדְשִׁין:
וּכְתַב תַּמָן עַל אַבְנַיָא יַת פַּתְשֶׁגֶן אוֹרַיְתָא דְמֹשֶׁה דִכְתַב קֳדָם בְּנֵי יִשְׂרָאֵל:
וְכָל יִשְׂרָאֵל וְסַבוֹהִי וְסַרְכוֹהִי וְדַיָנוֹהִי קַיְמִין מִכָּא וּמִכָּא לַאֲרוֹנָא קֳדָם כַּהֲנַיָא לֵוָאֵי נָטְלֵי אֲרוֹנָא דִקְיָמָא דַייָ כְּגִיוֹרָא כְּיַצִיבָא פַּלְגֵיהוֹן לָקֳבֵל טוּרָא דִגְרִיזִין וּפַלְגֵיהוֹן לָקֳבֵל טוּרָא דְעֵיבָל כְּמָא דְפַקֵיד מֹשֶׁה עַבְדָא דַייָ לְבָרָכָא יַת עַמָא יִשְׂרָאֵל בְּקַדְמֵיתָא:
וּבָתַר כֵּן קְרָא יַת כָּל פִּתְגָמֵי אוֹרַיְיתָא בִּרְכָן וּלְוָטִין כְּכָל דִכְתִיב בְּסִפְרָא דְאוֹרַיְתָא:
לָא הֲוָה פִּתְגָמָא מִכָּל דִי פַקֵיד מֹשֶׁה דְלָא קְרָא יְהוֹשֻׁעַ קֳדָם כָּל קְהָלָא דְיִשְׂרָאֵל וּנְשַׁיָא וְטַפְלָא וְגִיוֹרַיָא דְאָזְלִין בֵּינֵיהוֹן:

Chapter 9

וַהֲוָה כַּד שְׁמָעוּ כָּל מַלְכַיָא דִי בְּעִבְרָא דְיַרְדְנָא בְּטוּרָא וּבִשְׁפֶלְתָּא וּבְכָל סְפַר יַמָא רַבָּא דִי לָקֳבֵיל לִיבְנָן חִתָּאֵי וֶאֱמוֹרָאֵי כְּנַעֲנָאֵי פְּרִיזָאֵי חִוָאֵי וִיבוּסָאֵי:
וְאִתְכְּנָשׁוּ כַּחֲדָא לְאַגָחָא קְרָבָא עִם יְהוֹשֻׁעַ וְעִם יִשְׂרָאֵל סִיעָא חָדָא:
וְיַתְבֵי גִבְעוֹן שְׁמָעוּ יַת דִי עֲבַד יְהוֹשֻׁעַ לִירִיחוֹ וְלָעָי:
וַעֲבַדוּ אַף אִינוּן בְּחוּכְמָא וְאִזְדַוָדוּ וּנְסִיבוּ שַׂקִין בָּלָן לַחֲמָרֵיהוֹן וְזִיקִין דַחֲמָר בָּלָן וּמְבַזְעִין וּמְצָרְרִין:
וּמְסָנָן בָּלָן וּמְרַקְעִין בְּרִגְלֵיהוֹן וּכְסוּתָא בָלְיָא עֲלֵיהוֹן וְכָל לְחַם זְוָדֵיהוֹן יָבֵשׁ הֲוָה כִסְנִין:
וַאֲזַלוּ לְוַת יְהוֹשֻׁעַ לְמַשְׁרִיתָא לְגִלְגְלָא וַאֲמַרוּ לֵיהּ וּלְאֱנַשׁ יִשְׂרָאֵל מֵאַרְעָא רְחִיקָא אָתֵינָא וּכְעַן גְזָרוּ לָנָא קְיָם:
וַאֲמַרוּ אֱנַשׁ יִשְׂרָאֵל לְחִוָאָה דִלְמָא בֵּינָנָא אַתּוּן יַתְבוּן וְאֵיכְדֵין נִגְזַר לְכוֹן קְיָם:
וַאֲמַרוּ לִיהוֹשֻׁעַ עַבְדָךְ אֲנַחְנָא וַאֲמַר לְהוֹן יְהוֹשֻׁעַ מָן אַתּוּן וּמְנָן אֲתֵיתוּן:
וַאֲמַרוּ לֵיהּ מֵאַרַע רְחִיקָא לַחֲדָא אֲתוֹ עַבְדָךְ לִשְׁמָא דַייָ אֱלָהָךְ אֲרֵי שְׁמַעְנָא יַת שְׁמַע גְבוּרְתֵּיהּ וְיַת כָּל דַעֲבַד בְּמִצְרָיִם:
וְיַת כָּל דַעֲבַד לִתְרֵין מַלְכֵי אֱמוֹרָאֵי דִי בְּעִבְרָא דְיַרְדְנָא לְסִיחוֹן מַלְכָּא דְחֶשְׁבּוֹן וּלְעוֹג מַלְכָּא דְמַתְנַן דִבְעַשְׁתָּרוֹת:
וַאֲמַרוּ לָנָא סָבָנָא וְכָל יַתְבֵי אַרְעָנָא לְמֵימַר סָבוּ בְּיָדְכוֹן זְוָדִין לְאוֹרְחָא וֶאֱזִילוּ לִקֳדָמוּתְהוֹן וְתֵימְרוּן לְהוֹן עַבְדֵיכוֹן אֲנַחְנָא וּכְעַן גְזוּרוּ לָנָא קְיָם:
דֵין לַחְמָנָא כַּד חַמִים אִזְדְוָדְנָא יָתֵיהּ מִבָּתָּנָא בְּיוֹם מִפְקָנָא לְמֵיתֵי לְוַתְכוֹן וּכְעַן הָא יָבֵשׁ וַהֲוָה כִסְנִין:
וְאִלֵין זִיקִין דַחֲמַר דִי מְלֵינָא כַּד חַדְתִּין וְהָא אִתְבְּזָעוּ וְאִלֵין תּוֹתְבָנָא וּמְסָנָנָא בְּלִיאוּ מִסְגֵי אוֹרְחָא לַחְדָא:
וְקַבִּילוּ גַבְרַיָא לְפִּתְגָמֵיהוֹן וְאוּלְפַן מִן קֳדָם יְיָ לָא תְּבָעוּ:
וַעֲבַד לְהוֹן יְהוֹשֻׁעַ שְׁלָם וּגְזַר לְהוֹן קְיָם לְקַיָמוּתְהוֹן וְקַיְימוּ לְהוֹ רַבְרְבֵי כְּנִשְׁתָּא:
וַהֲוָה מִסוֹף תְּלָתָא יוֹמִין בָּתַר דִגְזָרוּ לְהוֹן קְיָם וּשְׁמָעוּ אֲרֵי קְרִיבִין אִינוּן לֵיהּ וּבֵינֵיהוֹן אִינוּן יָתְבִין:
וּנְטָלוּ בְנֵי יִשְׂרָאֵל וַאֲתוֹ לְקִרְוֵיהוֹן בְּיוֹמָא תְלִיתָאָה וְקִרְוֵיהוֹן גִבְעוֹן וּכְפִירָה וּבְאֵרוֹת וְקִרְיַת יְעָרִים:
וְלָא מְחוֹנוּן בְּנֵי יִשְׂרָאֵל אֲרֵי קַיְימוּ לְהוֹן רַבְרְבֵי כְנִשְׁתָּא בְּמֵימְרָא דַייָ אֱלָהָא דְיִשְׂרָאֵל וְאִתְרַעֲמוּ כָּל כְּנִשְׁתָּא עַל רַבְרְבַיָא:
וַאֲמַרוּ כָּל רַבְרְבַיָא לְכָל כְּנִשְׁתָּא אֲנַחְנָא קַיֵימְנָא לְהוֹן בְּמֵימְרָא דַייָ אֱלָהָא דְיִשְׂרָאֵל וּכְעַן לָא נִכּוּל לְאַנְזָקָא בְּהוֹן:
דָא נַעֲבֵּיד לְהוֹן וּנְקַיֵם יַתְהוֹן וְלָא יְהֵי עֲלָנָא רֻגְזָא עַל קְיָמָא דִי קַיָימְנָא לְהוֹן:
וַאֲמַרוּ לְהוֹן רַבְרְבַיָא יֵיחוֹן וַהֲווֹ מְלַקְטֵי אָעִין וּמְלָן מַיָא לְכָל כְּנִשְׁתָּא כְּמָא דְמַלִילוּ לְהוֹן רַבְרְבַיָא:
וּקְרָא לְהוֹן יְהוֹשֻׁעַ וּמַלֵיל עִמְהוֹן לְמֵימָר לְמָא שְׁקַרְתּוּן בָּנָא לְמֵימָר רְחִיקִין אֲנַחְנָא מִנְכוֹן לַחֲדָא וְאַתּוּן בֵּינָנָא יָתְבִין:
וּכְעַן לִיטִין אַתּוּן וְלָא יִפְסְקוּן מִנְכוֹן עַבְדִין וּמְלַקְטֵי אָעִין וּמָלָן מַיָא לְבֵית מַקְדְשָׁא דֶאֱלָהֵי:
וַאֲתִיבוּ יַת יְהוֹשֻׁעַ וַאֲמַרוּ אֲרֵי חַוָאָה אִתְחַוָא לְעַבְדָךְ יַת דְפַקֵד יְיָ אֱלָהָךְ יַת מֹשֶׁה עַבְדֵיהּ לְמִתַּן לְכוֹן יַת כָּל אַרְעָא וּלְשֵׁיצָאָה יַת כָּל יָתְבֵי אַרְעָא מִן קֳדָמֵיכוֹן וּדְחִילְנָא לַחֲדָא לְנַפְשָׁתָנָא מִן קֳדָמֵיכוֹן וַעֲבַרְנָא יַת פִּתְגָמָא הָדֵין:
וּכְעַן הָא אֲנַחְנָא מְסִירִין בִּידָךְ כְּמָא דְתַקִין וּכְמָא דְכָשַׁר בְּעֵינָךְ לְמֶעְבַּד לָנָא עַבָד:
וַעֲבַד לְהוֹן כֵן וְשֵׁזִיב יַתְהוֹן מִיַד בְּנֵי יִשְׂרָאֵל וְלָא קַטְלוּנוּן:
וּמַנִינוּן יְהוֹשֻׁעַ בְּיוֹמָא הַהוּא לָקְטֵי אָעִין וּמָלָן מַיָא לִכְנִשְׁתָּא וּלְמַדְבְּחָא דַייָ עַד יוֹמָא הָדֵין לְאַתְרָא דִי יִתְרְעֵי:

Chapter 10

וַהֲוָה כַּד שְׁמַע אֲדֹנִי צֶדֶק מַלְכָּא דִירוּשְׁלֵם אֲרֵי כְּבַשׁ יְהוֹשֻׁעַ יַת עַי וְגַמְרָהּ כְּמָא דַעֲבַד לִירִיחוֹ וּלְמַלְכָּהּ כֵּן עֲבַד לָעַי וּלְמַלְכָּה וַאֲרֵי אַשְׁלִימוּ יַתְבֵי גִבְעוֹן עִם יִשְׂרָאֵל וַהֲווֹ בֵּינֵיהוֹן:
וּדְחִילוּ לַחֲדָא אֲרֵי קַרְתָּא רַבְּתָא גִבְעוֹן כַּחֲדָא מִקִרְוֵי מַלְכְּוָתָא וַאֲרֵי הִיא רַבָּא מִן עָי וְכָל גַבְרָהָא גִבָּרִין:
וּשְׁלַח אֲדֹנִי צֶדֶק מַלְכָּא דִירוּשְׁלֵם לְוַת הוֹהָם מַלְכָּא דְחֶבְרוֹן וּלְוַת פִּרְאָם מַלְכָּא דְיַרְמוּת וּלְוַת יָפִיעַ מַלְכָּא דְלָכִישׁ וּלְוַת דְבִיר מַלְכָּא דְעֶגְלוֹן לְמֵימָר:
סַקוּ לְוָתִי וְסַעֲדוּנִי וְנִמְחֵי יַת יַתְבֵי גִבְעוֹן אֲרֵי אַשְׁלֵמַת עִם יְהוֹשֻׁעַ וְעִם בְּנֵי יִשְׂרָאֵל:
וְאִתְכְּנָשׁוּ וּסְלִיקוּ חַמְשָׁא מַלְכֵי אֱמוֹרָאָה מַלְכָּא דִירוּשְׁלֵם מַלְכָּא דְחֶבְרוֹן מַלְכָּא דְיַרְמוּת מַלְכָּא דְלָכִישׁ מַלְכָּא דְעֶגְלוֹן אִינוּן וְכָל מַשִׁרְיַתְהוֹן וּשְׁרוֹ עַל גִבְעוֹן וְאַגִיחוּ קְרָבָא עֲלָה:
וּשְׁלַחוּ אֲנָשֵׁי גִבְעוֹן לְוַת יְהוֹשֻׁעַ לְמַשְׁרִיתָא לְגִלְגָלָא לְמֵיְמַר לָא תְרַשֵׁל יְדָךְ מַעֲבְדָךְ סַק לְוָתָנָא בִּפְרִיעַ וּפְרוֹק לָנָא וְסַעֲדָנָא אֲרֵי אִתְכְּנָשׁוּ עֲלָנָא כָּל מַלְכֵי אֱמוֹרָאָה יַתְבֵי טוּרָא:
וּסְלֵיק יְהוֹשֻׁעַ מִן גִלְגָלָא הוּא וְכָל עַמָא עָבְדֵי קְרָבָא עִמֵיהּ וְכָל גִבָּרֵי חֵילָא:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ לָא תִדְחַל מִנְהוֹן אֲרֵי בִּידָךְ מְסַרְתִּינוּן לָא יִתְעַתַּד אֱנַשׁ מִנְהוֹן קָדָמָךְ:
וַאֲתָא לְוַתְהוֹן יְהוֹשֻׁעַ בִּתְכֵּיף כָּל לֵילְיָא סְלִיק מִן גִלְגְלָא:
וְתַבְּרִינוּן יְיָ קֳדָם יִשְׂרָאֵל וּמְחָנוּן מַחֲתָא רַבָּא בְּגִבְעוֹן וּרְדָפוּנוּן בְּאוֹרַח מַסְקְנָא דְבֵית חוֹרוֹן וּמְחָנוּן עַד עֲזֵקָה וְעַד מַקֵדָה:
וַהֲוָה בְּמִפְּכֵיהוֹן מִן קֳדָם יִשְׂרָאֵל אִינוּן בְּמַחְתָנָא דְבֵית חוֹרוֹן וּמִן קֳדָם יְיָ אִתְרְמִיאָה עֲלֵיהוֹן אַבְנִין רַבְרְבָן מִן שְׁמַיָא עַד דַאֲתוֹ לַעֲזֵקָה וּמִיתוּ סַגִיאִין דְמִיתוּ בְּאַבְנֵי בַרְדָא מִדְקַטוּלוּ בְּנֵי יִשְׂרָאֵל בְּחַרְבָּא:
בְּכֵן שַׁבַּח יְהוֹשֻׁעַ קֳדָם יְיָ בְּיוֹמָא דִמְסַר יְיָ יַת אֱמוֹרָאָה קֳדָם בְּנֵי יִשְׂרָאֵל וַאֲמַר לְעֵינֵי יִשְׂרָאֵל שִׁמְשָׁא בְּגִבְעוֹן אוֹרִיךְ וְסִיהֲרָא בְּמֵישַׁר אַיָלוֹן:
וְאוֹרְיַךְ שִׁמְשָׁא וְסִיהֲרָה קָם עַד דְאִתְפְּרַע עַמָא יִשְׂרָאֵל מִבַּעֲלֵי דְבָבֵיהוֹן הֲלָא הִיא כְתִיבָא עַל סִפְרָא דְאוֹרַיְיתָא וְקָם שִׁמְשָׁא בְּפַלְגוּת שְׁמַיָא וְלָא דָחַק לְמֵיעַל כְּיוֹם שְׁלִים:
וְלָא הֲוָה כְּיוֹמָא הַהוּא קֳדָמוֹהִי וּבַתְרוֹהִי דְאִתְקַבָּלַת קֳדָם יְיָ צְלוֹת אֱנָשׁ אֲרֵי בְּמֵימְרֵיהּ יְיָ מֵגִיחַ קְרָבָא לְיִשְׂרָאֵל:
וְתָב יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ לְמַשְׁרִיתָא לְגִלְגָלָא:
וְאַפָּכוּ חַמְשָׁא מַלְכַיָא הָאִלֵין וְאִטַמָרוּ בִּמְעַרְתָּא בְּמַקֵדָה:
וְאִתְחַוָא לִיהוֹשֻׁעַ לְמֵימָר אִשְׁתְּכָחוּ חַמְשָׁא מַלְכַיָא טְמִירִין בִּמְעַרְתָּא בְּמַקֵדָה:
וַאֲמַר יְהוֹשֻׁעַ קְרִיבוּ אַבְנִין רַבְרְבִין קֳדָם פּוּמָא דִמְעַרְתָּא וּמַנוֹ עֲלָהּ גַבְרִין לְמִטַרְהוֹן:
וְאַתּוּן לָא תְקוּמוּן רְדוּפוּ בָּתַר בַּעֲלֵי דְבָבֵיכוֹן וְתִדְבְּקוּן יַתְהוֹן לָא תִשְׁבְּקוּנוּן לְמֵיעַל לְקִרְוֵיהוֹן אֲרֵי מְסָרִינוּן יְיָ אֱלָהָכוֹן בִּידֵיכוֹן:
וַהֲוָה כַּד שֵׁיצֵי יְהוֹשֻׁעַ וּבְנֵי יִשְׂרָאֵל לְמִמְחֵיהוֹן מְחָתָא רַבְּתָא לַחֲדָא עַד דִשְׁלִימוּ וּמְשֵׁיזְבַיָא אִשְׁתֵּזִיבוּ מִנְהוֹן וְעָלוּ לְקִרְוִין כְּרִיכָן:
וְתָבוּ כָל עַמָא לְמַשְׁרִיתָא לְוַת יְהוֹשֻׁעַ לְמַקֵדָה בִּשְׁלָם לָא הֲוָה נִזְקָא לִבְנֵי יִשְׂרָאֵל לְמִדְחַק גְבַר יַת נַפְשֵׁיהּ:
וַאֲמַר יְהוֹשֻׁעַ פְּתָחוּ יַת פּוּמָא דִמְעַרְתָּא וְאַפִּיקוּ לְוָתִי יַת חַמְשָׁא מַלְכַיָא הָאִלֵין מִן מְעַרְתָּא:
וַעֲבַדוּ כֵן וְאַפִּיקוּ לְוָתֵיהּ יַת חַמְשָׁא מַלְכַיָא הָאִלֵין מִן מְעַרְתָּא יַת מַלְכָּא דִירוּשְׁלֵם יַת מַלְכָּא דְחֶבְרוֹן יַת מַלְכָּא דְיַרְמוּת יַת מַלְכָּא דְלָכִישׁ יַת מַלְכָּא דְעֶגְלוֹן:
וַהֲוָה כַּד אַפִּיקוּ יַת מַלְכַיָא הָאִלֵין לְוַת יְהוֹשֻׁעַ וּקְרָא יְהוֹשֻׁעַ לְכָל אֱנָשׁ יִשְׂרָאֵל וַאֲמַר לְשִׁלְטוֹנֵי גַבְרֵי מְגִיחֵי קְרָבָא דַאֲזַלוּ עִמֵיהּ קִרְבוּ שַׁווּ יַת רִגְלֵיכוֹן עַל צַוְארֵי מַלְכַיָא הָאִלֵין וּקְרִיבוּ וְשַׁוִיאוּ יַת רִגְלֵיהוֹן עַל צַוְארֵיהוֹן:
וַאֲמַר לְהוֹן יְהוֹשֻׁעַ לָא תִדַחֲלוּן וְלָא תִתַּבְּרוּן תַּקִיפוּ וַעֲלִימוּ אֲרֵי כְּדֵין יַעֲבֵּד יְיָ לְכָל בַּעֲלֵי דְבָבֵיכוֹן דְאַתּוּן מַגִיחִין קְרָבָא בְהוֹן:
וּמְחָנוּן יְהוֹשֻׁעַ בָּתַר כֵּן וְקַטְלִינוּן וְצַלְבִינוּן עַל חַמְשָׁא צְלִיבִין וַהֲווֹ צְלִיבִין עַל צְלִיבַיָא עַד רַמְשָׁא:
וַהֲוָה לְעִידָן מֵיעַל שִׁמְשָׁא פַּקֵד יְהוֹשֻׁעַ וְאַחִיתוּנוּן מֵעַל צְלִיבַיָא וּרְמוֹנוּן לִמְעַרְתָּא דְאִטַמָרוּ תַמָן וְשַׁוִיאוּ אַבְנִין רַבְרְבָן עַל פּוּמָא דִמְעַרְתָּא עַד כְּרַן יוֹמָא הָדֵין:
וְיַת מַקֵדָה כְּבַשׁ יְהוֹשֻׁעַ בְּיוֹמָא הַהוּא וּמְחָא לְפִתְגַם דְחָרֶב וְיַת מַלְכָּא גַמַר יַתְהוֹן וְיַת כָּל נַפְשָׁתָא דִי בָהּ לָא אַשְׁאֵר מְשֵׁזִיב וַעֲבַד לְמַלְכָּא דְמַקֵדָה כְּמָא דַעֲבַד לְמַלְכָּא דִירִיחוֹ:
וַעֲבַר יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ מִמַקֵדָה לְלִבְנָה וְאַגִיחַ קְרָבָא עִם לִבְנָה:
וּמְסַר יְיָ אַף יָתָהּ בִּידָא דְיִשְׂרָאֵל וְיַת מַלְכָּהּ וּמְחָהּ לְפִּתְגָם דְחָרֶב וְיַת כָּל נַפְשָׁתָא דִי בָּהּ לָא אַשְׁאַר בָּהּ מְשֵׁזִיב וַעֲבַד לְמַלְכָּא כְּמָא דַעֲבַד לְמַלְכָּהּ דִירִיחוֹ:
וַעֲבַר יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ מִלִבְנָה לְלָכִישׁ וּשְׁרָא עֲלָהּ וְאַגִיחַ קְרָבָא בָּהּ:
וּמְסַר יְיָ יַת לָכִישׁ בִּידָא דְיִשְׂרָאֵל וְכַבְשָׁהּ בְּיוֹמָא תִנְיָנָא וּמְחָהּ לְפִתְגַם דְחָרֶב וְיַת כָּל נַפְשָׁתָא דְבָהּ כְּכָל דַעֲבַד לְלִבְנָה:
בְּכֵן סְלִיק הוֹרָם מַלְכָּא דְגֶזֶר לְמִסְעַד יַת לָכִישׁ וּמְחָהִי יְהוֹשֻׁעַ וְיַת עַמֵיהּ עַד דְלָא אִשְׁתָּאַר לֵיהּ מְשֵׁזִיב:
וַעֲבַר יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ מִלָכִישׁ לְעֶגְלוֹן וּשְׁרוֹ עֲלָהּ וְאַגִיחוּ קְרָבָא עֲלָהּ:
וְכַבְשׁוּהָ בְּיוֹמָא הַהוּא וּמְחוּהָא לְפִתְגָם דְחָרֶב וְיַת כָל נַפְשָׁתָא דְבָהּ בְּיוֹמָא הַהוּא גַמַר כְּכֹל דַעֲבַד לְלָכִישׁ:
וּסְלִיק יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ מֵעֶגְלוֹן לְחֶבְרוֹן וְאַגִיחוּ קְרָבָא עֲלָהּ:
וְכַבְשׁוּהָא וּמְחוּהָא לְפִתְגָם דְחָרֶב וְיַת מַלְכָּהּ וְיַת כָּל קִרְוָהָא וְיַת כָּל נַפְשָׁתָא דְבָהּ לָא אַשְׁאַר מְשֵׁזִיב כְּכֹל דַעֲבַד לְעֶגְלוֹן וְגַמַר יָתָהּ וְיַת כָּל נַפְשָׁתָא דְבָהּ:
וְתַב יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ לִדְבִיר וְאַגִיחַ קְרָבָא עֲלָהּ:
וְכַבְשָׁהּ וְיַת מַלְכָּהּ וְיַת כָּל קִרְוָהָא וּמְחוֹנוּן לְפִּתְגַם דְחָרֶב וּגְמִירוּ יַת כָּל נַפְשָׁתָא דִי בָהּ לָא אַשְׁאַר מְשֵׁזִיב כְּמָא דַעֲבַד לְחֶבְרוֹן כֵּן עֲבַד לִדְבִיר וּלְמַלְכָּהּ וּכְמָא דַעֲבַד לְלִבְנָה וּלְמַלְכָּהּ:
וּמְחָא יְהוֹשֻׁעַ יַת כָּל אַרְעָא טוּרָא וְדָרוֹמָא וּשְׁפֶלְתָּא וּמַשְׁפַּךְ מְרָמָתָא וְיַת כָּל מַלְכֵיהוֹן לָא אַשְׁאַר מְשֵׁזִיב וְיַת כָּל נִשְׁמָא גַמַר כְּמָא דְפַקֵד יְיָ אֱלָהָא דְיִשְׂרָאֵל:
וּמְחָנוּן יְהוֹשֻׁעַ מֵרְקָם גֵיאָה וְעַד עַזָה וְיַת כָּל אֲרַע גּוֹשֶׁן וְעַד גִבְעוֹן:
וְיַת כָּל מַלְכַיָא הָאִילֵין וְיַת אַרְעֲהוֹן כְּבַשׁ יְהוֹשֻׁעַ בְּזִמְנָא חָדָא אֲרֵי יְיָ אֱלָהָא דְיִשְׂרָאֵל בְּמֵימְרֵיהּ אַגִיחַ קְרָבָא לְיִשְׂרָאֵל:
וְתַב יְהוֹשֻׁעַ וְכָל יִשְׂרָאֵל עִמֵיהּ לְמַשְׁרִיתָא לְגִלְגָלָא:

Chapter 11

וַהֲוָה כַּד שְׁמַע יָבִין מַלְכָּא דְחָצוֹר וּשְׁלַח לְוַת יוֹבָב מַלְכָּא דְמָדוֹן וּלְוַת מַלְכָּא דְשִׁמְרוֹן וּלְוַת מַלְכָּא דְאַכְשָׁף:
וּלְוַת מַלְכַיָא דְמִצְפּוּנָא בְּטוּרָא וּבְמֵישְׁרָא דָרוֹם גִינוֹסָר וּבִשְׁפֶלְתָּא וּבְפִלְכֵי דוֹר מַעֲרָבָא:
כְּנַעֲנָאָה מִמַדִינְחָא וּמִמַעֲרָבָא וֶאֱמוֹרָאָה וְחִתָּאָה וּפְרִיזָאָה וִיבוּסָאָה בְּטוּרָא וְחִוָאָה בְּשִׁפּוּלֵי חֶרְמוֹן בַּאֲרַע מִצְפַּיָא:
וּנְפָקוּ אִינוּן וְכָל מַשִׁרְיַתְהוֹן עִמְהוֹן עַם סַגִי כְּחָלָא דְעַל כֵּיף יַמָא לִסְגֵי וְסוּסְוַן וּרְתִיכוּן סַגִי לַחֲדָא:
וְאִזְדַמְנוּ כֹּל מַלְכַיָא הָאִילֵין וַאֲתוֹ וּשְׁרוֹ כַחְדָא עַל מֵי מֵרוֹם לְאַגָחָא קְרָבָא עִם יִשְׂרָאֵל:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ לָא תִדְחַל מִן קֳדָמֵיהוֹן אֲרֵי מְחַר כְּעִידָנָא הָדֵין אֲנָא מְסַר יַת כּוּלְהוֹן קְטִילִין קֳדָם יִשְׂרָאֵל יַת סוּסְוָתְהוֹן תְּעַקֵר וְיַת רְתִיכֵיהוֹן תּוֹקֵד בְּנוּרָא:
וַאֲתָא יְהוֹשֻׁעַ וְכָל עַמָא עָבְדֵי קְרָבָא עִמֵיהּ עֲלֵיהוֹן וְאִינוּן שָׁרָן עַל מֵי מֵרוֹם בִּתְכֵיף וּנְפַלוּ בְהוֹן:
וּמַסְרִינוּן יְיָ בִּידָא דְיִשְׂרָאֵל וּמְחוּנוּן וּרְדָפוּנוּן עַד צִידוֹן רַבְּתָא וְעַד חַרְצֵי יַמָא וְעַד בִּקְעַת מִצְפַּיָא מַדִינְחָא וּמְחוּנוּן עַד דְלָא אִישְׁתָּאַר לְהוֹן מְשֵׁזִיב:
וַעֲבַד לְהוֹן יְהוֹשֻׁעַ כְּמָא דַאֲמַר לֵיהּ יְיָ יַת סוּסְוָתְהוֹן עִקַר וְיַת רְתִיכֵיהוֹן אוֹקֵד בְּנוּרָא:
וְתַב יְהוֹשֻׁעַ בְּעִדָנָא הַהִיא וּכְבַשׁ יַת חָצוֹר וְיַת מַלְכָּהּ קְטַל בְּחַרְבָּא אֲרֵי חָצוֹר מִלְקַדְמִין הִיא רֵישׁ כָּל מַלְכְּוָתָא הָאִלֵין:
וּמְחוֹ יַת כָּל נַפְשָׁתָא דִי בָהּ לְפִתְגַם דְחָרֶב גְמִירָא לָא אַשְׁאַר כָּל נִשְׁמָא וְיַת חָצוֹר אוֹקֵד בְּנוּרָא:
וְיַת כָּל קִרְוֵי מַלְכַיָא הָאִילֵין וְיַת כָּל מַלְכֵיהוֹן כְּבַשׁ יְהוֹשֻׁעַ וּמְחָנוּן לְפִתְגַם דְחָרֶב גַמַר יַתְהוֹן כְּמָא דְפַקֵד מֹשֶׁה עַבְדָא דַיָי:
לְחוֹד כָּל קִרְוַיָא דְקַיְמִין עַל תּוּקְפֵיהוֹן לָא אוֹקְדִינוּן יִשְׂרָאֵל אֱלָהֵין יַת חָצוֹר בִּלְחוֹדָהּ אוֹקֵיד יְהוֹשֻׁעַ:
וְכָל עֲדֵי קִרְוַיָא הָאִלֵין וּבְעִירָא בְּזוֹ לְהוֹן בְּנֵי יִשְׂרָאֵל לְחוֹד יַת כָּל אֱנָשָׁא מְחוֹ לְפִתְגַם דְחָרֶב עַד דִי שֵׁיצִיאוּ יַתְהוֹן לָא אַשְׁאָרוּ כָּל נִשְׁמָא:
כְּמָא דְפַקֵיד יְיָ יַת מֹשֶׁה עַבְדֵיהּ כֵן פַּקֵד מֹשֶׁה יַת יְהוֹשֻׁעַ וְכֵן עֲבַד יְהוֹשֻׁעַ לָא בַטֵיל פִּתְגָמָא מִכָּל דְפַקֵיד יְיָ יַת מֹשֶׁה:
וְאַחְסִין יְהוֹשֻׁעַ יַת כָּל אַרְעָא הָדָא טוּרָא וְיַת כָּל דָרוֹמָא וְיַת כָּל אַרְעָא דְגוֹשֶׁן וְיַת שְׁפֶלְתָּא וְיַת מֵישְׁרָא וְיַת טוּרָא דְיִשְׂרָאֵל וּשְׁפֵלָתֵיהּ:
מִן טוּרָא פְּלִיגָא דְסָלִיק לְשֵׂעִיר וְעַד מֵישַׁר גָד בְּבִקְעַת לִבְנָן שְׁפּוּלֵי טוּרָא דְחֶרְמוֹן וְיַת כָּל מַלְכֵיהוֹן כְּבַשׁ וּמְחָנוּן וְקַטְלִינוּן:
יוֹמִין סַגִיאִין עֲבַד יְהוֹשֻׁעַ יַת כָּל מַלְכַיָא הָאִלֵין קְרָבָא:
לָא הֲוַת קַרְתָּא דְאַשְׁלֵימַת עִם בְּנֵי יִשְׂרָאֵל אֱלָהֵן חִוָאָה יַתְבֵי גִבְעוֹן יַת כּוֹלָא נְסִיבוּ בִּקְרָבָא:
אֲרֵי מִן קֳדָם יְיָ הֲוַת לְתַקָפָא יַת לִבְּהוֹן לְסַדָרָא קְרָבָא עִם יִשְׂרָאֵל בְּדִיל לְגַמָרוּתְהוֹן בְּדִיל דְלָא לְמֶהֱוֵי לְהוֹן רַחֲמִין אֲרֵי בְּדִיל לְשֵׁצָיוּתְהוֹן כְּמָא דְפַקֵד יְיָ יַת מֹשֶׁה:
וַאֲתָא יְהוֹשֻׁעַ בְּעִדָנָא הַהִיא וְשֵׁיצֵי יַת גִבָּרַיָא מִן טוּרָא מִן חֶבְרוֹן מִן דְבִיר מִן עֲנָב וּמִכֹּל טוּרָא דִיהוּדָה וּמִכֹּל טוּרָא דְיִשְׂרָאֵל עִם קִרְוֵיהוֹן גַמְרוּנוּן יְהוֹשֻׁעַ:
לָא אִשְׁתָּאַר גִבָּרַיָא בַּאֲרַע בְּנֵי יִשְׂרָאֵל לְחוֹד בְּעַזָה בְּגַת וּבְאַשְׁדוֹד אִשְׁתָּאֲרוּ:
וּנְסִיב יְהוֹשֻׁעַ יַת כָּל אַרְעָא כְּכֹל דִי מַלֵיל יְיָ עִם מֹשֶׁה וְיַהֲבָהּ יְהוֹשֻׁעַ לְאַחֲסָנָא לְיִשְׂרָאֵל כְּפַלְגוּתֵיהוֹן לְשִׁבְטֵיהוֹן וְאַרְעָא שָׁדוֹכָת מִלְמֶעְבַּד קְרָבָא:

Chapter 12

וְאִלֵין מַלְכֵי אַרְעָא דִמְחוֹ בְּנֵי יִשְׂרָאֵל וִירִיתוּ יַת אַרְעֲהוֹן בְּעִבְרָא דְיַרְדְנָא מַדְנַח שִׁמְשָׁא מִנַחֲלָא דְאַרְנוֹן עַד טוּרָא דְחֶרְמוֹן וְכָל מֵישְׁרָא מַדִינְחָא:
סִיחוֹן מַלְכָּא אֱמוֹרָאָה דְיָתֵיב בְּחֶשְׁבּוֹן שַׁלִיט מֵעֲרוֹעֵר דְעַל כֵּיף נַחֲלָא דְאַרְנוֹן וְגוֹ נַחֲלָא וּפַלְגוּת אֲרַע גִלְעָד וְעַד יוּבְקָא דְנַחֲלָא תְּחוּמָא דִבְנֵי עַמוֹן:
וּמֵישְׁרָא עַד יַם גִינוֹסַר מַדִינְחָא וְעַד יַמָא דְמֵישְׁרָא יַמָא דְמִלְחָא מַדִינְחָא אוֹרַח בֵּית יְשִׁמוֹת וּמִדָרוֹמָא תְּחוֹת מַשְׁפַּךְ מֵרָמָתָא:
וּתְחוּם עוֹג מַלְכָּא דְמַתְנָן מִשְׁאַר גִבָּרַיָא דְיָתֵיב בְּעַשְׁתָּרוֹת וּבְאֶדְרְעִי:
וְשַׁלִיט בְּטוּרָא דְחֶרְמוֹן וּבְסַלְכָה וּבְכָל מַתְנָן עַד תְּחוּם גְשׁוּרָאָה וְאַפְקֵירוֹס וּפַלְגוּת אֲרַע דְגִלְעָד תְּחוּם סִיחוֹן מַלְכָּא דְחֶשְׁבוֹן:
מֹשֶׁה עַבְדָא דַייָ וּבְנֵי יִשְׂרָאֵל מְחוּנוּן וְיַהְבָהּ מֹשֶׁה עַבְדָא דַייָ יְרוּתָא לְשִׁבְטָא דִרְאוּבֵן וּלְשִׁבְטָא דְגָד וּלְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה:
וְאִלֵין מַלְכֵי אַרְעָא דִי מְחָא יְהוֹשֻׁעַ וּבְנֵי יִשְׂרָאֵל בְּעִבְרָא דְיַרְדְנָא מַעַרְבָא מִמֵישַׁר גָד בְּבִקְעַת לִבְנָן וְעַד טוּרָא פְלִיגָא דְסָלִיק לְשֵׂעִיר וְיַהֲבָהּ יְהוֹשֻׁעַ לְשִׁבְטָא יִשְׂרָאֵל יְרוּתָא כְּפַלְגוּתְהוֹן:
בְּטוּרָא וּבַשְׁפֶלְתָּא וּבְמֵישְׁרָא וּבְמַשְׁפַּךְ מְרָמָתָא וּבְמַדְבְּרָא וּבְדָרוֹמָא חִתָּאֵי וֶאֱמוֹרָאֵי וּכְנַעֲנָאֵי וּפְרִיזָאֵי וְחִוָאֵי וִיבוּסָאֵי:
מַלְכָּא דִירִיחוֹ חָד מַלְכָּא דְעַי דְבִסְטַר בֵּית אֵל חָד:
מַלְכָּא דִירוּשְׁלֵם חָד מַלְכָּא דְחֶבְרוֹן חָד:
מַלְכָּא דְיַרְמוּת חָד מַלְכָּא דְלָכִישׁ חָד:
מַלְכָּא דְעֶגְלוֹן חָד מַלְכָא דְגֶזֶר חָד:
מַלְכָּא דִדְבִיר חָד מַלְכָּא דְגֶדֶר חָד:
מַלְכָּא דְחָרְמָה חָד מַלְכָּא דַעֲרָד חָד:
מַלְכָּא דְלִבְנָא חָד מַלְכָּא דַעֲדֻלָם חָד:
מַלְכָּא דְמַקֵדָה חָד מַלְכָּא דְבֵית אֵל חָד:
מַלְכָּא דְתַּפּוּחַ חָד מַלְכָּא דְחֵפֶר חָד:
מַלְכָּא דַאֲפֵק חָד מַלְכָּא דְלַשָׁרוֹן חָד:
מַלְכָּא דְמָדוֹן חָד מַלְכָּא דְחָצוֹר חָד:
מַלְכָּא דְשִׁמְרוֹן מְראוֹן חָד מַלְכָּא דְאַכְשָׁף חָד:
מַלְכָּא דְתַעְנַךְ חָד מַלְכָּא דִמְגִדוֹ חָד:
מַלְכָּא דְקֶדֶשׁ חָד מַלְכָּא דְיָקְנְעָם לְכַרְמְלָא חָד:
מַלְכָּא דְדוֹר לְפִלְכֵי דוֹר חָד מַלְכָּא דְעַמְמִין לְגִלְגָלָא חָד:
מַלְכָּא דְתִרְצָה חָד כֹּל מַלְכַיָא תְּלָתִין וְחָד:

Chapter 13

וִיהוֹשֻׁעַ סִיב עָל בְּיוֹמִין וַאֲמַר יְיָ לֵיהּ אַתְּ סַבְתָּא עַלְתָא בְיוֹמִין וְאַרְעָא אִשְׁתָּאֲרַת סַגִיאָה לַחֲדָא לִמֵירְתָהּ:
דָא אַרְעָא דְאִשְׁתָּאֲרַת כָּל תְּחוּמֵי פְּלִשְׁתָּאֵי וְכָל גְשׁוּרָאֵי:
מִן שִׁיחוֹר דִי עַל אַפֵּי מִצְרַיִם וְעַד תְּחוּם עֶקְרוֹן צִפּוּנָא עַל אַרְעָא כְנַעֲנָאֵי תִּתְמְנִי חַמְשָׁא טוּרְנֵי פְּלִשְׁתָּאֵי עֲזָתָאֵי וְאַשְׁדוֹדָאֵי אַשְׁקְלוֹנָאֵי גִתָּאֵי וְעֶקְרוֹנָאֵי וְעַוָאֵי:
מִדָרוֹמָא כָּל אַרְעָא כְנַעֲנָאֵי וּמְעַרְתָּא דִי לְצִידוֹנָאָה עַד אֲפֵק עַד תְחוּם אֱמוֹרָאָה:
וְאַרְעָא גִבְלִי וְכָל לִבְנָן מִדְנַח שִׁמְשָׁא מִמֵישַׁר גָד שִׁפּוּלֵי טוּרָא דְחֶרְמוֹן עַד מַעֲלָנָא דַחֲמָת:
כָּל יַתְבֵי טוּרָא מִן לִבְנַן עַד חַרְצֵי יַמָּא כָּל צִידוֹנָאֵי בְּמֵימְרִי אֲתָרְכִינוּן מִן קֳדָם בְּנֵי יִשְׂרָאֵל לְחוֹד פְּלִיגְהָא לְיִשְׂרָאֵל בְּאַחֲסָנָא כְּמָא דְפַקִידְתָּךְ:
וּכְעַן פַּלֵיג יַת אַרְעָא הָדָא בְּאַחְסָנָא לְתִשְׁעָה שִׁבְטִין וּפַלְגוּת שִׁבְטָא דִמְנַשֶׁה:
עִמֵיהּ שִׁבְטָא דִרְאוּבֵן וְשִׁבְטָא גָד קַבִּילוּ אַחֲסַנְתְּהוֹן דִיהַב לְהוֹן מֹשֶׁה בְּעִבְרָא דְיַרְדְנָא מַדִינְחָא כְּמָא דִיהַב לְהוֹן מֹשֶׁה עַבְדָא דַיָי:
מֵעֲרוֹעֵר דְעַל כֵּיף נַחֲלָא דְאַרְנוֹן וְקַרְתָּא דִי בְגוֹ נַחֲלָא וְכָל מֵישַׁר מֵידְבָא עַד דִיבוֹן:
וְכָל קִרְוֵי סִיחוֹן מַלְכָּא דֶאֱמוֹרָאָה דִי מְלַךְ בְּחֶשְׁבּוֹן עַד תְּחוּם בְּנֵי עַמוֹן:
וְאַרְעָא גִלְעָד וּתְחוּם גְשׁוּרָאָה וַאֲפִיקֵירוֹס וְכָל טוּרָא דְחֶרְמוֹן וְכָל מַתְנָן עַד סַלְכָה:
כָּל מַלְכוּת עוֹג בְּמַתְנָן דִי מְלַךְ בְּעַשְׁתָּרוֹת וּבְאֶדְרֶעִי וְהוּא אִשְׁתָּאַר מִשְׁאַר גִבָּרַיָא וּמְחָנוּן מֹשֶׁה וְתַרְכִינוּן:
וְלָא תָרִיכוּ בְּנֵי יִשְׂרָאֵל יַת גְשׁוּרָאָה וְיַת אֲפִיקֵירוֹס וִיתֵיב גְשׁוּרָאָה וַאֲפִיקֵירוֹס בְּגוֹ יִשְׂרָאֵל עַד יוֹמָא הָדֵין:
לְחוֹד לְשִׁבְטָא דְלֵוִי לָא יְהַב אַחְסָנָא קוּרְבָּנַיָא דַייָ אֱלָהָא דְיִשְׂרָאֵל אִינוּן אַחֲסַנְתֵּיהּ כְּמָא דְמַלֵל לֵיהּ:
וִיהַב מֹשֶׁה לְשִׁבְטָא דִבְנֵי רְאוּבֵן לְזַרְעֲיַתְהוֹן:
וַהֲוָה לְהוֹן תְּחוּמָא מֵעֲרוֹעֵר דִי עַל כֵּיף נַחֲלָא דְאַרְנוֹן וְקַרְתָּא דִי בְגוֹ נַחֲלָא וְכָל מֵישְׁרָא עַד מֵידְבָא:
חֶשְׁבּוֹן וְכָל קִרְוָהָא דִי בְמֵישְׁרָא דִיבוֹן וּבָמוֹת בַּעַל וּבֵית בַּעַל מְעוֹן:
וְיָהֲצָה וּקְדֵמוֹת וּמֵפָעַת:
וְקִרְיָתַיִם וְשִׂבְמָה וְצֶרֶת הַשַׁחַר בְּטוּרָא דְמֵישְׁרָא:
וּבֵית פְּעוֹר וּמַשְׁפַּךְ מְרָמָתָא וּבֵית יְשִׁמוֹת:
וְכָל קִרְוֵי מֵישְׁרָא וְכָל מַלְכְּוַת סִיחוֹן מַלְכָּא דֶאֱמוֹרָאָה דִמְלַךְ בְּחֶשְׁבּוֹן דִמְחָא מֹשֶׁה יָתֵיהּ וְיַת רַבְרְבֵי מִדְיָן יַת אֱוִי וְיַת רֶקֶם וְיַת צוּר וְיַת חוּר וְיַת רֶבַע רַבְרְבֵי סִיחוֹן יַתְבֵי אַרְעָא:
וְיַת בִּלְעָם בַּר בְּעוֹר קִסְמָא קְטָלוּ בְּנֵי יִשְׂרָאֵל בְּחַרְבָּא עַל קְטִילֵיהוֹן:
וַהֲוָה תְּחוּם בְּנֵי רְאוּבֵן יַרְדְנָא וּתְחוּמֵיהּ דָא אַחֲסָנַת בְּנֵי רְאוּבֵן לְזַרְעֲיַתְהוֹן קִרְוַיָא וּפַצְחֵיהֶן:
וִיהַב מֹשֶׁה לְשִׁבְטָא דְגָד לִבְנֵי גָד לְזַרְעֲיַתְהוֹן:
וַהֲוָה לְהוֹן תְּחוּמָא יַעְזֵר וְכָל קִרְוֵי גִלְעָד וּפַלְגוּת אַרְעָא בְּנֵי עַמוֹן עַד עֲרוֹעֵר דִי עַל אַפֵּי רַבָּה:
וּמֵחֶשְׁבּוֹן עַד רָמַת מִצְפַּיָא וּבְטוֹנִים וּמִמַחֲנָיִם עַד תְחוּם לִדְבִיר:
וּבְמֵישְׁרָא בֵּית הָרָם וּבֵית נִמְרָה וְסֻכּוֹת וְצִפוּנָא שְׁאָר מַלְכוּת סִיחוֹן מַלְכָּא דְחֶשְׁבּוֹן יַרְדְנָא וּתְחוּמֵיהּ עַד סְיָפֵי יַם גִינוֹסַר עַבְרָא דְיַרְדְנָא מַדִינְחָא:
דָא אַחֲסָנַת בְּנֵי גָד לְזַרְעֲיַתְהוֹן קִרְוַיָא וּפַצְחֵיהֶן:
וִיהַב מֹשֶׁה לְפַלְגוּת שִׁבְטַיָא דִמְנַשֶׁה וַהֲוָה לְפַּלְגוּת שִׁבְטַיָא דִבְנֵי מְנַשֶׁה לְזַרְעֲיַתְהוֹן:
וַהֲוָה תְּחוּמֵיהוֹן מִמַחֲנַיִם כָּל מַתְנָן כָּל מַלְכוּתֵיהּ דְעוֹג מַלְכָּא דְמַתְנָן וְכָל כַּפְרָנֵי יָאִיר דִבְמַתְנָן שִׁתִּין קִרְוִין:
וּפַלְגוּת אַרְעָא דְגִלְעָד וְעַשְׁתָּרוֹת וְאֶדְרֶעִי קִרְוַיָא מַלְכוּתֵיהּ דְעוֹג בְּמַתְנָן לִבְנֵי מָכִיר בַּר מְנַשֶׁה לְפַלְגוּת בְּנֵי מָכִיר לְזַרְעֲיַתְהוֹן:
אִלֵין דִי אַחֲסִין מֹשֶׁה בְּמֵישְׁרַיָא דְמוֹאָב מֵעִיבְרָא לְיַרְדְנָא דִירִיחוֹ מַדִינְחָא:
וּלְשִׁבְטָא דְלֵוִי לָא יְהַב מֹשֶׁה אַחֲסָנָא מַתְּנָן דִיהַב לְהוֹן יְיָ אֱלָהָא דְיִשְׂרָאֵל אִינוּן אַחֲסַנְתְּהוֹן כְּמָא דְמַלֵיל לְהוֹן:

Chapter 14

וְאִלֵין  דְאַחֲסִינוּ בְּנֵי יִשְׂרָאֵל בְּאַרְעָא דִכְנָעַן דְאַחֲסִינוּ יַתְהוֹן אֶלְעָזָר כַּהֲנָא וִיהוֹשֻׁעַ בַּר נוּן וְרֵישֵׁי אֲבָהַת שִׁבְטַיָא לִבְנֵי יִשְׂרָאֵל:
בְּעַדְבָא אִתְפַּלְגַת לְהוֹן אַחֲסַנְתְּהוֹן כְּמָא דְפַקֵד יְיָ בִּידָא דְמֹשֶׁה לְתִשְׁעַת שִׁבְטִין וּפַלְגוּת שִׁבְטָא:
אֲרֵי יְהַב מֹשֶׁה אַחֲסָנַת תְּרֵין שִׁבְטִין וּפַלְגוּת שִׁבְטָא מֵעִיבְרָא לְיַרְדְנָא וּלְלֵוָאֵי לָא יְהַב אַחֲסָנָא בֵּינֵיהוֹן:
אֲרֵי הֲווֹ בְנֵי יוֹסֵף תְּרֵין שִׁבְטִין מְנַשֶׁה וְאֶפְרָיִם וְלָא יְהָבוּ חוּלֵק לְלֵוָאֵי בְּאַרְעָא אֱלָהֵן קִרְוִין לְמִתַּב וּרְוָחֵיהוֹן לִבְעִירֵיהוֹן וּלְגֵיתֵיהוֹן:
כְּמָא דְפַקֵד יְיָ יַת מֹשֶׁה כֵּן עֲבָדוּ בְּנֵי יִשְׂרָאֵל וּפַלִיגוּ יַת אַרְעָא:
וּקְרִיבוּ בְנֵי יְהוּדָה לְוַת יְהוֹשֻׁעַ בְּגִלְגָלָא וַאֲמַר לֵיהּ כָּלֵב בַּר יְפֻנֶה קְנִיזָאָה אַתְּ יְדַעְתָּא יַת פִּתְגָמָא דְמַלֵיל יְיָ עִם מֹשֶׁה נְבִיָא דַייָ עַל עֵסַק דִילִי וְעַל עֵסַק דִילָךְ בִּרְקַם גֵיאָה:
בַּר אַרְבְּעִין שְׁנִין אֲנָא כַּד שְׁלַח מֹשֶׁה עַבְדָא דַייָ יָתִי מֵרְקַם גֵיאָה לַאֲלָלָא יַת אַרְעָא וַאֲתֵיבִית יָתֵיהּ פִתְגָמָא כְּמָא דַהֲוָה עִם לִבִּי:
וְאַחַי דִסְלִיקוּ עִמִי תְּבַרוּ יַת לִבָּא דְעַמָא וַאֲנָא אַשְׁלֵמִית בָּתַר דַחַלְתָּא דַייָ אֱלָהִי:
וְקַיֵם מֹשֶׁה בְּיוֹמָא הַהוּא לְמֵימָר אִם לָא אַרְעָא דִי דְרָכַת פַּרְסַת רִגְלָךְ בָּהּ דִילָךְ תְּהֵי לְאַחֲסָנָא וְלִבְנָךְ עַד עָלְמָא אֲרֵי אַשְׁלֵימְתָּא בָּתַר דַחַלְתָּא דַייָ אֱלָהִי:
וּכְעַן הָא קַיֵים יְיָ יָתִי כְּמָא דִי מַלֵיל דְנַן אַרְבְּעִין וַחֲמֵשׁ שְׁנִין מֵעִידָן דְמַלֵיל יְיָ יַת פִּתְגָמָא הָדֵין עִם מֹשֶׁה דַאֲזַל יִשְׂרָאֵל בְּמַדְבְּרָא וּכְעַן הָא אַנָא יוֹמָא דֵין בַּר תַּמְנָן וְחָמֵשׁ שְׁנִין:
עַד כְּעַן אֲנָא יוֹמָא דֵין תַּקִיף כִּדְבְּיוֹמָא דִשְׁלַח יָתִי מֹשֶׁה כְּחֵילִי כִּבְכֵן וְכֵן חֵילִי דִכְעָן לְאַגָחָא קְרָבָא וּלְמִפַּק וּלְמֵיעָל:
וּכְעַן הַב לִי יַת טוּרָא הָדֵין דְמַלֵיל יְיָ בְּיוֹמָא הַהוּא אֲרֵי אַתְּ שְׁמַעְתָּא בְּיוֹמָא הַהוּא אֲרֵי גִבָּרִין תַמָן וְקִרְוִין רַבְרְבָן כְּרִיכָן מָאִים יְהֵא מֵימְרָא דַייָ בְּסַעֲדִי וְאֵירְתִינוּן כְּמָא דִי מַלֵיל יְיָ:
וּבָרְכֵיהּ יְהוֹשֻׁעַ וִיהַב יַת חֶבְרוֹן לְכָּלֵב בַּר יְפֻנֶה לְאַחְסָנָא:
עַל כֵּן הֲוַת חֶבְרוֹן לְכָלֵב בַּר יְפֻנֶה קְנִזָאָה לְאַחֲסָנָא עַד יוֹמָא הָדֵין חֲלַף דְאַשְׁלֵם בָּתַר דַחַלְתָּא דַייָ אֱלָהָא דְיִשְׂרָאֵל:
וְשׁוּם חֶבְרוֹן מִלְקָדְמִין קִרְיַת אַרְבַּע אֱנָשׁ רַב בְּגִבָּרַיָא הוּא וְאַרְעָא שְׁדוֹכַת מִלְמֶעְבַּד קְרָבָא: 

Chapter 15

וַהֲוָה עַדְבָא לְשִׁבְטָא דִבְנֵי יְהוּדָה לְזַרְעֲיַתְהוֹן עַל תְּחוּם אֱדוֹם מַדְבְּרָא דְצִין דָרוֹמָא מִסְיָפֵי דָרוֹמָא:
וַהֲוָה לְהוֹן תְּחוּם דָרוֹמָא מִסְיָפֵי יַמָא דְמִלְחָא מִן כֵּיפָא דְמִתְפְּנֵי לְדָרוֹמָא:
וְנָפַק לְמִדָרוֹמָא לְמַסְקָנָא דְעַקְרַבִּין וַעֲבַר לְצִין וּסְלֵיק מִדָרוֹמָא לִרְקַם גֵיאָה וַעֲבַר לְחֶצְרוֹן וּסְלֵיק לְאַדָר וּמִסְתְּחַר לְקַרְקָעָה:
וַעֲבַר לְעַצְמוֹן וְנָפֵק לְנַחְלָא דְמִצְרַיִם וַהֲווֹ מַפְקָנוֹהִי דִתְחוּמָא לְמַעֲרָבָה דֵין יְהֵא לְכוֹן תְּחוּם דָרוֹמָא:
וּתְחוּם קִידוּמָא יַמָא דְמִלְחָא עַד סְיָפֵי יַרְדְנָא וּתְחוּמָא לְרוּחַ צִפּוּנָא מִכֵּיף יַמָא מִסְיָפֵי יַרְדְנָא:
וְסָלֵק תְּחוּמָא לְבֵית חָגְלָה וְעָבַר מִצִפּוּנָא לְבֵית מֵישְׁרָא וְסָלֵיק תְּחוּמָא לְאֶבֶן בּוֹהַן בַּר רְאוּבֵן:
וְסָלֵק תְּחוּמָא לִדְבִיר מִמֵישַׁר עָכוֹר וְצִפּוּנָא מִתְפְּנֵי לְגִלְגָלָא דְלָקֳבֵיל מַסְקָנָא דַאֲדֻמִים דְמִדָרוֹם לְנַחֲלָא וְעָבַר תְּחוּמָא לְמֵי עֵין שֶׁמֶשׁ וַהֲוָן מַפְקָנוֹהִי לְעֵין קַצְרָה:
וְסָלֵיק תְּחוּמָא לְחֵילַת בַּר הִנֹם דִלְעִיבַּר יְבוּס מִדָרוֹמָא הִיא יְרוּשְׁלֵם וְסָלֵק תְּחוּמָא לְרֵישׁ טוּרָא דְעַל אַפֵּי חֵילַת הִנֹם מַעַרְבָא דְבִּסְיָפֵי מֵישַׁר גִבָּרַיָא צִפּוּנָא:
וְיִסְחַר תְּחוּמָא מֵרֵישׁ טוּרָא לְמַבּוּעֵי מֵי נֶפְתּוֹחַ וְנָפֵק לְקִרְוֵי טוּרָא דְעֶפְרוֹן וְיִסְחַר תְּחוּמָא לְבַעֲלָה הִיא קִרְיַת יְעָרִים:
וּמִסְתַּחַר תְּחוּמָא מִבַּעֲלָה לְיַמָא לְטוּרָא דְשֵׂעִיר וַעֲבַר לְעִיבַּר טוּר יְעָרִים מִצִפּוּנָא הִיא כְסָלוֹן וְנָחִית לְבֵית שֶׁמֶשׁ וַעֲבַר לְתִמְנָא:
וְנָפַק תְּחוּמָא לְעִבַּר עֶקְרוֹן לְצִפּוּנָא וְיִסְחַר תְּחוּמָא לְשִׁכָּרוֹן וַעֲבַר לְטוּר בַּעֲלָה וְנָפֵק לְיַבְנְאֵל וַהֲווֹן מַפְקָנוֹהִי דִתְחוּמָא לְיַמָא:
וּתְחוּם מַעַרְבָא יַמָא רַבָּא וּתְחוּמָא דֵין תְּחוּם בְּנֵי יְהוּדָה סְחוֹר סְחוֹר לְזַרְעֲיַתְהוֹן:
וּלְכָלֵב בַּר יְפוּנֶה יְהַב חוּלְקָא בְּגוֹ בְּנֵי יְהוּדָה עַל מֵימְרָא דַייָ לִיהוֹשֻׁעַ יַת קִרְיַת אַרְבַּע אֲבוּהוֹן דְגִבָּרַיָא הִיא חֶבְרוֹן:
וְתָרֵיךְ מִתַּמָן כָּלֵב יַת תְּלָתָא בְּנֵי גִבָּרַיָא יַת שֵׁשַׁי וְיַת אֲחִימָן וְיַת תַּלְמַי בְנֵי גִבָּרַיָא:
וּסְלֵיק מִתַּמָן לְוַת יַתְבֵי דְבִר וְשׁוּם דְבַר מִלְקַדְמִין קִרְיַת אַרְכֵי:
וַאֲמַר כָּלֵב דִמְחֵי יַת קִרְיַת אַרְכֵי וְיִכְבְּשִׁינָהּ וְאֶתֵּן לֵיהּ יַת עַכְסָה בְּרַתִּי לְאִתּוּ:
וְכַבְּשָׁהּ עָתְנִיאֵל בַּר קְנַז אֲחוּהִי דְכָלֵב וִיהַב לֵיהּ יַת עַכְסָה בְּרַתֵּיהּ לְאִתּוּ:
וַהֲוָה בְּמֵיעֲלָהּ וְאַמְלַכְתֵּיהּ לְמִשְׁאַל מִן אֲבוּהָא אַחְסַנְתָּא וְאִתְרְכִינַת מֵעַל חַמְרָא וַאֲמַר לָהּ כָּלֵב מַה לִיךְ:
וַאֲמַרַת הַב לִי אַחְסַנְתָּא אֲרֵי לְאַרַע דָרוֹמָא יְהַבְתַּנִי וְתִתֶּן לִי אֲתַר בֵּית שַׁקְיָא דְמַיָא וִיהַב לָהּ יַת בֵּית שַׁקְיָא עִילָאָה וְיַת בֵּית שַׁקְיָא תַּחְתָּאָה:
דָא אַחְסָנַת שִׁבְטָא דִבְנֵי יְהוּדָה לְזַרְעֲיַתְהוֹן:
וַהֲוָה קִרְוַיָא מִסְיָפֵי לְשִׁבְטָא דִבְנֵי יְהוּדָה עַל תְּחוּם אֱדוֹם בְּדָרוֹמָא קַבְצְאֵל וָעֵדֶר וְיָגוּר:
וְקִינָה וְדִימוֹנָה וְעַדְעָדָה:
וְקֶרֶשׁ וְחָצוֹר וְיִתְנָן:
זִיף וָטֶלֶם וּבְעָלוֹת:
וְחָצוֹר חֲדַתָּה וּקְרִיוֹת חֶצְרוֹן הִיא חָצוֹר:
אֲמָם וּשְׁמָע וּמוֹלָדָה:
וַחֲצַר גַדָה וְחֶשְׁמוֹן וּבֵית פָּלֵט:
וַחֲצַר שׁוּעָל וּבְאֵר שֶׁבַע וּבִזְיוֹתְיָה:
בַּעֲלָה וְעִיִים וָעָצֶם:
וְאֶלְתּוֹלַד וּכְסִיל וְחָרְמָה:
וְצִקְלַג וּמַדְמַנָה וְסַנְסַנָה:
וּלְבָאוֹת וְשִׁלְחִים וְעַיִן וְרִמוֹן כָּל קִרְוַיָא עַשְׂרִין וּתְשַׁע וּפַצְחֵיהֶן:
בִּשְׁפֶלְתָּא אֶשְׁתָּאוֹל וְצָרְעָה וְאַשְׁנָה:
וְזָנוֹחַ וְעֵין גַנִים תַּפּוּחַ וְהָעֵינָם:
יַרְמוּת וַעֲדֻלָם שׁוֹכֹה וַעֲזֵקָה:
וְשַׁעֲרַיִם וַעֲדִיתַיִם וְהַגְדֵרָה וּגְדֵרוֹתָיִם קִרְוִין אַרְבַּע עַסְרֵי וּפַצְחֵיהֶן:
צְנָן וַחֲדָשָׁה וּמִגְדַל גָד:
וְדִלְעָן וְהַמִצְפֶּה וְיָקְתְאֵל:
לָכִישׁ וּבָצְקַת וְעֶגְלוֹן:
וְכַבּוֹן וְלַחְמָס וְכִתְלִישׁ:
וּגְדֵרוֹת בֵּית דָגוֹן וְנַעֲמָה וּמַקֵדָה קִרְוִין שִׁית עַסְרֵי וּפַצְחֵיהֶן:
לִבְנָה וָעֶתֶר וְעָשָׁן:
וְיִפְתָּח וְאַשְׁנָה וּנְצִיב:
וּקְעִילָה וְאַכְזִיב וּמָרֵאשָׁה קִרְוִין תְּשַׁע וּפַצְחֵיהֶן:
עֶקְרוֹן וְכַפְרָנָהָא וּפַצְחָהָא:
מֵעֶקְרוֹן וּמַעַרְבָא כָּל קִרְוַיָא דְעַל תְּחוּם אַשְׁדוֹד וּפַצְחֵיהֶן:
אַשְׁדוֹד כַּפְרָנָהָא וּפַצְחָהָא עַזָה כַּפְרָנָהָא וּפַצְחָהָא עַד נַחֲלָא דְמִצְרָיִם וְיַמָא רַבָּה וּתְחוּמֵיהּ:
וּבְטוּרָא שָׁמִיר וְיַתִּיר וְשׁוֹכֹה:
וְדַנָה וְקִרְיַת סַנָה הִיא דְבִיר:
וַעֲנָב וְאִשְׁתְּמֹה וְעָנִים:
וְגֹשֶׁן וְחֹלֹן וְגִלֹה קִרְוִין חֲדָא עַסְרֵי וּפַצְחֵיהֶן:
אֲרַב וְדוּמָה וְאֶשְׁעָן:
וְיָנוּם וּבֵית תַּפּוּחַ וַאֲפֵקָה:
וְחֻמְטָה וְקִרְיַת אַרְבַּע הִיא חֶבְרוֹן וְצִיעֹר קִרְוִין תְּשַׁע וּפַצְחֵיהֶן:
מָעוֹן כַּרְמֶל וָזִיף וְיוּטָה:
וְיִזְרְעֶאל וְיָקְדְעָם וְזָנוֹחַ:
הַקַיִן גִבְעָה וְתִמְנָה קִרְוִין עֲסַר וּפַצְחֵיהֶן:
חַלְחוּל בֵּית צוּר וּגְדוֹר:
וּמַעֲרָת וּבֵית עֲנוֹת וְאֶלְתְּקֹן קִרְוִין שִׁית וּפַצְחֵיהֶן:
קִרְיַת בַּעַל הִיא קִרְיַת יְעָרִים וְהָרַבָּה קִרְוִין תַּרְתֵּין וּפַצְחֵיהֶן:
בְּמַדְבְּרָא בֵּית מֵישְׁרָא מִדִין וּסְכָכָה:
וְנִבְשָׁן וְקִרְיַת מֶלַח וְעֵין גֶדִי קִרְוִין שִׁית וּפַצְחֵיהֶן:
וְיַת יְבוּסָאֵי יַתְבֵי יְרוּשְׁלֵם לָא יְכִילוּ בְּנֵי יְהוּדָה לְתָרָכוּתְהוֹן וִיתִיבוּ יְבוּסָאֵי עִם בְּנֵי יְהוּדָה בִּירוּשְׁלֵם עַד יוֹמָא הַדִין:

Chapter 16

וּנְפַק עַדְבָא לִבְנֵי יוֹסֵף מִיַרְדְנָא דִירִיחוֹ לְמַיָא יְרִיחוֹ מַדִינְחָא לְמַדְבְּרָא דְסָלִיק מִירִיחוֹ בְּטוּרָא לְבֵית אֵל:
וְנָפֵק מִבֵּית אֵל לְלוּז וַעֲבַר לִתְּחוּם אַרְכִּי לַעֲטָרוֹת:
וְנָחִית לְיַמָא לִתְחוּם יַפְלֵיט עַד תְּחוּם בֵּית חוֹרוֹן אַרְעָאָה וְעַד גָזֶר וַהֲווֹן מַפְקָנוֹהִי לְיַמָא:
וְאַחֲסִינוּ בְנֵי יוֹסֵף מְנַשֶׁה וְאֶפְרָיִם:
וַהֲוָה תְחוּם בְּנֵי אֶפְרַיִם לְזַרְעֲיַתְהוֹן וַהֲוָה תְחוּם אַחְסַנְתְּהוֹן מַדִינְחָא עַטְרוֹת אֲדָר עַד בֵּית חוֹרוֹן עִילָאָה:
וּנְפַק תְּחוּמָא לְיַמָא לְמִכְמְתָת מִצִפּוּנָא וּמִסְתְּחַר תְּחוּמָא מִמַדִינְחָא לְתַאֲנַת שִׁלֹה וְעָבַר לֵיהּ מִמַדְנַח לְיָנוֹחַ:
וְנָחֵית מִיָנוֹחַ לַעֲטָרוֹת וּלְנַעֲרָתָה וּמְעָרַע בִּירִיחוֹ וְנָפַק לְיַרְדְנָא:
מִתַּפּוּחַ אָזִיל תְּחוּמָא לְיַמָא לְנַחֲלָא דְקָנָה וַהֲווֹן מִפְקָנוֹהִי לְיַמָא דָא אַחֲסָנַת שִׁבְטָא דִבְנֵי אֶפְרַיִם לְזַרְעֲיַתְהוֹן:
וְקִרְוַיָא דְמַפְרְשָׁן לִבְנֵי אֶפְרַיִם בְּנוֹ אַחְסָנַת בְּנֵי מְנַשֶׁה כָּל קִרְוַיָא וּפַצְחֵיהֶן:
וְלָא תָרִיכוּ יַת כְּנַעֲנָאֵי דְיָתֵיב בְּגָזֶר וִיתִיבוּ כְּנַעֲנָאֵי בְּגוֹ בֵּית אֶפְרַיִם עַד יוֹמָא הָדֵין וַהֲווֹ לְמַסְקֵי מִסִין פַּלְחִין:

Chapter 17

וַהֲוָה עַדְבָא לְשִׁבְטָא דִמְנַשֶׁה אֲרֵי הוּא בּוּכְרָא דְיוֹסֵף לְמָכִיר בּוּכְרָא דִמְנַשֶׁה אֲבוּהִי דְגִלְעָד אֲרֵי הוּא הֲוָה גְבַר עָבֵיד קְרָבִין וַהֲוָה לֵיהּ אַרְעָא גִלְעָד וּמַתְנָן:
וַהֲוָה לִבְנֵי מְנַשֶׁה דְאִשְׁתָּאֲרוּ לְזַרְעֲיַתְהוֹן וְלִבְנֵי אֲבִיעֶזֶר וְלִבְנֵי חֵלֶק וְלִבְנֵי אַשְׂרִיאֵל וְלִבְנֵי שֶׁכֶם וְלִבְנֵי חֵפֶר וְלִבְנֵי שְׁמִידָע אִילֵן בְּנֵי מְנַשֶׁה בַּר יוֹסֵף דָכְרַיָא לְזַרְעֲיַתְהוֹן:
וְלִצְלָפְחָד בַּר חֵפֶר בַּר גִלְעָד בַּר מָכִיר בַּר מְנַשֶׁה לָא הֲווֹ לֵיהּ בְּנִין אֱלָהֵן בְּנָן וְאִלֵין שְׁמָהַת בְּנָתֵיהּ מַחְלָה וְנֹעָה חָגְלָה מִלְכָּה וְתִרְצָה:
וְקָרִיבָא לָקֳדָם אֶלְעָזָר כַּהֲנָא וְלִקֳדָם יְהוֹשֻׁעַ בַּר נוּן וְלִקֳדָם רַבְרְבַיָא לְמֵימַר יְיָ פַּקֵיד יַת מֹשֶׁה לְמִתַּן לָנָא אַחֲסָנָא בְּגוֹ אַחָנָא וִיהַב לְהוֹן עַל מֵימְרָא דַייָ אַחֲסָנָא בְּגוֹ אֲחֵי אֲבוּהֶן:
וּנְפָלוּ עַדְבֵי מְנַשֶׁה עַסְרָא בַּר מֵאֲרְעָא דְגִלְעָד וּמַתְנָן דְמֵעִיבְרָא לְיַרְדְנָא:
אֲרֵי בְּנַת מְנַשֶׁה אַחֲסִינוּ אַחֲסָנָא בְּגוֹ בְנוֹהִי וְאַרְעָא גִלְעָד הֲוַת לִבְנֵי מְנַשֶׁה דְאִשְׁתָּאֲרוּ:
וַהֲוָה תְּחוּם מְנַשֶׁה מֵאָשֵׁר לְמִכְמְתָת דִי עַל אַפֵּי שְׁכֶם וַאֲזַל תְּחוּמָא לִימִינָא לְיַתְבֵי עֵין תַּפּוּחַ:
לִמְנַשֶׁה הֲוַת אֲרַע תַּפּוּחַ וְתַפּוּחַ עַל תְּחוּם מְנַשֶׁה לִבְנֵי אֶפְרַיִם:
וְנָחִית תְּחוּמָא לְנַחֲלָא דְקָנָה לְדָרוֹמָא לְנַחֲלָא קִרְוַיָא הָאִלֵין לְאֶפְרַיִם בְּגוֹ קִרְוֵי מְנַשֶׁה וּתְחוּם מְנַשֶׁה מִצִפּוּנָא לְנַחֲלָא וַהֲוָן מַפְקָנוֹהִי לְיַמָא:
דָרוֹמָא לְאֶפְרַיִם וְצִפּוּנָא לִמְנַשֶׁה וַהֲוָה יַמָא תְּחוּמֵיהּ וּבְאָשֵׁר מְעַרְעִין מִצִפּוּנָא וּבְיִשָׂשכָר מִמַדִינְחָא:
וַהֲוָה לִמְנַשֶׁה בְּיִשָׂשכָר וּבְאָשֵׁר בֵּית שְׁאָן וְכַפְרָנָהָא וְיִבְלְעָם וְכַפְרָנָהָא וְיַת יַתְבֵי דֹאר וְכַפְרָנָהָא וְיַתְבֵי עֵין דֹר וְכַפְרָנָהָא וְיַתְבֵי תַעְנַךְ וְכַפְרָנָהָא וְיַתְבֵי מְגִדוֹ וְכַפְרָנָהָא תְּלָתָא פִּלְכִין:
וְלָא יָכִילוּ בְּנֵי מְנַשֶׁה לְתָרָכָא יַת קִרְוַיָא הָאִלֵין וּשְׁרֵי כְנַעֲנָאָה לְמִתַּב בְּאַרְעָא הָדָא:
וַהֲוָה כַּד תַּקִיפוּ בְּנֵי יִשְׂרָאֵל וּמַנִיאוּ יַת כְּנַעֲנָאֵי לְמַסְקֵי מִסִין וְתָרָכָא לָא תָרְכִינוּן:
וּמַלִילוּ בְּנֵי יוֹסֵף עִם יְהוֹשֻׁעַ לְמֵימָר מָא דֵין יְהָבְתָּ לִי אַחֲסָנָא עַדְבָא חָד וְחוּלַק חָד וַאֲנָא עַם סַגִי עַד לִסְגֵי בָּרְכַנִי יְיָ:
וַאֲמַר לְהוֹן יְהוֹשֻׁעַ אִם עַם סַגִי אַתּוּן סַקוּ לְכוֹן לְחוּרְשָׁא וּתְתַקֵן לָךְ תַמָן אֲתַר בְּאַרַע פְּרִזָאֵי וְגִבָּרַיָא אֲרֵי עָק לְכוֹן טוּרָא דְהַר אֶפְרָיִם:
וַאֲמַרוּ בְּנֵי יוֹסֵף לָא יַסְפֵּק לָנָא טוּרָא וּרְתִיכִין דְפַרְזְלָא בְּכָל כְּנַעֲנָאֵי דְיַתְבִין בְּאֲרַע מֵישְׁרָא לְדִי בְּבֵית שְׁאָן וְכַפְרָנָהָא וּלְדִי בְּמֵישַׁר יִזְרְעֶאל:
וַאֲמַר יְהוֹשֻׁעַ לְבֵית יוֹסֵף לְאֶפְרַיִם וְלִמְנַשֶׁה לְמֵימַר עַם סַגִי אַתּוּן וְחֵיל סַגִי לְכוֹן לָא יְהֵי לְכוֹן עַדְבָא חָד:
אֲרֵי טוּרָא יְהֵי לָךְ אֲרֵי חוּרְשָׁא הוּא וְתַתְקְנִינֵיהּ וִיהֵי לָךְ מַפְקָנוֹהִי אֲרֵי תְתַרְכוּן יַת כְּנַעֲנָאֵי אֲרֵי רְתִיכִין דְפַרְזְלָא לֵיהּ אֲרֵי תַּקִיפִין אִינוּן:

Chapter 18

וְאִתְכַּנָשׁוּ כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל לְשִׁלֹה וְאַשְׁרִיאוּ תַמָן יַת מַשְׁכַּן זִמְנָא וְאַרְעָא אִתְכְּבִישַׁת קֳדָמֵיהוֹן:
וְאִשְׁתָּאֲרוּ בִּבְנֵי יִשְׂרָאֵל דְלָא פַּלִיגוּ יַת אַחְסַנְתְּהוֹן שַׁבְעָה שִׁבְטִין:
וַאֲמַר יְהוֹשֻׁעַ לִבְנֵי יִשְׂרָאֵל עַד אֵימָתַי אַתּוּן מִתְרַשְׁלִין לְמֵיעַל לְמֵירַת יַת אַרְעָא דִיהַב לְכוֹן יְיָ אֱלָהָא דְאַבְהַתְכוֹן:
הָבוּ לְכוֹן תְּלָתָא גַבְרִין לְשִׁבְטָא וְאֶשְׁלָחִינוּן וִיקוּמוּן וִיהַלְכוּן בְּאַרְעָא וְיִכְתְּבוּן יָתָהּ לְפוּם אַחְסַנְתְּהוֹן וְיֵיתוּן לְוָתִי:
וִיפַלְגוּן יָתָהּ לְשִׁבְעָה חוּלְקִין דְבֵית יְהוּדָה יְקוּמוּן עַל תְּחוּמֵיהוֹן מִדָרוֹמָא וּדְבֵית יוֹסֵף יְקוּמוּן עַל תְּחוּמֵיהוֹן מִצִפּוּנָא:
וְאַתּוּן תִּכְתְּבוּן יַת אַרְעָא לְשִׁבְעָא חוּלְקִין וְתַיְתוּן לְוָתִי הַלְכָא וְאַרְמֵי לְכוֹן עַדְבָא הַלְכָא קֳדָם יְיָ אֱלָהָנָא:
אֲרֵי לֵית חוּלַק לְלֵוָאֵי בֵּינֵיכוֹן אֱלָהֵן מַתְּנָן דִיהַב לְהוֹן יְיָ אִינוּן אַחְסַנְתְּהוֹן וְגָד וּרְאוּבֵן וּפַלְגוּת שִׁבְטָא דִמְנַשֶׁה קַבִּילוּ אַחְסַנְתְּהוֹן מֵעִבְרָא לְיַרְדְנָא מַדִינְחָא דִיהַב לְהוֹן מֹשֶׁה עַבְדָא דַיָי:
וְקָמוּ גַבְרַיָא וַאֲזַלוּ וּפַקֵיד יְהוֹשֻׁעַ יַת דַאֲזַלוּ לְמִכְתַּב יַת אַרְעָא לְמֵימָר אֱזִילוּ וְהַלִיכוּ בְאַרְעָא וּכְתוּבוּ יָתָהּ וְתוּבוּ לְוָתִי וְהָכָא אֶרְמֵי לְכוֹן עַדְבִין קֳדָם יְיָ בְּשִׁלֹה:
וַאֲזַלוּ גִבָּרַיָא וַעֲבָרוּ בְּאַרְעָא וְכַתְבוּהָ לְקִרְוַיָא לְשַׁבְעָא חוּלְקִין עַל סִפְרָא וַאֲתוֹ לְוַת יְהוֹשֻׁעַ לְמַשְׁרִיתָא לְשִׁלֹה:
וּרְמָא לְהוֹן יְהוֹשֻׁעַ עַדְבָא בְּשִׁלֹה קֳדָם יְיָ וּפַלֵיג תַּמָן יְהוֹשֻׁעַ יַת אַרְעָא לִבְנֵי יִשְׂרָאֵל כְּפַלְגוּתְהוֹן:
וּסְלִיק עַדְבָא שִׁבְטָא דִבְנֵי בִנְיָמִן לְזַרְעֲיַתְהוֹן וּנְפַק תְּחוּם עַדְבֵיהוֹן בֵּין בְּנֵי יְהוּדָה וּבֵין בְּנֵי יוֹסֵף:
וַהֲוָה לְהוֹן תְּחוּמָא לְרוּחַ צִפּוּנָא מִן יַרְדְנָא וְסָלִיק תְּחוּמָא לְעִיבַר יְרִיחוֹ מִצִפּוּנָא וְסָלֵיק בְּטוּרָא לְיַמָא וַהֲוָן מַפְקָנוֹהִי לְמַדְבַּר בֵּית אָוֶן:
וַעֲבַר מִתַּמָן תְּחוּמָא לְלוּז דִלְעִבַּר דְלוּז דָרוֹמָא הִיא בֵּית אֵל וְנָחִית תְּחוּמָא עַטְרוֹת אַדָר עַל טוּרָא דִמִדָרוֹם לְבֵית חוֹרוֹן אַרְעָאָה:
וְתָאַר תְּחוּמָא וּמִסְתְּחַר לְרוּחַ מַעַרְבָא לְדָרוֹמָא מִן טוּרָא דִי עַל אַפֵּי בֵּית חוֹרוֹן דָרוֹמָא וַהֲווֹן מַפְקָנוֹהִי עַל קִרְיַת בַּעַל הִיא קִרְיַת יְעָרִים קַרְתָּא דִבְנֵי יְהוּדָה דָא רוּחַ מַעַרְבָא:
וְרוּחַ דָרוֹמָא מִסְיָפֵי קִרְיַת יְעָרִים וְנָפַק תְּחוּמָא לְיַמָא וְנָפֵיק לְמַבּוּעֵי מֵי נֶפְתּוֹחַ:
וְנָחִית תְּחוּמָא לִסְטַר טוּרָא דִי עַל אַפֵּי חֵילַת בַּר הִנֹם דִי בְּמֵישַׁר גִבָּרַיָא צִפוּנָא וְנָחִית לְחֵילַת בַּר הִנֹם לְעִיבַר יְבוּס דָרוֹמָא וְנָחִית לְעֵין קַצְרָא:
וְתָאַר מִצָפוֹן וְנָפַק לְעֵין שֶׁמֶשׁ וְנָפֵיק לִגְלִילָא דִי לָקֳבֵיל מַסְקָנָא דַאֲדֻמִים וְנָחִית לְאֶבֶן בּוֹהַן בַּר רְאוּבֵן:
וַעֲבַר לְעִיבְרָא דִלָקֳבֵיל מֵישְׁרָא צִפּוּנָא וְנָחִית לְעַרְבָתָה:
וַעֲבַר תְּחוּמָא לְעִיבַר בֵּית חָגְלָה לְצִפּוּנָא וַהֲווֹן מַפְקָנוֹהִי דִתְחוּמָה לְכֵיף יַמָא דְמִלְחָא צִפּוּנָא לִסְיָפֵי יַרְדְנָא דָרוֹמָא דֵין תְּחוּם דָרוֹמָא:
וְיַרְדְנָא תְּחוּמָא לֵיהּ לְרוּחַ קִידוּמָא דָא אַחְסָנַת בְּנֵי בִנְיָמִין לִתְחוּמָהָא סְחוֹר סְחוֹר לְזַרְעֲיַתְהוֹן:
וַהֲוָה קִרְוַיָא לְשִׁבְטָא דִבְנֵי בִנְיָמִין לְזַרְעֲיַתְהוֹן יְרִיחוֹ וּבֵית חָגְלָה וּמֵישַׁר קְצִיץ:
וּבֵית מֵישְׁרָא וּצְמָרַיִם וּבֵית אֵל:
וְהָעַוִים וְהַפָרָה וְעָפְרָה:
וּכְפַר הָעַמוֹנָה וְהָעָפְנִי וְגָבַע קִרְוִין תַּרְתָּא עַסְרֵי וּפַצְחַיהֶן:
גִבְעוֹן וְרָמָתָא וּבְאֵרוֹת:
וּמִצְפַּיָא וּכְפִירָה וְהַמֹצָה:
וְרֶקֶם וְיִרְפְּאֵל וְתַרְאֲלָה:

Chapter 19

וּנְפַק עַדְבָא תִנְיָנָא לְשִׁמְעוֹן לְשִׁבְטָא דִבְנֵי שִׁמְעוֹן לְזַרְעֲיַתְהוֹן וַהֲוָה אַחֲסַנְתְּהוֹן בְּגוֹ אַחֲסָנַת בְּנֵי יְהוּדָה:
וַהֲוָה לְהוֹן בְּאַחֲסַנְתְּהוֹן בְּאֵר שֶׁבַע וְשֶׁבַע וּמוֹלָדָה:
וַחֲצַר שׁוּעָל וּבָלָה וְעֶצֶם:
וְאֶלְתּוֹלַד וּבְתוּל וְחָרְמָה:
וְצִקְלַג וּבֵית הַמַרְכָּבֹת וַחֲצַר סוּסָה:

לְיִשָׂשכָר נְפַק עַדְבָא רְבִיעָאָה לִבְנֵי יִשָׂשכָר לְזַרְעֲיָתְהוֹן:

וַחַפָרַיִם וְשִׁיאוֹן וַאֲנָחֲרָת:
וְהָרַבִּית וְקִשְׁיוֹן וָאָבֶץ:

וּנְפַק עַדְבָא חֲמִישָׁאָה לְשִׁבְטָא דִבְנֵי אָשֵׁר לְזַרְעֲיַתְהוֹן:

דָא אַחֲסָנַת שִׁבְטָא דִבְנֵי אָשֵׁר לְזַרְעֲיַתְהוֹן קִרְוַיָא הָאִלֵין וּפַצְחֵיהֶן:

וַהֲוָה תְּחוּמְהוֹן מֵחֵלֶף מֵאַלוֹן בְּצַעֲנַנִים וַאֲדָמִי הַנֶקֶב וְיַבְנְאֵל עַד לַקוּם וַהֲווֹן מַפְקָנוֹהִי לְיַרְדְנָא:

וַאֲדָמָה וְהָרָמָה וְחָצוֹר:
וְקֶדֶשׁ וְאֶדְרֶעִי וְעֵין חָצוֹר:
וְיִרְאוֹן וּמִגְדַל אֵל חָרֵם וּבֵית עֲנָת וּבֵית שֶׁמֶשׁ קִרְוִין תְּשַׁע עַשְׂרֵי וּפַצְחֵיהֶן:
דָא אַחֲסָנַת שִׁבְטָא דִבְנֵי נַפְתָּלִי לְזַרְעֲיַתְהוֹן קִרְוַיָא וּפַצְחֵיהֶן:
לְשִׁבְטָא דִבְנֵי דָן לְזַרְעֲיַתְהוֹן נַפְקָא עַדְבָא שְׁבִיעָאָה:
וַהֲוָה תְּחוּם אַחְסַנְתְּהוֹן צָרְעָה וְאֶשְׁתָּאוֹל וְקִרְיַת שִׁמְשָׁא:
וְשַׁעֲלַבִּין וְאַיָלוֹן וְיִתְלָה:
וְאֵילוֹן וְתִמְנָתָה וְעֶקְרוֹן:
וְאֶלְתְּקֵה וְגִבְּתוֹן וּבַעֲלָת:
וִיהֻד וּבְנֵי בְרַק וְגַת רִמוֹן:
וּמֵי הַיַרְקוֹן וְרַקוֹן עִם תְּחוּמָא דִי לָקֳבֵיל יָפוֹ:
וּנְפַק תְּחוּם בְּנֵי דָן מִנְהוֹן וּסְלִיקוּ בְּנֵי דָן וְאַגִיחוּ קְרָבָא עִם לֶשֶׁם וּכְבַשׁוּ יָתָהּ וּמְחוֹ יָתָה לְפִתְגַם דְחֶרֶב וִירִיתוּ יָתָהּ וִיתִיבוּ בָהּ וּקְרוֹ לְלֶשֶׁם דָן כְּשׁוּם דָן אֲבוּהוֹן:
דָא אַחֲסָנַת שַׁבְטָא דִבְנֵי דָן לְזַרְעֲיַתְהוֹן קִרְוַיָא הָאִלֵין וּפַצְחֵיהֶן:
וְשֵׁיצִיאוּ לְאַחְסָנָא יַת אַרְעָא לִתְחוּמָהָא וִיהָבוּ בְנֵי יִשְׂרָאֵל אַחֲסָנָא לִיהוֹשֻׁעַ בַּר נוּן בֵּינֵיהוֹן:
עַל מֵימְרָא דַייָ יְהֲבוּ לֵיהּ יַת קַרְתָּא דִי שְׁאֵל יַת תִּמְנַת סָרַח בְּטוּרָא דְאֶפְרָיִם וּבְנָא יַת קַרְתָּא וִיתִיב בָּהּ:
אִילֵן אַחֲסַנְתַּיָא דִי אַחֲסִין אֶלְעָזָר כַּהֲנָא וִיהוֹשֻׁעַ בַּר נוּן וְרֵישֵׁי אַבְהָתָא לְשִׁבְטַיָא דִבְנֵי יִשְׂרָאֵל בְּעַדְבָא בְּשִׁלֹה קֳדָם יְיָ בִּתְרַע מַשְׁכַּן זִמְנָא וְשֵׁיצִיאוּ מִלְפַלָגָא יַת אַרְעָא:

Chapter 20

וּמַלֵל יְיָ עִם יְהוֹשֻׁעַ לְמֵימָר:
מַלֵיל עִם בְּנֵי יִשְׂרָאֵל לְמֵימַר הָבוּ לְכוֹן יַת קִרְוֵי שֵׁיזָבוּתָא דְמַלֵילִית עִמְכוֹן בִּידָא דְמֹשֶׁה:
לְמֵעֲרוֹק לְתַמָן קָטוֹלָא דְיִקְטוֹל נַפְשָׁא בְּשָׁלוּ בְּלָא מַדְעֵיה וְיֶהֱוְיָן לְכוֹן לְשֵׁזָבָא מִגָאֵל דְמָא:
וְיַעֲרוֹק לַחֲדָא מִקִרְוַיָא הָאִלֵין וִיקוּם בְּמַעֲלָנָא דִתְרַע קַרְתָּא וִימַלֵיל קֳדָם סָבֵי קַרְתָּא הַהִיא יַת פִּתְגָמוֹהִי וִיכַנְשׁוּן יָתֵיהּ לְקַרְתָּא לְוַתְהוֹן וְיִתְּנוּן לֵיהּ אֲתַר וִיתֵיב עִמְהוֹן:
וַאֲרֵי יִרְדוֹף גָאֵל דְמָא בַּתְרוֹהִי וְלָא יִמְסְרוּן יַת קָטוֹלָא בִּידֵיהּ אֲרֵי בְּלָא מַדָעֵיהּ קְטַל יַת חַבְרֵיהּ וְלָא שָׂנֵי הוּא לֵיהּ מֵאִתְמָלֵי וּמִדְקֳדָמוֹהִי:
וִיתֵיב בְּקַרְתָּא הַהִיא עַד דִיקוּם קֳדָם כְנִשְׁתָּא לְדִינָא עַד דִימוּת כַּהֲנָא רַבָּא דִי יְהֵי בְיוֹמַיָא הָאִינוּן בְּכֵן יְתוּב קְטוֹלָא וְיֵיעוֹל לְקַרְתֵּיהּ וּלְבֵיתֵיהּ לְקַרְתָּא דִי עֲרַק מִתַּמָן:

Chapter 22

בְּכֵן קְרָא יְהוֹשֻׁעַ לְשִׁבְטָא דִרְאוּבֵן וּלְשִׁבְטָא דְגָד וּלְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה:
וַאֲמַר לְהוֹן אַתּוּן נְטַרְתּוּן יַת כָּל דְפַקֵד יַתְכוֹן מֹשֶׁה עַבְדָא דַייָ וְקַבֵּילְתּוּן לְמֵימְרִי לְכֹל דִי פַקְדֵית יַתְכוֹן:
לָא שְׁבַקְתּוּן יַת אֲחוּכוֹן דְנַן יוֹמִין סַגִיאִין עַד יוֹמָא הָדֵין וּנְטַרְתּוּן יַת מַטְרַת תַּפְקֵידְתָּא מֵמְרָא דַייָ אֱלָהָכוֹן:
וּכְעַן אַנִיחַ יְיָ אֱלָהָכוֹן לַאֲחוּכוֹן כְּמָא דִי מַלֵיל לְהוֹן וּכְעַן אִתְפְּנִיאוּ וֶאֱזִילוּ לְכוֹן לְקִרְוֵיכוֹן לְאַרְעָא אַחְסַנְתְּכוֹן דִיהַב לְכוֹן מֹשֶׁה עַבְדָא דַייָ בְּעִבְרָא דְיַרְדְנָא:
לְחוֹד אִסְתַּמָרוּ לַחֲדָא לְמֶעֱבַּד יַת תַפְקֵדְתָּא וְיַת אוֹרַיְתָא דִי פַקֵיד יַתְכוֹן מֹשֶׁה עַבְדָא דַייָ לְמִרְחַם יַת יְיָ אֱלָהָכוֹן וְלִמְהַךְ בְּכָל אוֹרְחָן דְתַקְנָן קֳדָמוֹהִי וּלְמִטַר פִּקוּדֹהִי וּלְאִתְקְרָבָא לִדְחַלְתֵּיהּ וּלְמִפְלַח קֳדָמוֹהִי בְּכָל לִבְּכוֹן וּבְכָל נַפְשְׁכוֹן:
וּבָרְכִינוּן יְהוֹשֻׁעַ וְשַׁלְחִינוּן וַאֲזַלוּ לְקִרְוֵיהוֹן:
וּלְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה יְהַב מֹשֶׁה בְּמַתְנָן וּלְפַלְגֵיהוֹן יְהַב יְהוֹשֻׁעַ עִם אֲחֵיהוֹן בְּעִבְרָא דְיַרְדְנָא לְמַעֲרָבָא וְאַף אֲרֵי שַׁלְחִינוּן יְהוֹשֻׁעַ לְקִרְוֵיהוֹן וּבָרְכִינוּן:
וַאֲמַר לְהוֹן לְמֵימָר בְּנִכְסִין סַגִיאִין תּוּבוּ לְקִרְוֵיכוֹן וּבִבְעִירָא סַגִי לַחֲדָא בְּכַסְפָּא וּבְדַהֲבָא וּבִנְחָשָׁא וּבְפַרְזְלָא וּבִלְבוּשִׁין סַגִי לַחֲדָא פַּלִיגוֹ בִּזַת בַּעֲלֵי דְבָבֵיכוֹן עִם אֲחֵיכוֹן:
וְתָבוּ וַאֲזַלוּ בְּנֵי רְאוּבֵן וּבְנֵי גָד וּפַלְגוּת שִׁבְטָא דִמְנַשֶׁה מִלְוַת בְּנֵי יִשְׂרָאֵל מִשִׁלֹה דִי בְאַרְעָא דִכְנָעַן לְמֵיזַל לְאַרְעָא דְגִלְעָד לְאַרְעָא אַחְסַנְתְּהוֹן דְאַחֲסִינוּ בָהּ עַל מֵימְרָא דַייָ בִּידָא דְמֹשֶׁה:
וַאֲתוֹ לִגְלִילֵי יַרְדְנָא דִי בְּאַרְעָא דִכְנָעַן וּבְנוֹ בְנֵי רְאוּבֵן וּבְנֵי גָד וּפַלְגוּת שִׁבְטָא דִמְנַשֶׁה תַּמָן מַדְבְּחָא עַל יַרְדְנָא מַדְבַּח רַב לְמֶחֱזֵי:
וּשְׁמָעוּ בְנֵי יִשְׂרָאֵל לְמֵימָר הָא בְּנוֹ בְּנֵי רְאוּבֵן וּבְנֵי גָד וּפַלְגוּת שִׁבְטָא דִמְנַשֶׁה יַת מַדְבְּחָא לָקֳבֵיל אַרְעָא דִכְנָעַן בִּגְלִילֵי יַרְדְנָא דִלְעִבַּר בְּנֵי יִשְׂרָאֵל:
וּשְׁמָעוּ בְּנֵי יִשְׂרָאֵל וְאִתְכְּנִישׁוּ כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל לְשִׁלֹה לְמֵיסַק עֲלֵיהוֹן לְחֵילָא:
וּשְׁלָחוּ בְּנֵי יִשְׂרָאֵל לְוַת בְּנֵי רְאוּבֵן וּלְוַת בְּנֵי גָד וּלְוַת פַּלְגוּת שִׁבְטָא דִמְנַשֶׁה לְאַרְעָא גִלְעָד יַת פִּנְחָס בַּר אֶלְעָזָר כַּהֲנָא:
וְעַשְׂרָא רַבְרְבִין עִמֵיהּ רַבָּא חָד רַבָּא חָד לְבֵית אַבָּא לְכָל שִׁבְטַיָא דְיִשְׂרָאֵל וּגְבַר רֵישׁ בֵּית אֲבָהַתְהוֹן אִינוּן לְאַלְפַיָא דְיִשְׂרָאֵל:
וַאֲתוֹ לְוַת בְּנֵי רְאוּבֵן וּלְוַת בְּנֵי גָד וּלְוַת פַּלְגוּת שִׁבְטָא דִמְנַשֶׁה לְאַרְעָא דְגִלְעָד וּמַלִילוּ עִמְהוֹן לְמֵימָר:
כְּדֵין אֲמַרוּ כָּל כְּנִשְׁתָּא דַייָ מָא שִׁקְרָא הָדֵין דִי שְׁקַרְתּוּן בְּמֵימְרָא דֶאֱלָהָא יִשְׂרָאֵל לְמִיתַב יוֹמָא דֵין מִבָּתַר פּוּלְחָנָא דַייָ בְּמִבְנְכֹן לְכוֹן מַדְבְּחָא לְמִמְרַדְכוֹן יוֹמָא דֵין בְּמֵימְרָא דַיָי:
הַזְעֵיר לָנָא יַת חוֹבָא דִפְעוֹר דְלָא אִדַכִּינָא מִנֵיהּ עַד יוֹמָא הָדֵין וַהֲוַת מְחָתָא בִּכְנִשְׁתָּא דַיָי:
וְאַתּוּן תְּתוּבוּן יוֹמָא דֵין מִבָּתַר פּוּלְחָנָא דַייָ וִיהֵי אַתּוּן תִּמְרְדוּן יוֹמָא דֵין בְּמֵימְרָא דַייָ וּמְחַר עַל כָּל כְּנִשְׁתָּא דְיִשְׂרָאֵל יְהֵי רוּגְזָא:
וּבְרַם אִם מְסָאֲבָא אֲרַע אַחְסַנְתְּכוֹן עֲבָרוּ לְכוֹן לַאֲרַע אַחְסַנַת עַמָא דַייָ דִי שְׁרָא תַמָן מַשְׁכְּנָא דַייָ וְאַחֲסִינוּ בֵּינָנָא וּבְמֵימְרָא דַייָ לָא תִמְרְדוּן וּבָנָא לָא תִמְרְדוּן בְּמִבְנֵיכוֹן לְכוֹן מַדְבְּחָא בַּר מִמַדְבְּחָא דַייָ אֱלָהָנָא:
הֲלָא עָכָן בַּר זָרַח שְׁקַר שִׁקְרָא בְּחֶרְמָא וְעַל כָּל כְּנִשְׁתָּא דְיִשְׂרָאֵל הֲוָה רוּגְזָא וְהוּא גַבְרָא חָד לָא מִית בְּחוֹבֵיהּ:
וַאֲתִיבוּ בְּנֵי רְאוּבֵן וּבְנֵי גָד וּפַלְגוּת שִׁבְטָא דִמְנָשֶׁה וּמַלִילוּ עִם רֵישֵׁי אַלְפַיָא דְיִשְׂרָאֵל:
אֵל אֱלֹהִים יְיָ אֵל אֱלֹהִים יְיָ קֳדָמוֹהִי יְדִיעַ וְיִשְׂרָאֵל בְּסוֹפָא יֵדַע אִם בְּמֵרְדָא וְאִם בְּשִׁקְרָא שְׁקַרְנָא בְמֵימְרָא דַייָ לָא תִפְרְקִינָנָא יוֹמָא הָדֵין:
לְמִבְנָא לָנָא מַדְבְּחָא לְמֵיתַב מִבָּתַר פּוּלְחָנָא דַייָ וְאִם לְאַסָקָא עֲלוֹהִי עֲלָתָא וּמִנְחָתָא וְאִם לְמֶעְבַּד עֲלוֹהִי נִכְסַת קוּדְשִׁין יְיָ הוּא יִתְבַּע:
וְאִם לָא מְיוֹצְפָא מִפִּתְגַם עָבַדְנָא יַת דָא לְמֵימָר מְחַר יֵימְרוּן בְּנֵיכוֹן לִבְנָנָא לְמֵימָר לֵית לְכוֹן חוּלַק בְּמֵימְרָא דַייָ אֱלָהָא דְיִשְׂרָאֵל:
וּתְחוּמָא יְהַב יְיָ בֵּינָנָא וּבֵינֵיכוֹן בְּנֵי רְאוּבֵן וּבְנֵי גָד יַת יַרְדְנָא לֵית לְכוֹן חוּלַק בְּמֵימְרָא דַייָ וִיבַטְלוּן בְּנֵיכוֹן יַת בְּנָנָא בְּדִיל דְלָא לְמִדְחַל מִן קֳדָם יְיָ:
וַאֲמַרְנָא נַעְבֵּיד כְּעַן לָנָא לְמִבְנֵי יַת מַדְבְּחָא לָא לַעֲלָתָא וְלָא לְנִכְסַת קוּדְשַׁיָא:
אֲרֵי סָהִיד הוּא בֵּינָנָא וּבֵינֵיכוֹן וּבֵין דָרָנָא בַּתְרָנָא לְמִפְלַח יַת פּוּלְחָנָא דַייָ קֳדָמוֹהִי בְּעַלָוְתָנָא וּבְדִבְחָנָא וּבְנִכְסַת קוּדְשָׁנָא וְלָא יֵמְרוּן בְּנֵיכוֹן לִבְנָנָא מְחַר לֵית לְכוֹן חוּלַק בְּמֵימְרָא דַיָי:
וַאֲמַרְנָא וִיהֵי אֲרֵי יֵימְרוּן לָנָא וּלְדָרָנָא מְחַר וְנֵימַר חֲזוֹ יַת דְמוּת מַדְבְּחָא דַייָ דִי עֲבַדוּ אַבְהָתָנָא לָא לַעֲלָתָא וְלֹא לְנִכְסַת קוּדְשַׁיָא אֲרֵי סָהִיד הוּא בֵּינָנָא וּבֵינֵיכוֹן:
חַס לָנָא מִנָנָא מִלְמֶחְטֵי קֳדָמוֹהִי מִלְמֵרַד בְּמֵימְרָא דַייָ וּלְמִיתַב יוֹמָא דֵין מִבָּתַר פּוּלְחָנָא דַייָ לְמִבְנֵי מַדְבְּחָא לַעֲלָתָא לְמִנְחָתָא וּלְנִכְסַת קוּדְשַׁיָא בַּר מִמַדְבְּחָא דַייָ אֱלָהָנָא דִי קֳדָם מַשְׁכְּנֵיהּ:
וּשְׁמַע פִּינְחָס כַּהֲנָא וְרַבְרְבֵי כְנִשְׁתָּא וְרֵישֵׁי אַלְפַיָא דְיִשְׂרָאֵל דִי עִמֵיהּ יַת פִּתְגָמַיָא דִי מַלִילוּ בְּנֵי רְאוּבֵן וּבְנֵי גָד וּבְנֵי מְנַשֶׁה וּשְׁפַר בְּעֵינֵיהוֹן:
וַאֲמַר פִינְחָס בַּר אֶלְעָזָר כַּהֲנָא לִבְנֵי רְאוּבֵן וְלִבְנֵי גָד וְלִבְנֵי מְנַשֶׁה יוֹמָא דֵין יְדַעְנָא אֲרֵי בֵינָנָא שַׁרְיָא שְׁכִינְתָּא דַייָ דְלָא שְׁקַרְתּוּן בְּמֵימְרָא דַייָ שִׁקְרָא הָדֵין בְּכֵן שֵׁיזַבְתּוּן יַת בְּנֵי יִשְׂרָאֵל מִיַד מֵימְרָא דַיָי:
וְתַב פִינְחָס בַּר אֶלְעָזָר כַּהֲנָא וְרַבְרְבַיָא מִלְוַת בְּנֵי רְאוּבֵן וּמִלְּוַת בְּנֵי גָד מֵאַרְעָא גִלְעָד לְאַרְעָא דִכְנָעַן לְוַת בְּנֵי יִשְׂרָאֵל וַאֲתִיבוּ יַתְהוֹן פִּתְגָמָא:
וּשְׁפַר פִּתְגָמָא בְּעֵינֵי בְּנֵי יִשְׂרָאֵל וְאוֹדִיאוּ קֳדָם יְיָ בְּנֵי יִשְׂרָאֵל וְלָא אֲמַרוּ לְמִיסַק עֲלֵיהוֹן לְחֵילָא לְחַבָּלָא יַת אַרְעָא דִבְנֵי רְאוּבֵן וּבְנֵי גָד יַתְבִין בָּהּ:
וּקְרוֹ בְּנֵי רְאוּבֵן וּבְנֵי גָד לְמַדְבְּחָא אֲרֵי סָהִיד הוּא בֵּינָנָא אֲרֵי יְיָ הוּא אֱלֹהִים:

Chapter 23

וַהֲוָה לִזְמַן יוֹמִין סַגִיאִין בָּתַר דְאַנִיחַ יְיָ לְיִשְׂרָאֵל מִכָּל בַּעֲלֵי דְבָבֵיהוֹן מִסְחוֹר סְחוֹר וִיהוֹשֻׁעַ סִיב עַל בְּיוֹמִין:
וּקְרָא יְהוֹשֻׁעַ לְכָל יִשְׂרָאֵל לְסָבוֹהִי וּלְרֵישׁוֹהִי וּלְדַיָנוֹהִי וּלְסַרְכוֹהִי וַאֲמַר לְהוֹן אֲנָא סִיבִית עָלִית בְּיוֹמִין:
וְאַתּוּן חֲזֵיתוּן יַת כָּל דַעֲבַד יְיָ אֱלָהָכוֹן לְכָל עַמְמַיָא הָאִלֵין מִן קֳדָמֵיכוֹן אֲרֵי יְיָ אֱלָהָכוֹן מֵימְרֵיהּ מֵגִיחַ לְכוֹן:
חֲזוֹ דִי פַלְגִית לְכוֹן יַת עַמְמַיָא דִי אִשְׁתָּאֲרוּ הָאִלֵין בְּאַחֲסָנָא לְשִׁבְטֵיכוֹן מִן יַרְדְנָא וְכָל עַמְמַיָא דִי שֵׁיצֵיתִי וְיַמָא רַבָּא מַעֲלָנֵי שִׁמְשָׁא:
וַיָי אֱלָהָכוֹן הוּא יִתְבְּרִינוּן מִן קֳדָמֵיכוֹן וִיתָרֵךְ יַתְהוֹן מִן קֳדָמֵיכוֹן וְתֵרְתוּן יַת אַרְעֲהוֹן כְּמָא דִי מַלֵיל יְיָ אֱלָהָכוֹן לְכוֹן:
וְתִתְקְפוּן לַחֲדָא לְמִטַר וּלְמֶעְבַּד יַת כָּל דִכְתִיב בְּסֵפֶר אוֹרַיְתָא דְמֹשֶׁה בְּדִיל דְלָא לְמִיסְטֵי מִינֵיהּ לְיַמִינָא וְלִשְׂמָאלָא:
בְּדִיל דְלָא לְאִתְעֲרָבָא בְּעַמְמַיָא הָאִלֵין דִי אִשְׁתָּאֲרוּ הָאִלֵין עִמְכוֹן וּבְשׁוּם טַעֲוַתְהוֹן לָא תִדְכְּרוּן וְלָא תְקִימוּן וְלָא תִפְלְחוּנוּן וְלָא תִסְגְדוּן לְהוֹן:
אֱלָהֵן בְּדַחַלְתָּא דַייָ אֱלָהָכוֹן תִּדְבְּקוּן כְּמָא דַעֲבַדְתּוּן עַד יוֹמָא הָדֵין:
וְתָרֵיךְ יְיָ מִן קֳדָמֵיכוֹן עַמְמִין רַבְרְבִין וְתַקִיפִין וְאַתּוּן לָא קָם אֱנַשׁ קֳדָמֵיכוֹן עַד יוֹמָא הָדֵין:
גַבְרָא חָד מִנְכוֹן יִרְדוֹף אַלְפָא אֲרֵי יְיָ אֱלָהָכוֹן מֵימְרֵיהּ מֵגִיחַ לְכוֹן כְּמָא דִי מַלֵיל לְכוֹן:
וְתִסְתַּמְרוּן לַחֲדָא לְנַפְשָׁתֵיכוֹן לְמִרְחַם יַת יְיָ אֱלָהָכוֹן:
אֲרֵי אִם מֵיתַב תְּתוּבוּן וְתִדְבְּקוּן בִּשְׁאָר עַמְמַיָא הָאִלֵין דִי אִשְׁתָּאֲרוּ הָאִלֵין עִמְכוֹן וְתִתְחַתְּנוּן בְּהוֹן וְתִתְעַרְבוּן בְּהוֹן וְאִינוּן יִתְעַרְבוּן בְּכוֹן:
מִדַע תֵּדְעוּן אֲרֵי לָא יוֹסִיף מֵימְרָא דַייָ אֱלָהָכוֹן לְתָרָכָא יַת עַמְמַיָא הָאִלֵין מִן קֳדָמֵיכוֹן וִיהוֹן לְכוֹן לִתְבִיר וּלְתַקְלָא וּלְסִיעַן נַטְלָן זְיָן לְקִבְלֵיכוֹן וּלְמַשִׁרְיָן מַקְפַּנְכוֹן עַד דְתֵבְדוּן מֵעַל אַרְעָא טַבְתָא הָדָא דִי יְהַב לְכוֹן יְיָ אֱלָהָכוֹן:
וְהָא אֲנָא אָזֵיל יוֹמָא דֵין בְּאוֹרַח כָּל אַרְעָא וְתֵדְעוּן בְּכָל לִבְּכוֹן וּבְכָל נַפְשְׁכוֹן אֲרֵי לָא בַטִיל פִּתְגָמָא חָדָא מִכָּל פִּתְגָמַיָא תַקְנַיָא דִי מַלֵיל יְיָ אֱלָהָכוֹן עֲלֵיכוֹן כּוּלְהוֹן אִתְקַיְמוּ לְכוֹן לָא בָטֵיל מִנְכוֹן פִּתְגָמָא חָדָא:
וִיהֵי כְּמָא דְאִתְקַיָמוּ עֲלֵיכוֹן כָּל פִּתְגָמַיָא תַקְנַיָא דִי מַלֵיל יְיָ אֱלָהָכוֹן עֲלֵיכוֹן כֵּן יַיְתֵי יְיָ עֲלֵיכוֹן יַת כָּל פִּתְגָמַיָא בִּישַׁיָא עַד דִישֵׁיצֵי יַתְכוֹן מֵעַל אַרְעָא טַבְתָא הָדָא דִי יְהַב לְכוֹן יְיָ אֱלָהָכוֹן:
בְּמֶעְבַּרְכוֹן יַת קְיָמָא דַייָ אֱלָהָכוֹן דְפַקֵד יַתְכוֹן וְתַהֲכוּן וְתִפְלְחוּן לְטַעֲוַת עַמְמַיָא וְתִסְגְדוּן לְהוֹן וְיִתְקֵיף רוּגְזָא דַייָ בְּכוֹן וְתֵיבְדוּן בִּפְרִיעַ מֵעַל אַרְעָא טַבְתָא דִיהַב לְכוֹן:

Chapter 24

וּכְנַשׁ יְהוֹשֻׁעַ יַת כָּל שִׁבְטַיָא דְיִשְׂרָאֵל לִשְׁכֶם וּקְרָא לְסָבֵי יִשְׂרָאֵל וּלְרֵישׁוֹהִי וּלְדַיָנוֹהִי וּלְסַרְכוֹהִי וְאִתְעַתְּדוּ קֳדָם יְיָ:
וַאֲמַר יְהוֹשֻׁעַ לְכָל עַמָא כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל בְּעִבַּר פְּרָת יְתַבוּ אֲבָהַתְכוֹן מֵעַלְמָא תֶּרַח אֲבוּהִי דְאַבְרָהָם וַאֲבוּהִי דְנָחוֹר וּפְלָחוּ לְטַעֲוַת עַמְמַיָא:
וְדַבְּרֵית יַת אֲבוּכוֹן יַת אַבְרָהָם מֵעִבַּר פְּרָת וְאוֹבֵלִית יָתֵיהּ בְּכָל אַרְעָא דִכְנָעַן וְאַסְגֵיתִי יַת בְּנוֹהִי וִיהָבִית לֵיהּ יַת יִצְחָק:
וִיהָבִית לְיִצְחָק יַת יַעֲקֹב וְיַת עֵשָׂו וִיהָבִית לְעֵשָׂו יַת טוּרָא דְשֵׂעִיר לְמֵירַת יָתֵיהּ וְיַעֲקֹב וּבְנוֹהִי נְחָתוּ לְמִצְרָיִם:
וְשַׁלְחֵית יַת מֹשֶׁה וְיַת אַהֲרֹן וּמָחִית יַת מִצְרָאֵי כְּמָא דְעַבְדִית בֵּינֵיהוֹן וּבָתַר כֵּן אַפְקֵית יַתְכוֹן:
וְאַפְקֵית יַת אֲבָהַתְכוֹן מִמִצְרַיִם וַאֲתֵיתוּן לְיַמָא וּרְדָפוּ מִצְרָאֵי בָּתַר אֲבָהַתְכוֹן בִּרְתִיכִין וּבְפָרָשִׁין לְיַמָא דְסוּף:
וּצְעִיקוּ קֳדָם יְיָ וְשַׁוִי קְבָלָא בֵּינֵיכוֹן וּבֵין מִצְרָאֵי וְאַיְתֵי עֲלוֹהִי יַת יַמָא וַחֲפָנוּן וַחֲזָאָה עֵינֵיכוֹן יַת דִי עַבְדֵית בְּמִצְרַיִם וִיתֵיבְתּוּן בְּמַדְבְּרָא יוֹמִין סַגִיאִין:
וְאַעֲלֵית יַתְכוֹן לַאֲרַע אֱמוֹרָאָה דְיָתֵיב בְּעִבְרָא דְיַרְדְנָא וְאַגִיחוּ קְרָבָא עִמְכוֹן וּמְסָרִית יַתְהוֹן בְּיֶדְכוֹן וִירִיתְתּוּן יַת אַרְעֲהוֹן וְשֵׁיצִיתִינוּן מִן קֳדָמֵיכוֹן:
וְקָם בָּלָק בַּר צִפּוֹר מַלְכָּא דְמוֹאָב וְאַגִיחַ קְרָבָא בְּיִשְׂרָאֵל וּשְׁלַח וּקְרָא לְבִלְעָם בַּר בְּעוֹר לְמֵילַט יַתְכוֹן:
וְלָא אֲבִיתִי לְקַבָּלָא מִן בִּלְעָם וּבָרִיךְ בְּרָכָא יַתְכוֹן וְשֵׁיזְבוּת יַתְכוֹן מִן יְדֵיהּ:
וַעֲבַרְתּוּן יַת יַרְדְנָא וַאֲתֵיתוּן לִירִיחוֹ וְאַגִיחוּ קְרָבָא בְּכוֹן יַתְבֵי יְרִיחוֹ אֱמוֹרָאֵי וּפְּרִזָאֵי וּכְנַעֲנָאֵי וְחִתָּאֵי וְגִרְגָשָׁאֵי חִוָאֵי וִיבוּסָאֵי וּמַסְרֵית יַתְהוֹן בְּיֶדְכוֹן:
וּשְׁלָחִית קֳדָמֵיכוֹן יַת עַרְעִיתָא וְתָרְכֵית יַתְהוֹן מִן קֳדָמֵיכוֹן תְּרֵין מַלְכֵי אֱמוֹרָאֵי לָא בְחַרְבָּךְ וְלָא בְקַשְׁתָּךְ:
וְיַהֲבֵית לְכוֹן אַרְעָא דִי לָא לְאִיתוּן בָּהּ וְקִרְוִין דִי לָא בְנִיתוּן וִיתֵיבְתּוּן בְּהוֹן כַּרְמִין וְזֵיתִין דִי לָא נְצֵיבְתּוּן אַתּוּן אָכְלִין:
וּכְעַן דְחִילוּ מִן קֳדָם יְיָ וּפְלַחוּ קֳדָמוֹהִי בִּשְׁלָמוּתָא וּבְקוּשְׁטָא וְאַעֲדוּ יַת טַעֲוָתָא דִי פַלְחוּ אֲבָהַתְכוֹן בְּעִבַר פְּרָת וּבְמִצְרַיִם וּפְלַחוּ קֳדָם יְיָ:
וְאִם בִּישׁ בְּעֵינֵיכוֹן לְמִפְלַח קֳדָם יְיָ בְּחָרוּ לְכוֹן יוֹמָא דֵין קֳדָם מָן תִּפְלְחוּן אִם יַת טַעֲוָתָא דִי פְלַחוּ אֲבָהַתְכוֹן דִבְעִיבְרָא דִפְרָת וְאִם יַת טַעֲוַת אֱמוֹרָאֵי דִי אַתּוּן יַתְבִין בְּאַרְעֲהוֹן בְּרַם אֲנָא וֶאֱנַשׁ בֵּיתִי נִפְלַח קֳדָם יְיָ:
וַאֲתֵיב עַמָא וַאֲמַר חַס לָנָא מִלְמִשְׁבַּק יַת פּוּלְחָנָא דַייָ לְמִפְלַח לְטַעֲוַת עַמְמַיָא:
אֲרֵי יְיָ אֱלָהָנָא הוּא דְאַסֵיק יָתָנָא וְיַת אַבְהָתָנָא מֵאַרְעָא דְמִצְרַיִם מִבֵּית עַבְדוּתָא וְדִי עֲבַד לְעֵינָנָא יַת אָתַיָא וְיַת רַבְרְבָתָא הָאִלֵין וּנְטָרָנָא בְּכָל אוֹרְחָא דִי הֲלִיכְנָא בָּהּ וּבְכָל עַמְמַיָא דִי עָבַרְנָא בֵּינֵיהוֹן:
וְתָרֵיךְ יְיָ יַת כָּל עַמְמַיָא וְיַת אֱמוֹרָאָה יָתֵיב אַרְעָא מִן קֳדָמָנָא אַף אֲנַחְנָא נִפְלַח קֳדָם יְיָ אֲרֵי הוּא אֱלָהָנָא:
וַאֲמַר יְהוֹשֻׁעַ לְעַמָא לָא תוּכְלוּן לְמִפְלַח קֳדָם יְיָ אֲרֵי אֱלָהָא קַדִישָׁא הוּא אֵל קַנָא הוּא מִן קֳדָם דִין דִקְשׁוֹט לָא יִשְׁבּוֹק לְחוֹבֵיכוֹן וּלְחַטָאֵיכוֹן:
אֲרֵי תִשְׁבָּקוּן יַת פּוּלְחָנָא דַייָ וְתִפְלְחוּן טַעֲוַת עַמְמַיָא וִיתוּב וְיַבְאֵשׁ לְכוֹן וִישֵׁיצֵי יַתְכוֹן בָּתַר דְאוֹטֵיב לְכוֹן:
וַאֲמַר עַמָא לִיהוֹשֻׁעַ לָא אֱלָהֵן קֳדָם יְיָ נִפְלָח:
וַאֲמַר יְהוֹשֻׁעַ לְעַמָא סַהֲדִין אַתּוּן בְּנַפְשַׁתְכוֹן אֲרֵי אַתּוּן בְּחַרְתּוּן לְכוֹן יַת פּוּלְחָנָא דַייָ לְמִפְלַח קֳדָמוֹהִי וַאֲמַרוּ סַהֲדִין:
וּכְעַן אַעֲדוֹ יַת טַעֲוַת עַמְמַיָא דִי בֵינֵיכוֹן וְאַפְנוּ יַת לִבְּכוֹן לְפוּלְחָנָא דַייָ אֱלָהָא דְיִשְׂרָאֵל:
וַאֲמָרוּ עַמָא לִיהוֹשֻׁעַ יַת יְיָ אֱלָהָנָא נִפְלַח וּלְמֵימְרֵיהּ נְקַבֵּיל:
וּגְזַר יְהוֹשֻׁעַ קְיָם לְעַמָא בְּיוֹמָא הַהוּא וְשַׁוִי לֵיהּ קְיָם וְדִין בִּשְׁכֶם:
וּכְתַב יְהוֹשֻׁעַ יַת פִּתְגָמַיָא הָאִלֵין וְאַצְנְעִינוּן בְּסֵפֶר אוֹרַיְתָא דַייָ וּנְסִיב אַבְנָא רַבְּתָא וַאֲקִימָהּ תַּמָן תְּחוֹת אַלְתָא דִי בְּבֵית מַקְדְשָׁא דַיָי:
וַאֲמַר יְהוֹשֻׁעַ לְכָל עַמָא הָא אַבְנָא הָדָא תְּהֵי לָנָא כִּתְרֵין לוּחֵי אֶבֶן קְיָמָא אֲרֵי יָתָהּ עֲבַדְנָא לְסַהֲדוּ אֲרֵי פִּתְגָמַיָא דִכְתִיבִין עֲלָהּ מֵעֵין כָּל פִּתְגָמַיָא דִי מַלֵיל עִמָנָא וּתְהֵי בְכוֹן לְדָכְרָן וּלְסַהֲדוּ דִילְמָא תְכַדְבוּן קֳדָם אֱלָהָכוֹן:
וּשְׁלַח יְהוֹשֻׁעַ יַת עַמָא גְבַר לְאַחְסַנְתֵּיהּ:
וַהֲוָה בָּתַר פִּתְגָמַיָא הָאִלֵין וּמִית יְהוֹשֻׁעַ בַּר נוּן עַבְדָא דַייָ בַּר מְאָה וַעֲשַׂר שְׁנִין:
וּקְבַרוּ יָתֵיהּ בִּתְחוּם אַחֲסַנְתֵּיהּ בְּתִמְנַת סֶרַח דִי בְּטוּרָא דְבֵית אֶפְרַיִם מִצִפּוּנָא לְטוּר גָעַשׁ:
וּפְלַח יִשְׂרָאֵל קֳדָם יְיָ כָּל יוֹמֵי יְהוֹשֻׁעַ וְכָל יוֹמֵי סָבַיָא דִי אוֹרִיכוּ יוֹמִין בָּתַר יְהוֹשֻׁעַ וְדִי יְדָעוּ יַת כָּל עוֹבְדָא דַייָ דִי עֲבַד לְיִשְׂרָאֵל:
וְיַת גַרְמֵי יוֹסֵף דְאַסִיקוּ בְּנֵי יִשְׂרָאֵל מִמִצְרַיִם קְבָרוּ בִּשְׁכֶם בְּאַחְסָנַת חַקְלָא דִקְנָא יַעֲקֹב מִבְּנֵי חֲמוֹר אֲבוּהִי דִשְׁכֶם בְּמָא חוּרְפַן וַהֲווֹ לִבְנֵי יוֹסֵף לְאַחֲסָנָא:
וְאֶלְעָזָר בַּר אַהֲרֹן מִית וּקְבָרוּ יָתֵיהּ בְּגִבְעֲתָא דְפִּנְחָס בְּרֵיהּ דְאִתְיְהִיבַת לֵיהּ בְּטוּרָא דְבֵית אֶפְרָיִם:
