תרגום יונתן על שופטים



Chapter 1

וַהֲוָה בָּתַר דְמִית יְהוֹשֻׁעַ וּשְׁאִילוּ בְּנֵי יִשְׂרָאֵל בְּמֵימְרָא דַייָ לְמֵימַר מַן יִסַק לָנָא לְוַת כְּנַעֲנָאָה בְּקַדְמֵיתָא לְאַגָחָא קְרָבָא בֵּיהּ:
וַאֲמַר יְיָ יְהוּדָה יִסַק הָא מַסְרִית יַת יַתְבֵי אַרְעָא בִּידֵיהּ:
וַאֲמַר יְהוּדָה לְשִׁמְעוֹן אֲחוּהִי סַק עִמִי בְּעַדְבִי וּנְגִיחַ קְרָבָא עִם כְּנַעֲנָאָה וְאֵהַךְ אַף אֲנָא עִמָךְ בְּעַדְבָךְ וַאֲזַל עִמֵיהּ שִׁמְעוֹן:
וּסְלִיקוּ דְבֵית יְהוּדָה וּמְסַר יְיָ יַת כְּנַעֲנָאָה וּפְרִיזָאָה בִּידֵיהוֹן וּמְחָנוּן בְּבֶזֶק עַסְרָא אַלְפִין גַבְרָא:
וְאַשְׁכָּחוּ יַת אֲדֹנִי בֶזֶק בְּבֶזֶק וְאַגִיחוּ קְרָבָא בֵּיהּ וּמְחוֹ יַת כְּנַעֲנָאֵי וְיַת פְּרִיזָאֵי:
וַעֲרַק אֲדֹנִי בֶזֶק וּרְדָפוּ בַּתְרוֹהִי וַאֲחַדוּ יָתֵיהּ וְקַצִיצוּ יַת קַרְסֻלֵי יְדוֹהִי וְרַגְלוֹהִי:
וַאֲמַר אֲדֹנִי בֶזֶק שַׁבְעִין מַלְכִין קַרְסֻלֵי יְדֵיהוֹן וְרַגְלֵיהוֹן מְקַצְצִין הֲווֹ מְלַקְטִין לַחְמָא תְּחוֹת פָּתוֹרִי כְּמָא דְעַבְדִית כֵּן שַׁלֵים לִי יְיָ וְאַיְתוֹהִי לִירוּשְׁלֵם וּמִית תַּמָן:
וְאַגִיחוּ קְרָבָא בְנֵי יְהוּדָה בִּירוּשְׁלֵם וּכְבַשׁוּ יָתָהּ וּמְחוֹהָא לְפִתְגַם דְחָרֶב וְיַת קַרְתָּא אוֹקִידוּ בְנוּרָא:
וּבָתַר כֵּן נְחָתוּ בְּנֵי יְהוּדָה לְאַגָחָא קְרָבָא בִּכְנַעֲנָאֵי יָתֵיב טוּרָא וְדָרוֹמָא וּשְׁפֶלְתָּא:
וַאֲזַלוּ דְבֵית יְהוּדָה לְוַת כְּנַעֲנָאֵי דְיַתְבִין בְּחֶבְרוֹן וְשׁוּם חֶבְרוֹן מִלְקַדְמִין קִרְיַת אַרְבַּע וּמְחוֹ יַת שֵׁשַׁי וְיַת אֲחִימָן וְיַת תַּלְמָי:
וַאֲזַלוּ מִתַּמָן לְוַת יַתְבֵי דְבִיר וְשׁוּם דְבִיר מִלְקַדְמִין קִרְיַת אַרְכֵי:
וַאֲמַר כָּלֵב דְיִמְחֵי יַת קִרְיַת אַרְכֵי וְיִכְבְּשִׁינָהּ וְאֶתֵּן לֵיהּ יַת עַכְסָה בְּרַתִּי לְאִתּוּ:
וְכַבְּשָׁהּ עָתְנִיאֵל בַּר קְנַז אֲחוֹהִי דְכָלֵב דִזְעֵיר מִנֵיהּ וִיהַב לֵיהּ יַת עַכְסָה בְרַתֵּיהּ לְאִתּוּ:
וַהֲוָה בְּמֵעֲלָהּ וְאַמְלַכְתֵּיהּ לְמִשְׁאַל מִן אֲבוּהָא אַחְסַנְתָּא וְאִתְרְכִינַת מֵעַל חֲמָרָא וַאֲמַר לָהּ כָּלֵב מַה לִיךְ:
וַאֲמָרַת לֵיהּ הַב לִי אַחֲסַנְתָּא אֲרֵי לַאֲרַע דָרוֹמָא יְהַבְתָּנִי וְתִתֶּן לִי אֲתַר בֵּית שַׁקְיָא דְמַיָא וִיהַב לָהּ כָּלֵב יַת בֵּית שַׁקְיָא עִלָאָה וְיַת בֵּית שַׁקְיָא תַּחְתָּאָה:
וּבְנֵי שַׁלְמָאָה חָמוּהִי דְמֹשֶׁה סְלִיקוּ מִן קַרְתָּא יְרִיחוֹ עִם בְּנֵי יְהוּדָה מִדְבַּר יְהוּדָה דִי בְּדָרוֹם עֲרָד וַאֲזַלוּ וִיתַבוּ עִם עַמָא:
וַאֲזַל יְהוּדָה עִם שִׁמְעוֹן אֲחוֹהִי וּמְחוֹ יַת כְּנַעֲנָאָה יָתֵיב צְפַת וְגַמַרוּ יָתָהּ וּקְרָא יַת שְׁמָא דְקַרְתָּא חָרְמָה:
וּכְבַשׁ יְהוּדָה יַת עַזָה וְיַת תְּחוּמָא וְיַת אַשְׁקְלוֹן וְיַת תְּחוּמָא וְיַת עֶקְרוֹן וְיַת תְּחוּמָא:
וַהֲוָה מֵימְרָא דַייָ בְּסַעֲדֵיהּ דְבֵית יְהוּדָה וְתָרִיכוּ יַת יַתְבֵי טוּרָא וּבָּתַר כֵּן דְחָבוּ לָא יְכִילוּ לְתָרָכָא יַת יַתְבֵי מֵישְׁרָא אֲרֵי דְתִיכִין דְבַרְזְלָא לְהוֹן:
וִיהַבוּ לְכָלֵב יַת חֶבְרוֹן כְּמָא דְמַלֵיל מֹשֶׁה וְתָרֵיךְ מִתַּמָן יַת תְּלָתָא בְּנֵי גִבָּרַיָא:
וְיַת יְבוּסָאֵי יַתְבֵי יְרוּשְׁלֵם לָא תָרִיכוּ בְּנֵי בִנְיָמִין וִיתִיבוּ יְבוּסָאֵי עִם בְּנֵי בִנְיָמִין בִּירוּשְׁלֵם עַד יוֹמָא הָדֵין:
וּסְלִיקוּ דְבֵית יוֹסֵף אַף אִנוּן לְבֵית אֵל וּמֵימְרָא דַייָ בְּסַעְדְהוֹן:
וּשְׁלַחוּ מְאַלְלִין דְבֵית יוֹסֵף לְבֵית אֵל וְשׁוּם קַרְתָּא מִלְקַדְמִין לוּז:
וַחֲזוֹ נָטְרַיָא גַבְרָא נָפֵיק מִן קַרְתָּא וַאֲמַרוּ לֵיהּ אַחֲזֵינָא כְעַן יַת מַעֲלָנָא דְקַרְתָּא וְנַעְבֵּד עִמָךְ טִיבוּ:
וְאַחֲזִינוּן יַת מַעֲלָנָא דְקַרְתָּא וּמְחוֹ יַת קַרְתָּא לְפִתְגַם דְחָרֶב וְיַת גַבְרָא וְיַת כָּל זַרְעִיתֵיהּ שֵׁזִיבוּ:
וַאֲזַל גַבְרָא לְאַרְעָא חִתָּאֵי וּבְנָא קַרְתָּא וּקְרָא שְׁמָהּ לוּז הוּא שְׁמָהּ עַד יוֹמָא הָדֵין:
וְלָא תָרִיכוּ דְבֵית מְנַשֶׁה יַת בֵּית שְׁאָן וְיַת כַּפְרְנָהָא וְיַת תַּעֲנָךְ וְיַת כַּפְרְנָהָא וְיַת יַתְבֵי דוֹר וְיַת כַּפְרְנָהָא וְיַת יַתְבֵי יִבְלְעָם וְיַת כַּפְרְנָהָא וְיַת יַתְבֵי מְגִדוֹ וְיַת כַּפְרְנָהָא וּשְׁרֵי כְנַעֲנָאָה לְמִתַּב בְּאַרְעָא הָדָא:
וַהֲוָה כַּד תַּקֵיף יִשְׂרָאֵל וְשַׁוֵי יַת כְּנַעֲנָאָה לְמַסְקֵי מִסִין וְתָרָכָא לָא תָרְכִינוּן:
וּדְבֵית אֶפְרַיִם לָא תָרִיךְ יַת כְּנַעֲנָאֵי דְיַתְבִין בְּגָזֶר וִיתִבוּ כְּנַעֲנָאֵי בֵינֵיהוֹן בְּגָזֶר:
דְּבֵית זְבוּלוּן לָא תָרִיכוּ יַת יַתְבֵי קִטְרוֹן וְיַת יַתְבֵי נַהֲלֹל וִיתִבוּ כְנַעֲנָאֵי בֵּינֵיהוֹן וַהֲווֹ לְמַסְקֵי מִסִין:
דְבֵית אָשֵׁר לָא תָרִיכוּ יַת יַתְבֵי עַכּוֹ וְיַת יַתְבֵי צִידוֹן וְיַת אַחְלָב וְיַת אַכְזִיב וְיַת חֶלְבָּה וְיַת אֲפִיק וְיַת רְחוֹב:
וִיתִיבוּ דְבֵית אָשֵׁר בְּגוֹ כְּנַעֲנָאֵי יַתְבֵי אַרְעָא אֲרֵי לָא תָרְכִינוּן:
דְבֵית נַפְתָּלִי לָא תָרִיכוּ יַת יַתְבֵי בֵית שֶׁמֶשׁ וְיַת יַתְבֵי בֵית עֲנָת וִיתִיבוּ בְּגוֹ כְנַעֲנָאֵי יַתְבֵי אַרְעָא וְיַתְבֵי בֵית שֶׁמֶשׁ וּבֵית עֲנָת הֲווֹ לְהוֹן לְמַסְקֵי מִסִין:
וּדְחַקוּ אֱמוֹרָאֵי יַת בְּנֵי דָן לְטוּרָא אֲרֵי לָא שַׁבְקוּנוּן לְמֵחַת לְמֵישְׁרָא:
וּשְׁרִי אֱמוֹרָאָה לְמִתַּב בְּטוּר חֶרֶס בְּאַיָלוֹן וּבְשַׁעַלְבִים וּתְקֵיפַת יַד בֵּית יוֹסֵף וַהֲווֹ לְמַסְקֵי מִסִין:
וּתְחוּם אֱמוֹרָאָה מִמַסְקְנֵי עַקְרַ בִּים מִן כֵּיפָא וּלְעֵילָא:

Chapter 2

וּסְלִיק נְבִיָא בִּשְׁלִיחוּת מִן קֳדָם יְיָ מִן גִלְגְלָא לַבּוֹכִים וַאֲמַר אֲסֵקִית יַתְכוֹן מִמִצְרַיִם וְאַעֲלִית יַתְכוֹן לְאַרְעָא דִי קַיֵמִית לַאֲבָהַתְכוֹן וַאֲמַרִית לָא אַשְׁנֵי קְיָמַי עִמְכוֹן לְעֳלָם:
וְאַתּוּן לָא תִגְזְרוּן קְיָם לְיַתְבֵי אַרְעָא הָדָא אֲגוֹרֵיהוֹן תְּתָרְעוּן וְלָא קַבֵּלְתּוּן לְמֵימְרִי מָא דָא עֲבַדְתּוּן:
וְאַף אַמְרֵית לָא אֲתָרֵיךְ יַתְהוֹן מִן קֳדָמֵיכוֹן וִיהוֹן לְכוֹן לִמְעִיקִין וְטַעֲוַתְהוֹן יְהוֹן לְכוֹן לְתַקָלָא:
וַהֲוָה כַּד מַלֵיל נְבִיָא דַייָ יַת פִּתְגָמַיָא הָאִלֵין לְכָל בְּנֵי יִשְׂרָאֵל וַאֲרִימוּ עַמָא יַת קַלְהוֹן וּבְכוֹ:
וּקְרוֹ שְׁמָא דְאַתְרָא הַהוּא בּוֹכִים וּנְכִיסוּ תַמָן נִכְסַת קוּדְשַׁיָא קֳדָם דַיָי:
וּשְׁלַח יְהוֹשֻׁעַ יַת עַמָא וַאֲזַלוּ בְּנֵי יִשְׂרָאֵל גְבַר לְאַחְסַנְתֵּיה לְמֵירַת יַת אַרְעָא:
וּפְלַח עַמָא קֳדָם יְיָ כָּל יוֹמֵי יְהוֹשֻׁעַ וְכֹל יוֹמֵי סָבַיָא דִי אוֹרִיכוּ יוֹמִין בָּתַר יְהוֹשֻׁעַ דַחֲזוֹ יַת כָּל עוֹבָדָא דַייָ רַבָּא דִי עֲבַד לְיִשְׂרָאֵל:
וּמִית יְהוֹשֻׁעַ בַּר נוּן עַבְדָא דַייָ בַּר מְאָה וַעֲשַׂר שְׁנִין:
וּקְבַרוּ יָתֵיהּ בִּתְחוּם אַחְסַנְתֵּיהּ בְּתִמְנַת חֶרֶס בְּטוּרָא דְבֵית אֶפְרַיִם מִצִיפּוּנָא לְטוּר גָעַשׁ:
וְאַף כָּל דָרָא הַהוּא אִתְכְּנִישׁוּ לְוַת אֲבָהָתוֹהִי וְקָם דָרָא אוֹחֲרָנָא בַּתְרֵיהוֹן דְלָא יַדְעִין לְמִדְחַל מִן קֳדָם יְיָ וְאַף יַת עוֹבָדָא דִי עֲבַד לְיִשְׂרָאֵל:
וַעֲבַדוּ בְּנֵי יִשְׂרָאֵל יַת דְבִישׁ קֳדָם יְיָ וּפְלַחוּ יַת בַּעֲלַיָא:
וּשְׁבַקוּ יַת פּוּלְחָנָא דַייָ אֱלָהָא דַאַבָהַתְהוֹן דְאַפֵּיק יַתְהוֹן מֵאַרְעָא דְמִצְרַיִם וּטְעוֹ בָּתַר טַעֲוַת עַמְמַיָא מִטַעֲוַת עַמְמַיָא דִי בְסַחֲרָנֵיהוֹן וּסְגִידוּ לָהֶן וְאַרְגִיזוּ קֳדָם יְיָ:
וּשְׁבַקוּ יַת פּוּלְחָנָא דַייָ וּפְלַחוּ לְבַעֲלָא וּלְעַשְׁתָּרָתָא:
וּתְקֵף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְּיַד בוֹזְזֵיהֶן וּבְזוֹ יַתְהוֹן וּמַסְרִינוּן בְּיַד בַּעֲלֵי דְבָבֵיהוֹן מִסְחוֹר סְחוֹר וְלָא יָכִילוּ עוֹד לְמֵיקָם קֳדָם בַּעֲלֵי דְבָבֵיהוֹן:
בְּכָל אֲתַר דִי נְפַקוּ מָחָא מִן קֳדָם יְיָ הֲוַת בְּהוֹן לְבִישָׁא כְּמָא דְמַלֵיל יְיָ וּכְמָא דְקַיֵם יְיָ לְהוֹן וְעָקַת לְהוֹן לַחֲדָא:
וַאֲקֵים יְיָ נְגִידִין וּפַרְקִינוּן מִיַד בָּזְזֵיהוֹן:
וְאַף מִן נְגִידֵיהוֹן לָא קַבִּילוּ אֲרֵי טָעוּ בָּתַר טַעֲוַת עַמְמַיָא וּסְגִידוּ לְהוֹן סָטוּ בִּפְרִיעַ מִן אוֹרְחָא דַאֲזַלוּ אֲבָהַתְהוֹן לְקַבָּלָא פִּקוּדַיָא דַייָ לָא עֲבַדוּ כֵן:
וַאֲרֵי אָקֵים יְיָ לְהוֹן נְגִידִין וַהֲוָה מֵימְרָא דַייָ בְּסַעֲדֵיהּ דִנְגִידָא וּפָרִיק לְהוֹן מִיַד בַּעֲלֵי דְבָבֵיהוֹן כָּל יוֹמֵי נְגִידָא אֲרֵי מְתִיב מִמֵימְרָא דַאֲמַר יְיָ וּמְקַבֵּל צְלוֹתְהוֹן וּפָרִיק לְהוֹן מִן קֳדָם דַחֲקֵיהוֹן וּמְשַׁעְבְּדֵיהוֹן:
וַהֲוָה כַּד מָאִית נְגִידָא תַּיְבִין וּמְחַבְּלִין מֵאֲבָהַתְהוֹן לִמְהַךְ בָּתַר טַעֲוַת עַמְמַיָא לְמִפְלַחְהֶן וּלְמִסְגַד לָהֶן לָא שַׁבְקִין מֵעוֹבְדֵיהוֹן בִּישַׁיָא וּמֵאוֹרְחַתְהוֹן מְקַלְקְלָתָא:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וַאֲמַר חֲלַף דִי עֲבָרוּ עַמָא הָדֵין עַל קַיָמִי דִי פַקֵדִית יַת אֲבָהַתְהוֹן וְלָא קַבִּילוּ לְמֵימְרִי:
אַף אֲנָא לָא אוֹסִיף לְתָרָכָא אֱנָשׁ מִן קֳדָמֵיהוֹן מִן עַמְמַיָא דִי שְׁבַק יְהוֹשֻׁעַ וּמִית:
בְּדִיל לְנַסָאָה בְּהוֹן יַת יִשְׂרָאֵל הֲנַטְרִין אִינוּן יַת אוֹרְחָן דְתַקְנַן קֳדָם יְיָ לִמְהַךְ בְּהוֹן כְּמָא דִי נְטַרוּ אֲבָהַתְהוֹן אִם לָא:
וּשְׁבַק יְיָ יַת כָּל עַמְמַיָא הָאִלֵין בְּדִיל דְלָא לְתָרָכוּתְהוֹן בִּפְרִיעַ וְלָא מַסְרִינוּן בִידָא דִיהוֹשֻׁעַ:


Chapter 3

וְאִלֵין עַמְמַיָא דִי שְׁבַק יְיָ לְנַסָאָה בְהוֹן יַת יִשְׂרָאֵל יַת כָּל דְלָא הֲווֹ יָדְעִין יַת כָּל קְרָבֵי כְנָעַן:
לְחוֹד בְּדִיל דְיֵדְעוּן דָרַיָא דִבְנֵי יִשְׂרָאֵל לַאֲלָפוּתְהוֹן קְרָבָא לְחוֹד דְמִלְקַדְמִין לָא הֲווֹ יָדְעִין:
חַמְשָׁא טוּרְנֵי פְּלִשְׁתָּאֵי וְכָל כְּנַעֲנָאֵי וְצִידוֹנָאֵי וְחִוָאֵי יַתְבֵי טוּרָא דִלְבָנוֹן מִטוּר מֵישַׁר חֶרְמוֹן עַד מַעֲלָנָא דַחֲמָת:
וַהֲווֹ לְנַסָאָה בְּהוֹן יַת יִשְׂרָאֵל לְמִידַע הַיְקַבְּלוּן יַת פִּקוּדַיָא דַייָ דְפַקֵד יַת אֲבָהַתְהוֹן בִּידָא דְמֹשֶׁה:
וּבְנֵי יִשְׂרָאֵל יְתִיבוּ בְּגוֹ כְנַעֲנָאֵי חִתָּאֵי וֶאֱמוֹרָאֵי וּפְרִיזָאֵי וְחִוָאֵי וִיבוּסָאֵי:
וּנְסִיבוּ יַת בְּנָתֵיהוֹן לְהוֹן לִנְשִׁין וְיַת בְּנַתְהוֹן יְהִיבוּ לִבְנֵיהוֹן וּפְלַחוּ יַת טַעֲוַתְהוֹן:
וַעֲבַדוּ בְנֵי יִשְׂרָאֵל יַת דְבִישׁ קֳדָם יְיָ וְאִתְנְשִׁיאוּ יַת פּוּלְחָנָא דַייָ אֱלָהָהוֹן וּפְלַחוּ יַת בְּעָלַיָא וְלַאֲשָׁרָתָא:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְּיַד כּוּשַׁן חַיָבָא מַלְכָּא דַאֲרָם דִי עַל פְּרָת וּפְלַחוּ בְּנֵי יִשְׂרָאֵל יַת כּוּשַׁן חַיָבָא תְּמָנֵי שְׁנִין:
וּזְעִיקוּ בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ וְאֲקֵים יְיָ פָרִיק לִבְנֵי יִשְׂרָאֵל וּפַרְקִינוּן יַת עָתְנִיאֵל בַּר קְנַז אֲחוּהִי דְכָלֵב דִזְעֵיר מִנֵיהּ:
וּשְׁרַת עֲלוֹהִי רוּחַ נְבוּאָה מִן קֳדָם יְיָ וְדָן יַת יִשְׂרָאֵל וּנְפַק לְאַגָחָא קְרָבָא וּמְסַר יְיָ בִּידוֹהִי יַת כּוּשַׁן חַיָבָא מַלְכָּא דַאֲרָם דִי עַל פְרָת וּתְקֵפַת יְדֵיהּ עַל כּוּשַׁן חַיָבָא:
וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל אַרְבְּעִין שְׁנִין וּמִית עָתְנִיאֵל בַּר קְנַז:
וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וּתְקֵיף יְיָ יַת עֶגְלוֹן מַלְכָּא דְמוֹאָב עַל יִשְׂרָאֵל עַל דַעֲבָדוּ יַת דְבִישׁ קֳדָם יְיָ:
וּכְנַשׁ לְוָתֵיה יַת בְּנֵי עַמוֹן וַעֲמָלֵק וַאֲזַל וּמְחָא יַת יִשְׂרָאֵל וִירִיתוּ יַת קַרְתָּא יְרִיחוֹ:
וּפְלַחוּ בְּנֵי יִשְׂרָאֵל יַת עֶגְלוֹן מַלְכָּא דְמוֹאָב תַּמְנֵי עַסְרֵי שְׁנִין:
וּזְעִיקוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ וַאֲקֵים יְיָ לְהוֹן פָּרִיק יַת אֵהוּד בַּר גֵרָא בַּר שִׁבְטָא דְבִנְיָמִין גַבְרָא גְמִיד בִּידֵיהּ דְיַמִינָא וְשַׁדָרוּ בְנֵי יִשְׂרָאֵל בִּידֵיהּ תִּקְרוּבְתָּא לְעֶגְלוֹן מַלְכָּא דְמוֹאָב:
וַעֲבַד לֵיהּ אֵהוּד חַרְבָּא וְלָהּ תְּרֵין פּוּמִין גַרְמִידָא אָרְכָּא וְזָרֵיז יָתָהּ מִלְרַע לִלְבוּשׁוֹהִי עַל יַרְכֵיהּ דְיַמִינֵיהּ:
וְקָרֵיב יַת תִּקְרוּבְתָּא לְעֶגְלוֹן מַלְכָּא דְמוֹאָב וְעֶגְלוֹן גְבַר פַטִים לַחֲדָא:
וַהֲוָה כַּד שֵׁיצֵי לְקָרָבָא יַת תִּקְרוּבְתָּא וּשְׁלַח יַת עַמָא נַטְלֵי תִּקְרוּבְתָּא:
וְהוּא תַב מִן מַחֲצָבַיָא דְעִם גִלְגָלָא וַאֲמַר פִּתְגָמָא דְסִתְרָא אִית לִי לְמַלָלָא עִמָךְ מַלְכָּא וַאֲמַר סְלִיק וּנְפַקוּ מֵעִלַווֹהִי כָּל דְקַיְמִין עֲלוֹהִי:
וְאֵהוּד עָאל לְוָתֵיהּ וְהוּא יָתֵיב בַּעֲלִיַת בֵּית קַיְטָא דִילֵיהּ בִּלְחוֹדוֹהִי וַאֲמַר אֵהוּד פִּתְגָמָא דַייָ לִי לְמַלָלָא עִמָךְ וְקָם מֵעַל כּוּרְסֵיהּ:
וְאוֹשִׁיט אֵהוּד יַת יַד שְׂמָאלֵיהּ וּנְסִיב יַת חַרְבָּא מֵעַל יַרְכָא דְיַמִינֵיהּ וּקְבָעָה בִּמְעוֹהִי:
וְעַל אַף דִסְתְּקָא בָּתַר שְׁנָנָא וְטִמְטֵם שׁוּמְנָא בְּאַפֵּי שְׁנָנָא אֲרֵי לָא שְׁלַף חַרְבָּא מִמְעוֹהִי וּנְפַק אוּכְלֵיהּ שְׁפִיךְ:
וּנְפַק אֵהוּד לְאַכְסַדְרָא וַאֲחַר דָשֵׁי עִילִיתָא בְאַפּוֹהִי וְאַגָף:
וְהוּא נְפַק וְעַבְדוֹהִי עֲלוֹ וַחֲזוֹ וְהָא דָשֵׁי עִילִיתָא אֲחִידָן וַאֲמַרוּ בְּרַם עָבִיד הוּא יַת צוֹרְכֵיהּ בְּאִדְרוֹן בֵּית קַיְטָא:
וְאוֹרִיכוּ עַד סַגִי וְהָא לֵיתוֹהִי פָּתַח דָשֵׁי עִלִיתָא וּנְסִיבוּ יַת מַפְתְּחָא וּפְתַחוּ וְהָא רִבּוֹנְהוֹן טָרִיף רְמֵי עַל אַרְעָא מִית:
וְאֵהוּד אִשְׁתְּזִיב עַד אִתְעַכְּבֵיהוֹן וְהוּא עֲבַר יַת מַחְצְבַיָא וְאִשְׁתְּזֵיב לִשְׂעִירָתָא:
וַהֲוָה בְּמֵיתוֹהִי וּתְקַע בְּשׁוֹפָרָא בְּטוּרָא דְבֵית אֶפְרָיִם וּנְחָתוּ עִמֵיהּ בְּנֵי יִשְׂרָאֵל מִן טוּרָא וְהוּא קֳדָמֵיהוֹן:
וַאֲמַר לְהוֹן רְדִיפוּ בַתְרַי אֲרֵי מְסַר יְיָ יַת בַּעֲלֵי דְבָבֵיכוֹן יַת מוֹאֲבָאֵי בִּידֵיכוֹן וּנְחָתוּ בַתְרוֹהִי וַאֲחַדוּ יַת מְגִיזַת עָבְרָאֵי יַרְדְנָא עַל מוֹאָב וְלָא שְׁבַקוּ אֱנַשׁ לְמֶעְבַר:
וּקְטַלוּ יַת מוֹאָב בְּעִדָנָא הַהִיא כְּעַסְרָא אַלְפִין גַבְרָא כָּל אֵימְתָן וְכָל גְבַר גִבַּר חֵילָא וְלָא אִישְׁתְּזֵיב אֱנָשׁ:
וְאִתַּבָּרוּ מוֹאֲבָאֵי בְּיוֹמָא הַהוּא תְּחוֹת יַד אֱנַשׁ יִשְׂרָאֵל וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל תְּמָנָן שְׁנִין:
וּבַתְרוֹהִי הֲוָה שַׁמְגַר בַּר עֲנָת וּקְטַל יַת פְּלִשְׁתָּאֵי שִׁית מְאָה גַבְרִין בִּפְרַשׁ תּוֹרַיָא וּפְרַק אַף הוּא יַת יִשְׂרָאֵל:

Chapter 4

וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וְאֵהוּד מִית:
וּמְסַרְנוּן יְיָ בְּיַד יָבִין מַלְכָּא דִכְנַעַן דִי מְלַךְ בְּחָצוֹר וְרַב חֵילֵיה סִיסְרָא וְהוּא יָתֵיב בִּתְקוֹף כְּרַכֵּי עַמְמַיָא:
וּצְעָקוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ אֲרֵי תְשַׁע מֵאָה רְתִיכִין דְבַרְזְלָא לֵיהּ וְהוּא דְחַק יַת בְּנֵי יִשְׂרָאֵל בְּתוּקְפָא עַסְרִין שְׁנִין:
וּדְבוֹרָה אִתְּתָא נְבִיאֲתָא אִיתַּת לַפִּידוֹת הִיא דַיְנָא יַת יִשְׂרָאֵל בְּעִדָנָא הַהִיא:
וְהִיא יַתְבָה בְּקַרְתָּא בַּעֲטָרוֹת דְבוֹרָה מִתְפַּרְנְסָא מִן דִילָהּ וְלָה דִקְלִין בִּירִיחוֹ פַּרְדְסִין בְּרָמָתָא זֵיתִין עָבְדִין מְשַׁח בְּבִקְעֲתָא בֵּית שַׁקְיָא בְּבֵית אֵל עֲפַר חִוַר בְּטוּר מַלְכָּא וְסַלְקִין לְוָתָהּ בְנֵי יִשְׂרָאֵל לְדִינָא:
וּשְׁלָחַת וּקְרַת לְבָרָק בַּר אֲבִינוֹעַם מִקֶדֶשׁ דְבֵית נַפְתָּלִי וַאֲמַרַת לֵיהּ הֲלָא פָּקַד יְיָ אֱלָהָא דְיִשְׂרָאֵל אֱזֵיל וְתִתְנְגִיד בְּטוּרָא דְתָבוֹר וְתִדְבַּר עִמָךְ עַסְרָא אַלְפִין גַבְרָא מִבְּנֵי נַפְתָּלִי וּמִבְּנֵי זְבוּלֻן:
וֶאֱתַתֵּיב עַל יְדָךְ לְנַחְלָא דְקִישׁוֹן יַת סִיסְרָא רַב חֵילָא דְיָבִין וְיַת רְתִיכוֹהִי וְיַת מַשִׁרְיָתֵיהּ וְאֶמְסְרִנֵהּ בִידָךְ:
וַאֲמַר לָהּ בָּרָק אִם תֵּיזְלִין עִמִי וְאֵיזִיל וְאִם לָא תֵּיזְלִין עִמִי לָא אֵיזִיל:
וַאֲמַרַת מֵיזַל אֵיזִיל עִמָךְ לְחוֹד אֲרֵי לָא תְהֵי תוּשְׁבַּחְתָּךְ עַל אוֹרְחָא דְאַתְּ אָזֵיל אֲרֵי בְּיַד אִתְּתָא יִמְסוֹר יְיָ יַת סִיסְרָא וְקָמַת דְבוֹרָה וַאֲזַלַת עִם בָּרָק לְקֶדֶשׁ:
וּכְנַשׁ בָּרָק יַת שִׁבְטָא דִזְבוּלֻן וְיַת שִׁבְטָא דְנַפְתָּלִי לְקֶדֶשׁ וּסְלִיקוּ עִמֵיהּ עַסְרָא אַלְפִין גַבְרָא וּסְלִיקַת עִמֵיהּ דְבוֹרָה:
וְחֶבֶר שַׁלְמָאָה אִתְפָּרַשׁ מִשַׁלְמָאֵי מִבְּנֵי חוֹבָב חֲמוּהִי דְמֹשֶׁה וּפַרְסֵיהּ לְמַשְׁכְּנֵיהּ עַד מֵישַׁר אַגְנַיָא דְעִם קֶדֶשׁ:
וְחַוִיאוּ לְסִיסְרָא אֲרֵי סְלִיק בָּרָק בַּר אֲבִינוֹעַם לְטוּר תָּבוֹר:
וּכְנַשׁ סִיסְרָא יַת כָּל רְתִיכֵיהּ תְּשַׁע מְאָה רְתִיכִין דְבַרְזְלָא וְיַת כָּל עַמָא דְעִמֵיהּ מִתְקוֹף כְּרַכֵּי עַמְמַיָא לְנַחֲלָא דְקִישׁוֹן:
וַאֲמַרַת דְבוֹרָה לְבָרָק קוּם אֲרֵי דֵין יוֹמָא דִמְסַר יְיָ יַת סִיסְרָא בִּידָךְ הֲלָא מַלְאָכָא דַייָ נְפַק לְאַצְלָחָא קֳדָמָךְ וּנְחַת בָּרָק מִטוּרָא דְתָבוֹר וְעַסְרָא אַלְפִין גַבְרָא בַּתְרוֹהִי:
וְתַבֵּר יְיָ יַת סִיסְרָא וְיַת כָּל רְתִיכַיָא וְיַת כָּל מַשְׁרִיתָא לְפִתְגַם דְחֶרֶב קֳדָם בָּרָק וּנְחַת סִיסְרָא מֵעַל רְתִיכָא וְאַפַּךְ בְּרַגְלוֹהִי:
וּבָרָק רְדַף בָּתַר רְתִיכָא וּבָתַר מַשְׁרִיתָא עַד תְּקוֹף כְּרַכֵּי עַמְמַיָא וְאִתְקְטָלָא כָּל מַשִׁרְיַת סִיסְרָא לְפִתְגַם דְחָרֶב לָא אִשְׁתָּאַר עַד חָד:
וְסִיסְרָא אַפֵּךְ בְּרַגְלוֹהִי לְמַשְׁכְּנָא דְיָעֵל אִתַּת חֶבֶר שַׁלְמָאָה אֲרֵי שְׁלָמָא בֵּין יָבִין מַלְכָּא דְחָצוֹר וּבֵין בֵּית חֶבֶר שַׁלְמָאָה:
וְנַפְקַת יָעֵל לִקְדָמוּת סִיסְרָא וַאֲמַרַת לֵיהּ זוּר רִבּוֹנִי זוּר לְוָתִי לָא תִדְחַל וְזַר לְוָתָהּ לְמַשְׁכְּנָא וְכַסִיתֵיהּ בְּגוּנְכָא:
וַאֲמַר לָהּ אַשְׁקִינִי כְעַן זְעֵיר מַיָא אֲרֵי צָחֵי אֲנָא לְמִשְׁתֵּי וּפְתַחַת יַת נוֹדָא דְחֶלְבָּא וְאַשְׁקִיתֵיהּ וכַסִיתֵיהּ:
וַאֲמַר לָה קוּמִי בִּתְרַע מַשְׁכְּנָא וִיהֵי אִם אֱנַשׁ יֵיתֵי וְיִשְׁאֲלִנָיךְ וְיֵמַר הֲאִית כָּא גַבְרָא וְתֵימְרִין לָא:
וּנְסִיבַת יָעֵל אִתַּת חֶבֶר יַת סִכַּת מַשְׁכְּנָא וְשַׁוִיאַת יַת אַרְזַפְתָּא בִּידָהּ וַעֲלַת לְוָתֵיהּ בְּרָז וּנְקִישַׁת יַת סִכְּתָא בְּצִדְעֵיהּ וּנְעָצַת בְּעַרְעָא וְהוּא דְמוּך וְאִשְׁתַּלְהִי וּמִית:
וְהָא בָרָק רְדַף יַת סִיסְרָא וְנַפְקַת יָעֵל לִקְדָמוּתֵיהּ וַאֲמַרַת לֵיהּ אִיתָא וְאַחֲזִינָךְ יַת גַבְרָא דְאַתְּ בָּעֵי וְעַל לְוָתָהּ וְהָא סִיסְרָא טָרִיף וּמִית וְסִכְּתָא נְעִיצָא בְּצִדְעֵיהּ:
וּתְבַר יְיָ בְּיוֹמָא הַהוּא יַת יָבִין מַלְכָּא דִכְנָעַן קֳדָם בְּנֵי יִשְׂרָאֵל:
וְאַזְלַת יַד בְּנֵי יִשְׂרָאֵל אָזְלָא וְתָקְפָא עַל יָבִין מַלְכָּא דִכְנָעַן עַד דִי שֵׁצִיאוּ יַת יָבִין מַלְכָּא דִכְנָעַן:

Chapter 5

וְשַׁבַּחַת דְבוֹרָה וּבָרָק בַּר אֲבִינוֹעַם עַל נִסָא וּפוּרְקָנָא דְאִתְעֲבִידָא לְיִשְׂרָאֵל בְּיוֹמָא הַהוּא לְמֵימָר:
כַּד מְרָדוּ בֵּית יִשְׂרָאֵל בְּאוֹרַיְתָא אֲתוֹ עֲלֵיהוֹן עַמְמַיָא וּטְרָדוּנוּן מִקְרְוֵיהוֹן וְכַד תָּבוּ לְמֶעְבַד אוֹרַיְתָא אִתְגַבָּרוּ אִנוּן עַל בַּעֲלֵי דְבָבֵיהוֹן וְסָרְכוּנוּן מִכָּל תְּחוּם אַרְעָא דְיִשְׂרָאֵל בְּכֵן עַל פּוּרְעָנוּת תְּבַר סִיסְרָא וְכָל מַשִׁרְיָתֵהּ וְעַל נִסָא וּפוּרְקָנָא דְאִתְעֲבֵיד לְהוֹן לְיִשְׂרָאֵל בְּכֵן תָּבוּ חַכִּימַיָא לְמֵתַב בְּבָתֵּי כְנִשְׁתָּא בְּרֵישׁ גְלֵי וּלְאַלְפָא יַת עַמָא פִּתְגָמֵי אוֹרַיְתָא בְּכֵן בָּרִיכוּ וְאוֹדוּ קֳדָם יְיָ:
שְׁמָעוּ מַלְכַיָא דַאֲתוֹ עִם סִיסְרָא לִקְרָבָא אֲצִיתוּ שִׁלְטוֹנַיָא דַהֲווֹ עִם יָבִין מַלְכָּא דִכְנָעַן לָא בְּחֵילַתְכוֹן וְלָא בִּגְבוּרַתְכוֹן אִתְגַבַּרְתּוּן וּסְלֵקְתּוּן עַל דְבֵית יִשְׂרָאֵל אֲמַרַת דְבוֹרָה בִּנְבוּאָה קֳדָם יְיָ אֲנָא מְשַׁבְּחָא מוֹדָא וּמְבָרְכָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל:
יְיָ אוֹרַיְתָךְ דִיהַבְתָּא לְיִשְׂרָאֵל כַּד עָבְרִין עֲלָהּ הֲוָה שַׁלְטִין בְּהוֹן עַמְמַיָא וְכַד תַּיְבִין לָהּ מִתְגַבְּרִין אִנוּן עַל בַּעֲלֵי דְבָבֵיהוֹן יְיָ בְּיוֹם אִתְגַלֵיתָךְ לְמִתְּנָהּ לְהוֹן מִשֵׂעִיר בְּהוֹן אִתְגַלֵיתָא בְּהוֹפָעוֹת יְקָרָךְ עַל תְּחוּמֵי אֱדוֹם אַרְעָא זָעַת אַף שְׁמַיָא מָכוּ אַף עֲנָנַיָא נְגִידוּ מִטְרָא:
טוּרַיָא זָעוּ מִן קֳדָם יְיָ טוּרָא דְתָבוֹר טוּרָא דְחֶרְמוֹן וְטוּרָא דְכַרְמְלָא מִתְרַגְשִׁין דֵין עִם דֵין וְאָמְרִין דֵין לְדֵין דֵין אֲמַר עֲלַי תִּשְׁרֵי שְׁכִנְתֵּיהּ וְלִי חַזְיָא וְדִין לְדֵין אֲמַר עֲלַי תִּשְׁרֵי שְׁכִנְתֵּיהּ וְלִי חַזְיָא אַשְׁרֵי שְׁכִנְתֵּיהּ עַל טוּרָא דְסִינַי דְהוּא חֲלָשׁ וּזְעֵיר מִכָּל טוּרַיָא דֵין סִינַי מִתְרְגִיף וּמִתְרְגִישׁ וְסָלִיק תְּנָנֵיהּ כִּתְנָנָא דְאַתּוּנָא מִן קֳדָם דְאִתְגְלֵי עַלוֹהִי יְקָרָא דַייָ אֱלָהָא דְיִשְׂרָאֵל:
כַּד חָבוּ בְּיוֹמֵי שַׁמְגַר בַּר עֲנָת בְּיוֹמֵי יָעֵל פְּסַקוּ עָדֵי אוֹרְחָן דַהֲווֹן מְהַלְכִין בִּשְׁבִילִין דְתַקְנִין תָּבוּ לְמֶהֱוֵי אָזְלִין בְּאוֹרְחָן מְגַנְבָן:
חֲרוֹבָא קִרְוֵי פַצְחַיָא דַהֲווֹ יַתְבִין בְּאַרְעָא דְיִשְׂרָאֵל אִצְטְדִיאָה וְאִטַלְטַלוּ דַיָרֵיהוֹן עַד דְאִשְׁתַּלְחֵית אֲנָא דְבוֹרָה אִשְׁתַּלְחֵית לְהִתְנַבָּאָה עַל דְבֵית יִשְׂרָאֵל:
כַּד אִתְרְעִיאוּ בְנֵי יִשְׂרָאֵל לְמִפְלַח לְטַעֲוָתָא חֲדַתָּן דְמִקָרֵב אִתְעֲבִידָא דְלָא אִיתְעַסְקוּ בְהוֹן אֲבָהַתְהוֹן אֲתוּ עֲלֵיהוֹן עַמְמַיָא וּטְרָדוּנוּן מִקִרְוֵיהוֹן וְכַד תָּבוּ לְמֶעְבַּד אוֹרַיְתָא לָא יְכִילוּ לְהוֹן עַד דְאִתְגַבְּרוּ וּסְלִיק עֲלֵיהוֹן סִיסְרָא סַנְאָה וּמְעִיקָא בְּאַרְבְּעִין אַלְפִין רֵישֵׁי מַשִׁרְיָן בְּחַמְשִׁין אַלְפִין אַחֲדֵי סַיְפִין בְּשִׁתִּין אַלְפִין אַחֲדֵי רוֹמְחִין בְּשִׁבְעִין אַלְפִין אַחֲדֵי תְּרֵיסִין בִּתְמָנָן אַלְפִין מְחַצְצֵי גִירַיָא בַּר מִתְּשַׁע מְאָה רְתִיכִין דְבַרְזְלָא דַהֲווֹ עִמֵיהּ וּרְתִיכוֹהִי כָּל אִלֵין אַלְפַּיָא וְכָל אִלֵין מַשִׁרְיָתָא לָא יָכִילוּ לְמֵיקַם קֳדָם בָרָק וְקָדָם עַשְׂרָא אַלְפִין גַבְרָא דְעִמֵיהּ:
אֲמַרַת דְבוֹרָה בִּנְבוּאָה אֲנָא שְׁלִיחָא לְשַׁבָּחָא לְסַפְרֵי יִשְׂרָאֵל דְכַד הֲוַת עַקְתָא הַהִיא לָא פְּסָקוּ מִלְמִדְרַשׁ בְּאוֹרַיְתָא וּכְדוּ יָאֵי לְהוֹן דְיַתְבִין בְּבָתֵּי כְנִשְׁתָּא בְּרֵישׁ גְלֵי וּמְאַלְפִין יַת עַמָא פִּתְגָמֵי אוֹרַיְתָא וּמְבָרְכִין וּמוֹדִין קֳדָם יְיָ:
דַהֲווֹ מְבַטְלִין עִסְקֵיהוֹן רְכִיבִין עַל אַתְנָן דַחֲשִׁיקָן בְּכָל מִינֵי צִיוּרִין וּמְהַלְכִין בְּכָל תְּחוּם אַרְעָא דְיִשְׂרָאֵל וּמִתְחַבְּרִין לְמִתַּב עַל דִינָא דַהֲווֹ אָזְלִין בְּאוֹרְחַתְהוֹן וּמִשְׁתָּעִין עַל גְבוּרָן דְאִתְעֲבִידָא לְהוֹן בְּאַרְעָא דְיִשְׂרָאֵל:
מַאֲתַר דַהֲווֹ אָנְסִין לְהוֹן וְנַסְבִּין מַה דְבִידֵיהוֹן בֵּית מְכוּנַת לִסְטִין וּמְתוּבַת מוּכְסִין עַל גוּבִין בֵּית שַׁקְיָא דְמַיָא לַאֲתַר דַהֲוָאָה נַפְקָן בְּנָתָא דְיִשְׂרָאֵל לְמִמְלֵי מַיָא דְלָא הֲוָאָה יָכְלִין לְאַשְׁמָעָא קַל טַרְפַת פַּרְסַת רַגְלֵיהוֹן מִן קֳדָם סַנְאָה וּמְעִיקָא תַּמָן יוֹדוּן וְעַל זַכְוָתֵיהּ דַייָ וְעַל זַכְוָתֵיה דְיָתֵב קִרְוֵי פַּצְחַיָא בְּאַרְעָא דְיִשְׂרָאֵל בְּכֵן נְחָתוּ מִכְּרַכֵּי תוּקְפָא לְמִתַּב בְּקִרְוֵי פַּצְחַיָא עַמֵיהּ דַיָי:
שַׁבָּחִי שַׁבָּחִי דְבוֹרָה שַׁבָּחִי וְאוֹדִי מַלִילִי תּוּשְׁבְּחָא קוּם בָּרָק וּשְׁבִי שִׁבְיָךְ בַּר אֲבִינוֹעַם:
בְּכֵן נְחַת חַד מִמַשְׁרִיַת יִשְׂרָאֵל וְתַבַּר תְּקוֹף גִבָּרֵי עַמְמַיָא הָא לָא מִגְבוּרָא דִדְהוֹן הֲוַת דָא אֱלָהֵן יְיָ תְּבַר קֳדָם עַמֵיהּ תְּקוֹף גִבָּרֵי סַנְאֵיהוֹן:
מִדְבֵית אֶפְרַיִם קָם יְהוֹשֻׁעַ בַּר נוּן בְּקַדְמֵיתָא אַגִיחַ קְרָבָא בִּדְבֵית עֲמָלֵק בַּתְרוֹהִי קָם מַלְכָּא שָׁאוּל מִדְבֵית בִּנְיָמִין וְקַטֵל יַת דְבֵית עֲמָלֵק וְאַגִיחַ קְרָבָא בִּשְׁאַר עַמְמַיָא מִדְבֵית מָכִיר נַחֲתוּ כַּד מְרַשְׁמִין בִּקְרָבָא מִשִׁבְטָא זְבוּלֻן כַּתְבִין בְּקוֹלְמוֹס דְסֵפֶר חָד:
וְרַבְרְבֵי יִשָׂשכָר מִשְׁתַּמְעִין לְמִילֵי דְבוֹרָה וּשְׁאַר שִׁבְטָא דְיִשָׂשכָר מְשַׁמְשִׁין קֳדָם בָּרָק מִשְׁתַּלְחִין בְּקִרְוֵי מַשְׁרָא לְכָל אֲתַר דְצָרִיךְ לְתַּמָן בְּתוּשְׁלַחְתֵּיה אַזְלִין בְּזַרְעִית רְאוּבֵן סַגִיאִין נִבְלֵי לִבָּא:
לְמָא תַבְתּוּן מִמַשִׁרְיַת קְרָבָא לְמִתַּב בֵּין תְּחוּמִין בְּפָרָשַׁת אָרְחָא לְמִשְׁמַע בְּשׂוֹרְתָּא דָא מִן דָא לְבָרָק אַתּוּן אָמְרִין דִילָךְ אֲנַחְנָא לְסִיסְרָא אַתּוּן אָמְרִין דִילָךְ אֲנַחְנָא לְמִשְׁמַע בְּשׂוֹרָא לְמִידַע אֵידָא הִיא מַשְׁרִיתָא נַצְחָא בִּקְרָבָא לְמֶהֱוֵי עִמֵיהּ אֵיכְדֵין כָּשֵׁר לְמֶעְבַּד לְכוֹן דְבֵית רְאוּבֵן הֲלָא יָדְעִיתוּן דִקְדָמוֹהִי גַלְיָן מַחְשְׁבוֹת לִבָּא:
דְבֵית גִלְעָד בְּעִבְרָא דְיַרְדְנָא שָׁרָן וּדְבֵית דָן גָזוּ עָבְרוּ יַרְדְנָא וְשַׁוִיאוּ נִכְסֵיהוֹן בִּסְפִינָן דְבֵית אָשֵׁר עַל סְפַר יַמַיָא שְׁרוֹ קִרְוֵי עַמְמַיָא דְפַּגָרוּ תָּבוּ וּבָנְיָן וִיתִיבוּ בְהוֹן:
דְבֵית זְבוּלֻן לָקֳבֵיל עַמְמַיָא דְגַדִיפוּ קֳדָם יְיָ מְסַרוּ נַפְשֵׁיהוֹן לְקַטָלָא אִנוּן וּדְבֵית נַפְתָּלִי יְשַׁבְּחוּנוּן כָּל עַמְמֵי אַרְעָא:
אֲתוֹ מַלְכַיָא דַהֲווֹ עִם סִיסְרָא וְאַגִיחוּ קְרָבָא בְּכֵן אֲגִיחוּ קְרָבָא מַלְכֵי כְנַעַן בְּתַעְנָךְ הֲווֹ שָׁרָן וּמָטָן עַל מֵי מְגִדוֹ מָמוֹן דִכְסַף לָא קַבִּילוּ:
מִן שְׁמַיָא אִתְגַח עִמְהוֹן קְרָבָא מַאֲתַר דְנַפְקִין כּוֹכְבַיָא מְכַבְשֵׁי דַבָּרֵיהוֹן תַּמָן אִתָּגַח קְרָבָא עִם סִיסְרָא:
נַחֲלָא דְקִישׁוֹן תַּבְּרִינוּן נַחֲלָא דְאִתְעֲבִידוּ בֵיהּ נִסִין וּגְבוּרָן מִלְקַדְמֵין לְיִשְׂרָאֵל הוּא נַחֲלָא דְקִישׁוֹן תַּמָן דוּשִׁישַׁת נַפְשִׁי קְטִילֵי גִבָּרֵיהוֹן בִּתְקוֹף:
בְּכֵן אִשְׁתַּלְפָא טוּפְרֵי סוּסְוַתְהוֹן מִפִּנוּכָא דַהֲוָה מְפַנֵיךְ קֳדָם רְתִיכֵי גִבָּרוֹהִי:
לוּטוּ מֵרוֹז אֲמַר בָּרָק נְבִיָא דַייָ לוּטוּ וְתַבָּרוּ יַת יַתְבָהָא אֲרֵי לָא אֲתוֹ לְסַעְדָן עַמָא דַייָ כַּד אַגִיחַ קְרָבָא בְּגִבָּרַיָא:
תִּתְבָּרַךְ מִבִּרְכַת נְשַׁיָא טָבָתָא יָעֵל אִתַּת חֶבֶר שַׁלְמָאָה כַּחֲדָא מִנְשַׁיָא דִמְשַׁמְשִׁין בְּבָתֵּי מֶדְרָשִׁין תִּתְבָּרַךְ:
מַיָא שְׁאַל מִנָה סִיסְרָא רַשִׁיעָא חַלְבָא אַשְׁקִיתֵיהּ לְמִדַע אִם אִית רַעֲיוֹנוֹהִי עֲלוֹהִי בְּפַיְלֵי גִבָּרַיָא קְרִיבַת לָקֳדָמוֹהִי שְׁמַן דְגוּבְנִין:
טַבְתָא יָעֵל אִתַת חֶבֶר שַׁלְמָאָה דְקַיְמַת מַה דִכְתִיב בִּסְפַר אוֹרַיְתָא דְמֹשֶׁה לָא יֱהֶוֵי תִקוּן זֵין דִגְבַר עַל אִתְּתָא וְלָא יְתַקֵן גְבַר בְּתִקוּנֵי אִתְּתָא אֱלָהֵן יְדָהָא לְסִכְתָא אוֹשִׁיטַת וִימִינָא לְאַרְזַפְתָּא דְנַפְחִין לְמִתְבַּר רַשִׁיעִין וַאֲנוּסִין מַחֲתֵיהּ לְסִיסְרָא תַּבְרַת רֵישֵׁיהּ פַּצְעַת מוֹחֵיהּ אַעֲבָרַת סִכְתָא בְּצִדְעֵיהּ:
בֵּין רִגְלָהָא כְּרַע נְפַל שְׁכִיב בֵּין רִגְלָהָא כְּרַע נְפַל בַּאֲתַר דִכְרַע תַּמָן נְפַל בְּזִיזָא סִיסְרָא:
מִן חֲרַכָּא אִסְתְּכִיאַת וּמְדַיְקָא אִמֵיהּ דְסִיסְרָא מִבֵּינֵי אַעֲיָתָא אֲמַרַת מָא דֵין אוֹחֲרָא רְתִיכוֹהִי דְסִיסְרָא בְּרִי מִלְמֵיתִי מָא דֵין אִתְעֲכָּבוּ רְהָטַיָא דַהֲווֹ מַיְתָן לִי אִגֶרֶת נִצְחָנִין:
חֲכִּימַת פְּרִיסְתְּוָתָהָא עַנְיָן לָהּ אַף הִיא בְּחוּכְמָא מְתִיבָא וְאָמְרָה לְמֵימַר לָהֶן:
הֲלָא מִדְמַשְׁכְּחִין מְפַלְגִין בִּזְתָא יַהֲבִין גְבַר וּבֵיתֵיהּ כָּל חַד וְחַד בִּזָא סַגִי קֳדָם סִיסְרָא בִּזַת צִיוּרֵי צִבְעוֹנִין עַל צַוְרֵיהּ נִכְסֵי עַתִּירַיָא וּמָנֵי חֲמוּדָא קֳדָם גִבָּרוֹהִי דְבָזוּ:
כְּסִיסְרָא כֵּן יֵיבְדוּן כָּל סַנְאֵי עַמָךְ יְיָ וְרַחֲמוֹהִי צַדִיקַיָא יְהוֹן עֲתִידִין לְאַזְהָרָא כְּזֵיהוֹר יְקָרֵיהּ עַל חַד תְּלַת מְאָה וְאַרְבְּעִין וּתְלָתָא כְּמִפַּק שִׁמְשָׁא בִּגְבוּרְתֵּיה וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל מִלְמֶעְבֵּד קְרָבָא אַרְבְּעִין שְׁנִין:

Chapter 6

וַעֲבָדוּ בְנֵי יִשְׂרָאֵל דְבִישׁ קֳדָם יְיָ וּמַסְרִינוּן יְיָ בִּידָא דְמִדְיָנָאֵי שְׁבַע שְׁנִין:
וּתְקֵיפַת יַד מִדְיָנָאֵי עַל יִשְׂרָאל מִן קֳדָם מִדְיָנָאֵי עֲבָדוּ לְהוֹן בְּנֵי יִשְׂרָאֵל יַת מַטְמוֹרִיתָא דִבְטוּרַיָא וְיַת מְעָרָתָא וְיַת מְצוֹדָתָא:
וַהֲוָה כַּד זְרַע יִשְׂרָאֵל וְסַלְקִין מִדְיָנָאֵי וַעֲמַלְקָאֵי וּבְנֵי מַדִינְחָא וְסַלְקִין עֲלוֹהִי:
וּשְׁרוֹ עֲלֵיהוֹן וּמְחַבְּלִין יַת עֲלַלְתָּא דְאַרְעָא עַד מַעְלָנָא דְעַזָה וְלָא מְשַׁאֲרִין מָזוֹן לְקַיָימָא נְפַשׁ בְּיִשְׂרָאֵל וְאִמַר וְתוֹר וַחֲמָר:
אֲרֵי אִינוּן וְגֵיתֵיהוֹן סַלְקִין וּמַשְׁכְּנֵיהוֹן וְאָתָן כְּמִסַת גוֹבָא לִסְגֵי וּלְהוֹן וְלִגְמַלֵיהוֹן לֵית מִנְיָן וְעָלִין בְּאַרְעָא לְחַבָּלוּתָהּ:
וּמְאַךְ יִשְׂרָאֵל לַחֲדָא מִן קֳדָם מִדְיָנָאֵי וּזְעִיקוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ:
וַהֲוָה כַּד זְעִיקוּ בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ עַל עֵסַק מִדְיָנָאֵי:
וּשְׁלַח יְיָ גַבְרָא נְבִיָא לְוַת בְּנֵי יִשְׂרָאֵל וַאֲמַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל אֲנָא אַסְקֵית יַתְכוֹן מִמִצְרַיִם וְאַפְקֵית יַתְכוֹן מִבֵּית עַבְדוּתָא:
וְשֵׁיזְבֵית יַתְכוֹן מִידָא דְמִצְרָאֵי וּמִידָא כָּל דַחֲקֵיכוֹן וְתָרְכֵית יַתְהוֹן מִן קֳדָמֵיכוֹן וְיַהֲבֵית לְכוֹן יַת אַרְעֲהוֹן:
וַאֲמָרִת לְכוֹן אֲנָא יְיָ אֱלָהָכוֹן לָא תִדְחֲלוּ יַת טַעֲוַת אֱמוֹרָאֵי דְאַתּוּן יַתְבִין בְּאַרְעֲהוֹן וְלָא קַבֵּלְתּוּן לְמֵימְרִי:
וַאֲתָא מַלְאָכָא דַייָ וִיתֵב תְּחוֹת בּוּטְמָא דִי בְעָפְרָה דִי לְיוֹאָשׁ אֲבוּהִי דְעֶזְרִי וְגִדְעוֹן בְּרֵיהּ חָבֵיט חִטִין בְּמַעֲצַרְתָּא לַעֲרָקָא מִן קֳדָם מִדְיָנָאֵי:
וְאִתְגְלֵי לֵיהּ מַלְאָכָא דַייָ וַאֲמַר לֵיהּ מֵימְרָא דַייָ בְּסַעְדָךְ גִבַּר חֵילָא:
וַאֲמַר לֵיהּ גִדְעוֹן בְּבָעוּ רִבּוֹנִי וְאִית שְׁכִנְתָּא דַיְיָ בְּסַעְדָנָא וּלְמָא עַרְעָנָא כָּל דָא וְאָן כָּל פְּרִשְׁוָתֵיהּ דְאִשְׁתְּעִיאוּ לָנָא אַבְהָתָנָא לְמֵימַר הֲלָא מִמִצְרַיִם אַסְקָנָא יְיָ וּכְעַן רַטְשָׁנָא יְיָ וּמַסְרָנָא בִּידָא דְמִדְיָנָאֵי:
וְאִתְפְּנֵי לֵיהּ יְיָ וַאֲמַר אֱזִיל בְּחֵילָךְ דֵין וְתִפְרוֹק יַת יִשְׂרָאֵל מִידָא דְמִדְיָנָאֵי הֲלָא שְׁלַחְתִּיךְ:
וַאֲמַר לֵיהּ בְּבָעוּ יְיָ בְּמָא אֶפְרוֹק יַת יִשְׂרָאֵל הֲלָא הָא זַרְעִיתִי זְעִירָא בְּשֵׁבֶט מְנַשֶׁה וַאֲנָא חֲלַשׁ בְּבֵית אַבָּא:
וַאֲמַר לֵיהּ יְיָ אֲרֵי יְהֵי מֵימְרִי בְּסַעֲדָךְ וְתִקְטוֹל יַת מִדְיָן כְּגַבְרָא חָד:
וַאֲמַר לֵיהּ אִם כְּעַן אַשְׁכָּחִית רַחֲמִין בְעֵינָךְ וְתַעְבֵּיד לִי אָת דִי אַתְּ מְמַלֵיל עִמִי:
לָא כְעַן תַּעֲדֵי מִכָּא עַד מֵתָאִי לְוָתָךְ וְאַפֵק יַת תִּקְרוּבְתִּי וְאַנִיחַ קֳדָמָךְ וַאֲמַר אֲנָא אוֹרִיךְ עַד דִתְתוּב:
וְגִדְעוֹן אָתָא וַעֲבַד גַדְיָא בַּר עִזֵי וּמְכִילְתָּא דְקִמְחָא אָפָה פַּטֵיר בִּסְרָא שַׁוִי בְּסַלָא וְתַבְשִׁילָא שַׁוִי בְּקִדְרָא וְאַפֵּיק לְוָתֵיה לִתְחוֹת אִילָנָא וְקָרֵב:
וַאֲמַר לֵיהּ מַלְאָכָא דַיְיָ סַב יַת בִּסְרָא וְיַת פַּטִירָא וְאָחֵית לְכֵיפָא דֵיכִי וְיַת תַּבְשִׁילָא שְׁפוֹךְ וַעֲבַד כֵּן:
וְאוֹשֵׁיט מַלְאָכָא דַייָ יַת רֵישׁ שׁוֹטִיתָא דְבִידֵיהּ וְקָרֵיב בְּבִסְרָא וּבְפַּטִירָא וּסְלֵיקַת אֶשְׁתָּא מִן טִנָרָא וְאַכְלַת יַת בִּסְרָא וְיַת פַּטִירָא וּמַלְאָכָא דַייָ אִסְתַּלֵק מִלְקֳבֵיל עֵינוֹהִי:
וַחֲזָא גִדְעוֹן אֲרֵי מַלְאָכָא דַייָ הוּא וַאֲמַר גִדְעוֹן בְּבָעוּ יְיָ אֱלֹהִים אֲרֵי עַל כֵּן חֲזֵיתִי מַלְאָכָא דַייָ אַפִּין בְּאַפִּין:
וַאֲמַר לֵיהּ יְיָ שְׁלַם לָךְ לָא תִדְחַל לָא תְמוּת:
וּבְנָא תַמָן גִדְעוֹן מַדְבְּחָא קֳדָם יְיָ וּפְלַח עֲלוֹהִי קֳדָם יְיָ דַעֲבַד לֵיהּ שְׁלָם עַד יוֹמָא הָדֵין עַד כְּעַן אִיתוֹהִי בְּעָפְרַת אֲבוּהִי דְעֵזֶר:
וַהֲוָה בְּלֵילְיָא הַהוּא וַאֲמַר לֵיהּ יְיָ סַב יַת תּוֹר בַּר תּוֹרֵי דִי לַאֲבוּךְ וְתוֹר תִּנְיָנָא דְאִתְפַּטִים שְׁבַע שְׁנִין וּתְפַגֵר יַת אֱגוֹרָא דְבַעֲלָא דִי לַאֲבוּךְ וְיַת אֲשֶׁרְתָּא דִי עֲלוֹהִי תְּקוֹץ:
וְתִבְנֵי מַדְבְּחָא קֳדָם יְיָ אֱלָהָךְ עַל רֵישׁ תּוּקְפָא הָדֵין בְּסִדְרָא וְתִסַב יַת תּוֹרָא תִּנְיָנָא וְתַסֵיק עֲלָתָא בְּאָעֵי אֲשֶׁרְתָּא דִי תְקוֹץ:
וּדְבַר גִדְעוֹן עַסְרָאן גַבְרִין מֵעַבְדוֹהִי וַעֲבַד כְּמָא דְמַלֵיל עִמֵיהּ יְיָ וַהֲוָה מִדְדָחִיל מִן בֵּית אֲבוּהִי וּמִן אֱנָשֵׁי קַרְתָּא מִלְמֶעְבַּד בִּימָמָא וַעֲבַד לֵילְיָא:
וְאַקְדִימוּ אֱנָשֵׁי קַרְתָּא בְּצַפְרָא וְהָא מִתְרַע אֱגוֹרָא דְבַעֲלָא וַאֲשֶׁרְתָּא דִי עֲלוֹהִי קְצִיצָא וְיַת תּוֹרָא תִנְיָנָא מִסַק עַל מַדְבְּחָא דְבָנוּי:
וַאֲמַרוּ גְבַר לְחַבְרֵיהּ מָן עֲבַד פִּתְגָמָא הָדֵין וּתְבַעוּ וּבְעוֹ וַאֲמַרוּ גִדְעוֹן בַּר יוֹאָשׁ עֲבַד פִּתְגָמָא הָדֵין:
וַאֲמַרוּ אֱנָשֵׁי קַרְתָּא לְיוֹאָשׁ אַפֵיק יַת בְּרָךְ וְיִתְקְטַל אֲרֵי תְרַע יַת אֵיגוֹרָא דְבַעֲלָא וַאֲרֵי קָץ אֲשֶׁרְתָּא דִי עֲלוֹהִי:
וַאֲמַר יוֹאָשׁ לְכָל דְקָמוּ עֲלוֹהִי הֲאַתּוּן תִּתְפַּרְעִין לְבַעֲלָא אִם אַתּוּן תִּפְרְקוּן יָתֵיהּ דְיִתְפְּרַע לֵיהּ יִתְקְטֵל בְּרַם אַרְכָּא יִתְיְהֵב לֵיהּ עַד צַפְרָא אִם צְרוֹךְ אִית בֵּיהּ יִתְפְּרַע לֵיהּ בַּעֲלָא אֲרֵי תְּרַע יַת אֱגוֹרֵיהּ:
וּקְרָא לֵיהּ בְּיוֹמָא הַהוּא יְרֻבַּעַל לְמֵימָר יִתְפְּרַע לֵיהּ בַּעֲלָא אֲרֵי תְּרַע יַת אֱגוֹרֵיהּ:
וְכָל מִדְיָנָאֵי וַעֲמַלְקָאֵי וּבְנֵי מַדִינְחָא אִתְכְּנִישׁוּ כַּחֲדָא וַעֲבָרוּ וּשְׁרוֹ בְּמֵישַׁר יִזְרְעֶאל:
וְרוּחַ גְבוּרָא מִן קֳדָם יְיָ לְבִישַׁת יַת גִדְעוֹן וּתְקַע בְּשׁוֹפָרָא וְאִתְכְּנִישׁוּ אֱנַשׁ אֲבִיעֶזֶר בַּתְרוֹהִי:
וְאִזְגַדִין שָׁלַח בְּכָל תְּחוּם מְנַשֶׁה וְאִתְכְּנִישׁוּ אַף אִינוּן בַּתְרוֹהִי וְאִזְגַדִין שָׁלַח בְּשֵׁבֶט אָשֵׁר וּבְשֵׁבֶט זְבוּלֻן וּבְשֵׁבֶט נַפְתָּלִי וּסְלִיקוּ לָקֳדָמוּתְהוֹן:
וַאֲמַר גִדְעוֹן קֳדָם יְיָ אִם אִיתָךְ פָּרִיק בִּידִי יַת יִשְׂרָאֵל כְּמָא דְמַלֵילְתָּא:
הָא אֲנָא מָחֵת יַת גִזְתָא דְעִמְרָא בְּבֵית אִדְרֵי אִם טַלָא יְהֵי עַל גִזְתָא בִּלְחוֹדָהּ וְעַל כָּל אַרְעָא יוּבְשָׁא וְאִדַע אֲרֵי תִפְרוֹק בִּידִי יַת יִשְׂרָאֵל כְּמָא דְמַלֵלְתָּא:
וַהֲוָה כֵן וְאַקְדֵים מִיוֹמָא דְבַתְרוֹהִי וַעֲצַר יַת גִזְתָא וּנְפַץ טַלָא מִן גִזְתָא מְלֵי לַקְנָא דְמַיָא:
וַאֲמַר גִדְעוֹן קֳדָם יְיָ לָא יִתְקֵיף רוּגְזָךְ בִּי וַאֲמַלֵל בְּרַם זִמְנָא הָדָא אֲנַסֵי כְעַן לְחוֹד זִמְנָא בְּגִזְתָא יְהֵי כְעַן יוּבְשָׁא עַל גִזְתָא בִּלְחוֹדָהּ וְעַל כָּל אַרְעָא יְהֵי טַלָא:
וַעֲבַד יְיָ כֵּן בְלֵילְיָא הַהוּא וַהֲוָה יוּבְשָׁא עַל גִזְתָא בִלְחוֹדָהּ וְעַל כָּל אַרְעָא הֲוָה טַלָא:

Chapter 7

וְאַקְדֵים יְרֻבַּעַל הוּא גִדְעוֹן וְכָל עַמָא דִי עִמֵיהּ וּשְׁרוֹ עַל עֵין חֲרוֹד וּמַשִׁרְיַת מִדְיָנָאֵי שַׁרְיָא לְקִבְלֵיהּ מִצִפּוּנָא מִגִבְעֲתָא דְמִסְתַּכְיָא לְמֵישְׁרָא:
וַאֲמַר יְיָ לְגִדְעוֹן סַגִי עַמָא דִי עִמָךְ מִלְמִמְסַר יַת מִדְיָנָאֵי בִּידֵיהוֹן דִלְמָא יִתְרַבְרְבוּן קֳדָמַי יִשְׂרָאֵל לְמֵימַר יְדָנָא פְרָקַת לָנָא:
וּכְעַן אַכְלֵי כְעַן קֳדָם עַמָא לְמֵימָר מָן דְדָחֵיל וְזָע יְתוּב וְיִתְבְּחַר מִטוּרָא דְגִלְעָד וְתַב מִן עַמָא עַסְרִין וּתְרֵין אַלְפִין וְעַסְרָא אַלְפִין אִשְׁתַּאֲרוּ:
וַאֲמַר יְיָ לְגִדְעוֹן עוֹד עַמָא סַגִי אָחִית יַתְהוֹן לְמַיָא וְאֶבְחָרִינוּן לָךְ תַּמָן וִיהֵי דִי אֵימַר לָךְ דֵין יֵיזִיל עִמָךְ הוּא יֵיזִיל עִמָךְ וְכֹל דִי אֵימַר לָךְ דֵין לָא יֵיזִיל עִמָךְ הוּא לָא יֵיזִיל:
וַאֲחֵית יַת עַמָא לְמַיָא וַאֲמַר יְיָ לְגִדְעוֹן כֹּל דִי לָחֵיךְ בְּלִישָׁנֵיהּ מִן מַיָא כְּמָא דִמְלָחֵיךְ כַּלְבָּא תְּקִים יָתֵיהּ בִּלְחוֹדֵיה וְכֹל דִכְרַע עַל בִּרְכּוֹהִי לְמִשְׁתֵּי:
וַהֲוָה מִנְיַן דְשָׁדָן בִּידֵיהוֹן לְפוּמְהוֹן תְּלַת מְאָה גַבְרָא וְכָל שְׁאָר עַמָא כְּרָעוּ עַל בִּרְכֵּיהוֹן לְמִשְׁתֵּי מַיָא:
וַאֲמַר יְיָ לְגִדְעוֹן בִּתְלַת מְאָה גוּבְרִין דְשָׁדַן בִּידֵיהוֹן לְפוּמְהוֹן אֶפְרוֹק יַתְכוֹן וְאֶמְסַר יַת מִדְיָנָאֵי בִּידָךְ וְכָל ַעמָא יֵיזְלוּן גְבַר לְאַתְרֵיה:
וּנְסִיבוּ יַת זְוָדֵי עַמָא בִּידֵיהוֹן וְיַת שְׁפָרֵיהוֹן וְיַת כָּל אֱנָשׁ יִשְׂרָאֵל שְׁלַח גְבַר לְקִרְווֹהִי וּבִתְלַת מְאָה גַבְרָא אַתְקַף וּמַשִׁרְיַת מִדְיָנָאֵי שַׁרְיָא לְקִבְּלֵיהּ מִשִׁפּוּלֵי מֵישְׁרָא:
וַהֲוָה בְּלֵילְיָא הַהוּא וַאֲמַר לֵיהּ יְיָ קוּם חוֹת בְּמַשְׁרִיתָא אֲרֵי מְסַרְתִּינוּן בִּידָךְ:
וְאִם דָחִיל אַתְּ לְמֵיחַת חוֹת אַתְּ וּפוּרָה עוּלֵמָךְ לְמַשְׁרִיתָא:
וְתִשְׁמַע מַה יְמַלְלוּן וּבָתַר כֵּן יִתְקוֹף יְדָךְ וְתֵחוֹת בְּמַשְׁרִיתָא וּנְחַת הוּא וּפוּרָה עוּלֵימֵיהּ לִסְטַר מְזַרְזַיָא דִי בְּמַשְׁרִיתָא:
וּמִדְיָנָאֵי וַעֲמַלְקָאֵי וְכָל בְּנֵי מַדִינְחָא רָמָן בְּמֵישְׁרָא כְּגוֹבָאֵי לִסְגֵי וְלִגְמַלֵיהוֹן לֵית מִנְיַן כְּחָלָא דִי עַל כֵּיף יַמָא לִסְגֵי:
וַאֲתָא גִדְעוֹן וְהָא גַבְרָא מִשְׁתָּעֵי לְחַבְרֵיהּ חֶלְמָא וַאֲמַר הָא חֶלְמָא חַלְמֵית וְהָא חֲרַר דִלְחֵם שְׂעוֹרִין מִתְהַפִּיךְ בְּמַשִׁרְיַת מִדְיָנָאֵי וַאֲתָא עַד מַשְׁכְּנָא וּמְחָהִי וּנְפַל וְאַפְכֵיהּ לְעֵילָא וּנְפַל מַשְׁכְּנָא:
וַאֲתֵיב חַבְרֵיהּ וַאֲמַר לֵית דָא אֱלָהֵן חַרְבָּא דְגִדְעוֹן בַּר יוֹאָשׁ אֱנָשׁ יִשְׂרָאֵל מְסַר יְיָ בִּידֵיהּ יַת מִדְיָנָאֵי וְיַת כָּל מַשְׁרִיתָא:
וַהֲוָה כַּד שְׁמַע גִדְעוֹן יַת שׁוּעִית חֶלְמָא וְיַת פִּשְׁרֵיהּ וְאוֹדֵי וְתַב לְמַשְׁרִיתָא דְיִשְׂרָאֵל וַאֲמַר קוּמוּ אֲרֵי מְסַר יְיָ בִּידֵיכוֹן יַת מַשִׁרְיַת מִדְיָנָאֵי:
וּפַלֵג יַת תְּלַת מְאָה גַבְרָא לִתְלַת מַשִׁרְיָן וִיהַב שׁוֹפָרַיָא בְּיַד כּוּלְהוֹן וְכַדִין רֵיקָנִין וּבְעוּרַיָא בְּגוֹ כַדִין:
וַאֲמַר לְהוֹן מִנִי תֶחֱזוּן וְכֵן תַּעְבְּדוּן וְהָא אֲנָא אָתֵי בִּסְיָפֵי מַשְׁרִיתָא וִיהֵי כְּמָא דְאַעֲבֵּיד כֵּן תַּעְבְּדוּן:
וְאֶתְקַע בְּשׁוֹפָרָא אֲנָא וְכָל דְעִמִי וְתִתְקְעוּן בְּשׁוֹפָרַיָא אַף אַתּוּן סַחֲרָנוּת כָּל מַשְׁרִיתָא וְתֵימְרוּן חַרְבָּא דְמִקַטְלָא מִן קֳדָם יְיָ וְנִצְחָנָא עַל יְדֵי גִדְעוֹן:
וַאֲתָא גִדְעוֹן וּמְאָה גַבְרָא דְעִמֵיהּ בִּסְיָפֵי מַשְׁרִיתָא בְּרֵישׁ מַטַרְתָּא מְצִיעֲתָא בְּרַם אֲקָמָא אֲקִימוּ יַת נָטְרַיָא וּתְקָעוּ בְּשׁוֹפָרַיָא וּנְפָצוּ כַדִין דְבִידֵיהוֹן:
וּתְקָעוּ תְּלַת מַשִׁרְיָן בְּשׁוֹפָרַיָא וְתַבָּרוּ כַדַיָא וְאַתְקִיפוּ בְּיַד סְמָלְהוֹן בִּבְעוּרַיָא וּבְיַד יְמִינְהוֹן בְּשׁוֹפָרַיָא לְמִתְקַע וַאֲמַרוּ חַרְבָּא מִן קֳדָם יְיָ וְנַצְחָנָא עַל יְדֵי גִדְעוֹן:
וְקָמוּ גְבַר בְּאַתְרֵיהּ סְחוֹר סְחוֹר לְמַשְׁרִיתָא וּרְהָטַת כָּל מַשְׁרִיתָא וְיַבִּיבוּ וְאַפָכוּ:
וּתְקָעוּ תְּלַת מְאָה שׁוֹפָרַיָא וְשַׁוִי יְיָ יַת חֶרֶב גְבַר בְּחַבְרֵיהּ וּבְכָל מַשְׁרִיתָא וְאַפְּכַת מַשְׁרִיתָא עַד בֵּית שִׁיטָה לִצְרֵירַת עַד כֵּיף מֵישַׁר אָבֵל מְחוֹלָה לְטַבָּת:
וְאִתְכְּנִישׁ אֱנַשׁ יִשְׂרָאֵל מִנַפְתָּלִי וּמִן שִׁבְטָא דְאָשֵׁר וּמִן כָּל שִׁבְטָא דִמְנַשֶׁה וּרְדָפוּ בָּתַר מִדְיָנָאֵי:
וְאִזְגַדִין שְׁלַח גִדְעוֹן בְּכָל טוּרָא דְבֵית אֶפְרַיִם לְמֵימָר חוּתוּ לָקֳדָמוּת מִדְיָנָאֵי וַאֲחָדוּ עֲלֵיהוֹן יַת מִגִיזָתָא עַד בֵּית בָּרָה וְיַת יַרְדְנָא וְאִתְכְּנִישׁ כָּל אֱנָשׁ אֶפְרַיִם וַאֲחָדוּ יַת מְגִיזָתָא עַד בֵּית בָּרָה וְיַת יַרְדָנָא:
וַאֲחָדוּ תְּרֵין רַבְרְבֵי מִדְיָן יַת עוֹרֵב וְיַת זְאֵב וּקְטָלוּ יַת עוֹרֵב בִּשְׁקִיף עוֹרֵב וְיַת זְאֵב קְטָלוּ בְּמֵישַׁר זְאֵב וּרְדָפוּ בָּתַר מִדְיָנָאֵי וְרֵישׁ עוֹרֵב וּזְאֵב אַיְתִיאוּ לְוַת גִדְעוֹן מֵעִבְרָא לְיַרְדְנָא:

Chapter 8

וַאֲמַרוּ לֵיהּ אֱנָשׁ אֶפְרַיִם מָא פִּתְגָמָא הָדֵין עֲבַדְתָּא לָנָא בְּדִיל מַה לָא קְרֵיתָא לָנָא כִּדְאַזַלְתָּא לְאַגָחָא קְרָבָא בְּמִדְיָן וְדַיְנִין עִמֵיהּ בִּתְקוֹף:
וַאֲמַר לְהוֹן מֶה עַבְדֵית כְּעַן כְּוַתְכוֹן הֲלָא טָבִין חַלָשַׁיָא דְבֵית אֶפְרַיִם מִתַּקְפַיָא דְבֵית אֲבִיעֶזֶר:
בְּיֶדְכוֹן מְסַר יְיָ יַת רַבְרְבֵי מִדְיָן יַת עוֹרֵב וְיַת זְאֵב וּמַה יְכֵלִית לְמֶעְבַּד כְּוַתְכוֹן בְּכֵן נָחִת רוּחֵיהוֹן מִנֵיהּ בְּמַלָלוּתֵיה פִּתְגָמָא הָדֵין:
וַאֲתָא גִדְעוֹן לְיַרְדְנָא עֲבַר הוּא וּתְלַת מְאָה גַבְרַיָא דְעִמֵיהּ כַּד מְשַׁלְהֵן וְרָדְפִין:
וַאֲמַר לֶאֱנָשֵׁי סֻכּוֹת הָבוּ כְעַן פִּתָּא דְלַחְמָא לְעַמָא דִי עִמִי אֲרֵי מְשַׁלְהָן אִנוּן וַאֲנָא רָדִיף בָּתַר זֶבַח וְצַלְמֻנָע מַלְכֵי מִדְיָן:
וַאֲמַר רַבְרְבֵי סֻכּוֹת הָא כְּאִלּוּ זֶבַח וְצַלְמֻנָע כְעַן מְסִירִין בִּידָךְ אֲרֵי נִתַּן לְחֵילָךְ לַחֲמָא:
וַאֲמַר גִדְעוֹן בְּכֵן כַּד יִמְסַר יְיָ יַת זֶבַח וְיַת צַלְמֻנָע בִּידִי וְאָדוּשׁ יַת בִּסְרֵיכוֹן עַל כּוּבֵי מַדְבְּרָא וְעַל בַּרְקָנַיָא:
וּסְלִיק מִתַּמָן לִפְנוּאֵל וּמַלֵל לְוַתְהוֹן כְּדָא וַאֲתִיבוּ יָתֵיה אֱנָשֵׁי פְּנוּאֵל כְּמָא דַאֲתִיבוּ אֱנָשֵׁי סֻכּוֹת:
וַאֲמַר אַף לֶאֱנָשֵׁי פְּנוּאֵל לְמֵימַר כַּד אֵיתוּב בִּשְׁלָם אֶתְרַע יַת מַגְדְלָא הָדֵין:
וְזֶבַח וְצַלְמֻנָע בַּקַרְקֹר וּמַשִׁרְיָתֵיהוֹן עִמְהוֹן כַּחֲמֵשֶׁת עֲסַר אַלְפִין כֹּל דְאִשְׁתָּאֲרוּ מִכֹּל מַשִׁרְיַת בְּנֵי מַדִנְחָא וּדְאִתְקַטָלוּ מְאָה וְעַשְׂרִין אַלְפִין גַבְרָא שְׁלִיפֵי חַרְבָּא:
וּסְלֵיק גִדְעוֹן בְּאוֹרַח מַשִׁרְיַת עֲרָבָאֵי דְשָׁרָן בִּשְׁכוּנִין בְּמַדְבְּרָא מִמַדִינְחָא לְנוֹבַח וְרָמָתָא וּמְחָא יַת מַשְׁרִיתָא וּמַשְׁרִיתָא שַׁרְיָא לְרוֹחְצָן:
וְאַפָכוּ זֶבַח וְצַלְמֻנָע וּרְדַף בַּתְרֵיהוֹן וַאֲחַד יַת תְּרֵין מַלְכֵי מִדְיָן יַת זֶבַח וְיַת צַלְמֻנָע וְכָל מַשְׁרִיתָא אָזִיעַ:
וְתַב גִדְעוֹן בַּר יוֹאָשׁ מִן קְרָבָא עַד לָא מֵעַל שִׁמְשָׁא:
וַאֲחַד עוּלֵימָא מֵאֱנָשֵׁי סֻכּוֹת וְשַׁאֲלֵיהּ וּכְתַב לֵיהּ יַת רַבְרְבֵי סֻכּוֹת וְיַת סָבָהָא שַׁבְעִין וְשִׁבְעָה גוּבְרִין:
וַאֲתָא לְוַת אֱנָשֵׁי סֻכּוֹת וַאֲמַר הָא זֶבַח וְצַלְמֻנָע דְחַסִדְתּוּן יָתִי לְמֵימָר הָא כְאִלוּ זֶבַח וְצַלְמֻנָע כְּעַן מְסִירִין בִּידָךְ אֲרֵי נִתַּן לְגַבְרָךְ דִמְשַׁלְהֵן לַחְמָא:
וּדְבַר יַת סָבֵי קַרְתָּא וְיַת כּוּבֵי מַדְבְּרָא וְיַת בַּרְקָנַיָא וְתַבַּר עֲלֵיהוֹן יַת אֱנָשֵׁי סֻכּוֹת:
וְיַת מִגְדַל פְּנוּאֵל תְּרַע וּקְטַל יַת אֱנָשֵׁי קַרְתָּא:
וַאֲמַר לְזֶבַח וּלְצַלְמֻנָע הֵיכָא גַבְרַיָא דִקְטַלְתּוּן בְּתָבוֹר וַאֲמַרוּ כְּוָתָךְ כֵּן אַף אִינוּן חַד מִנְהוֹן כְּחֶזְוָא דִבְנֵי מַלְכָּא:
וַאֲמַר אַחַי בְּנֵי אִמָא אִנוּן קַיָם הוּא יְיָ אֵלוּ קַיֵמְתּוּן יַתְהוֹן לָא קַטְלֵית יַתְכוֹן:
וַאֲמַר לְיֶתֶר בּוּכְרֵיהּ קוּם קְטוֹל יַתְהוֹן וְלָא שְׁלַף עוּלֵימָא חַרְבֵּיהּ אֲרֵי דָחִיל אֲרֵי עַד כְּעַן יָנִיק:
וַאֲמַר זֶבַח וְצַלְמֻנָע קוּם אַתְּ וּשְׁלַט בָּנָא אֲרֵי כְגַבְרָא כֵּן גְבוּרְתֵּיהּ וְקָם גִדְעוֹן וּקְטַל יַת זֶבַח וְיַת צַלְמֻנָע וּנְסִיב יַת עִנְקַיָא דִי בְצַוְארֵי גְמַלֵיהוֹן:
וַאֲמַרוּ אֱנָשׁ יִשְׂרָאֵל לְגִדְעוֹן מְלוֹךְ עֲלָנָא אַף אַתְּ אַף בְּרָךְ אַף בַּר בְּרָךְ אֲרֵי פְרַקְתָּנָא מִן יַד מִדְיָנָאֵי:
וַאֲמַר לְהוֹן גִדְעוֹן לָא אֶמְלוֹךְ אֲנָא עֲלֵיכוֹן וְלָא יִמְלוֹךְ בְּרִי עֲלֵיכוֹן יְיָ יִמְלוֹךְ עֲלֵיכוֹן:
וַאֲמַר לְהוֹן גִדְעוֹן שָׁאֵיל אֲנָא מִנְכוֹן שְׁאֵילָה וְהָבוּ לִי גְבַר קָדָשֵׁיהּ מִן בִּזְתָא אֲרֵי קָדָשֵׁי דְדַהֲבָא לְהוֹן אֲרֵי עֲרָבָאֵי אִינוּן:
וַאֲמַרוּ מִתַּן נִתָּן וּפָרְשׁוּ יַת כְּסוּתָא וּרְמוּ לְתַמָן גְבַר קָדְשֵׁיהּ מִן בִּזְתָא:
וַהֲוָה מַתְקַל קָדָשֵׁי דְדַהֲבָא דִשְׁאַל אֶלֶף וּשְׁבַע מְאָה דַהֲבָא בַּר מִן סַהֲרָנַיָא וּכְלִילָא וּלְבוּשֵׁי אַרְגְוָנָא דִי עַל מַלְכֵי מִדְיָן וּבַר מִן עֲנָקַיָא דִי בְּצַוְרֵי גְמַלֵיהוֹן:
וַעֲבַד יָתֵיה גִדְעוֹן לְאֵיפוֹדָא וַאֲקֵים יָתֵיהּ בְּקַרְתֵּיהּ בְּעָפְרָה וּטְעוֹ כָל יִשְׂרָאֵל בַּתְרוֹהִי לְתַמָן וַהֲוָה לְגִדְעוֹן וּלְבֵיתֵיהּ לְתַקָלָא:
וְאִתְּבַרוּ מִדְיָנָאֵי קֳדָם בְּנֵי יִשְׂרָאֵל וְלָא אוֹסִיפוּ לַאֲרָמָא רֵישֵׁיהוֹן וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל אַרְבְּעִין שְׁנִין בְּיוֹמֵי גִדְעוֹן:
וַאֲזַל יְרֻבַּעַל בַּר יוֹאָשׁ וִיתֵיב בְּבֵיתֵיהּ:
וּלְגִדְעוֹן הֲווֹ שַׁבְעִין בְּנִין נַפְקֵי יַרְכֵיהּ אֲרֵי נְשִׁין סַגִיאִין הֲוָאָה לֵיהּ:
וּלְחֵנָתֵיהּ דִי בִשְׁכֶם יְלִידַת לֵיהּ אַף הִיא בַּר וְשַׁוִי יַת שְׁמֵיהּ אֲבִימֶלֶךְ:
וּמִית גִדְעוֹן בַּר יוֹאָשׁ בְּסֵיבוּ טָבָא וְאִתְקְבַר בְּקִבְרָא דְיוֹאָשׁ אֲבוּהִי בְּעָפְרָה דַאֲבוּהִי דְעָזֶר:
וַהֲוָה כַּד מִית גִדְעוֹן וְתָבוּ בְּנֵי יִשְׂרָאֵל וְטָעוּ בָּתַר בַּעֲלַיָא וְשַׁוִיאוּ לְהוֹן בָּעֵיל קְיָם לְטַעֲוָתָא:
וְלָא אִדְכָּרוּ בְּנֵי יִשְׂרָאֵל יַת פּוּלְחָנָא דַייָ אֱלָהָהוֹן דִי שֵׁיזִיב יַתְהוֹן מִיַד כָּל בַּעֲלֵי דְבָבֵיהוֹן מִסְחוֹר סְחוֹר:
וְלָא עֲבָדוּ טֵיבוּ עִם בֵּית יְרֻבַּעַל גִדְעוֹן כְּכָל טַבְתָא דִי עֲבַד עִם יִשְׂרָאֵל:

Chapter 9

וַאֲזַל אֲבִימֶלֶךְ בַּר יְרֻבַּעַל לִשְׁכֶם לְוַת אֲחֵי אִמֵיהּ וּמַלֵיל עִמְהוֹן וְעִם כָּל זַרְעִית בֵּית אֲבוּהָא דְאִימֵיהּ לְמֵימָר:
מַלִילוּ כְּעַן קֳדָם כָּל יַתְבֵי שְׁכֶם מַה טַב לְכוֹן דְיִמְלְכוּן עֲלֵיכוֹן שַׁבְעִין גַבְרָא כָּל בְּנֵי יְרֻבַּעַל אוֹ דְיִמְלוֹךְ עֲלֵיכוֹן גַבְרָא חָד וְתִדְכְּרוּן אֲרֵי קָרִיבְכוֹן וּבִשְׂרְכוֹן אֲנָא:
וּמַלִילוּ אֲחֵי אִמֵיהּ עֲלוֹהִי קֳדָם כָּל יַתְבֵי שְׁכֶם יַת כָּל פִּתְגָמַיָא הָאִלֵין וְאִתְפְּנֵי לִבְּהוֹן בָּתַר אֲבִימֶלֶךְ אֲרֵי אֲמַרוּ אֲחוּנָא הוּא:
וִיהָבוּ לֵיהּ שַׁבְעִין כְּסָף מִבֵּית בָּעֵיל קְיָם וַאֲגַר בְּהוֹן אֲבִימֶלֶךְ גַבְרִין סְרִיקִין וּבְסִירִין וַאֲזַלוּ בַּתְרוֹהִי:
וְעַל לְבֵית אֲבוּהִי לְעָפְרָה וּקְטַל יַת אֲחוֹהִי בְנֵי יְרֻבַּעַל שַׁבְעִין גַבְרָא עַל אַבְנָא חֲדָא וְאִשְׁתָּאַר יוֹתָם בַּר יְרֻבַּעַל זְעֵירָא אֲרֵי אִיטַמָר:
וְאִתְכְּנִישׁוּ כָּל יַתְבֵי שְׁכֶם וְכָל בֵּית מִלוֹא וַאֲזָלוּ וְאַמְלִיכוּ יַת אֲבִימֶלֶךְ לְמֶהֱוֵי מַלְכָּא עִם מֵישַׁר קָמָתָא דִי בִּשְׁכֶם:
וְחַוִיאוּ לְיוֹתָם וַאֲזַל וְקָם בְּרֵישׁ טוּרָא דִגְרִיזִין וַאֲרֵים קָלֵיהּ וּקְרָא וַאֲמַר לְהוֹן קַבִּילוּ מִנִי יַתְבֵי שְׁכֶם וִיקַבֵּל צְלוֹתְכוֹן יְיָ:
הֲמֵיזַל אֲזָלוּ אִילָנַיָא לְמַנָאָה עֲלֵיהוֹן מַלְכָּא וְאֲמָרוּ לְזֵיתָא מְלוֹךְ עֲלָנָא:
וַאֲמַר לְהוֹן זֵיתָא הַאֶשְׁבּוֹק יַת מִשְׁחִי דְמִנֵיהּ מְיַקְרִין קֳדָם יְיָ וּבֵיהּ מִתְפַּנְקִין בְּנֵי אֱנָשָׁא וְאֵהַךְ לְמֶעְבַּד מַלְכוּ עַל אִילָנַיָא:
וַאֲמָרוּ אִלָנַיָא לְתֵינָתָא אִיתָא אַתְּ מְלוֹכִי עֲלָנָא:
וַאֲמָרַת לְהוֹן תֵינָתָא הֲאֶשְׁבוֹק יַת חוּלְיִי וְיַת דְבֵילְתִּי שַׁפִּירְתָּא וְאֵהַךְ לְמֶעְבַּד מַלְכוּ עַל אִילָנַיָא:
וַאֲמָרוּ אִילָנַיָא לְגוּפְנָא אִיתָא אַתְּ מְלוֹכִי עֲלָנָא:
וַאֲמַרַת לְהוֹן גוּפְנָא הֲאֶשְׁבוֹק יַת חַמְרִי דְמִנֵיהּ מְנַסְכִין קֳדָם יְיָ וּבֵיהּ חָדָן רַבְרְבַיָא וְאֵיהַךְ לְמֶעְבַּד מַלְכוּ עַל אִילָנַיָא:
וַאֲמַרוּ כָל אִילָנַיָא לְאַטָדָא אִיתָא אַתְּ מְלוֹךְ עֲלָנָא:
וַאֲמַר אֲטָדָא לְאִילָנַיָא אִם בִּקְשׁוֹט אַתּוּן מָשְׁחִין יָתִי לְמֶהֱוֵי מַלְכָּא עֲלֵיכוֹן עוּלוּ שְׁרוֹ בְטוּלִי וְאִם לָא תִפּוֹק אֶשְׁתָּא מִן אֲטָדָא וְתֵיכוּל יַת אַרְזֵי לִבְנָן:
וּכְעַן אִם בְּקוּשְׁטָא וּבִשְׁלֶמְתָּא עֲבַדְתּוּן וְאַמְלַכְתּוּן יַת אֲבִימֶלֶךְ וְאִם טַבְתָא עֲבַדְתּוּן עִם יְרוּבַּעַל וְעִם בֵּיתֵיהּ וְאִם כְּגוּמְלָא דִידוֹהִי עֲבַדְתּוּן לֵיהּ:
דְאַגִיחַ אַבָּא עֲלֵיכוֹן וּמְסַר יַת נַפְשֵׁיהּ כְּעַל גַב לְאִתְקְטָלָא וְשֵׁזִיב יַתְכוֹן מִיַד מִדְיָנָאֵי:
וְאַתּוּן קַמְתּוּן עַל בֵּית אַבָּא יוֹמָא דֵין וּקְטַלְתּוּן יַת בְּנוֹהִי שַׁבְעִין גַבְרָא עַל אַבְנָא חֲדָא וְאַמְלַכְתּוּן יַת אֲבִימֶלֶךְ בַּר אַמְתֵיהּ עַל יַתְבֵי שְׁכֶם אֲרֵי אֲחוּכוֹן הוּא:
וְאִם בְּקוּשְׁטָא וּבִשְׁלֶמְתָּא עֲבַדְתּוּן עִם יְרוּבַּעַל וְעִם בֵּיתֵיהּ יוֹמָא דֵין חֲדוּ בַּאֲבִימֶלֶךְ וְיֶחֱדֵי אַף הוּא בְּכוֹן:
וְאִם לָא תִפּוֹק אֶשְׁתָּא מֵאֲבִימֶלֶךְ וְתֵיכוּל יַת יַתְבֵי שְׁכֶם וְיַת בֵּית מִלוֹא וְתִפּוֹק אֶשְׁתָּא מִיַתְבֵי שְׁכֶם וּמִבֵּית מִלוֹא וְתֵיכוּל יַת אֲבִימֶלֶךְ:
וַאֲפֵּךְ יוֹתָם וַעֲרַק וַאֲזַל לִבְאֵר וִיתֵיב תַּמָן מִן קֳדָם אֲבִימֶלֶךְ אֲחוֹהִי:
וּמְלַךְ אֲבִימֶלֶךְ עַל יִשְׂרָאֵל תְּלַת שְׁנִין:
וּשְׁלַח יְיָ רוּחָא בִּישָׁא בֵּין אֲבִימֶלֶךְ וּבֵין יַתְבֵי שְׁכֶם וְשַׁקָרוּ יַתְבֵי שְׁכֶם בַּאֲבִימֶלֶךְ:
לְמֵיתֵי חֲטוֹף שַׁבְעִין בְּנֵי יְרֻבַּעַל וּדְמֵיהוֹן לְשַׁוָאָה עַל אֲבִימֶלֶךְ אֲחוּהוֹן דִקְטַל יַתְהוֹן וְעַל יַתְבֵי שְׁכֶם דְתַקִיפוּ יַת יְדוֹהִי לְמִקְטַל יַת אֲחוֹהִי:
וְשַׁוִיאוּ לֵיהּ יַתְבֵי שְׁכֶם כְּמַנִין עַל רֵישֵׁי טוּרַיָא וְאַנְסִין יַת כָּל דִי עֲבַר עֲלֵיהוֹן בְּאוֹרְחָא וְאִתְחַוָא לַאֲבִימֶלֶךְ:
וַאֲתָא גַעַל בַּר עֶבֶד וַאֲחוֹהִי וַעֲבָרוּ בִּשְׁכֶם וְאִתְרְחִיצוּ עֲלוֹהִי יָתְבֵי שְׁכֶם:
וּנְפַקוּ לְחַקְלָא וּקְטַפוּ יַת כַּרְמֵיהוֹן וַעֲצָרוּ וַעֲבַדוּ חִנְגִין וַעֲלוּ לְבֵית טַעֲוַתְהוֹן וַאֲכַלוּ וּשְׁתִיאוּ וּלְטִיטוּ יַת אֲבִימֶלֶךְ:
וַאֲמַר גַעַל בַּר עֶבֶד מַן אֲבִימֶלֶךְ וּמַן שְׁכֶם אֲרֵי נִפְלְחִנֵיהּ הֲלָא בַּר יְרוּבַּעַל וּזְבֻל שׁוּלְטְנֵיהּ פְּלַחוּ יַת אֱנָשֵׁי חֲמוֹר אֲבוּהִי דִשְׁכֶם וּמָא דֵין נִפְלְחִנֵיהּ אֲנַחְנָא:
וּמַן יִמְסַר יַת עַמָא הָדֵין בִּידִי וְאַעֲדִי יַת אֲבִימֶלֶךְ וַאֲמַר לַאֲבִימֶלֶךְ טַקֵיס יַת מַשִׁרְיָתָךְ וּפוּק:
וּשְׁמַע זְבוּל רַב קַרְתָּא יַת פִּתְגָמֵי גַעַל בַּר עֶבֶד וּתְקֵיף רוּגְזֵיהּ:
וּשְׁלַח אִזְגַדִין לְוַת אֲבִימֶלֶךְ בְּרָז לְמֵימָר הָא גַעַל בַּר עֶבֶד וַאֲחוֹהִי אָתָן לִשְׁכֶם וְהָא אִנוּן צָיְרִין יַת קַרְתָּא עֲלָךְ:
וּכְעַן קוּם בְּלֵילְיָא אַתְּ וְעַמָא דְעִמָךְ וּכְמוֹן בְּחַקְלָא:
וִיהֵי בְצַפְרָא כְּמַדְנַח שִׁמְשָׁא תַּקְדֵים וְתִתְנְגֵיד עַל קַרְתָּא וְהָא הוּא וְעַמָא דִי עִמֵיהּ נַפְקִין לָקֳדָמוּתָךְ וְתַעְבֵּיד לֵיהּ כְּמָא דְתַשְׁכַּח יְדָךְ:
וְקָם אֲבִימֶלֶךְ וְכָל עַמָא דִי עִמֵיהּ בְּלֵילְיָא וּכְמָנוּ עַל שְׁכֶם אַרְבָּעָא מַשִׁרְיָן:
וּנְפַק גַעַל בַּר עֶבֶד וְקָם בְּמַעֲלָנָא דִתְרַע קַרְתָּא וְקָם אֲבִימֶלֶךְ וְעַמָא דִי עִמֵיהּ מִן כְּמָנָא:
וַחֲזָא גַעַל יַת עַמָא וַאֲמַר לִזְבוּל הָא עַמָא נָחִית מֵרֵישֵׁי טוּרַיָא וַאֲמַר לֵיהּ זְבוּל יַת טוּלָא דְטוּרַיָא אַתְּ חָזֵי כְּגַבְרַיָא:
וְאוֹסֵף עוֹד גַעַל לְמַלָלָא וַאֲמַר הָא עַמָא נָחְתִין מִן תּוּקְפָא דְאַרְעָא וּמַשְׁרִיתָא חֲדָא אַתְיָא מֵאוֹרַח מֵישַׁר מְעוֹנְנַיָא:
וַאֲמַר לֵיהּ זְבוּל אָן אִנוּן פִּתְגָמָךְ דִי אַתְּ אֲמַר מַן אֲבִימֶלֶךְ אֲרֵי נִפְלְחִינֵיהּ הֲלָא דֵין עַמָא דְקַצְתָּא בֵּיהּ פּוּק כְעַן וַאֲגַח קְרָבָא בֵּיהּ:
וּנְפַק גַעַל קֳדָם יַתְבֵי שְׁכֶם וַאֲגַח קְרָבָא בַּאֲבִימֶלֶךְ:
וְרַדְפֵיהּ אֲבִימֶלֶךְ וְאַפַּךְ מִן קֳדָמוֹהִי וּנְפַלוּ קְטוֹלִין סַגִיאִין עַד מַעֲלָנָא דְתַרְעָא:
וִיתֵיב אֲבִימֶלֶךְ בָּארוּמָא וְתָרֵיךְ זְבוּל יַת גַעַל וְיַת אֲחוֹהִי מִלְמֵיתַב בִּשְׁכֶם:
וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וּנְפַק עַמָא לְחַקְלָא וְחַוִיאוּ לַאֲבִימֶלֶךְ:
וְדַבֵּר יַת עַמָא וּפַלְגִנוּן לִתְלַת מַשִׁרְיָן וּכְמַן בְּחַקְלָא וַחֲזָא וְהָא עַמָא נָפֵיק מִן קַרְתָּא וְקָם עֲלֵיהוֹן וּמְחָנוּן:
וַאֲבִימֶלֶךְ וְכָל מַשִׁרְיָן דְעִמֵיהּ אִתְנְגִידוּ וְקָמוּ בְּמַעֲלָנָא דִתְרַע קַרְתָּא וְתַרְתֵּין מַשִׁרְיָן אִתְנְגִידוּ עַל כָּל דִי בְחַקְלָא וְקַטְלוּנוּן:
וַאֲבִימֶלֶךְ אַגִיחַ קְרָבָא בְקַרְתָּא כָּל יוֹמָא הַהוּא וּכְבַשׁ יַת קַרְתָּא וְיַת עַמָא דִי בָהּ קְטַל וּתְרַע יַת קַרְתָּא וְזַרְעָה מִלְחָא:
וּשְׁמָעוּ כָּל יַתְבֵי מִגְדַל שְׁכֶם וַאֲתוֹ לִצְרִיחַ בֵּית אֵל לְמִגְזַר קְיָם:
וְאִתְחֲוָאָה לַאֲבִימֶלֶךְ אֲרֵי אִתְכַּנָשׁוּ כָּל יַתְבֵי מִגְדַל שְׁכֶם:
וּסְלֵיק אֲבִימֶלֶךְ לְטוּרָא דְצַלְמוֹן הוּא וְכָל עַמָא דִי עִמֵיהּ וּנְסֵיב אֲבִימֶלֶךְ יַת כּוּלְבַיָא בִּידֵיהּ וְקָץ סוֹכַת אָעִין וּנְטָלָהּ וְשַׁוֵי עַל כִּתְפוֹהִי וַאֲמַר לְעַמָא דִי עִמֵיהּ מַה חֲזֵיתוּן עַבָדִית אוֹחוּ עֲבִידוּ כְּוָתִי:
וְקָצוּ אַף כָּל עַמָא גְבַר סוֹכֵיה וַאֲזַלוּ בָּתַר אֲבִימֶלֶךְ וְשַׁוִיאוּ עַל צְרִיחַ וְאַדְלִיקוּ עֲלֵיהוֹן יַת צְרִיחַ בְּנוּרָא וְאִתְקְטָלוּ אַף כָּל אֱנָשֵׁי מִגְדַל שְׁכֶם כְּאֶלֶף גְבַר וְאִתְּתָא:
וַאֲזַל אֲבִימֶלֶךְ לְתֵבֵץ וּשְׁרָא בְּתֵבֵץ וְכַבְּשָׁהּ:
וּמִגְדַל תַּקִיף הֲוָה בְּגוֹ קַרְתָּא וְאַפִיכוּ לְתַמָן כָּל גַבְרַיָא וּנְשַׁיָא וְכָל יַתְבֵי קַרְתָּא וְאַגִיפוּ בְּאַפֵּיהוֹן וּסְלִיקוּ עַל אִגַר מִגְדָלָא:
וַאֲתָא אֲבִימֶלֶךְ עַד מִגְדָלָא וְאַגִיחַ קְרָבָא בֵּיהּ וְקָרֵיב עַד תְּרַע מִגְדָלָא לְאוֹקָדוּתֵיה בְּנוּרָא:
וּרְמַת אִתְּתָא חֲדָא פַּלְגוּת רְכַב רְחַיָא עַל רֵישָׁא דַאֲבִימֶלֶךְ וְרַעַת יַת גוּלְגַלְתֵּיהּ:
וּקְרָא בִּפְרִיעַ לְעוּלֵימָא נָטִיל זְיָנֵיהּ וַאֲמַר לֵיהּ שְׁלוֹף חַרְבָּךְ וּקְטוֹלְנִי דִלְמָא יֵימְרוּן עָלֵי אִתְּתָא קְטַלְתֵּיהּ וְקַטְלֵיהּ עוּלֵימֵיהּ וּמִית:
וַחֲזוֹ גְבַר יִשְׂרָאֵל אֲרֵי מִית אֲבִימֶלֶךְ וַאֲזָלוּ אֱנַשׁ לְאַתְרֵיהּ:
וַאֲתֵיב יְיָ יַת בִּישַׁת אֲבִימֶלֶךְ דַעֲבַד לַאֲבוּהִי לְמִקְטַל יַת שַׁבְעִין אֲחוֹהִי:
וְיַת כָּל בִּישַׁת אֱנָשֵׁי שְׁכֶם אֲתֵיב יְיָ בְּרֵישֵׁיהוֹן וַאֲתָת לְהוֹן לְוָטַת יוֹתָם בַּר יְרֻבַּעַל:

Chapter 10

וְקָם בָּתַר אֲבִימֶלֶךְ לְמִפְרַק יַת יִשְׂרָאֵל תּוֹלָע בַּר פּוּאָה בַּר דּוֹדוֹ גַבְרָא מִשִׁבְטָא דְבֵית יִשָׂשכָר וְהוּא יָתֵיב בְּשָׁמִיר בְּטוּרָא דְבֵית אֶפְרָיִם:
וְדָן יַת יִשְׂרָאֵל עַסְרִין וּתְלַת שְׁנִין וּמִית וְאִתְקְבַר בְּשָׁמִיר:
וְקָם בַּתְרוֹהִי יָאִיר גִלְעָדָאָה וְדָן יַת יִשְׂרָאֵל עַסְרִין וְתַרְתֵּין שְׁנִין:
וַהֲווֹ לֵיהּ תְּלָתִין בְּנִין רְכִיבִין עַל תְּלָתִין עוּלִין וּתְלָתִין קִרְוִין לְהוֹן לְהוֹן קְרַן כַּפְּרָנֵי יָאִיר עַד יוֹמָא הָדֵין דִי בְּאַרְעָא דְגִלְעָד:
וּמִית יָאִיר וְאִתְקְבַר בְּקָמוֹן:
וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וּפְלָחוּ יַת בַּעֲלַיָא וְיַת עַשְׁתְּרָתָא וְיַת טַעֲוַת אֲרָם וְיַת טַעֲוַת צִידוֹן וְיַת טַעֲוַת מוֹאָב וְיַת טַעֲוַת בְּנֵי עַמוֹן וְיַת טַעֲוַת פְּלִשְׁתָּאֵי וּשְׁבַקוּ יַת פּוּלְחָנָא דַייָ וְלָא פְּלָחוּ קֳדָמוֹהִי:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְּיַד פְּלִשְׁתָּאֵי וּבְיַד בְּנֵי עַמוֹן:
וּדְחַקוּ וְשַׁעֲבִּידוּ יַת בְּנֵי יִשְׂרָאֵל בְּשַׁתָּא הַהִיא תְּמַנְיָא עַסְרֵי שְׁנִין יַת כָּל בְּנֵי יִשְׂרָאֵל דִבְעִבְרָא דְיַרְדְנָא בְּאַרְעָא אֱמוֹרָאָה דִי בְגִלְעָד:
וַעֲבָרוּ בְנֵי עַמוֹן יַת יַרְדְנָא לְאַגָחָא קְרָבָא אַף בִּיהוּדָה וּבִדְבֵית בִנְיָמִין וּבִדְבֵית אֶפְרַיִם וְעָקַת לְיִשְׂרָאֵל לַחֲדָא:
וּזְעִיקוּ בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ לְמֵימָר חַבְנָא קֳדָמָךְ וַאֲרֵי שְׁבַקְנָא יַת פּוּלְחָנָא דַייָ אֱלָהָנָא וּפְלַחְנָא יַת בַּעֲלַיָא:
וַאֲמַר יְיָ לִבְנֵי יִשְׂרָאֵל הֲלָא מִמִצְרַיִם וּמִן אֱמוֹרָאָה וּמִן בְּנֵי עַמוֹן וּמִפְּלִשְׁתָּאֵי:
וְצִידוֹנָאֵי וַעֲמַלְקָאֵי וֶאֱנָשֵׁי מָעוֹן דָחֲקוּ יַתְכוֹן וּבָעִיתוּן קֳדָמַי וּפַרְקֵית יַתְכוֹן מִידֵיהוֹן:
וְאַתּוּן שְׁבַקְתּוּן פּוּלְחָנִי וּפְלַחְתּוּן לְטַעֲוַת עַמְמַיָא בְּכֵן לָא אוֹסִיף לְמִפְרַק יַתְכוֹן:
אֱזִילוּ וּבְעוֹ מִן טַעֲוָתָא דְאִיתְרְעִיתוּן בְּהוֹן הַאִם יָכְלִין לְמִפְרַקְכוֹן בְּעִידָן עֲקַתְכוֹן:
וַאֲמַרוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ חַבְנָא עֲבֵיד אַתְּ לָנָא כְּכָל דְתַקִין קֳדָמָךְ בְּרַם שֵׁיזַבְנָא כְעַן יוֹמָא דֵין:
וְאַעֲדִיאוּ יַת טַעֲוַת עַמְמַיָא מִבֵּינֵיהוֹן וּפְלָחוּ קֳדָם יְיָ וְעָקַת נַפְשֵׁיהּ בַּעֲמַל יִשְׂרָאֵל:
וְאִתְכְּנִישׁוּ בְּנֵי עַמוֹן וּשְׁרוֹ בַּגִלְעָד וְאִתְכְּנִישׁוּ בְּנֵי יִשְׂרָאֵל וּשְׁרוֹ בְּמִצְפָּה:
וַאֲמַרוּ עַמָא רַבְרְבֵי גִלְעָד גְבַר לְחַבְרֵיהּ מַן גַבְרָא דְיִשְׁרֵי לְאַגָחָא קְרָבָא בִּבְנֵי עַמוֹן יְהֵי לְרֵישׁ לְכֹל יַתְבֵי גִלְעָד:


Chapter 11

וְיִפְתָּח גִלְעָדָא הֲוָה גִבַּר חֵילָא וְהוּא בַּר אִתְּתָא פּוּנְדְקִיתָא וְאוֹלִיד גִלְעָד יַת יִפְתָּח:
וִילֵידַת אִתַּת גִלְעָד לֵיהּ בְּנִין וּרְבִיאוּ בְנֵי אִתְּתָא וְתָרִיכוּ יַת יִפְתָּח וַאֲמַרוּ לֵיהּ לָא תַחְסֵין בְּבֵית אֲבוּנָא אֲרֵי בַּר אִתְּתָא אוֹחֲרִי אַתְּ:
וַעֲרַק יִפְתַּח מִן קֳדָם אֲחוֹהִי וִיתֵיב בְּאַרְעָא טָבָא וְאִתְכְּנִישׁוּ לְוַת יִפְתַּח גַבְרִין סְרִיקִין וּנְפָקוּ עִמֵיהּ:
וַהֲוָה מִזְמַן יוֹמִין וְאַגִיחוּ קְרָבָא בְּנֵי עַמוֹן עִם יִשְׂרָאֵל:
וַהֲוָה כַּד אַגִיחוּ קְרָבָא בְּנֵי עַמוֹן עִם יִשְׂרָאֵל וַאֲזַלוּ סָבֵי גִלְעָד לְמִדְבַּר יַת יִפְתַּח מֵאֲרַע טָב:
וַאֲמַרוּ לְיִפְתַּח אִיתָא וּתְהֵי לָנָא לְרַבָּא וּנְגִיחַ קְרָבָא בִּבְנֵי עַמוֹן:
וַאֲמַר יִפְתָּח לְסָבֵי גִלְעָד הֲלָא אַתּוּן סְנֵיתוּן יָתִי וְתָרִיכְתּוּנִי מִבֵּית אַבָּא וּמָא דֵין אֲתֵיתוּן לְוָתִי כְּעַן כַּד עָקַת לְכוֹן:
וַאֲמַרוּ סָבֵי גִלְעָד לְיִפְתָּח בְּכֵן כְּעַן תַּבְנָא לְוָתָךְ וּתְהַךְ עִמָנָא וְתַגִיחַ קְרָבָא בִּבְנֵי עַמוֹן וּתְהֵי לָנָא לְרֵישׁ לְכֹל יַתְבֵי גִלְעָד:
וַאֲמַר יִפְתָּח לְסָבֵי גִלְעָד אִם מְתִיבִין אַתּוּן יָתִי לְאַגָחָא קְרָבָא בִּבְנֵי עַמוֹן וְיִמְסַר יְיָ יַתְהוֹן קֳדָמַי אֲנָא אֱהֶוֵי לְכוֹן לְרֵישׁ:
וַאֲמַרוּ סָבֵי גִלְעָד לְיִפְתָּח מֵימְרָא דַייָ יְהֵי סָהִיד בֵּינָנָא אִם לָא כְּפִתְגָמָךְ כֵּן נַעְבֵּד:
וַאֲזַל יִפְתָּח עִם סָבֵי גִלְעָד וּמַנִיאוּ עַמָא יָתֵיהּ עֲלֵיהוֹן לְרֵישׁ וּלְרַבָּא וּסְדַר יִפְתָּח יַת כָּל פִּתְגָמוֹהִי קֳדָם יְיָ בְּמִצְפַּיָא:
וּשְׁלַח יִפְתָּח אִזְגַדִין לְוַת מַלְכָּא דִבְנֵי עַמוֹן לְמֵימָר מַה לִי וְלָךְ אֲרֵי אֲתֵיתָא לְוָתִי לְאַגָחָא קְרָבָא בְּאַרְעִי:
וַאֲמַר מַלְכָּא דִבְנֵי עַמוֹן לְאִזְגַדֵי יִפְתָּח אֲרֵי נְסִיב יִשְׂרָאֵל יַת אַרְעִי בְּמִסְקֵיהּ מִמִצְרַיִם מֵאַרְנוֹנָא וְעַד יוּבְקָא וְעַד יַרְדְנָא וּכְעַן אֲתֵיב יַתְהֶן בִּשְׁלָם:
וְאוֹסִיף עוֹד יִפְתָּח וּשְׁלַח אִזְגַדִין לְוַת מַלְכָא דִבְנֵי עַמוֹן:
וַאֲמַר לֵיהּ כִּדְנַן אֲמַר יִפְתָּח לָא נִסִיב יִשְׂרָאֵל יַת אַרְעָא דְמוֹאָב וְיַת אַרְעָא בְּנֵי עַמוֹן:
אֲרֵי בְּמִסַקְהוֹן מִמִצְרָיִם וַאֲזַל יִשְׂרָאֵל בְּמַדְבְּרָא עַד יַמָא דְסוּף וַאֲתָא לִרְקָם:
וּשְׁלַח יִשְׂרָאֵל אִזְגַדִין לְוַת מַלְכָּא דֶאֱדוֹם לְמֵימָר אֶעְבָּר כְּעַן בְּאַרְעָךְ וְלָא שְׁמַע מַלְכָּא דְאֱדוֹם וְאַף לְוַת מַלְכָּא דְמוֹאָב שְׁלַח וְלָא קַבֵּל וִיתֵיב יִשְׂרָאֵל בִּרְקָם:
וַאֲזַל בְּמַדְבְּרָא וְאַקֵיף יַת אֲרַע אֱדוֹם וְיַת אֲרַע מוֹאָב וַאֲתוֹ מִמַדְנַח שִׁמְשָׁא לְאַרְעָא דְמוֹאָב וּשְׁרוֹ בְּעִיבְרָא דְאַרְנוֹן וְלָא עָלוּ בִּתְחוּם מוֹאָב אֲרֵי אַרְנוֹן תְּחוּם מוֹאָב:
וּשְׁלַח יִשְׂרָאֵל אִזְגַדִין לְוַת סִיחוֹן מַלְכָּא דֶאֱמוֹרָאָה מַלְכָּא דְחֶשְׁבּוֹן וַאֲמַר לֵיהּ יִשְׂרָאֵל נֶעְבַּר כְּעַן בְּאַרְעָךְ עַד אַתְרִי:
וְלָא הֵימִין סִיחוֹן יַת יִשְׂרָאֵל לְמֶעְבַּר בִּתְחוּמֵיהּ וּכְנַשׁ סִיחוֹן יַת כָּל עַמֵיהּ וּשְׁרוֹ בְּיָהְצָה וְאַגִיחַ קְרָבָא עִם יִשְׂרָאֵל:
וּמְסַר יְיָ אֱלָהָא דְיִשְׂרָאֵל יַת סִיחוֹן וְיַת כָּל עַמֵיהּ בִּידָא דְיִשְׂרָאֵל וּמְחוֹנוּן וִירֵית יִשְׂרָאֵל יַת כָּל אֲרַע אֱמוֹרָאָה יָתֵיב אַרְעָא הַהִיא:
וִירִיתוּ יַת כָּל תְּחוּם אֱמוֹרָאָה מֵאַרְנוֹנָא וְעַד יוּבְקָא וּמִן מַדְבְּרָא וְעַד יַרְדְנָא:
וּכְעַן יְיָ אֱלָהָא דְיִשְׂרָאֵל תָּרִיךְ יַת אֱמוֹרָאָה מִן קֳדָם עַמֵיהּ יִשְׂרָאֵל וְאַתְּ מְדַמֵי לְמֵירְתֵיהּ:
הֲלָא יַת דְיַחְסְנִנָךְ כְּמוֹשׁ טַעֲוָתָךְ יָתֵיהּ תַּחְסִין וְיַת כָּל דְתָרֵיךְ יְיָ אֱלָהָנָא מִן קֳדָמָנָא יָתֵיהּ נִירָת:
וּכְעַן הֲטַב טַב אַתְּ מִבָּלָק בַּר צִפּוֹר מַלְכָּא דְמוֹאָב הֲמֵידַן דַן עִם יִשְׂרָאֵל אִם קְרָבָא אַגִיחַ בְּהוֹן:
כַּד יָתֵיב יִשְׂרָאֵל בְּחֶשְׁבּוֹן וּבְכַפְרָנָהָא וּבְעַרְעוֹר וּבְכַפְרָנָהָא וּבְכָל קִרְוַיָא דְעַל תְּחוּמֵי אַרְנוֹן תְּלַת מְאָה שְׁנִין וּמָא דֵין לָא שְׁזַבְתּוּן בְּעִדָנָא הַהִיא:
וַאֲנָא לָא חָטֵית לָךְ וְאַתְּ עָבֵיד עִמִי בִּישָׁא לְאַגָחָא קְרָבָא בִי יְדִין יְיָ דַעֲבַד דִינָא יוֹמָא דֵין בֵּין בְּנֵי יִשְׂרָאֵל וּבֵין בְּנֵי עַמוֹן:
וְלָא קַבֵּיל מַלְכָּא דִבְנֵי עַמוֹן לְפִתְגָמֵי יִפְתָּח דִשְׁלַח לְוָתֵיהּ:
וּשְׁרַת עַל יִפְתָּח רוּחַ גְבוּרָא מִן קֳדָם יְיָ וַעֲבַר יַת גִלְעָד וְיַת מְנַשֶׁה וַעֲבַר יַת מִצְפַּיָא דְגִלְעָד וּמִמִצְפַּיָא דְגִלְעָד עֲבַר בְּנֵי עַמוֹן:
וְקַיֵם יִפְתָּח קְיָם קֳדָם יְיָ וַאֲמַר אִם מִמְסַר תִּמְסַר יַת בְּנֵי עַמוֹן בִּידִי:
וִיהֵי דְיִפּוֹק מֵפַק בַּר מִדָשֵׁי בֵיתִי לָקֳדָמוּתִי כַּד אֵתוּב בִּשְׁלָם מִבְּנֵי עַמוֹן וִיהֵי קֳדָם יְיָ וְאַסְקִינֵיהּ עֲלָתָא:
וַעֲבַר יִפְתָּח לְוַת בְּנֵי עַמוֹן לְאַגָחָא קְרָבָא בְהוֹן וּמְסָרִינוּן יְיָ בִּידֵיהּ:
וּמְחָנוּן מֵעֲרוֹעֵר עַד מַעֲלָנָא דְמִנִית עַשְׂרִין קִרְוִין וְעַד מֵישַׁר כְּרָמַיָא מַחֲתָא רַבְּתָא לַחֲדָא וְאִתְבָּרוּ בְּנֵי עַמוֹן מִן קֳדָם בְּנֵי יִשְׂרָאֵל:
וַאֲתָא יִפְתָּח לְמִצְפַּיָא לְבֵיתֵיהּ וְהָא בְרַתֵּיהּ נַפְקַת לָקֳדָמוּתֵיהּ בְּתֻפִין וּבְחִנְגִין וּלְחוֹד הִיא יְחִידָה לֵית לֵיהּ מִנֵיהּ בַּר אוֹ בְרַת:
וַהֲוָה כַּד חֲזֵי יָתָהּ וּבְזַע יַת לְבוּשׁוֹהִי וַאֲמַר וַוי בְּרַתִּי אַכְרָעָא אַכְרַעְתִּינִי וְאַתְּ הֲוֵית בְּעָכְרִי וַאֲנָא פַּתְחֵית פּוּמִי בִּנְדַר קֳדָם יְיָ וְלָא אֵיכוּל לְמֵיתָב:
וַאֲמָרַת לֵיהּ אַבָּא פְּתַחְתָּא יַת פּוּמָךְ בִּנְדַר קֳדָם יְיָ ָעֲבֵיד לִי כְמָא דִי נְפַק מִפּוּמָךְ בָּתַר דַעֲבַד לָךְ יְיָ פּוּרְעָנוּתָא מִבַּעֲלֵי דְבָבָךְ מִבְּנֵי עַמוֹן:
וַאֲמַרַת לַאֲבוּהָא יִתְעֲבֵיד לִי פִּתְגָמָא הָדֵין שְׁבוֹק מִנִי תְּרֵין יַרְחִין וְאֵהַךְ וְאִתְנְגִיד עַל טוּרַיָא וְאֶבְכֶּה עַל בְּתוּלַי אֲנָא וְחַבְרָתִי:
וַאֲמַר אֱזִילִי וּשְׁלַח יָתָהּ תְּרֵין יַרְחִין וַאֲזַלַת הִיא וְחַבְרָתָהָא וּבְכַת עַל בְּתוּלָהָא עַל טוּרַיָא:
וַהֲוָה מִסּוֹף תְּרֵין יַרְחִין וְתָבַת לְוַת אֲבוּהָא וַעֲבַד לָהּ יַת נִדְרֵיהּ דִי נְדַר וְהִיא לָא יָדְעַת גְבַר וַהֲוַת לִגְזֵירָא בְּיִשְׂרָאֵל: [בְּדִיל דְלָא לְאַסָקָא גְבַר יַת בְּרֵיהּ וְיַת בְּרַתֵּיהּ לַעֲלָתָא כְּמָא דַעֲבַד יִפְתַּח גִלְעָדָאָה וְלָא שָׁאֵיל לְפִינְחָס כַּהֲנָא וְאִילוּ שָׁאֵיל לְפִינְחָס כַּהֲנָא הֲוָה פָרִיק יָתָהּ בְּדָמִין:]
מִזְמַן לִזְמַן אַזְלַן בְּנַת יִשְׂרָאֵל לַאֲלָאָה לְבַת יִפְתָּח גִלְעָדָאָה אַרְבְּעָא יוֹמִין בְּשַׁתָּא:

Chapter 12

וְאִתְכְּנַשׁ אֱנַשׁ אֶפְרַיִם וַעֲבַר לְצִפּוּנָא וַאֲמַרוּ לְיִפְתָּח מָא דֵין עֲבַרְתָּא לְאַגָחָא קְרָבָא בִּבְנֵי עַמוֹן וְלָנָא לָא קְרֵיתָא לְמֵיזַל עִמָךְ בֵּיתָךְ נוֹקֵד עֲלָךְ בְּנוּרָא:
וַאֲמַר יִפְתָּח לְהוֹן אֱנַשׁ דִין הֱוֵיתִי אֲנָא וְעַמִי וּבְנֵי עַמוֹן לַחֲדָא וּבְעֵית אֲנָא מִנְכוֹן וְלָא פְרַקְתּוּן יָתִי מִיֶדְהוֹן:
וַחֲזֵית אֲרֵי לֵיתָךְ פָּרִיק וּמָסְרִית נַפְשִׁי כְּעַל גַב לְאִתְקְטָלָא וְעַבְרֵית לְוַת בְּנֵי עַמוֹן וּמַסְרִינוּן יְיָ בִּידִי וּלְמָא סְלֵיקְתּוּן לְוָתִי יוֹמָא דֵין לְאַגָחָא קְרָבָא בִי:
וּכְנַשׁ יִפְתָּח יַת כָּל אֱנָשֵׁי גִלְעָד וְאַגִיחַ קְרָבָא עִם דְבֵית אֶפְרָיִם וּמְחוֹ אֱנָשֵׁי גִלְעָד יַת דְבֵית אֶפְרַיִם אֲרֵי אֲמַרוּ מְשֵׁיזְבַיָא דְבֵית אֶפְרַיִם מָה אַתּוּן חֲשִׁיבִין דְבֵית גִלְעָד בְּנֵי בֵית אֶפְרַיִם בְּגוֹ בֵית מְנַשֶׁה:
וַאֲחַדוּ דְבֵית גִלְעָד יַת מְגִיזַת יַרְדְנָא עַל דְבֵית אֶפְרַיִם וַהֲוָה כַּד אֲמַר חַד מִמְשֵׁיזְבַיָא דְבֵית אֶפְרַיִם אֶעְבַר וְאָמְרִין לֵיהּ אֱנָשֵׁי גִלְעָד אֶפְרָתִי אַתְּ וַאֲמַר לָא:
וְאָמְרִין לֵיהּ אֱמַר כְּעַן שׁוּבַּלְתָּא וַאֲמַר סוּבַּלְתָא וְלָא מְתַקֵין לְמַלָלָא כֵּן וְאַחֲדִין לֵיהּ וְנַכְסִין לֵיהּ בִמְזִיגַת יַרְדְנָא וְאִתְקְטָלוּ בְעִידָנָא הַהִיא מֵאֶפְרַיִם אַרְבְּעִין וּתְרֵין אַלְפִין:
וְדָן יִפְתָּח יַת יִשְׂרָאֵל שִׁית שְׁנִין וּמִית יִפְתָּח גִלְעָדָאָה וְאִתְקְבַר בְּקִרְוֵי גִלְעָד:
וְדָן בַּתְרוֹהִי יַת יִשְׂרָאֵל אִבְצָן דְמִבֵּית לֶחֶם:
וַהֲוָה לֵיהּ תְּלָתִין בְּנִין וּתְלָתִין בְּנָן שִׁלַח לְבָרָא וּתְלָתִין בְּנָן כַּלָן אָעֵיל לִבְנוֹהִי מִן בְּרָא וְדָן יַת יִשְׂרָאֵל שְׁבַע שְׁנִין:
וּמִית אִבְצָן וְאִתְקְבַר בְּבֵית לֶחֶם:
וְדָן בַּתְרוֹהִי יַת יִשְׂרָאֵל אֵילוֹן דְמִבֵּית זְבוּלֻן וְדָן יַת יִשְׂרָאֵל עֲסַר שְׁנִין:
וּמִית אֵילוֹן דְבֵית זְבוּלֻן וְאִתְקְבַר בְּאַיָלוֹן בְּאֲרַע זְבוּלֻן:
וְדָן בַּתְרוֹהִי יַת יִשְׂרָאֵל עַבְדוֹן בַּר הִלֵל דְמִפִּרְעָתוֹן:
וַהֲוָה לֵיהּ אַרְבְּעִין בְּנִין וּתְלָתִין בְּנֵי בְנִין רְכִיבִין עַל שַׁבְעִין עַיְלִין וְדָן יַת יִשְׂרָאֵל תְּמַנְיָא שְׁנִין:
וּמִית עַבְדוֹן בַּר הִלֵל דְמִפִּרְעָתוֹן וְאִתְקְבַר בְּפִרְעָתוֹן בְּאַרַע שִׁבְטָא דְאֶפְרַיִם בְּטוּרָא דְבֵית עֲמָלֵק:


Chapter 13

וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וּמַסְרִינוּן יְיָ בִידָא דִפְלִשְׁתָּאֵי אַרְבְּעִין שְׁנִין:
וַהֲוָה גַבְרָא חַד מִצָרְעָה מִזַרְעִית דָן וּשְׁמֵיהּ מָנוֹחַ וְאִתְּתֵיהּ עֲקָרָה לֵית לָהּ וְלָד:
וְאִתְחֲזֵי מַלְאָכָא דַייָ לְאִתְּתָא וַאֲמַר לָהּ הָא כְעַן אַתְּ עֲקְרָא וְלֵית לִיךְ וְלָד וְתַעֲדִין וְתֵלְדִין בַּר:
וּכְעַן אִסְתַּמָרִי כְעַן וְלָא תִשְׁתִּין חֲמַר חֲדַת וְעַתִּיק וְלָא תֵיכְלִין כָּל מְסָאָב:
אֲרֵי הָא אַתְּ מְעַדְיָא וְתֵלְדִין בַּר וּמִסְפַּר לָא יֶעְבַר עַל רֵישֵׁיהּ אֲרֵי נְזִירָא דַייָ יְהֵי רַבְיָא מִן בִּטְנָא וְהוּא יִשְׁרֵי לְמִפְרַק יַת יִשְׂרָאֵל מִידָא דִפְלִשְׁתָּאֵי:
וַאֲתַת אִתְּתָא וַאֲמַרַת לְבַעֲלָהּ לְמֵימַר נְבִיָא דַייָ אֲתָא לְוָתִי וְחֶזְוֵיהּ כְּחֵיזוּ מַלְאָכָא דַייָ חָסִין לַחֲדָא וְלָא שְׁאִלְתֵּיהּ אֵי מִדֵין הוּא וְיַת שְׁמֵיהּ לָא חֲוֵי לִי:
וַאֲמַר לִי הָא אַתְּ מְעַדְיָא וּתְלִידִין בַּר וּכְעַן לָא תִשְׁתִּין חֲמַר חֲדַת וְעַתִּיק וְלָא תֵיכְלִין כָּל מְסָאָב אֲרֵי נְזִירָא דַיְיָ יְהֵי רַבְיָא מִן בִּטְנָא עַד יוֹם מוֹתֵיהּ:
וְצַלִי מָנוֹחַ קֳדָם יְיָ וַאֲמַר בְבָעוּ יְיָ נְבִיָא דַייָ דִשְׁלַחְתָּא יֵיתֵי כְעַן עוֹד לְוָתָנָא וְיַלְפִינָנָא מַה נַעְבֵּד לְרַבְיָא דְאִתְיְלִיד:
וּשְׁמִיעַ קֳדָם יְיָ צְלוֹתֵיהּ דְמָנוֹחַ וַאֲתָא מַלְאָכָא דַייָ עוֹד לְוַת אִתְּתָא וְהִיא יַתְבָא בְּחַקְלָא וּמָנוֹחַ בַּעֲלָהּ לֵית עִמָהּ:
וְאוֹחִיאַת אִתְּתָא וּרְהָטַת וְחַוִיאַת לְבַעֲלָהּ וַאֲמַרַת לֵיהּ הָא אִתְחַזֵי לִי גַבְרָא דִי אָתָא בְיוֹמָא דֵיכִי לְוָתִי:
וְקָם וַאֲזַל מָנוֹחַ בָּתַר אִתְּתֵיהּ וַאֲתָא לְוַת גַבְרָא וַאֲמַר לֵיהּ הַאַתְּ גַבְרָא דִי מַלַלְתָּא עִם אִתְּתָא וַאֲמַר אֲנָא:
וַאֲמַר מָנוֹחַ כְּעַן יִתְקַיְמוּן פִּתְגָמָךְ מַה יְהֵא דַחַזֵי לְרַבְיָא וּמַה נַעְבֵּיד לֵיהּ:
וַאֲמַר מַלְאָכָא דַייָ לְמָנוֹחַ מִכֹּל דַאֲמַרִית לְאִתְּתָא תִּסְתַּמָר:
מִכֹּל דְיִפּוֹק מִגוּפְנָא דְחַמְרָא לָא תֵיכוּל וַחֲמַר חֲדַת וְעַתִּיק לָא תִשְׁתֵּי וְכָל מְסָאָב לָא תֵיכוּל כָּל דִי פַקֵידְתָּהּ תִּטָר:
וַאֲמַר מָנוֹחַ לְמַלְאָכָא דַייָ נֵיחוֹד כְּעַן יָתָךְ וְנַעְבֵּיד קֳדָמָךְ גַדְיָא בַּר עִזֵי:
וַאֲמַר מַלְאָכָא דַייָ לְמָנוֹחַ אִם תַּחֲדִינַנִי לָא אֵיכוּל בְּלַחְמָךְ וְאִם תַּעְבֵּיד עֲלָתָא קֳדָם יְיָ תַּסְקִינָהּ אֲרֵי לָא יְדַע מָנוֹחַ אֲרֵי מַלְאָכָא דַייָ הוּא:
וַאֲמַר מָנוֹחַ לְמַלְאָכָא דַייָ מַה שְׁמָךְ אֲרֵי יִתְקַיְמוּן פִּתְגָמָךְ וּנְיַקְרִינָךְ:
וַאֲמַר לֵיהּ מַלְאָכָא דַייָ לְמָא דְנַן אַתְּ שָׁאֵיל לִשְׁמִי וְהוּא מְפָרָשׁ:
וּנְסִיב מָנוֹחַ יַת גַדְיָא בַּר עִזֵי וְיַת תִּקְרוֹבְתָּא וְאַסִיק עַל טִינָרָא קֳדָם יְיָ וּמַפְרַשׁ לְמֶעְבַּד וּמָנוֹחַ וְאִתְּתֵיהּ חָזָן:
וַהֲוָה בְּמֵיסַק שַׁלְהוֹבִיתָא מֵעִלַוֵי מַדְבְּחָא לְצֵית שְׁמַיָא וּסְלֵיק מַלְאָכָא דַייָ בְּשַׁלְהוֹבִית מַדְבְּחָא וּמָנוֹחַ וְאִתְּתֵיהּ חָזָן וּנְפַלוּ עַל אַפֵּיהוֹן עַל אַרְעָא:
וְלָא אוֹסִיף עוֹד מַלְאָכָא דַייָ לְאִתְחַזָאָה לְמָנוֹחַ וּלְאִתְּתֵיהּ בְּכֵן יְדַע מָנוֹחַ אֲרֵי מַלְאָכָא דַייָ הוּא:
וַאֲמַר מָנוֹחַ לְאִתְּתֵיהּ מֵמַת נְמוּת אֲרֵי מַלְאָכָא דַייָ חֲזֵינָא:
וַאֲמַרַת לֵיהּ אִתְּתֵיהּ אִלוּ רַעֲוָא קֳדָם יְיָ דִנְמוּת לָא קַבֵּיל מִנָנָא עֲלָתָא וּמִנְחָתָא וְלָא אַחֲזִינָנָא יַת כָּל אִלֵין וּכְעִידַן פּוֹן לָא אַשְׁמְעִינָנָא כַּהֲדָא:
וִילֵדַת אִתְּתָא בַּר וּקְרַת יַת שְׁמֵיהּ שִׁמְשׁוֹן וּרְבָא רַבְיָא וּבָרְכֵיהּ יְיָ:
וּשְׁרִיאַת רוּחַ גְבוּרָא מִן קֳדָם יְיָ לְתַקָפוּתֵיהּ בְּמַשִׁרְיַת דָן בֵּין צָרְעָה וּבֵין אֶשְׁתָּאֹל:

Chapter 14

וּנְחַת שִׁמְשׁוֹן לְתִמְנָת וַחֲזָא אִתְּתָא בְּתִמְנָת מִבְנַת פְּלִשְׁתָּאֵי:
וּסְלִיק וְחַוִי לַאֲבוּהִי וּלְאִמֵיהּ וַאֲמַר אִתְּתָא חֲזֵיתִי בְּתִמְנָת מִבְּנַת פְּלִשְׁתָּאֵי וּכְעַן סָבוּ יָתָהּ לִי לְאִתּוּ:
וַאֲמַרוּ לֵיהּ אֲבוּהִי וְאִמֵיהּ הָא לֵית בִּבְנַת אַחָךְ וּבְכָל עַמִי אִתְּתָא דְאַתְּ אָזֵיל לְמִיסַב אִתְּתָא מִפְּלִשְׁתָּאֵי עֲרֵלַיָא וַאֲמַר שִׁמְשׁוֹן לַאֲבוּהִי יָתָהּ סַב לִי אֲרֵי הִיא כַּשְׁרַת בְּעֵינָי:
וַאֲבוּהִי וְאִמֵיהּ לָא יָדְעִין אֲרֵי מִן קֳדָם יְיָ הוּא אֲרֵי תוּסְקָפָא הוּא בָעֵי לְאִתְגְרָאָה בִּפְלִשְׁתָּאֵי וּבְעִידָנָא הַהִיא פְּלִשְׁתָּאֵי שַׁלְטִין בְּיִשְׂרָאֵל:
וּנְחַת שִׁמְשׁוֹן וַאֲבוּהִי וְאִמֵיהּ לְתִמְנַת וַאֲתוֹ עַד כַּרְמֵי תִמְנָת וְהָא אַרְיָא בַּר אַרְיְוָן נָהִים לָקֳדָמוּתֵיהּ:
וּשְׁרַת עֲלוֹהִי רוּחַ גְבוּרָא מִן קֳדָם יְיָ וְהַדְמֵיהּ כְּמָא דִמְהַדְמִין גַדְיָא וּמִדַעַם לֵית בִּידֵיהּ וְלָא חַוִי לַאֲבוּהִי וּלְאִמֵיהּ יַת דַעֲבָד:
וּנְחַת וּשְׁאִיל בְּאִתְּתָא וְכַשְׁרַת בְּעֵינֵי שִׁמְשׁוֹן:
וְתַב לִזְמַן יוֹמִין לְמִסְבָּהּ וְסָר לְמֶחֱזֵי יַת פִּגְרָא דְאַרְיָא וְהָא קִינָא דְדַבַּרְיָתָא בְּפִגְרָא דְאַרְיָא וּבֵיהּ דוּבְשָׁא:
וְנַסְחֵיהּ בִּידֵיהּ וַאֲזַל מֵיזַל וְאָכִיל וַאֲזַל לְוָת אָבוּהִי וּלְוַת אִמֵיהּ וִיהַב לְהוֹן וַאֲכַלוּ וְלָא חֲוִי לְהוֹן אֲרֵי מִפִּגְרָא דְאַרְיָא נְסִיחַ דוּבְשָׁא:
וּנְחַת אֲבוּהִי עַל עֵיסַק אִתְּתָא וַעֲבַד תַּמָן שִׁמְשׁוֹן מִשְׁתַּיָא אֲרֵי כֵּן עָבְדִין עוּלֵמַיָא:
וַהֲוָה כַּד חֲזוֹ יָתֵיהּ וְדַבָּרוּ תְּלָתִין חֲבֵרִין וַהֲווֹ עִמֵיהּ:
וַאֲמַר לְהוֹן שִׁמְשׁוֹן אֵיחוּד כְּעַן לְכוֹן חוּדִיתָא אִם חַוָאָה תְחַווּן יָתָהּ לִי שַׁבְעַת יוֹמֵי מִשְׁתַּיָא וְתַשְׁכְּחוּן וְאֶתֵּן לְכוֹן תְּלָתִין פִּלְדָסִין וּתְלָתִין אִצְטָלַוָן דִלְבוּשָׁא:
וְאִם לָא תוּכְלוּן לְחַוָאָה לִי וְתִתְּנוּן אַתּוּן לִי תְּלָתִין פִּלְדָסִין וּתְלָתִין אִצְטָלַוָן דִלְבוּשָׁא וַאֲמַרוּ לֵיהּ חוּד חִידָתָךְ וְנַשְׁמְעִינָהּ:
וַאֲמַר לְהוֹן מֵאָכְלָא נְפַק מֵיכְלָא וּמִתַּקִיפָא נְפַק חוּלְיָא וְלָא יְכִילוּ לְחַוָאָה חוּדִיתָא תְּלָתָא יוֹמִין:
וַהֲוָה בְּיוֹמָא שְׁבִיעָאָה וַאֲמַרוּ לְאִתַּת שִׁמְשׁוֹן שַׁדִילִי יַת בַּעֲלִיךְ וִיחַוֵי לָנָא יַת חוּדִיתָא דִלְמָא נוֹקִיד יָתִיךְ וְיַת בֵּית אָבוּךְ בְּנוּרָא הַלְמַסְכְּנוּתָנָא קְרֵיתוּן יָתָנָא הַלְכָא:
וּבְכַת אִתַּת שִׁמְשׁוֹן עֲלוֹהִי וַאֲמַרַת לְחוֹד שְׂנֵאתַנִי וְלָא רְחַמְתָּנִי חוּדְתָּא חַדְתָּא לִשבְנֵי עַמִי וְלִי לָא חֲוֵיתָא וַאֲמַר לָהּ הָא לְאַבָּא וּלְאִמָא לָא חֲוֵיתִי וְלִיךְ אֵכְדֵין אֲחַוֵי:
וּבְכַת עֲלוֹהִי שַׁבְעָא יוֹמַיָא דַהֲוָה לְהוֹן מִשְׁתַּיָא וַהֲוָה בְּיוֹמָא שְׁבִיעָאָה וְחַוִי לָהּ אֲרֵי דְחַקְתֵּיהּ וְחַוִיאַת חוּדִיתָא לִבְנֵי עַמָהּ:
וַאֲמַרוּ לֵיהּ אֱנָשֵׁי קַרְתָּא בְּיוֹמָא שְׁבִיעָאָה עַד לָא מֵעַל שִׁמְשָׁא מַה חַלֵי מִדוּבְשָׁא וּמַה תַּקִיף מֵאַרְיָא וַאֲמַר לְהוֹן אִילוּלָפוֹן בְּדַקְתּוּן בְּאִתְּתִי לָא אַשְׁכַּחְתּוּן חוּדָתִי:
וּשְׁרַת עֲלוֹהִי רוּחַ גְבוּרָא מִן קֳדָם יְיָ וּנְחַת לְאַשְׁקְלוֹן וּקְטַל מִנְהוֹן תְּלָתִין גַבְרָא וּנְסִיב יַת זְרָזֵהוֹן וִיהַב אִצְטְלָוַיָא לִדְחַוִיאוּ חוּדִיתָא וּתְקֵיף רוּגְזֵיהּ וּסְלִיק לְבֵית אָבוּהִי:
וַהֲוַת אִתַּת שִׁמְשׁוֹן לִדְחַבְרֵיהּ דַהֲוָה שׁוֹשְׁבִינֵיהּ:

Chapter 15

וַהֲוָה לִזְמַן יוֹמִין בְּיוֹמֵי חֲצַד חִטִין וְאַדְכַּר שִׁמְשׁוֹן יַת אִתְּתֵיהּ בְּגַדְיָא בַר עִזֵי וַאֲמַר אֵיעוּל לְוַת אִתְּתִי לְאִדְרוֹן בֵּית מִשְׁכְּבָא וְלָא שַׁבְקֵיהּ אֲבוּהָא לְמֵיעַל:
וַאֲמַר אֲבוּהָא מֵימַר אֲמַרִית אֲרֵי מִסְנָא סְנֵיתָהּ וִיהָבִיתָהּ לְחַבְרָךְ הֲלָא אֲחָתָהּ זְעֶרְתָּא שַׁפִירָא מִנָהּ תְּהֵי כְעַן לָךְ חִלוּפָא:
וַאֲמַר לְהוֹן שִׁמְשׁוֹן זַכַּאי אֲנָא זִמְנָא הָדָא מִפְּלִשְׁתָּאֵי אֲרֵי עָבִיד אֲנָא עִמְהוֹן בִּישָׁא:
וַאֲזַל שִׁמְשׁוֹן וַאֲחַד תְּלַת מֵאָה תַעֲלִין וּנְסִיב בְּעוּרַיָא וּקְטַר דַנְבָא בְּרֵישׁ דַנְבָא וְשַׁוִי בְעוּרָא חָד בֵּין תְּרֵין דִנְבַיָא בִּמְצִיעָן:
וְאַדְלֵיק אֶשְׁתָּא בִּבְעוּרַיָא וּשְׁלַח בְּקָמַת פְּלִשְׁתָּאֵי וְאַדְלֵיק מִגָדִישַׁיָא וְעַד כַּרְמַיָא וְעַד זֵיתָא:
וַאֲמַרוּ פְלִשְׁתָּאֵי מַן עֲבַד דָא וַאֲמַרוּ שִׁמְשׁוֹן חֲתָנֵיהּ דְתִמְנִי אֲרֵי נְסִיב יַת אִתְּתֵיהּ וְיַהֲבָהּ לְחַבְרֵיהּ וּסְלִיקוּ פְלִשְׁתָּאֵי וְאוֹקִידוּ יָתָהּ וְיַת אֲבוּהָא בְּנוּרָא:
וַאֲמַר לְהוֹן שִׁמְשׁוֹן אִם תַּעְבְדוּן כַּהֲדָא אֱלָהֵן כַּד אֶתְפְּרַע מִנְכוֹן וּבָתַר כֵּן אֶתְמְנָע:
וּמְחָא יָתְהוֹן פָּרָשִׁין עִם רִגְלָאִין מְחָא סַגִיאָה וּנְחַת וִיתֵיב בִּשְׁקֵיף כֵּיף עֵיטָם:
וּסְלִיקוּ פְלִשְׁתָּאֵי וּשְׁרוֹ בְּאַרְעָא דְבֵית יְהוּדָה וְאִתְרְטִישׁוּ עַל לֶחִי:
וַאֲמַרוּ אֱנָשֵׁי יְהוּדָה לְמָא סְלֵקְתּוּן עֲלָנָא וַאֲמַרוּ לְמֵיסַר יַת שִׁמְשׁוֹן סְלִיקְנָא לְמֶעְבַּד לֵיהּ כְּמָא דַעֲבַד לָנָא:
וּנְחָתוּ תְּלָתָא אַלְפִין גַבְרָא מִדְבֵית יְהוּדָה לִשְׁקֵיף כֵּיף עֵיטָם וַאֲמַרוּ לְשִׁמְשׁוֹן הֲלָא יְדַעְתָּא אֲרֵי שַׁלְטִין בָּנָא פְּלִשְׁתָּאֵי וּמַה דָא עֲבַדְתְּ לָנָא וַאֲמַר לְהוֹן כְּמָא דַעֲבַדוּ לִי כֵּן עַבְדֵית לְהוֹן:
וַאֲמַרוּ לֵיהּ לְמֵיסְרָךְ נְחַתְנָא לְמִמְסְרָךְ בִּידָא דִפְלִשְׁתָּאֵי וַאֲמַר לְהוֹן שִׁמְשׁוֹן קַיִימוּ לִי דִלְמָא תִשְׁלְטוּן בִּי אַתּוּן:
וַאֲמַרוּ לֵיהּ לְמֵימָר לָא אֲרֵי מֵיסַר נִיסְרִינָךְ וְנִמְסְרִינָךְ בִּידֵיהוֹן וּמִקְטַל לָא נִקְטְלִנָךְ וְאַסְרוּהִי בְּתַרְתֵּין גְדִילַן חַדְתִּין וְאַסְקוּהִי מִן כֵּיפָא:
הוּא אֲתָא עַד לֶחִי וּפְלִשְׁתָּאֵי יַבִּיבוּ לָקֳדָמוּתֵיהּ וּשְׁרַת עֲלוֹהִי רוּחַ גְבוּרָא מִן קֳדָם יְיָ וַהֲוָאָה גְדִילָתָא דְעַל דַרְעוֹהִי כְּכִתָּנָא דְאָרַח בֵּיהּ נוּרָא וְאִתְמְסִיוּ אֱסָרוֹהִי מֵעַל יְדוֹהִי:
וְאַשְׁכַּח לוֹעָא דַחֲמָרָא בְּטִינָתָא וְאוֹשִׁיט יְדֵיהּ וְנַסְבָהּ וּקְטַל בָּהּ אֲלַף גַבְרָא:
וַאֲמַר שִׁמְשׁוֹן בְּלוֹעָא דַחֲמָרָא רְמִיתִנוּן דְגוֹרִין בְּלוֹעָא דַחֲמָרָא קְטָלִית אֲלַף גַבְרָא:
וַהֲוָה כְּשֵׁיצָיוּתֵיהּ לְמַלָלָא וּרְמָא לוֹעָא מִן יְדֵיהּ וּקְרָא לְאַתְרָא הַהוּא רָמַת לְחִי:
וּצְחִי לַחֲדָא וְצַלִי קֳדָם יְיָ וַאֲמַר אַתְּ עֲבַדְתָּ עַל יַד עַבְדָךְ יַת פּוּרְקָנָא רַבָּא הָדָא וּכְעַן אֵימוּת בְּצָחוּתָא וְאִתְמְסַר בִּידָא דַעֲרֵלַיָא:
וּבְזַע יְיָ יַת כֵּיפָא דִי בְּלוֹעָא וּנְפַקוּ מִנֵיהּ מַיָא וּשְׁתִי וְתָבַת רוּחֵיהּ וַחֲיָא עַל כֵּן אִתְקְרֵי שְׁמָהּ עֵינָא דְאִתְיְהִיבַת בִּצְלוֹתָא דְשִׁמְשׁוֹן הִיא קַיְמָא בַּלֶחִי עַד יוֹמָא הָדֵין:
וְדָן יַת יִשְׂרָאֵל בְּיוֹמֵי פְּלִשְׁתָּאֵי עַסְרִין שְׁנִין: 

Chapter 16

וַאֲזַל שִׁמְשׁוֹן לְעַזָתָא וַחֲזָא תַמָן אִתְּתָא פּוּנְדְקִיתָא וְעַל לְוָתָהּ:
לְעַזָתָאֵי אִתְחַוָה לְמֵימָר אֲתָא שִׁמְשׁוֹן הַלְכָא וְאַסְחֲרוּ וּכְמָנוּ לֵיהּ כָּל לֵילְיָא בְּמַעֲלָנָא דִתְרַע קַרְתָּא וּשְׁתִיקוּ כָּל לֵילְיָא לְמֵימָר עַד נֵיהוֹר צַפְרָא וְנִקְטְלִינֵיהּ:
וּשְׁכַב שִׁמְשׁוֹן עַד פַּלְגוּת לֵילְיָא וְקָם בְּפַלְגוּת לֵילְיָא וַאֲחַד בְּדָשֵׁי תְּרַע קַרְתָּא וּבִתְרֵין סִפַּיָא וְנַטְלִינוּן עִם עַבְּרָא וְשַׁוִי עַל כְּתֵפוֹהִי וְאַסְקִינוּן לְרֵישׁ טוּרָא דְעַל אַפֵּי חֶבְרוֹן:
וַהֲוָה בָּתַר כֵּן וּרְחֵים אִתְּתָא בְּנַחֲלָא דְשׂוֹרֵק וּשְׁמָהּ דְלִילָהּ:
וּסְלִיקוּ לְוָתָהּ טוּרְנֵי פְּלִשְׁתָּאֵי וַאֲמַרוּ לָהּ שְׁדִילִי יָתֵיהּ וַחֲזִי בַּמֶה חֵילֵיהּ רַבָּא וּבַמֶה נֵיכוּל לֵיהּ וְנִסְרִינֵיהּ לְעַנָיוּתֵיהּ וַאֲנַחְנָא נִתַּן לָךְ גְבַר אֲלַף וּמְאָה סַלְעִין דִכְסַף:
וַאֲמַרַת דְלִילָה לְשִׁמְשׁוֹן חֲוֵי כְעַן לִי בְּמֶה חֵילָךְ סַגִי וּבְמָה תִתְאַסַר לְעַנְיוּתָךְ:
וַאֲמַר לָהּ שִׁמְשׁוֹן אִם יֵיסְרוּנַנִי בְּשַׁבְעָא יְתָרִין רְטִיבִין דְלָא יַבִּישׁוּ וְאֶחֱלוֹשׁ וְאֶהֱוֵי כְּחַד מִן בְּנֵי אֱנָשָׁא:
וְאַסִיקוּ לָהּ טוּרְנֵי פְּלִשְׁתָּאֵי שַׁבְעָא יְתָרִין רְטִיבִין דְלָא יַבִּישׁוּ וַאֲסַרְתֵּיהּ בְּהוֹן:
וּכְמָנָא יְתַב לָהּ בְּאִדְרוֹן בֵּית מִשְׁכְּבָא וַאֲמַרַת לֵיהּ פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וּפְסַק יַת יְתָרַיָא כְּמָא דְמִתְפְּסַק חוּטָא דְכִתָּנָא דַאֲרַח בֵיהּ נוּרָא וְלָא אִתְיְדַע חֵילֵיהּ:
וַאֲמַרַת דְלִילָה לְשִׁמְשׁוֹן הָא שְׁקַרְתָּ בִּי וּמַלֵילְתָּא עִמִי כַּדְבִין כְּעַן חַוֵי כְעַן לִי בְּמַה תִּתְאַסָר:
וַאֲמַר לָהּ אִם מֵיסַר יֵיסְרוּנַנִי בִּגְדִילָן חַדְתָּן דְלָא אִתְעֲבִידָא בְהוֹן עֲבִידְתָּא וְאֶחֱלוֹשׁ וְאֶהֱוֵי כְּחַד מִבְּנֵי אֱנָשָׁא:
וּנְסִיבַת דְלִילָה גְדִילָן חַדְתָּן וַאֲסַרְתֵּיהּ בְּהוֹן וַאֲמַרַת לֵיהּ פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וּכְמָנָא יָתֵיב בְּאִידְרוֹן בֵּית מִשְׁכְּבָא וּפַסְקִינוּן מֵעַל דַרְעוֹהִי כְּחוּטָא:
וַאֲמַרַת דְלִילָה לְשִׁמְשׁוֹן עַד הָכָא שְׁקַרְתָּ בִּי וּמַלֵילְתָּא עִמִי כַּדְבִין חַוִי כְעַן לִי בַּמֶה תִּתְאַסָר וַאֲמַר לָהּ אִם תִּשְׁתִּין יַת שְׁבַע גְדִילַת רֵישִׁי עִם אַכְסְנָא דְמַשְׁתֵּיתָא::
וְנִקְשַׁת בְסִכְתָא וַאֲמַרַת לֵיהּ פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וְאִתְעַר מִשִׁנְתֵיהּ וּנְטַל יַת אַכְסַן דְגַרְדָאִין וְיַת מַשְׁתֵּיתָא:
וַאֲמַרַת לֵיהּ אֵיכְדֵין תֵּימַר רְחִימְתִּיךְ וְלִבָּךְ לֵית עִמִי דְנַן תְּלַת זִמְנִין שַׁקַרְתָּא בִּי וְלָא חַוֵיתָ לִי בְּמָה חֵילָךְ סַגִי:
וַהֲוָה כַּד אֲעִיקַת לֵיהּ בְּפִתְגָמָהָא כָּל יוֹמַיָא וּדְחַקְתֵּיהּ וְעָקַת נַפְשֵׁיהּ לִמְמָת:
וְחַוֵי לָהּ יַת כָּל לִבֵּיהּ וַאֲמַר לָהּ מִסְפַּר לָא עֲדָא עַל רֵישִׁי אֲרֵי נְזִירָא דַייָ אֲנָא מִבִּטְנָא דְאִמָא אִם אִתְגְלַח וְיַדְעִי מִנִי חֵילִי וְאֶחֱלוֹשׁ וְאֶהֱוֵי כְּחַד מִן בְּנֵי אֱנָשָׁא:
וַחֲזַת דְלִילָה אֲרֵי חַוֵי לָהּ יַת כָּל לִבֵּיהּ וּשְׁלָחַת וּקְרַת לְטוּרְנֵי פְּלִשְׁתָּאֵי לְמֵימָר סַקוּ בְּרַם זִמְנָא הָדָא אֲרֵי חֲוֵי לִי יַת כָּל לִבֵּיהּ וּסְלִיקוּ לְוָתָהּ טוּרְנֵי פְּלִשְׁתָּאֵי וְאַסִיקוּ כַסְפָּא בִּידֵיהוֹן:
וְאַשְׁכַּבְתֵּיהּ עַל בִּרְכָּהָא וּקְרַת לְגַבְרָא וּגְדַעַת יַת שְׁבַע גְדִילַת רֵישֵׁיהּ וּשְׁרִיאַת לְעַנָיוּתֵיהּ וַעֲדָא חֵילֵיהּ מִנֵיהּ:
וַאֲמַרַת פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וְאִתְעַר מִשִׁנְתֵיהּ וַאֲמַר אֶפּוֹק כִּזְמַן בִּזְמַן וְאִתְגַבַּר וְהוּא לָא יְדַע אֲרֵי גְבוּרְתָּא דַייָ עֲדָת מִנֵיהּ:
וַאֲחָדוּהוּ פְּלִשְׁתָּאֵי וְנַקְרוּ יַת עֵינוֹהִי וְאַחִיתוּ יָתֵיהּ לְעַזָת וַאֲסָרוּהוּ בְּשַׁלְשְׁלָן דִנְחָשׁ וַהֲוָה טָחִין רֵחַיָא בְּבֵית אֲסִירַיָא:
וּשְׁרֵי שְׂעַר רֵישֵׁיהּ לְצַמָחָא בָּתַר דְאִתְגַלָח:
וְטוּרְנֵי פְלִשְׁתָּאֵי אִתְכְּנִישׁוּ לְדַבָּחָא דְבַח רַבָּא לְדָגוֹן טַעֲוַתְהוֹן וּלְחֶדְוָא וַאֲמַרוּ מְסָרַת דְחַלְתָּנָא בִּידָנָא יַת שִׁמְשׁוֹן בְּעֵיל דְבָבָנָא:
וַחֲזוֹ יָתֵיהּ עַמָא וְשַׁבָּחוּ יַת טַעֲוַתְהוֹן אֲרֵי אֲמַרוּ מְסָרַת דְחַלְתָּנָא בִּידָנָא יַת בְּעֵיל דְבָבָנָא וְיַת מַחֲרִיב אַרְעָנָא וְדִי מַסְגֵי יַת קְטִילָנָא:
וַהֲוָה כַּד שְׁפַר לִבְּהוֹן וַאֲמַרוּ קְרוּ לְשִׁמְשׁוֹן וִיחַיִךְ לָנָא וּקְרוֹ לְשִׁמְשׁוֹן מִבֵּית אֲסִירַיָא וְחַיִיךְ קֳדָמֵיהוֹן וַאֲקִימוּ יָתֵיהּ בֵּין עַמוּדַיָא:
וַאֲמַר שִׁמְשׁוֹן לְעוּלֵימָא דְמַתְקֵיף בִּידֵיהּ שְׁבוֹק מִנִי וַאֲמִישֵׁנִי יַת עַמוּדַיָא דְבֵיתָא מְתַקֵן עִלַוֵיהוֹן וְאִסְתְּמִיךְ עִלַוֵיהוֹן:
וּבֵיתָא מָלֵי גַבְרַיָא וּנְשַׁיָא וְתַמָן כָּל טוּרְנֵי פְלִישְׁתָּאֵי וְעַל אִגְרָא כִּתְלָתָא אַלְפִין גַבְרָא וְאִתְּתָא דְחָזָן בְּחוֹכָא דְשִׁמְשׁוֹן:
וְצַלִי שִׁמְשׁוֹן קֳדָם יְיָ וַאֲמַר יְיָ אֱלֹהִים יֵיעוּל דָכְרָנִי קֳדָמָךְ וְתַקֵפְנִי כְעַן בְּרַם זִמְנָא הָדָא יְיָ וְאִתְפְּרַע פּוּרְעָנוּתָא חָדָא מִתַּרְתֵּי עֵינַי דְנַקְרוּ לִי פְּלִשְׁתָּאֵי:
וַאֲחַד שִׁמְשׁוֹן יַת תְּרֵין עַמוּדֵי מְצִיעָאֵי דְבֵיתָא מַתְקַן עִלַוֵיהוֹן וְאִסְתְּמִיךְ עֲלֵיהוֹן חַד בִּימִינֵיהּ וְחַד בִּסְמָלֵיהּ:
וַאֲמַר שִׁמְשׁוֹן תְּמוּת נַפְשִׁי עִם פְּלִשְׁתָּאֵי וּנְגַד בְּחֵילָא וּנְפַל בֵּיתָא עַל טוּרְנַיָא וְעַל כָּל עַמָא דִי בֵיהּ וַהֲווֹ קְטִילַיָא דִי קְטַל בְּמוֹתֵיהּ סַגִיאִין מִדִי קְטַל בְּחַיוֹהִי:
וּנְחָתוּ אֲחוֹהִי וְכָל בֵּית אֲבוּהִי וּנְטָלוּ יָתֵיהּ וּסְלִיקוּ וּקְבָרוּ יָתֵיהּ בֵּין צָרְעָה וּבֵין אֶשְׁתָּאֹל בְּקִבְרָא דְמָנוֹחַ אֲבוּהִי וְהוּא דָן יַת יִשְׂרָאֵל עַסְרִין שְׁנִין:


Chapter 17

וַהֲוָה גַבְרָא מִטוּרָא דְבֵית אֶפְרַיִם וּשְׁמֵיהּ מִיכָיְהוּ:
וַאֲמַר לְאִמֵיהּ אֲלַף וּמְאָה סַלְעִין דִכְסַף דְאִתְנְסִיב לִיךְ וְאַתְּ אוֹמִית וְאַף אֲמַרְתְּ קֳדָמַי הָא כַסְפָּא עִמִי אֲנָא נְסִבְתֵּיהּ וַאֲמַרַת אִמֵיהּ בְּרִיךְ בְּרִי קֳדָם יְיָ:
וַאֲתֵיב יַת אֲלַף וּמְאָה סַלְעִין דִכְסַף לְאִמֵיהּ וַאֲמַרַת אִמֵיהּ אַקְדָשָׁא אַקְדֵשִׁית יַת כַּסְפָּא קֳדָם יְיָ מִן יְדִי לִבְרִי לְמֶעְבַּד צְלַם וּמַתְּכָא וּכְעַן אֲתֵבִינֵיהּ לָךְ:
וַאֲתֵיב יַת כַּסְפָּא לְאִמֵיהּ וּנְסִיבַת אִמֵיהּ מָאתָן סַלְעִין דִכְסַף וִיהַבְתֵּיהּ לְקִינָאָה וְעַבְדֵיהּ צְלַם וּמַתְּכָא וַהֲוָה בְּבֵית מִיכָיְהוּ:
וְגַבְרָא מִיכָה לֵיהּ בֵּית טַעֲוָתָא וַעֲבַד אֵפוֹד וּדְמָאִין וּקְרֵיב יַת קוּרְבַּן חָד מִבְּנוֹהִי וַהֲוָא לֵיהּ לְכוּמָרָא:
בְּיוֹמַיָא הָאִינוּן לֵית מַלְכָּא בְּיִשְׂרָאֵל גְבַר דְכָשַׁר בְּעֵינוֹהִי יַעְבֵד:
וַהֲוָה עוּלֵימָא מִבֵּית לֶחֶם יְהוּדָה מִזַרְעִית יְהוּדָה וְהוּא לֵוִי וְהוּא דָר תַמָן:
וַאֲזַל גַבְרָא מִקַרְתֵּיהּ מִבֵּית לֶחֶם יְהוּדָה לְמֵידַר בַּאֲתַר דְיַשְׁכַּח וַאֲתָא לְטוּרָא דְבֵית אֶפְרַיִם עַד בֵּית מִיכָה לְמֶעְבַּד אוֹרְחֵיהּ:
וַאֲמַר לֵיהּ מִיכָה מִנָן אַתְּ אָתֵי וַאֲמַר לֵיהּ לֵיוָאָה אֲנָא מִבֵּית לֶחֶם יְהוּדָה וִאֲנָא אָזֵיל לְמֵידַר בַּאֲתַר דְאַשְׁכַּח:
וַאֲמַר לֵיהּ מִיכָה תִּיב עִמִי וֶהֱוֵי לִי לְאַב וּלְכָהִין וַאֲנָא אֶתֵּן לָךְ עֲסַר סַלְעִין דִכְסַף לִזְמַן עִידַן וְזוּג לְבוּשִׁין וּפַּרְנָסָתָךְ וַאֲזַל לֵיוָאָה:
וּצְבִי לֵיוָאָה לְמִיתַּב עִם גַבְרָא וַהֲוָה עוּלֵימָא לֵיהּ כְּחַד מִבְּנוֹהִי:
וְקָרֵיב מִיכָה יַת קוּרְבַּן לֵיוָאָה וַהֲוָה לֵיהּ עוּלֵימָא לְכָהִין וַהֲוָה בְּבֵית מִיכָה:
וַאֲמַר מִיכָה כְּעַן יָדַעְנָא אֲרֵי אוֹטֵיב יְיָ לִי אֲרֵי הֲוָה לִי לֵוָאָה לְכוּמְרָא:


Chapter 18

בְּיוֹמַיָא הָאִינוּן לֵית מַלְכָּא בְּיִשְׂרָאֵל וּבְיוֹמַיָא הָאִינוּן שִׁבְטָא דְבֵית דָן בָּעָן לְהוֹן אַחֲסַנְתָּא לְמִתַּב אֲרֵי לָא אִתְפַּלְגַת לְהוֹן עַד יוֹמָא הַהוּא מִגוֹ שִׁבְטַיָא דְיִשְׂרָאֵל בְּאַחְסָנָא:
וּשְׁלֵחוּ בְנֵי דָן מִזַרְעֲיַתְהוֹן חַמְשָׁא גוּבְרִין מִקְצַתְהוֹן גַבְרִין בְּנֵי חֵילָא מִצָרְעָה וּמֵאֶשְׁתָּאוֹל לְאַלָלָא יַת אַרְעָא וּלְמִבְדְקָא וַאֲמַרוּ לְהוֹן אֶזִילוּ בְּדוּקוּ יַת אַרְעָא וַאֲתוֹ לְטוּרָא דְבֵית אֶפְרַיִם עַד בֵּית מִיכָה וּבָתוּ תַמָן:
אִינוּן עִם בִּיתָא דְמִיכָה וִאִינוּן אִשְׁתְּמוֹדָעוּ יַת קַל עוּלֵימָא לִיוָאָה וְזַרוּ לְתַמָן וַאֲמַרוּ לֵיהּ מַן אַיְתָךְ הַלִכָא וּמָא אַתְּ עָבֵיד הָכָא וּמַה לָךְ כְּדֵין:
וַאֲמַר לְהוֹן כְּדֵין וּכְדֵין עֲבַד לִי מִיכָה וְאַגִרַנִי וַהֲוֵית לֵיהּ לְכוּמְרָא:
וַאֲמַרוּ לֵיהּ שְׁאֵיל כְּעַל בְמֵימְרָא דַייָ וְנֵידַע הַתַצְלַח אוֹרְחָנָא דִי אֲנַחְנָא אָזְלִין עֲלָהּ:
וַאֲמַר לְהוֹן כַּהֲנָא אֱזִילוּ לִשְׁלָם אַתְקִין יְיָ אוֹרְחַתְכוֹן דִי תְהָכוּן בָּהּ:
וַאֲזַלוּ חַמְשָׁא גוּבְרִין וַאֲתוֹ לְלָיִשׁ וַחֲזוֹ יַת עַמָא דִי בְגַוָהּ יַתְבִין לְרוֹחֲצָן כְּהִלְכַת צִידוֹנָאֵי דְשָׁקְטִין וּשְׁרָן לְרוֹחֲצָן וְלֵית דְמַנְזַק פִּתְגָמָא בְּאַרְעָא יְרוּתִין זְעִירִין וּרְחִיקִין אִינוּן מִצִידוֹנָאֵי וּפִתְגַם לֵית לְהוֹן עִם אֱנָשׁ:
וַאֲתוֹ לְוַת אֲחוּהוֹן צָרְעָה וְאֶשְׁתָּאוֹל וַאֲמַרוּ לְהוֹן אֲחוּהוֹן מָה אַתּוּן:
וַאֲמַרוּ קוּמוּ וְנִיסַק עֲלֵיהוֹן אֲרֵי חֲזֵינָא יַת אַרְעָא וְהָא טָבָא לַחֲדָא וְאַתּוּן שַׁתְקִין לָא תִתְרַשְׁלוּן לְמֵיזַל לְמֵיעַל לְמֵירַת יַת אַרְעָא:
כְּמֵיתֵיכוֹן תַּעֲלוּן לְוַת עַמָא דְשָׁרָן לְרוֹחֲצָן וְאַרְעָא פַּתְיוּת יְדִין אֲרֵי מַסְרָא יְיָ בִידֵיכוֹן אַתְרָא דְלֵית תַּמָן חֻסְרָן כָּל מִדַעַם דִי בְאַרְעָא:
וּנְטָלוּ מִתַּמָן מִזַרְעִית דָן מִצָרְעָה וּמֵאֶשְׁתָּאֹל שִׁית מְאָה גַבְרִין מְזָרְזִין בְמָנֵי קְרָבָא:
וּסְלִיקוּ וּשְׁרוֹ בְּקִרְיַת יְעָרִים בְּאַרְעָא דְבֵית יְהוּדָה עַל כֵּן קְרוֹ לְאַתְרָא הַהוּא מַשִׁרְיַת דָן עַד יוֹמָא הָדֵין הָא הוּא אֲחוֹרֵי קִרְיַת יְעָרִים:
וַעֲבָרוּ מִתַּמָן לְטוּרָא דְבֵית אֶפְרַיִם וַאֲתוֹ עַד בֵּית מִיכָה:
וַאֲתִיבוּ חַמְשָׁא גוּבְרַיָא דַאֲזַלוּ לְאַלָלָא יַת אַרְעָא לַיִשׁ וַאֲמַרוּ לַאֲחוּהוֹן הַיְדַעְתּוּן אֲרֵי אִית בְּבָתַּיָא הָאִלֵין אֵפוֹד וּדְמָאִין וּצְלֵים וּמַתְּכָא וּכְעַן דְעוּ מַה תַעְבְּדוּן:
וְזָרוּ לְתַמָן וְעָלוּ לְבֵית עוּלֵימָא לְוָאָה בֵּית מִיכָה וּשְׁאִילוּ לֵיהּ לִשְׁלָם:
וְשִׁית מְאָה גַבְרִין מְזָרְזִין בְּמָנֵי קְרָבֵיהוֹן קַיְמִין בְּמַעֲלָנָא דִתְרַע דְמִשֵׁבֶט בְּנֵי דָן:
וּסְלִיקוּ חַמְשָׁא גַבְרַיָא דַאֲזָלוּ לְאַלָלָא יַת אַרְעָא עָלוּ לְתַמָן וּנְסִיבוּ יַת צַלְמָנָא וְיַת אֵפוֹדָא וְיַת דְמָאִין וְיַת מַתְּכָא וְכַהֲנָא קָאִים בְּמַעֲלָנָא דְתַרְעָא וְשִׁית מְאָה גַבְרֵי דִמְזָרְזֵי בְּמָנֵי קְרָבָא:
וְאִלֵין עָלוּ לְבֵית מִיכָה וּנְסִיבוּ יַת צְלַם אֵפוֹדָא וְיַת דְמָאִין וְיַת מַתְּכָא וַאֲמַר לְהוֹן כַּהֲנָא מָה אַתּוּן עָבְדִין:
וַאֲמַרוּ לֵיהּ שְׁתוֹק שַׁוֵי יְדָךְ עַל פּוּמָךְ וֶאֱזֵיל עִמָּנָא וֶהֱוֵי לָנָא לְאָב וּלְכַהֲנָא הֲטַב דִתְהֵי כָהִין לְבֵית גַבְרָא חָד אוֹ דִתְהֵי כָהִין לְשִׁבְטָא וּלְזַרְעִית בְּיִשְׂרָאֵל:
וּשְׁפַר לִבָּא דְכַהֲנָא וּנְסִיב יַת אֵפוֹדָא וְיַת דְמָאִין וְיַת צַלְמָא וְעַל בְּגוֹ עַמָא:
וְאִתְפְּנִיאוּ וַאֲזַלוּ וְשַׁוִיאוּ יַת טַפְלַיָא וְיַת בְּעִירָא וְיַת קִנְיָנָא קֳדָמֵיהוֹן:
אִינוּן אַרְחִיקוּ מִבֵּית מִיכָה וְגַבְרַיָא דִבְבָתַּיָא דְעִם בֵּית מִיכָה אִתְכְּנִישׁוּ וְאַדְבִּיקוּ יַת בְּנֵי דָן:
וּקְרוֹ לִבְנֵי דָן וְאַסְחֲרוּ אַפֵּיהוֹן וַאֲמַרוּ לְמִיכָה מַה לָךְ אֲרֵי אִתְכְּנִישְׁתָּא:
וַאֲמַר יַת דַחַלְתִּי דִי עֲבָדִית נְסֵיבְתּוּן וְיַת כַּהֲנָא וַאֲזַלְתּוּן וּמַה לִי עוֹד וּמַה דֵין תֵּימְרוּן לִי מַה לָךְ:
וַאֲמַרוּ לֵיהּ בְּנֵי דָן לָא תַשְׁמַע קָלָךְ עִמָנָא דִלְמָא יִשְׁלְטוּן בְּכוֹן גַבְרִין מְרִירַן נַפְשָׁן וְתָסֵיף נַפְשָׁךְ וּנְפַשׁ אֱנָשׁ בֵיתָךְ:
וַאֲזַלוּ בְּנֵי דָן לְאוֹרְחֵיהוֹן וַחֲזָא מִיכָה אֲרֵי תַקִיפִין אִינוּן מִינֵיהּ וְאִתְפְּנֵי וְתַב לְבֵיתֵיהּ:
וְאִינוּן נְסִיבוּ יַת דַעֲבַד מִיכָה וְדַבֵּרוּ יַת כַּהֲנָא דִי הֲוָה לֵיהּ וַאֲתוֹ לְלַיִשׁ לְעַמָא דְשָׁקְטִין וְשָׁרָן לְרוֹחֲצָן וּמְחוֹ יַתְהוֹן לְפִּתְגָם דְחָרֶב וְיַת קַרְתָּא אוֹקִידוּ בְּנוּרָא:
וְלֵית דִמְשֵׁיזִיב אֲרֵי רְחִיקָא הִיא מִצִידוֹן וּפִתְגַם לֵית לְהוֹן עִם אֱנַשׁ וְאִינוּן בְּמֵשַׁר דִלְבֵית רְחוֹב וּבְנוֹ יַת קַרְתָּא וִיתִיבוּ בָהּ:
וּקְרוֹ שְׁמָא דְקַרְתָּא דָן בְּשׁוּם דָן אֲבוּהוֹן דְאִתְיְלִיד לְיִשְׂרָאֵל וּבְרַם לַיִשׁ שְׁמָא דְקַרְתָּא מִלְקַדְמִין:
וַאֲקִימוּ לְהוֹן בְּנֵי דָן יַת צַלְמָא וִיהוֹנָתָן בַּר גֵרְשֹׁם בַּר מְנַשֶׁה הוּא וּבְנוֹהִי הֲווֹ כוּמָרִין לְשִׁבְטָא דְבֵית דָן עַד יוֹמָא דִי גְלוֹ יַתְבֵי אַרְעָא:
וְשַׁוִיאוּ לְהוֹן יַת צְלָם מִיכָה דִי עֲבַד כָּל יוֹמַיָא דַהֲוָה קַיָם בֵּית מַקְדְשָׁא דַייָ בְּשִׁלֹה:

Chapter 19

וַהֲוָה בְיוֹמַיָא הָאִינוּן וּמַלְכָּא לֵית בְּיִשְׂרָאֵל וַהֲוָה גְבַר לְוָאָה דָר בִּסְיָפֵי טוּרָא דְבֵית אֶפְרַיִם וּנְסֵיב לֵיהּ אִתְּתָא לְחֵינָתָא מִבֵּית לֶחֶם דְבֵית יְהוּדָה:
וּבְסָרַת עֲלוֹהִי לְחֵינָתֵיהּ וַאֲזָלַת מִלְוָתֵיהּ לְבֵית אֲבוּהָא לְבֵית לֶחֶם דְבֵית יְהוּדָה וַהֲוָת תַּמָן יוֹמִין אַרְבְּעָא יַרְחִין:
וְּקָם בַּעֲלָהּ וַאֲזַל בַּתְרָהָא לְמַלֵיל עַל לִבָּהּ לַאֲתָבוּתָהּ וְעוּלֵימֵיהּ עִמֵיהּ וְזוּג חֲמָרִין וְאַעֲלָתֵיהּ לְבֵית אֲבוּהָא וַחֲזָהוּ אֲבוּהָא דְעוּלֶמְתָּא וַחֲדִי לָקֳדָמוּתֵיהּ:
וְאַתְקֵיף בֵּיהּ חֲמוּהִי אֲבוּהָא דְעוּלֶמְתָּא וִיתֵיב עִמֵיהּ תְּלָתָא יוֹמִין וַאֲכָלוּ וּשְׁתִיאוּ וּבָתוּ תַמָן:
וַהֲוָה בְּיוֹמָא רְבִיעָאָה וְאַקְדִימוּ בְּצַפְרָא וְקָם לְמֵיזַל וַאֲמַר אֲבוּהָא דְעוּלֶמְתָּא לַחֲתָנֵיהּ סְעֵיד לִבָּךְ פִּתָא דְלַחְמָא וּבָתַר כֵּן תֵּיזְלוּן:
וְאַסְחֲרוּ וַאֲכַלוּ תַרְוֵיהוֹן כַּחֲדָא וּשְׁתִיאוּ וַאֲמַר אֲבוּהָא דְעוּלֶמְתָּא לְגַבְרָא שְׁרֵי כְעַן וּבִית וְיִשְׁפַּר לִבָךְ:
וְקָם גַבְרָא לְמֵיזַל וְאַתְקֵיף בֵּיהּ חֲמוּהִי וְתַב וּבָת תַּמָן:
וְאַקְדֵים בְּצַפְרָא בְּיוֹמָא חֲמִישָׁאָה לְמֵיזַל וַאֲמַר אֲבוּהָא דְעוּלֶמְתָּא סְעוֹד כְּעַן לִבָּךְ וְאִתְעַכְּבוּ עַד מִתְפְּנֵי יוֹמָא וַאֲכָלוּ תַּרְוֵיהוֹן:
וְקָם גַבְרָא לְמֵיזַל הוּא וּלְחֵינָתֵיהּ וְעוּלֵימֵיהּ וַאֲמַר לֵיהּ חֲמוֹהִי אֲבוּהָא דְעוּלֶמְתָּא הָא כְעַן פְּנָא יוֹמָא לְמֶעֱרַב בִּיתוּ כְעַן הָכָא לְחוֹד יוֹמָא דֵין בִּית הָכָא וְיִשְׁפַּר לִבָּךְ וּתְקַדְמוּן מְחָר לְאוֹרְחַתְכוֹן וּתְהַךְ לְקִירְוָךְ:
וְלָא אֲבָא גַבְרָא לִמְבָת וְקָם וַאֲזַל וַאֲתָא עַד לָקֳבֵיל יְבוּס הִיא יְרוּשְׁלֵם וְעִמֵיהּ זוּג חֲמָרִין חֲשִׁיקִין וּלְחֵינָתֵיהּ עִמֵיהּ:
אִינוּן עִם יְבוּס וְיוֹמָא מְאַךְ לַחֲדָא וַאֲמַר עוּלֵימָא לְרִבּוֹנֵיהּ אִיתָא כְעַן וּנְזוּר לְקַרְתָּא דִיבוּסָאֵי הָדָא וּנְבֵית בָּהּ:
וַאֲמַר לֵיהּ רִבּוֹנֵיהּ לָא נְזוּר לְקַרְתָּא בְּנֵי עַמְמַיָא דְלָא מִבְּנֵי יִשְׂרָאֵל אִינוּן וְנִתְמְטֵי עַד גִבְעָתָא:
וַאֲמַר לְעוּלֵימֵיהּ אִיתָא וּנְקָרֵיב בְּחַד מִן אַתְרַיָא וּנְבִית בְּגִבְעָתָא אוֹ בְרָמָתָא:
וַעֲבָרוּ וַאֲזַלוּ וְעָלַת לְהוֹן שִׁמְשָׁא בִּסְטַר גִבְעֲתָא דִי לְשֵׁבֶט בִנְיָמִין:
וְזָרוּ לְתַמָן לְמֵיעַל לְמֵבַת בְּגִבְעָתָא וַאֲתָא וִיתֵיב בִּרְחוֹבָא דְקַרְתָּא וְלֵית אֱנָשׁ דִכְנֵישׁ יַתְהוֹן לְבֵיתָא לִמְבָת:
וְהָא גַבְרָא סָבָא אָתָא מִן עִיבִידְתֵּיהּ מִן חַקְלָא בְּרַמְשָׁא וְגַבְרָא מִטוּרָא דְבֵית אֶפְרַיִם וְהוּא דָר בְּגִבְעָתָא וֶאֱנָשֵׁי אַתְרָא מִן שִׁבְטָא דְבֵית בִּנְיָמִין:
וּזְקֵיף עֵינוֹהִי וַחֲזָא יַת גַבְרָא דַאֲרַח בִּרְחוֹב קַרְתָּא וַאֲמַר גַבְרָא סָבָא לְאָן אַתְּ אָזִיל וּמְנָן אַתְּ אָתֵי:
וַאֲמַר לֵיהּ עָבְרִין אֲנַחֲנָא מִבֵּית לֶחֶם דְבֵית יְהוּדָה עַד סְיָפֵי טוּרָא דְבֵית אֶפְרַיִם מִתַּמָן אֲנָא וַאֲנָא אָזֵיל עַד בֵּית לֶחֶם דְבֵית יְהוּדָה וּלְבֵית מַקְדְשָׁא דַייָ אֲנָא אָזִיל וְלֵית אֱנַשׁ דְכָנֵישׁ יָתִי לְבֵיתָא:
וְאַף תִּבְנָא וְאַף כִּסְתָא אִית לַחֲמָרָנָא וְאַף לְחַם וַחֲמָר אִית לִי וּלְאַמְתָךְ וּלְעוּלֵימָא דְעִם עַבְדָךְ לֵית חוּסְרַן כָּל מִדָעַם:
וַאֲמַר גַבְרָא סָבָא שְׁלָם לָךְ לְחוֹד כָּל חוֹסְרָנָךְ עָלַי לְחוֹד בִּרְחוֹבָא לָא תְבִית:
וְאַעֲלֵיהּ לְבֵיתֵיהּ וּרְמָא כִּסְתָא לַחֲמָרֵיהּ וְאַסְחוּ רַגְלֵיהוֹן וְאָכָלוּ וְשָׁתִיאוּ:
אִינוּן מְטִיבִין יַת לִבְּהוֹן וְהָא אֱנָשֵׁי קַרְתָּא גַבְרִין בְּנֵי רִשְׁעָא אַקִיפוּ יַת בֵּיתָא דַחֲקִין לְמִתְבַּר דָשָׁא וַאֲמַרוּ לְגַבְרָא מָרֵיהּ דְבֵיתָא סָבָא לְמֵימַר אַפֵּיק יַת גַבְרָא דְעַל לְבֵיתָךְ וְנֵדְעִינֵיהּ:
וּנְפַק לְוַתְהוֹן גַבְרָא מָרֵיהּ דְבֵיתָא וַאֲמַר לְהוֹן בְּבָעוּ אֲחַי לָא תַבְאֲשׁוּן כְּעַן בָּתַר דְעַל גַבְרָא הָדֵין לְבֵיתִי לָא תַעְבְּדוּן יַת קְלָנָא הָדֵין:
הָא בְרַתִּי בְתוּלְתָּא וּלְחֵינָתֵיהּ אַפֵּיק כְּעַן יַתְהֶן וְעַנוּ יַתְהֶן וַעֲבִידוּ לְהֵן דְתַקִין בְּעֵינֵיכוֹן וּלְגַבְרָא הָדֵין לָא תַעְבְדוּן עִיסַק קְלָנָא הָדֵין:
וְלָא אָבוּ גַבְרַיָא לְקַבָּלָא מִנֵיהּ וְאַתְקֵיף גַבְרָא בִּלְחֵינָתֵיהּ וְאַפֵּיק לְוַתְהוֹן לְבָּרָא וִידָעוּ יָתָהּ וְאִתְלְעָבוּ בָהּ כָּל לֵילְיָא עַד צַפְרָא וְשַׁלְחוּהָא כְּמֵיסַק צַפְרָא:
וַאֲתַת אִתְּתָא לְמִפְנֵי צַפְרָא וּנְפַלַת בִּתְרַע בֵּית גַבְרָא דִי רִבּוֹנָהּ תַּמָן עַד דִי נְהוֹר:
וְקָם רִבּוֹנָהָא בְּצַפְרָא וּפְתַח דָשֵׁי בֵיתָא וּנְפַק לְמֵיזַל לְאָרְחֵיהּ וְהָא אִתְּתָא לְחֵינָתֵיהּ רַמְיָא בִּתְרַע בֵיתָא וִידָהָא מַחֲתִין עַל סִיפָא:
וַאֲמַר לָהּ קוּמִי וְנֵיזֵיל וְלָא מְתִיבָא וְנַסְבָהּ עַל חֲמָרָא וְקָם גַבְרָא וְאַזַל לְאַתְרֵיהּ:
וַאֲתָא לְבֵיתֵיהּ וּנְסִיב יַת סַכִּינָא וְאַתְקֵיף בִּלְחֵינָתֵיהּ וְהַדְמָה לְאֵיבְרָהָא לִתְרֵי עֲשַׂר אֵיבָרִין וְשַׁלְחָהּ בְּכָל תְּחוּם אֲרַע דְיִשְׂרָאֵל:
וַהֲוָה כָּל דְחָזֵי וַאֲמַר לָא הֲוַת וְלָא אִתְחֲזִיאַת כַּהֲדָא לְמִן יוֹמָא דִי סְלִיקוּ בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרָיִם עַד יוֹמָא הָדֵין שַׁווּ לִבְּכוֹן עֲלָהּ אִתְיַעְצוּ וּמַלִילוּ:

Chapter 20

וּנְפַקוּ כָּל בְּנֵי יִשְׂרָאֵל וְאִתְכְּנִישַׁת כְּנִשְׁתָּא כְגַבְרָא חָד לְמִדָן וְעַד בְּאֵר שֶׁבַע וְאַרְעָא גִלְעָד לָקֳדָם יְיָ לְמִצְפַּיָא:
וְאִתְעַתָּדוּ רֵישֵׁי כָל עַמָא כָּל שִׁבְטַיָא דְיִשְׂרָאֵל בִּקְהַל עַמָא דַייָ אַרְבַּע מְאָה אַלְפִין גְבַר רִגְלֵי שַׁלְפֵי סַיְפָא:
וּשְׁמָעוּ בְּנֵי בִנְיָמִין אֲרֵי סַלִיקוּ בְנֵי יִשְׂרָאֵל לְמִצְפַּיָא וַאֲמַרוּ בְּנֵי יִשְׂרָאֵל מַלִילוּ אֵיכְדֵין הֲוַת בִּישָׁתָא הָדָא:
וַאֲתֵיב גַבְרָא לֵוָאָה בַּעֲלָא דְאִתְּתָא קְטִלְתָּא וַאֲמַר לְגִבְעָתָא דִי לְשִׁבְטָא בִנְיָמִין עָלִית אֲנָא וּלְחֵינָתִי לִמְבָת:
וְקָמוּ עֲלַי יַתְבֵי גִבְעָתָא וְאַקִיפוּ עֲלַי יַת בֵּיתָא בְּלֵילְיָא עָלַי חֲשִׁיבוּ לְמִקְטַל וְיַת לְחֵינָתִי עַנְיוּהָא וּמִיתַת:
וַאֲחָדִית בִּלְחֵינָתִי וְהַדְמִיתָהּ וּשְׁלַחְתָּהּ בְּכָל תְּחוּם אַחְסָנַת אַרְעָא דְיִשְׂרָאֵל אֲרֵי עֲבָדוּ עֵיצַת חֶטְאִין וּדְלָא כָשַׁר בְּיִשְׂרָאֵל:
הָא כוּלְכוֹן בְּנֵי יִשְׂרָאֵל הָבוּ לְכוֹן פִּתְגָם וְעֵצָה הַלְכָא:
וְקָם כָּל עַמָא כְּגַבְרָא חָד לְמֵימָר לָא נֵיזִיל גְבַר לְמַשְׁכְּנֵיהּ וְלָא נְזוּר גְבַר לְבֵיתֵיהּ:
וּכְעַן דֵין פִּתְגָם דְנַעֲבֵיד לְגִבְעָתָא נִתְמַנֵי עֲלָהּ בְּעַדְבָא:
וּנְדַבֵּר עַשְׂרָא גַבְרִין לִמְאָה לְכָל שִׁבְטַיָא דְיִשְׂרָאֵל וּמְאָה לְאַלְפָא וְאַלְפָא לְרִבּוֹתָא לְמִיסַב זְוָדִין לְעַמָא לְמֶעֱבַּד לְאַיְתָיוּתְהוֹן לְגִבְעָתָא דְבֵית בִּנְיָמִין כְּכָל קְלָנָא דְאִתְעֲבִיד בְּיִשְׂרָאֵל:
וְאִתְכְּנִישׁ כָּל אֱנַשׁ יִשְׂרָאֵל לְקַרְתָּא כְּגַבְרָא חַד שָׁוָן:
וּשְׁלָחוּ שִׁבְטַיָא דְיִשְׂרָאֵל גַבְרִין בְּכָל שִׁבְטָא דְבִנְיָמִין לְמֵימָר מַה בִּישְׁתָא הָדָא דַהֲוָת בְּכוֹן:
וּכְעַן אַיְתוּ יַת גַּבְרַיָא בְּנֵי רִשְׁעָא דִי בְגִבְעָתָא וְנִקְטְלִינוּן וּנְפַלֵי עָבְדֵי בִּישְׁתָא מִיִשְׂרָאֵל וְלָא אָבוּ בְנֵי בִּנְיָמִין לְקַבָּלָא לִמֵימַר אֲחוּהוֹן בְּנֵי יִשְׂרָאֵל:
וְאִתְכְּנִישׁוּ בְנֵי בִנְיָמִין מִן קִרְוַיָא לְגִבְעָתָא לְמִפַּק לְאַגָחָא קְרָבָא עִם בְּנֵי יִשְׂרָאֵל:
וְאִתְמְנִיאוּ בְנֵי בִנְיָמִין בְּיוֹמָא הַהוּא מִן קִרְוַיָא עַסְרִין וְשִׁיתָּא אַלְפִין גַבְרָא שַׁלְפֵי סַיְפָא בַּר מִיַתְבֵי גִבְעֲתָא אִתְמְנִיאוּ שְׁבַע מְאָה גְבַּר עוּלֵים:
מִכָּל עַמָא הָדֵין שְׁבַע מְאָה גְבַר עוּלֵים גַבְרָא גְמִיד בִּידֵיהּ דְיַמִינָא כָּל אִלֵין דְשָׁדָן אַבְנָא בְקִלְעָא וּמַרְמָן בְּבִינַת שַׂעֲרָא וְלָא מְשַׁנָן:
וֶאֱנַשׁ יִשְׂרָאֵל אִתְמַנִיאוּ בַּר מִדְבֵית בִּנְיָמִין אַרְבַּע מְאָה אַלְפִין גַבְרָא שַׁלְפֵי סַיְפָא כָּל אִלֵין גַבְרֵי עָבְדֵי קְרָבָא:
וְקָמוּ וּסְלִיקוּ לְבֵית אֵל וּשְׁאִילוּ בְמֵימְרָא דַייָ וַאֲמַרוּ בְּנֵי יִשְׂרָאֵל מַן יִסַק לָנָא בְּקַדְמֵיתָא לְאַגָחָא קְרָבָא עִם בְּנֵי בִּנְיָמִין וַאֲמַר יְיָ יְהוּדָה בְּקַדְמֵיתָא:
וְקָמוּ בְּנֵי יִשְׂרָאֵל בְּצַפְרָא וּשְׁרוֹ עַל גִבְעָתָא:
וּנְפַק אֱנַשׁ יִשְׂרָאֵל לְאַגָחָא קְרָבָא עִם דְבֵית בִּנְיָמִין וְסַדַרוּ עִמְהוֹן אֱנַשׁ יִשְׂרָאֵל קְרָבָא עַל גִבְעָתָא:
וּנְפַקוּ בְנֵי בִנְיָמִין מִן גִבְעָתָא וְחַבִּילוּ בְּיִשְׂרָאֵל בְּיוֹמָא הַהוּא עַסְרִין וּתְרֵין אַלְפִין גַבְרָא קְטִילִין רְמָן לְאַרְעָא:
וְאִתְקֵף עַמָא אֱנַשׁ יִשְׂרָאֵל וְאוֹסִיפוּ לְסַדָרָא קְרָבָא בְּאַתְרָא דְסַדַרוּ תַּמָן בְּיוֹמָא קַדְמָאָה:
וּסְלִיקוּ בְנֵי יִשְׂרָאֵל וּבְכוֹ קֳדָם יְיָ עַד רַמְשָׁא וּשְׁאִילוּ בְמֵימְרָא דַייָ לְמֵימַר הַאוֹסִיף לְמִקְרַב לְאַגָחָא קְרָבָא עִם בְּנֵי בִנְיָמִין אֲחִי וַאֲמַר יְיָ סַקוּ עֲלֵיהוֹן:
וּקְרִיבוּ בְּנֵי יִשְׂרָאֵל לְוַת בְּנֵי בִנְיָמִין בְּיוֹמָא תִּנְיָנָא:
וּנְפַק בִּנְיָמִין לִקֳדָמוּתְהוֹן מִן גִבְעֲתָא בְּיוֹמָא תִּנְיָנָא וְחַבִּילוּ בִּבְנֵי יִשְׂרָאֵל עוֹד תַּמְנַת עֲסַר אַלְפִין גַבְרָא קְטִילִין רְמָן לְאַרְעָא כָּל אִלֵין שַׁלְפֵי סַיְפָא:
וּסְלִיקוּ כָּל בְּנֵי יִשְׂרָאֵל וְכָל עַמָא וַאֲתוֹ לְבֵית אֵל וּבְכוֹ וִיתִיבוּ תַמָן קֳדָם יְיָ וְצָמוּ בְּיוֹמָא הַהוּא עַד רַמְשָׁא וַאֲסִיקוּ עֲלַוָן וְנִכְסַת קוּדְשִׁין קֳדָם יְיָ:
וּשְׁאִילוּ בְנֵי יִשְׂרָאֵל בְּמֵימְרָא דַייָ וְתַמָן אֲרוֹן קְיָמָא דַייָ בְּיוֹמַיָא הָאִנוּן:
וּפִינְחָס בַּר אֶלְעָזָר בַּר אַהֲרֹן קָאִים קֳדָמוֹהִי בְּיוֹמַיָא הָאִנוּן לְמֵימָר הַאוֹסֵף עוֹד לְמִפַּק לְאַגָחָא קְרָבָא עִם בְּנֵי בִּנְיָמִין אָחִי אִם אִתְמְנַע וַאֲמַר יְיָ סַקוּ אֲרֵי מְחָר אֶמְסָרִינוּן בִּידָךְ:
וְשַׁוִי יִשְׂרָאֵל כְּמָנִין עַל גִבְעֲתָא סְחוֹר סְחוֹר:
וּסְלִיקוּ בְנֵי יִשְׂרָאֵל לְוַת בְּנֵי בִנְיָמִין בְּיוֹמָא תְּלִיתָאָה וְסַדָרוּ קְרָבָא עַל גִבְעֲתָא כִּזְמַן בִּזְמָן:
וּנְפָקוּ בְּנֵי בִּנְיָמִין לִקְדָמוּת עַמָא אִתְנְגִידוּ מִן קַרְתָּא וּשְׁרִיאוּ לְקַטָלָא מֵעַמָא קְטִילִין כִּזְמַן בִּזְמָן בִּכְבֵשׁ אָרְחָתָא דַחֲדָא סַלְקָא לְבֵית אֵל וַחֲדָא לְגִבְעֲתָא בְּחַקְלָא כִּתְלָתִין גַבְרָא בְּיִשְׂרָאֵל:
וַאֲמַרוּ בְּנֵי בִּנְיָמִין תְּבִירִין אִנוּן קֳדָמָנָא כִּדְבְּקַדְמֵיתָא וּבְנֵי יִשְׂרָאֵל אֲמַרוּ נְפַךְ וְנִגוֹדִנּוּן מִן קַרְתָּא לְכִבְשַׁיָא:
וְכָל אֱנַשׁ יִשְׂרָאֵל קָמוּ מֵאַתְרֵיהוֹן וְסַדָרוּ קְרָבָא בְּמֵישְׁרֵי יְרִיחוֹ וּכְמָנָא יִשְׂרָאֵל מַגִיחַ מֵאַתְרֵיהּ מִמֵישַׁר גִבְעָתָא:
וַאֲתוֹ מִדָרוֹם לְגִבְעָתָא עַסְרָא אַלְפִין גְבַר עוּלֵים מִכָּל יִשְׂרָאֵל וְעָבְדֵי קְרָבָא תַקִיפוּ וְאִנוּן לָא יָדְעִין אֲרֵי קְרִיבַת דְתֵיתֵי לְהוֹן בִּישְׁתָּא:
וְתַבֵּר יְיָ יַת בִּנְיָמִין קֳדָם יִשְׂרָאֵל וְחַבִּילוּ בְּנֵי יִשְׂרָאֵל בִּדְבֵית בִּנְיָמִין בְּיוֹמָא הַהוּא עַסְרִין וְחַמְשָׁא אַלְפִין וּמֵאָה גַבְרָא כָּל אִלֵין שַׁלְפֵי סַיְפָא:
וַחֲזוֹ בְנֵי בִנְיָמִין אֲרֵי אִתְבָּרוּ וִיהֲבוּ אֱנַשׁ יִשְׂרָאֵל אֲתַר לְבִנְיָמִין אֲרֵי אִתְרְחִיצוּ עַל כְּמָנָא דְשַׁוִיאוּ עַל גִבְעֲתָא:
וּכְמָנָא אוֹחִי וְאִתְנְגִיד עַל גִבְעֲתָא וְאִתְנְגִיד כְּמָנָא וּמְחָא יַת כָּל קַרְתָּא לְפִּתְּגָם דְחָרֶב:
וְזִמְנָא הֲוָה מְתַקֵן לֶאֱנַשׁ יִשְׂרָאֵל עִם כְּמָנָא דְיִסְגוּן לְאַסָקוּתְהוֹן יְטוּר דִתְנָן מִן קַרְתָּא:
וְאִתְחֲזַר אֱנַשׁ יִשְׂרָאֵל בִּקְרָבָא וּדְבֵית בִּנְיָמִין שְׁרִיאוּ לְקַטָלָא קְטִילַיָא בֶאֱנַשׁ יִשְׂרָאֵל כִּתְלָתִין גַבְרָא אֲרֵי אֲמַרוּ בְּרַם מִתְּבַר תְּבִירִין אִנוּן קֳדָמָנָא כִּקְרָבָא קַדְמָאָה:
וּתְנָנָא שְׁרֵי לְאַסָקָא מִן קַרְתָּא יְטוּר דִתְנָן וְאִתְפְּנִיאוּ דְּבֵית בִּנְיָמִין בַּתְרֵיהוֹן וְהָא סְלִיק תְּנָנָא דְקַרְתָּא לְצֵית שְׁמַיָא:
וֶאֱנַשׁ יִשְׂרָאֵל אִתְחַזֵר וְאִתְבְּהֵיל אֱנַשׁ בִּנְיָמִין אֲרֵי חֲזָא אֲרֵי קְרִיבַת דִתֵיתֵי לְהוֹן בִּישְׁתָּא:
וְאִתְפְּנִיאוּ קֳדָם אֱנָשׁ יִשְׂרָאֵל לְאוֹרַח מַדְבְּרָא וְעָבְדֵי קְרָבָא אַדְבִּיקוּנוּן וֶאֱנָשׁ כְּמָנָא קָמוּ מִקִרְוֵיהוֹן קָטְלִין בְּהוֹן מִכָּא וּמִכָּא:
טַלְטִילוּ יַת דְבֵית בִּנְיָמִין רְדָפוּנוּן מִבֵּית נְיַחֲהוֹן טְרָדוּנוּן עַד לָקֳבֵיל גִבְעֲתָא מִמַדְנַח שִׁמְשָׁא:
וְאִתְקְטָלוּ מִדְבֵית בִּנְיָמִין תַּמְנַת עֲסַר אַלְפִין גַבְרָא יַת כָּל אִלֵין גַבְרָא עַבְדֵי קְרָבָא:
וְאִתְפְּנִיאוּ וְאַפִּיכוּ לְמַדְבְּרָא לְכֵיף רִמוֹן וְאִטַמָרוּ בְּכִיבְשַׁיָא חַמְשָׁא אַלְפִין גַבְרָא וְאַדְבִּיקוּ בַתְרֵיהוֹן עַד גִדְעֹם וּקְטִילוּ מִנְהוֹן תְּרֵין אַלְפִין גַבְרָא:
וַהֲווֹ כָּל דְאִתְקְטִילוּ מִדְבֵית בִּנְיָמִין עַסְרִין וְחַמְשָׁא אַלְפִין גַבְרָא שַׁלְפֵי סַיְפָא בְּיוֹמָא הַהוּא יַת כָּל אִלֵין גַבְרֵי עָבְדֵי קְרָבָא:
וְאִתְפְּנִיאוּ וְאַפִּיכוּ לְמַדְבְּרָא לְכֵיף רִמוֹן שִׁית מְאָה גַבְרָא וִיתִיבוּ בְּכֵיף רִמוֹן אַרְבְּעָא יַרְחִין:
וֶאֱנַשׁ יִשְׂרָאֵל תָּבוּ לְוַת בְּנֵי בִנְיָמִין וּמְחוֹנוּן לְפִּתְגַם דְחָרֶב מִקִרְוֵיהוֹן גַמְרִינוּן עַד בְּעִירָא עַד כָּל דְאִשְׁתְּכַח אַף כָּל קִרְוַיָא דְאִשְׁתַּכָּחָא אוֹקִידוּ בְּנוּרָא:

Chapter 21

וֶאֱנַשׁ יִשְׂרָאֵל קַיֵם בְּמִצְפַּיָא לְמֵימָר אֱנַשׁ מִנָנָא לָא יִתֵּן בְּרַתֵּיהּ לִדְבֵית בִּנְיָמִין לְאִתּוּ:
וַאֲתָא עַמָא לְבֵית אֵל וִיתִיבוּ תַמָן עַד רַמְשָׁא קֳדָם יְיָ וַאֲרִימוּ קָלְהוֹן וּבְכוֹ בְּכִי סַגְיָא:
וַאֲמֵרוּ לְמָא יְיָ אֱלָהָא דְיִשְׂרָאֵל הֲוַת דָא בְּיִשְׂרָאֵל לְמִשְׁגֵי יוֹמָא דֵין מִיִשְׂרָאֵל שִׁבְטָא חָד:
וַהֲוָה מִיוֹמָא דְבַתְרוֹהִי וְאַקְדִימוּ עַמָא וּבְנוֹ תַמָן מַדְבְּחָא וְאַסִיקוּ עֲלַוָן וְנִכְסַת קוּדְשִׁין:
וַאֲמַרוּ בְּנֵי יִשְׂרָאֵל מַן דְלָא סְלִיק בִּקְהָלָא מִכָּל שִׁבְטַיָא דְיִשְׂרָאֵל לָקֳדָם יְיָ אֲרֵי שְׁבוּעֲתָא רַבְּתָא הֲוַת לִדְלָא סְלִיק לִקְדָם יְיָ לְמִצְפַּיָא לְמֵימַר אִתְקְטָלָא יִתְקְטֵיל:
וְאִתְנְחָמוּ בְּנֵי יִשְׂרָאֵל עַל דְבֵית בִּנְיָמִין אֲחוּהוֹן וַאֲמַרוּ אִתְמְנַע יוֹמָא דֵין שִׁבְטָא חַד מִיִשְׂרָאֵל:
מַה נַעְבֵּיד לְהוֹן לִדְאִשְׁתַּאֲרוּ לִנְשִׁין וַאֲנַחְנָא קַיֵמְנָא בְּמֵימְרָא דַייָ בְּדִיל דְלָא לְמִתַּן לְהוֹן מִבְּנָתָנָא לִנְשִׁין:
וַאֲמַרוּ מַן חַד מִשִׁבְטַיָא דְיִשְׂרָאֵל דְלָא סְלִיק לָקֳדָם יְיָ לְמִצְפַּיָא וְהָא לָא אֲתָא אֲנַשׁ לְמַשְׁרִיתָא מֵיָבֵשׁ גִלְעָד לִקְהָלָא:
וְאִתְמְנִיאוּ עַמָא וְהָא לֵית תַּמָן אֱנַשׁ מִיַשְׁבֵי יָבֵשׁ גִלְעָד:
וּשְׁלַחוּ תַמָן כְּנִשְׁתָּא תְּרֵי עֲסַר אַלְפִין גַבְרָא מְזָרְזֵי חֵילָא וּפַקִידוּ יַתְהוֹן לְמֵימָר אֱזִילוּ וְתִמְחוּן יַת יַתְבֵי יָבֵשׁ גִלְעָד לְפִתְגַם דְחָרֶב וּנְשַׁיָא וְטַפְלָא:
וְדֵין פִּתְגָמָא דִי תַעְבְּדוּן כָּל דְכוּרָא וְכָל אִתְּתָא דִידַעַת מִשְׁכְּבֵי דְכוּרָא תִּגְמְרוּן:
וְאַשְׁכָּחוּ מִיַתְבֵי יָבֵשׁ גִלְעָד אַרְבַּע מְאָה עוּלֵימָן בְּתוּלָן דְלָא יָדְעָא גְבַר מִשְׁכְּבֵי דְכוּרָא וְאַיְתִיאוּ יַתְהוֹן לְמַשְׁרִיתָא לְשִׁילוֹ דִי בְּאַרְעָא דִכְנָעַן:
וּשְׁלָחוּ כָּל כְּנִשְׁתָּא וּמַלִילוּ עִם בְּנֵי בִנְיָמִין דִי בְּכֵיף רִמוֹן וּקְרוֹ לְהוֹן מִלִין דִשְׁלָם:
וְתָבוּ בְנֵי בִנְיָמִין בְּעִידָנָא הַהִיא וִיהָבוּ לְהוֹן נְשִׁין דְקַיְמוּ מִנְשֵׁי יָבֵשׁ גִלְעָד וְלָא סַפִּיקוּ לְהוֹן כֵּן:
וְעַמָא אִתְנַחֲמוּ עַל דְבֵית בִּנְיָמִן אֲרֵי עֲבַד יְיָ תַּרְעָתָא בְּשִׁבְטַיָא דְיִשְׂרָאֵל:
וַאֲמַרוּ סָבֵי כְנִשְׁתָּא מַה נַעְבֵּד לִדְאִשְׁתַּאֲרוּ לִנְשִׁין אֲרֵי פְסָקַת מִדְבֵית בִּנְיָמִן אִתְּתָא:
וַאֲמַרוּ יְרוּתַת שְׁזֵיבְתָּא לִבְנֵי בִנְיָמִן וְלָא יִתְמְחֵי שִׁבְטָא מִיִשְׂרָאֵל:
וַאֲנַחְנָא לָא נֵיכוּל לְמִתַּן לְהוֹן נְשִׁין מִבְּנָתָנָא אֲרֵי קַיְימוּ בְנֵי יִשְׂרָאֵל לְמֵימָר לִיט דִי יִתֵּן אִתְּתָא לִדְבֵית בִּנְיָמִין:
וַאֲמַרוּ הָא חַגָא קֳדָם יְיָ בְּשִׁלוֹ מִזְמַן מוֹעֵד לְמוֹעֵד דִי מִצִיפּוּנָא לְבֵית אֵל מַדְּנַח שִׁמְשָׁא לְכִבְשָׁא דְסָלֵיק מִבֵּית אֵל לִשְׁכֶם וּמִדָרוֹמָא לִלְבוֹנָתָא:
וּפַקִידוּ יַת בְּנֵי בִנְיָמִין לְמֵימָר אֱזִילוּ וְתִכְמְנוּן בְּכַרְמַיָא:
וְתֶחְזוּן וְהָא אִם יִפְקוּן בְּנַת שִׁילוֹ לְמֵיחַג בְּחִנְגַיָא וְתִפְקוּן מִן כַרְמַיָא וּתְחַטְפוּן לְכוֹן גְבַר אִתְּתֵיהּ מִבְּנַת שִׁילוֹ וּתְהָכוּן לְאַרְעָא שִׁבְטָא דְבִנְיָמִן:
וִיהֵי אֲרֵי יֵיתוּן אֲבָהַתְהוֹן אוֹ אֲחֵיהוֹן לְמִידַן קֳדָמָנָא וְנֵימַר לְהוֹן חֲנִינוּנוּן לְהוֹן אֲרֵי לָא סַפִּיקְנָא לְמִתַּן גְבַר אִתְּתֵיהּ בִּקְרָבָא אֲרֵי לָא אַתּוּן יַהֲבִין לְהוֹן כְּעִידַן פּוֹן תְּחוֹבוּן:
וַעֲבָדוּ כֵּן בְּנֵי בִנְיָמִן וּנְסִיבוּ נְשִׁין לְמִנְיָנֵיהוֹן מִן חִנְגַיָא דִי אֲנָסוּ וַאֲזַלוּ וְתָבוּ לְאַחְסַנְתְּהוֹן וּבְנוֹ יַת קִרְוַיָא וִיתִיבוּ בְהוֹן:
וְהַלִיכוּ מִתַּמָן בְּנֵי יִשְׂרָאֵל בְּעִידָנָא הַהִיא גְבַר לְשִׁבְטֵיהּ וּלְזַרְעִיתֵיהּ וּנְפַקוּ מִתַּמָן גְבַר לְאַחְסַנְתֵּיהּ:
בְּיוֹמַיָא הָאִינוּן לֵית מַלְכָּא בְּיִשְׂרָאֵל גְבַר דְכָשַׁר בְּעֵינוֹהִי יַעְבַּד:
