Chapter 1



Chapter 2




וַיְדַבְּרוּ הַכַּשְׂדִּים לַמֶּלֶךְ אֲרָמִית: הַמֶּלֶךְ, לְעוֹלָמִים חֱיֵה! אֱמֹר אֶת הַחֲלוֹם לַעֲבָדֶיךָ, וְאֶת הַפִּתְרוֹן נַגִּידָה.
עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר אֶל הַכַּשְׂדִּים: הַדָּבָר וַדַּאי מִמֶּנִּי, אִם לֹא תּוֹדִיעוּנִי אֶת הַחֲלוֹם וְאֶת פִּתְרוֹנוֹ, נְתָחִים תֵּעָשׂוּ, וּבָתֵּיכֶם אַשְׁפָּה יוּשָׂמוּ.
וְאִם אֶת הַחֲלוֹם וְאֶת פִּתְרוֹנוֹ תַּגִּידוּ, מַתָּנוֹת וּתְשׁוּרוֹת וְכָבוֹד רַב תְּקַבְּלוּ מִלְּפָנַי; אֶלָּא אֶת הַחֲלוֹם וְאֶת פִּתְרוֹנוֹ הַגִּידוּ לִי.
עָנוּ שֵׁנִית וְאוֹמְרִים: הַמֶּלֶךְ, אֶת הַחֲלוֹם יֹאמַר לַעֲבָדָיו, וְאֶת הַפִּתְרוֹן נַגִּידָה.
עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: נָכוֹן יוֹדֵעַ אָנֹכִי, כִּי זְמַן אַתֶּם קוֹנִים, כָּל־עֻמַּת שֶׁרְאִיתֶם כִּי וַדַּאי מֵאִתִּי הַדָּבָר.
שֶׁאִם אֶת הַחֲלוֹם לֹא תּוֹדִיעוּנִי, אֶחָד הוּא מִשְׁפַּטְכֶם, וְדָבָר כּוֹזֵב וּמָשְׁחָת הִתְכּוֹנַנְתֶּם לֵאמֹר לְפָנַי, עַד שֶׁהַזְּמַן יִשְׁתַּנֶּה; אֶלָּא אֶת הַחֲלוֹם אִמְרוּ לִי, וְאֵדַע כִּי אֶת פִּתְרוֹנוֹ תַּגִּידוּנִי.
עָנוּ הַכַּשְׂדִּים לִפְנֵי הַמֶּלֶךְ וְאוֹמְרִים: אֵין אָדָם עַל הַיַּבָּשָׁה אֲשֶׁר אֶת דְּבַר הַמֶּלֶךְ יוּכַל לְהַגִּיד, כָּל־עֻמַּת שֶׁכָּל מֶלֶךְ גָּדוֹל וְשַׁלִּיט, דָּבָר כָּזֶה לֹא שָׁאַל מִכֹּל חַרְטֹם, אַשָּׁף וְכַשְׂדִּי.
וְהַדָּבָר אֲשֶׁר הַמֶּלֶךְ שׁוֹאֵל כָּבֵד הוּא, וְאַחֵר אַיִן אֲשֶׁר יַגִּידֶנּוּ לִפְנֵי הַמֶּלֶךְ, אֶלָּא אֱלֹהִים אֲשֶׁר אֵין מְדוֹרָם עִם בָּשָׂר.
כָּל־עֻמַּת זֶה הִתְאַנַּף הַמֶּלֶךְ וְקָצַף מְאֹד, וַיֹּאמֶר לְהַאֲבִיד אֶת כָּל חַכְמֵי בָבֶל.
וְהַדָּת יָצְאָה, וְהַחֲכָמִים נֶהֱרָגִים, וְעָמְדוּ דָּנִיֵּאל וַחֲבֵרָיו לְהֵהָרֵג.
אֲזַי דָּנִיֵּאל הֵשִׁיב עֵצָה וָטַעַם לְאַרְיוֹךְ שַׁר הַטַּבָּחִים שֶׁל הַמֶּלֶךְ, אֲשֶׁר יָצָא לַהֲרֹג אֶת חַכְמֵי בָבֶל.
עוֹנֶה וְאוֹמֵר לְאַרְיוֹךְ הַמְּמֻנֶּה שֶׁל הַמֶּלֶךְ: לָמָה הַדָּת עַזָּה מִלִּפְנֵי הַמֶּלֶךְ? אֲזַי הוֹדִיעַ אַרְיוֹךְ אֶת הַדָּבָר לְדָנִיֵּאל.
וְדָנִיֵּאל נִכְנַס וּבִקֵּשׁ מִן הַמֶּלֶךְ, כִּי יִתֶּן לוֹ זְמָן, וְאֶת הַפִּתְרוֹן יַגִּיד לַמֶּלֶךְ.
אֲזַי דָּנִיֵּאל לְבֵיתוֹ הָלַךְ, וְלַחֲנַנְיָה, מִישָׁאֵל וַעֲזַרְיָה חֲבֵרָיו אֶת הַדָּבָר הוֹדִיעַ.
וְרַחֲמִים לְבַקֵּשׁ מִלִּפְנֵי אֱלֹהֵי הַשָּׁמַיִם עַל הָרָז הַזֶּה, שֶׁלֹּא יֹאבְדוּ דָּנִיֵּאל וַחֲבֵרָיו עִם שְׁאָר חַכְמֵי בָבֶל.
אֲזַי נִגְלָה הָרָז לְדָנִיֵּאל בְּחֶזְיוֹן הַלַּיְלָה, אֲזַי בֵּרַךְ דָּנִיֵּאל אֶת אֱלֹהֵי הַשָּׁמַיִם.
עוֹנֶה דָּנִיֵּאל וְאוֹמֵר: יְהִי שֵׁם הָאֱלֹהִים מְבֹרָךְ, מִן הָעוֹלָם וְעַד הָעוֹלָם, אֲשֶׁר הַחָכְמָה וְהַגְּבוּרָה שֶׁלּוֹ הִיא.
וְהוּא מְשַׁנֶּה אֶת הָעִתִּים וְאֶת הַזְּמַנִּים, מַעֲבִיר מְלָכִים וּמֵקִים מְלָכִים, נוֹתֵן אֶת הַחָכְמָה לַחֲכָמִים, וְאֶת הַמַּדָּע לְיוֹדְעֵי בִינָה.
הוּא מְגַלֶּה אֶת הָעֲמֻקּוֹת וְאֶת הַנִּסְתָּרוֹת, יוֹדֵעַ מַה שֶׁבַּחֹשֶׁךְ, וְהָאוֹר עִמּוֹ שָׁרוּי.
לְךָ אֱלֹהֵי אֲבוֹתַי, מוֹדֶה וּמְשַׁבֵּחַ אֲנִי, אֲשֶׁר אֶת הַחָכְמָה וְהַגְּבוּרָה נָתַתָּ לִי, וְכָעֵת הוֹדַעְתַּנִי אֵת אֲשֶׁר בִּקַּשְׁנוּ מִמֶּךָּ, שֶׁאֶת דְּבַר הַמֶּלֶךְ הוֹדַעְתָּנוּ.
כָּל־עֻמַּת זֶה דָּנִיֵּאל נִכְנַס אֶל אַרְיוֹךְ, אֲשֶׁר מִנָּה הַמֶּלֶךְ לְהַאֲבִיד אֶת חַכְמֵי בָבֶל, הָלַךְ וְכֵן אָמַר לוֹ: אֶת חַכְמֵי בָבֶל אַל תַּאֲבֵד, הַכְנִיסֵנִי לִפְנֵי הַמֶּלֶךְ, וְאֶת הַפִּתְרוֹן לַמֶּלֶךְ אַגִידָה.
אֲזַי הִכְנִיס אַרְיוֹךְ בְּהִתְבַּהֲלוּת אֶת דָּנִיֵּאל לִפְנֵי הַמֶּלֶךְ, וְכֵן אָמַר לוֹ: מָצָאתִי אִישׁ מִבְּנֵי גָּלוּת יְהוּדָה, אֲשֶׁר יוֹדִיעַ אֶת הַפִּתְרוֹן לַמֶּלֶךְ.
עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל, אֲשֶׁר שְׁמוֹ בֵּלְטְשַׁאצַּר: הֲיֶשְׁךָ יָכוֹל לְהוֹדִיעֵנִי אֶת הַחֲלוֹם אֲשֶׁר רָאִיתִי וְאֶת פִּתְרוֹנוֹ?
עוֹנֶה דָּנִיֵּאל לִפְנֵי הַמֶּלֶךְ וְאוֹמֵר: הָרָז אֲשֶׁר הַמֶּלֶךְ שׁוֹאֵל אֵין חֲכָמִים, אַשָּׁפִים, חַרְטֻמִּים וְקוֹסְמִים יְכוֹלִים לְהַגִּיד לַמֶּלֶךְ.
אוּלָם, יֵשׁ אֱלוֹהַּ בַּשָּׁמַיִם מְגַלֶּה רָזִים, וְהוֹדִיעַ לַמֶּלֶךְ נְבוּכַדְנֶצַּר, מַה שֶׁיִּהְיֶה בְּאַחֲרִית הַיָּמִים, חֲלוֹמְךָ וְחֶזְיוֹנוֹת רֹאשְׁךָ עַל מִשְׁכָּבְךָ זֶה הוּא.
אַתָּה, הַמֶּלֶךְ, מַחְשְׁבוֹתֶיךָ עַל מִשְׁכָּבְךָ עָלוּ, מַה שֶׁיִּהְיֶה אַחֲרֵי כֵן, וּמְגַלֵּה הָרָזִים הוֹדִיעֲךָ מַה שֶׁיִּהְיֶה.
וַאֲנִי, לֹא בְחָכְמָה שֶׁיֶּשׁ בִּי מִכָּל הַחַיִּים נִגְלָה לִי רָז זֶה, אֶלָּא עַל דִּבְרַת, שֶׁאֶת הַפִּתְרוֹן לַמֶּלֶךְ יוֹדִיעוּ, וְאֶת רַעְיוֹנוֹת לִבְּךָ תֵּדָע.
אַתָּה הַמֶּלֶךְ, רוֹאֶה הָיִיתָ וְהִנֵּה צֶלֶם אֶחָד גָּדוֹל, צֶלֶם זֶה גָּדוֹל הוּא וַהֲדָרוֹ רַב, עוֹמֵד נֶגְדְּךָ וּמַרְאֵהוּ נוֹרָא.
הַצֶּלֶם הַהוּא, רֹאשׁוֹ שֶׁל זָהָב טוֹב, חָזֵהוּ וּזְרוֹעוֹתָיו שֶׁל כֶּסֶף, מֵעָיו וִירֵכָיו שֶׁל נְחֹשֶׁת.
שׁוֹקָיו שֶׁל בַּרְזֶל, רַגְלָיו, מֵהֶן שֶׁל בַּרְזֶל וּמֵהֶן שֶׁל חָרֶשׂ.
רוֹאֶה הָיִיתָ, עַד שֶׁנִּכְרְתָה אֶבֶן שֶׁלֹּא בְיָדַיִם, וַתַּךְ אֶת הַצֶּלֶם עַל רַגְלָיו שֶׁל בַּרְזֶל וְחֶרֶשׁ, וַתָּדֶק אוֹתָן.
אֲזַי הוּדַקוּ הַבַּרְזֶל, הַחֶרֶשׂ, הַנְּחֹשֶׁת, הַכֶּסֶף וְהַזָּהָב, וְהָיוּ כְמֹץ מִגָּרְנוֹת קַיִץ וַתִּשָּׂא אוֹתָם הָרוּחַ, וְכָל מָקוֹם לֹא נִמְצָא לָהֶם, וְהָאֶבֶן אֲשֶׁר הִכְּתָה אֶת הַצֶּלֶם הָיְתָה לְהַר גָּדוֹל, וּמָלְאָה כָּל הָאָרֶץ.
זֶה הַחֲלוֹם, וְאֶת פִּתְרוֹנוֹ נֹאמַר לִפְנֵי הַמֶּלֶךְ.
אַתָּה הַמֶּלֶךְ, מֶלֶךְ הַמְּלָכִים, אֲשֶׁר אֱלֹהֵי הַשָּׁמַיִם נָתַן לְךָ אֶת הַמְּלוּכָה, אֶת הָעֹז, וְאֶת הַתֹּקֶף וְאֶת הַכָּבוֹד.
וּבְכָל שֶׁדָּרִים בְּנֵי הָאָדָם, אֶת חַיַּת הַשָּׂדֶה וְעוֹף הַשָּׁמַיִם נָתַן בְּיָדְךָ וְהִשְׁלִיטְךָ בְּכֻלָּם, אַתָּה הוּא רֹאשׁ הַזָּהָב.
וְאַחֲרֶיךָ תָּקוּם מַלְכוּת אַחֶרֶת שְׁפֵלָה מִמְּךָ, וּמַלְכוּת שְׁלִישִׁית אַחֶרֶת מִנְּחֹשֶׁת, אֲשֶׁר תִּמְשֹׁל בְּכָל הָאָרֶץ.
וּמַלְכוּת רְבִיעִית תִּהְיֶה חֲזָקָה כַּבַּרְזֶל, כָּל־עֻמַּת שֶׁהַבַּרְזֶל מֵדֵק וְטוֹחֵן אֶת הַכֹּל, וְכַבַּרְזֶל הַרוֹעֵעַ – אֶת כָּל אֵלֶּה תָּדֵק וְתָרֹעַ.
וַאֲשֶׁר רָאִיתָ הָרַגְלַיִם וְהָאֶצְבָּעוֹת, מֵהֶן חֶרֶשׂ הַיּוֹצֵר וּמֵהֶן בַּרְזֶל, מַלְכוּת חֲלוּקָה תִּהְיֶה, וּמֵעֹצֶם הַבַּרְזֶל יִהְיֶה בָּהּ, כָּל־עֻמַּת שֶׁרָאִיתָ בַּרְזֶל מְעֹרָב בְּחֶרֶשׂ הַטִּין.
וְאֶצְבְּעוֹת הָרַגְלַיִם, מֵהֶן בַּרְזֶל וּמֵהֶן חֶרֶשׁ, מִקְצָת הַמַּלְכוּת תִּהְיֶה חֲזָקָה, וּמִמֶּנָּה תִּהְיֶה שְׁבוּרָה.
וַאֲשֶׁר רָאִיתָ הַבַּרְזֶל מְעֹרָב בְּחֶרֶשׂ הַטִּין, מִתְעָרְבִים יִהְיוּ בְּזֶרַע הָאֲנָשִׁים, וְלֹא יִהְיוּ דְּבֵקִים זֶה בָזֶה, כְּמוֹ שֶׁהַבַּרְזֶל לֹא יִתְעָרֵב עִם הֶחָרֶשׂ.
וּבִימֵיהֶם שֶׁל הַמְּלָכִים הָהֵם, יָקִים אֱלֹהֵי הַשָּׁמַיִם מַלְכוּת, אֲשֶׁר לְעוֹלָם לֹא תִּשָּׁחֵת, וּמַלְכוּתָהּ לְעַם אַחֵר לֹא תֵּעָזֵב, תָּדֵק וּתְכַלֶּה כָּל אֵלֶּה הַמַּלְכֻיּוֹת, וְהִיא תַּעֲמֹד לְעוֹלָמִים.
כָּל־עֻמַּת שֶׁרָאִיתָ, כִּי מִן הָהָר נִכְרְתָה אֶבֶן שֶׁלֹּא בְיָדַיִם, וְהֵדֵקָּה אֶת הַבַּרְזֶל, וְאֶת הַנְּחֹשֶׁת, וְאֶת הַחֶרֶשׂ, וְאֶת הַכֶּסֶף, וְאֶת הַזָּהָב, הָאֵל הַגָּדוֹל הוֹדִיעַ אֶת הַמֶּלֶךְ, מַה שֶׁיִּהְיֶה אַחֲרֵי זֶה, וְנָכוֹן הַחֲלוֹם וְנֶאֱמָן פִּתְרוֹנוֹ.
אֲזַי הַמֶּלֶךְ נְבוּכַדְנֶצַּר נָפַל עַל פָּנָיו, וּלְדָנִיֵּאל סָגַד, וּמִנְחָה וְנִיחוֹחִים אָמַר לְנַסֵּךְ לוֹ.
עוֹנֶה הַמֶּלֶךְ לְדָנִיֵּאל וְאוֹמֵר: אֱמֶת כִּי אֱלֹהֵיכֶם הוּא אֱלֹהֵי הָאֱלֹהִים, וַאֲדוֹנֵי הָאֲדוֹנִים וּמְגַלֶּה רָזִים, אֲשֶׁר יָכֹלְתָּ לְגַלּוֹת אֶת הָרָז הַזֶּה.
אֲזַי הַמֶּלֶךְ אֶת דָּנִיֵּאל גִּדֵּל, וּמַתָּנוֹת רַבּוֹת גְּדוֹלוֹת נָתַן לוֹ, וְיַשְׁלִיטֵהוּ עַל כָּל מְדִינַת בָּבֶל, וְרַב סְגָנִים עַל כָּל חַכְמֵי בָבֶל.
וְדָנִיֵּאל בִּקֵּשׁ מִן הַמֶּלֶךְ, וַיְמַן עַל עֲבוֹדַת מְדִינַת בָּבֶל אֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, וְדָנִיֵּאל בְּשַׁעַר הַמֶּלֶךְ.


Chapter 3

נְבוּכַדְנֶצַּר הַמֶּלֶךְ עָשָׂה צֶלֶם זָהָב, גָּבְהוֹ אַמּוֹת שִׁשִּׁים, רָחְבּוֹ אַמּוֹת שֵׁשׁ, הֱקִימוֹ בְּבִקְעַת דּוּרָא בִּמְדִינַת בָּבֶל.
וּנְבוּכַדְנֶצַּר הַמֶּלֶךְ, שָׁלַח לְכַנֵּס אֶת הָאֲחַשְׁדַּרְפְּנִים, הַסְּגָנִים, הַפַּחוֹת, הַיּוֹעֲצִים, הַגִּזְבָּרִים, הַשּׁוֹפְטִים, הַשּׁוֹטְרִים וְכָל שַׁלִּיטֵי הַמְּדִינוֹת, לָבוֹא לַחֲנֻכַּת הַצֶּלֶם, אֲשֶׁר הֵקִים נְבוּכַדְנֶצַּר הַמֶּלֶךְ.
אֲזַי מִתְכַּנְּסִים הָאֲחַשְׁדַּרְפְּנִים, הַסְּגָנִים וְהַפַּחוֹת, הַיּוֹעֲצִים, הַגִּזְבָּרִים, הַשּׁוֹפְטִים, הַשּׁוֹטְרִים, וְכָל שַׁלִּיטֵי הַמְּדִינוֹת לַחֲנֻכַּת הַצֶּלֶם, אֲשֶׁר הֵקִים נְבוּכַדְנֶצַּר הַמֶּלֶךְ, וְעוֹמְדִים לְעֻמַּת הַצֶּלֶם, אֲשֶׁר הֵקִים נְבוּכַדְנֶצַּר.
וְהַכָּרוֹז קוֹרֵא בְכֹחַ: לָכֶם אוֹמְרִים – הָעַמִּים הָאֻמּוֹת וְהַלְּשׁוֹנוֹת.
בָּעֵת אֲשֶׁר תִּשְׁמְעוּ אֶת קוֹל הַקֶּרֶן, הֶחָלִיל, הַנֵּבֶל, הַכִּנּוֹר, הַמִּנִּים, הָעוּגָב וְכָל מִינֵי הַזֶּמֶר, תִּפְּלוּ וְתִסְגְּדוּ לְצֶלֶם הַזָּהָב, אֲשֶׁר הֵקִים נְבוּכַדְנֶצַּר הַמֶּלֶךְ.
וּמִי שֶׁלֹּא יִפֹּל וְיִסְגֹּד, בָּהּ בַּשָּׁעָה יֻשְׁלַךְ לְתוֹךְ כִּבְשַׁן הָאֵשׁ הַבּוֹעָרֶת.
כָּל־עֻמַּת זֶה בּוֹ בַּזְּמַן, כְּשֶׁשּׁוֹמְעִים כָּל הָעַמִּים אֶת קוֹל הַקֶּרֶן, הֶחָלִיל, הַנֵּבֶל, הַכִּנּוֹר, הַמִּנִּים וְכָל מִינֵי הַזֶּמֶר, נוֹפְלִים כָּל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת, סוֹגְדִים לְצֶלֶם הַזָּהָב, אֲשֶׁר הֵקִים נְבוּכַדְנֶצַּר הַמֶּלֶךְ.
כָּל־עֻמַּת זֶה בּוֹ בַּזְּמַן, קָרְבוּ אֲנָשִׁים כַּשְׂדִּים וְהוֹצִיאוּ דִּבָּתָם שֶׁל הַיְּהוּדִים.
עָנוּ וְאוֹמְרִים לִנְבוּכַדְנֶצַּר הַמֶּלֶךְ: הַמֶּלֶךְ לְעוֹלָמִים חֱיֵה.
אַתָּה הַמֶּלֶךְ שַׂמְתָּ צַו, שֶׁכָּל אִישׁ שֶׁיִּשְׁמַע אֶת קוֹל הַקֶּרֶן, הֶחָלִיל, הַנֵּבֶל, הַכִּנּוֹר, הַמִּנִּים וְהָעוּגָב וְכָל מִינֵי הַזֶּמֶר, יִפֹּל וְיִסְגֹּד לְצֶלֶם הַזָּהָב.
וּמִי שֶׁלֹּא יִפֹּל וְיִסְגֹּד, יֻשְׁלַךְ לְתוֹךְ כִּבְשַׁן הָאֵשׁ הַבּוֹעָרֶת.
יֵשׁ אֲנָשִׁים יְהוּדִים, אֲשֶׁר מִנִּיתָ אוֹתָם עַל עֲבוֹדַת מְדִינַת בָּבֶל – שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, הָאֲנָשִׁים הָאֵלֶּה לֹא שָׂמוּ עָלֶיךָ הַמֶּלֶךְ לֵב, אֶת אֱלֹהֶיךָ אֵינָם עוֹבְדִים, וּלְצֶלֶם הַזָּהָב אֲשֶׁר הֲקִימוֹתָ אֵינָם סוֹגְדִים.
אֲזַי נְבוּכַדְנֶצַּר בְּרֹגֶז וְחֵמָה, אָמַר לְהָבִיא אֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ; אֲזַי אֶת הָאֲנָשִׁים הָאֵלֶּה הֵבִיאוּ לִפְנֵי הַמֶּלֶךְ.
עוֹנֶה נְבוּכַדְנֶצַּר וְאוֹמֵר לָהֶם: הֲבֶאֱמֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, אֶת אֱלֹהַי אֵינְכֶם עוֹבְדִים וּלְצֶלֶם הַזָּהָב אֲשֶׁר הֲקִימוֹתִי אֵינְכֶם סוֹגְדִים?
עַתָּה אִם יֶשְׁכֶם עֲתִידִים, שֶׁבָּעֵת שֶׁתִּשְׁמְעוּ אֶת קוֹל הַקֶּרֶן, הֶחָלִיל, הַנֵּבֶל הַכִּנּוֹר, הַמִּנִּים וְהָעוּגָב וְכָל מִינֵי הַזֶּמֶר, תִּפְּלוּ וְתִסְגְּדוּ לַצֶּלֶם אֲשֶׁר עָשִׂיתִי; וְאִם לֹא תִּסְגְּדוּ, בָּהּ בַּשָּׁעָה תָּשְׁלְכוּ לְתוֹךְ כִּבְשַׁן הָאֵשׁ הַבּוֹעֶרֶת, וּמִי הוּא אֱלוֹהַּ אֲשֶׁר יַצִּילְכֶם מִיָּדָי.
עָנוּ שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ וְאוֹמְרִים לַמֶּלֶךְ: נְבוּכַדְנֶצַּר, לֹא צְרִיכִים אֲנַחְנוּ עַל דָּבָר זֶה לַהֲשִׁיבֶךָּ.
אִם יֵשׁ אֱלֹהֵינוּ, אֲשֶׁר אֲנַחְנוּ עוֹבְדִים, יָכוֹל לְהַצִּילֵנוּ מִכִּבְשַׁן הָאֵשׁ הַבּוֹעֶרֶת, וּמִיָּדְךָ הַמֶּלֶךְ יַצִּיל.
וְאִם לֹא, יָדוּעַ יִהְיֶה לְךָ, הַמֶּלֶךְ, כִּי לֵאלֹהֶיךָ אֵין אֲנַחְנוּ עוֹבְדִים, וּלְצֶלֶם הַזָּהָב אֲשֶׁר הֲקִימוֹתָ לֹא נִסְגֹּד.
אֲזַי נְבוּכַדְנֶצַּר נִתְמַלֵּא חֵמָה, וּדְמוּת פָּנָיו הִשְׁתַּנְּתָה, עַל שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ; עוֹנֶה וְאוֹמֵר לְהַסִּיק אֶת הַכִּבְשָׁן עַל אַחַת שֶׁבַע, עַל שֶׁרָאוּי לְהַסִּיקוֹ.
וְלַאֲנָשִׁים גִּבּוֹרֵי חַיִל אֲשֶׁר בְּחֵילוֹ אָמַר לִכְפֹּת אֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, לְהַשְׁלִיךְ לְכִבְשַׁן הָאֵשׁ הַבּוֹעָרֶת.
אֲזַי הָאֲנָשִׁים הָאֵלֶּה נִכְפְּתוּ בְּשִׂמְלוֹתֵיהֶם, כָּתְנוֹתֵיהֶם, מִגְבְּעוֹתֵיהֶם וּלְבוּשֵׁיהֶם וְהָשְׁלְכוּ לְתוֹךְ כִּבְשַׁן הָאֵשׁ הַבּוֹעָרֶת.
כָּל־עֻמַּת זֶה, מִשֶּׁדְּבַר הַמֶּלֶךְ עַז וְהַכִּבְשָׁן הֻסַּק בְּיוֹתֵר, הָאֲנָשִׁים הָאֵלֶּה, אֲשֶׁר הֶעֱלוּ אֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, הֵמִית אוֹתָם שְׁבִיב הָאֵשׁ.
וְהָאֲנָשִׁים הָאֵלֶּה שְׁלָשְׁתָּם: שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, נָפְלוּ לְתוֹךְ כִּבְשַׁן הָאֵשׁ הַבּוֹעָרֶת כְּפוּתִים.
אֲזַי נְבוּכַדְנֶצַּר הַמֶּלֶךְ תָּמַהּ וְקָם בְּהִתְבַּהֲלוּת, עוֹנֶה וְאוֹמֵר לְיוֹעֲצָיו: הֲלֹא אֲנָשִׁים שְׁלֹשָׁה הִשְׁלַכְנוּ לְתוֹךְ הָאֵשׁ כְּפוּתִים! עוֹנִים וְאוֹמְרִים לַמֶּלֶךְ: נָכוֹן הַמֶּלֶךְ.
עוֹנֶה וְאוֹמֵר: הִנֵּה אֲנִי רוֹאֶה אֲנָשִׁים אַרְבָּעָה מֻתָּרִים, מְהַלְּכִים בְּתוֹךְ הָאֵשׁ וְחַבָּלָה אֵין בָּהֶם, וּמַרְאֵהוּ שֶׁל הָרְבִיעִי דּוֹמֶה לְבֶן אֱלֹהִים.
אֲזַי קָם נְבוּכַדְנֶצַּר לְשַׁעַר כִּבְשַׁן הָאֵשׁ הַבּוֹעֶרֶת, עוֹנֶה וְאוֹמֵר: שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, עֲבָדָיו שֶׁל הָאֵל הָעֶלְיוֹן צְאוּ וּבֹאוּ! אֲזַי יוֹצְאִים שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ מִתּוֹךְ הָאֵשׁ.
וּמִתְכַּנְּסִים הָאֲחַשְׁדַּרְפְּנִים, הַסְּגָנִים וְהַפַּחוֹת וְיוֹעֲצֵי הַמֶּלֶךְ, רוֹאִים אֶת הָאֲנָשִׁים הָאֵלֶּה, אֲשֶׁר לֹא שָׁלְטָה הָאֵשׁ בְּגוּפָם וּשְׂעַר רֹאשָׁם לֹא נֶחֱרַךְ, וְשִׂמְלוֹתֵיהֶם לֹא נִשְׁתַּנּוּ, וְרֵיחַ אֵשׁ לֹא עָבַר בָּהֶם.
עוֹנֶה נְבוּכַדְנֶצַּר וְאוֹמֵר: בָּרוּךְ אֱלֹהֵיהֶם שֶׁל שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, אֲשֶׁר שָׁלַח מַלְאָכוֹ וְהִצִּיל אֶת עֲבָדָיו אֲשֶׁר בָּטְחוּ עָלָיו, וּדְבַר הַמֶּלֶךְ שִׁנּוּ וְנָתְנוּ אֶת גּוּפָם, שֶׁלֹּא יַעַבְדוּ וְלֹא יִסְגְּדוּ לְכָל אֱלוֹהַּ, אֶלָּא לֵאלֹהֵיהֶם.
וּמִמֶּנִּי הוּשַׂם צַו, אֲשֶׁר כָּל עַם, אֻמָּה וְלָשׁוֹן, אֲשֶׁר יֹאמַר תּוֹעָה עַל אֱלֹהֵיהֶם שֶׁל שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹא, נְתָחִים יֵעָשֶׂה וּבֵיתוֹ אַשְׁפָּה יוּשַׂם, כָּל־עֻמַּת שֶׁאֵין אֱלוֹהַּ אַחֵר אֲשֶׁר יוּכַל לְהַצִּיל כָּזֶה.
אֲזַי הַמֶּלֶךְ, הִצְלִיחַ אֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ בִּמְדִינַת בָּבֶל.
נְבוּכַדְנֶצַּר הַמֶּלֶךְ, לְכָל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת, שֶׁדָּרִים בְּכָל הָאָרֶץ, שְׁלוֹמְכֶם יִרְבֶּה.
הָאוֹתוֹת וְהַנִּפְלָאוֹת, שֶׁעָשָׂה עִמִּי אֵל עֶלְיוֹן, נָאֶה לְפָנַי לְהַגִּיד.
אוֹתוֹתָיו כַּמָּה גְּדוֹלִים, וְנִפְלְאוֹתָיו כַּמָּה תַקִּיפוֹת! מַלְכוּתוֹ מַלְכוּת עוֹלָם, וְשִׁלְטוֹנוֹ עִם דּוֹר וָדוֹר.


Chapter 4

אֲנִי נְבוּכַדְנֶצַּר, שָׁלֵו הָיִיתִי בְּבֵיתִי, וְרַעֲנָן בְּהֵיכָלִי.
חֲלוֹם רָאִיתִי וְיַפְחִידֵנִי, וְהִרְהוּרִים עַל מִשְׁכָּבִי, וְחֶזְיוֹנוֹת רֹאשִׁי יְבַהֲלוּנִי.
וּמִמֶּנִּי הוּשַׂם צַו, לְהַכְנִיס לְפָנַי אֶת כָּל חַכְמֵי בָבֶל, שֶׁפִּתְרוֹן הַחֲלוֹם יוֹדִיעוּנִי.
אֲזַי נִכְנָסִים הַחַרְטֻמִּים, הַכַּשְׂדִּים וְהַקּוֹסְמִים, וְאֶת הַחֲלוֹם אֲנִי אוֹמֵר לִפְנֵיהֶם, וּפִתְרוֹנוֹ אֵין מוֹדִיעִים לִי.
וְעַד לָאַחֲרוֹנָה נִכְנַס לְפָנַי דָּנִיֵּאל, אֲשֶׁר שְׁמוֹ בֵּלְטְשַׁאצַּר כְּשֵׁם אֱלֹהַי, וַאֲשֶׁר רוּחַ אֱלֹהִים קְדוֹשִׁים בּוֹ, וְאֶת הַחֲלוֹם לְפָנָיו אָמַרְתִּי.
בֵּלְטְשַׁאצַּר רַב הַחַרְטֻמִּים, אֲשֶׁר אֲנִי יָדַעְתִּי כִּי רוּחַ אֱלֹהִים קְדוֹשִׁים בָּךְ, וְכָל רָז אֵינוֹ מוֹנֵעַ אוֹתְךָ, חֶזְיוֹנוֹת חֲלוֹמִי אֲשֶׁר רָאִיתִי וּפִתְרוֹנוֹ אֱמֹר.
וְחֶזְיוֹנוֹת רֹאשִׁי עַל מִשְׁכָּבִי, רוֹאֶה הָיִיתִי וְהִנֵּה אִילָן בְּתוֹךְ הָאָרֶץ, וְרוּמוֹ רָב.
גָּדַל הָאִילָן וְחָזַק, וְרוּמוֹ יַגִּיעַ לַשָּׁמַיִם, וּמַרְאֵהוּ לְסוֹף כָּל הָאָרֶץ.
עָפְיוֹ יָפֶה וּפִרְיוֹ רַב, וּמָזוֹן לַכֹּל בּוֹ, תַּחְתָּיו תִּמְצָא צֵל חַיַּת הַבָּר, וּבַעֲנָפָיו תָּדֹרְנָה צִפֳּרֵי הַשָּׁמִים, וּמִמֶּנּוּ יִזּוֹן כָּל בָּשָׂר.
רוֹאֶה הָיִיתִי בְּחֶזְיוֹנוֹת רֹאשִׁי עַל מִשְׁכָּבִי, וְהִנֵּה מַלְאָךְ וְקָדוֹשׁ מִשָּׁמַיִם יוֹרֵד.
קוֹרֵא בְכֹחַ וְכֵן אוֹמֵר: גֹּזוּ הָאִילָן וְקַצְּצוּ עֲנָפָיו, הַשִּׁירוּ עָפְיוֹ וּפַזְּרוּ פִרְיוֹ, תָּגוּד הַחַיָּה מִתַּחְתָּיו, וְהַצִּפֳּרִים מֵעֲנָפָיו.
אוּלָם עִקַּר שָׁרָשָׁיו בָּאָרֶץ הַנִּיחוּ, וּבְאֵסוּר שֶׁל בַּרְזֶל וּנְחֹשֶׁת בְּדֶשֶׁא שֶׁל הַבָּר, וּבְטַל הַשָּׁמַיִם יִטְבֹּל, וְעִם הַחַיָּה חֶלְקוֹ בְּעֵשֶׂב הָאָרֶץ.
לִבּוֹ מֵאֱנוֹשׁ יְשַׁנּוּ, וְלֵב חַיָּה יִנָּתֵן לוֹ, וְשִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלָיו.
בִּגְזֵרַת מַלְאָכִים הַדָּבָר, וּמַאֲמַר קְדוֹשִׁים הַמִּשְׁאָלָה, עַל דִּבְרַת שֶׁיֵּדְעוּ הַחַיִּים, כִּי שׁוֹלֵט הָעֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וּלְמִי שֶׁיִּרְצֶה יִתְּנֶנָּה, וּשְׁפַל אֲנָשִׁים יָקִים עָלֶיהָ.
זֶה הַחֲלוֹם רָאִיתִי, אֲנִי הַמֶּלֶךְ נְבוּכַדְנֶצַּר, וְאַתָּה בֵּלְטְשַׁאצַּר פִּתְרוֹנוֹ אֱמֹר, כָּל־עֻמַּת שֶׁכָּל חַכְמֵי מַלְכוּתִי אֵינָם יְכוֹלִים לְהוֹדִיעֵנִי הַפִּתְרוֹן, וְאַתָּה יָכוֹל, כִּי רוּחַ אֱלֹהִים קְדוֹשִׁים בָּךְ.
אֲזַי דָּנִיֵּאל, אֲשֶׁר שְׁמוֹ בֵּלְטְשַׁאצַּר, הִשְׁתּוֹמֵם כְּשָׁעָה אַחַת וְרַעְיוֹנָיו יְבַהֲלוּהוּ. עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: בֵּלְטְשַׁאצַּר, הַחֲלוֹם וּפִתְרוֹנוֹ אַל יְבַהֲלֶךָ! עוֹנֶה בֵּלְטְשַׁאצַּר וְאוֹמֵר: אֲדוֹנִי, הַחֲלוֹם לְשׂוֹנְאֶיךָ וּפִתְרוֹנוֹ לְצָרֶיךָ.
הָאִילָן אֲשֶׁר רָאִיתָ, אֲשֶׁר גָּדַל וְחָזַק, וְרוּמוֹ מַגִּיעַ לַשָּׁמַיִם, וּמַרְאֵהוּ לְכָל הָאָרֶץ.
וְעָפְיוֹ יָפֶה וּפִרְיוֹ רַב, וּמָזוֹן לַכֹּל בּוֹ; תַּחְתָּיו תָּדוּר חַיַּת הַבָּר, וּבַעֲנָפָיו תִּשְׁכֹּנָּה צִפֳּרֵי הַשָּׁמַיִם.
אַתָּה הוּא הַמֶּלֶךְ, אֲשֶׁר גָּדַלְתָּ וְחָזַקְתָּ, וּגְדֻלָּתְךָ גָּדְלָה וְהִגִּיעָה לַשָּׁמַיִם וְשִׁלְטוֹנְךָ לְסוֹף הָאָרֶץ.
וַאֲשֶׁר רָאָה הַמֶּלֶךְ, מַלְאָךְ וְקָדוֹשׁ יוֹרֵד מִן הַשָּׁמַיִם וְאוֹמֵר: גֹּזוּ הָאִילָן וְהַשְׁחִיתוּהוּ, אוּלָם עִקַּר שָׁרָשָׁיו בָּאָרֶץ הַנִּיחוּ, וּבְאֵסוּר שֶׁל בַּרְזֶל וּנְחֹשֶׁת בְּדֶשֶׁא שֶׁל הַבָּר, וּבְטַל הַשָּׁמַיִם יִטְבֹּל, וְעִם חַיַּת הַבָּר חֶלְקוֹ, עַד אֲשֶׁר שִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלָיו.
זֶה הַפִּתְרוֹן, הַמֶּלֶךְ, וּגְזֵרַת עֶלְיוֹן הִיא, שֶׁהִגִּיעָה אֶל אֲדוֹנִי הַמֶּלֶךְ.
וְאוֹתְךָ מְגָרְשִׁים מֵאֱנוֹשׁ, וְעִם חַיַּת הַבָּר יִהְיֶה מְדוֹרְךָ, וְעֵשֶׂב כִּשְׁוָרִים אוֹתְךָ יַאֲכִילוּ, וּמִטַּל הַשָּׁמַיִם יִטְבְּלוּךָ, וְשִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלֶיךָ עַד שֶׁתֵּדַע, כִּי שׁוֹלֵט עֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וּלְמִי שֶׁיִּרְצֶה יִתְּנֶנָּה.
וַאֲשֶׁר אָמְרוּ לְהַנִּיחַ עִקַּר שָׁרָשָׁיו שֶׁל הָאִילָן, הַמַּלְכוּת לְךָ קַיֶּמֶת, מִשֶּׁתֵּדַע כִּי שׁוֹלְטִים הַשָּׁמַיִם.
אֶלָּא הַמֶּלֶךְ, עֲצָתִי תִּיטַב עָלֶיךָ, וַחֲטָאֶיךָ בִּצְדָקָה פְדֵה וַעֲוֹנוֹתֶיךָ בְּחֹן עֲנִיִּים, אִם תִּהְיֶה אַרְכָּה לְשַׁלְוָתֶךָ.
הַכֹּל הִגִּיעַ אֶל נְבוּכַדְנֶצַּר הַמֶּלֶךְ.
לְקֵץ יְרָחִים שְׁנֵים עָשָׂר, עַל הֵיכַל הַמַּלְכוּת שֶׁל בָּבֶל מְהַלֵּךְ הָיָה.
עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: הֲלֹא זוֹ הִיא בָבֶל הַגְּדוֹלָה, אֲשֶׁר אֲנִי בְנִיתִיהָ לְבֵית מַלְכוּת, בְּתֹקֶף חָסְנִי וְלִכְבוֹד הֲדָרִי.
עוֹד הַדָּבָר בְּפִי הַמֶּלֶךְ, קוֹל מִן הַשָּׁמַיִם נָפַל: לְךָ אוֹמְרִים נְבוּכַדְנֶצַּר הַמֶּלֶךְ, הַמַּלְכוּת עָבְרָה מִמֶּךָּ.
וּמֵאֱנוֹשׁ אוֹתְךָ מְגָרְשִׁים, וְעִם חַיַּת הַבָּר מְדוֹרְךָ, עֵשֶׂב כִּשְׁוָרִים אוֹתְךָ יַאֲכִילוּ, וְשִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלֶיךָ עַד שֶׁתֵּדַע, כִּי שׁוֹלֵט עֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וּלְמִי שֶׁיִּרְצֶה יִתְּנֶנָּה.
בָּהּ בַּשָּׁעָה, הַדָּבָר כָּלָה עַל נְבוּכַדְנֶצַּר, וּמֵאֱנוֹשׁ גֹּרַשׁ, וְעֵשֶׂב כִּשְׁוָרִים יֹאכַל, וּמִטַּל הַשָּׁמַיִם גּוּפוֹ יִטָּבֵל עַד שֶׁשְּׂעָרוֹ כִּנְשָׁרִים גָּדַל וְצִפָּרְנָיו כְּצִפֳּרִים.
וּלְקֵץ הַיָּמִים, אֲנִי נְבוּכַדְנֶצַּר, עֵינַי לַשָּׁמַיִם נָשָׂאתִי, וְדַעְתִּי עָלַי תָּשׁוּב, וְאֶת הָעֶלְיוֹן בֵּרַכְתִּי, וְאֶת חַי הָעוֹלָם שִׁבַּחְתִּי וּפֵאַרְתִּי, אֲשֶׁר שִׁלְטוֹנוֹ שִׁלְטוֹן עוֹלָם וּמַלְכוּתוֹ עִם דּוֹר וָדוֹר.
וְכָל דָּרֵי הָאָרֶץ כְּאַיִן חֲשׁוּבִים, וְכִרְצוֹנוֹ עוֹשֶׂה בְּחֵיל הַשָּׁמַיִם וְדָרֵי הָאָרֶץ, וְאֵין אֲשֶׁר יְמַחֶה בְיָדוֹ וְיֹאמַר לוֹ מָה עָשִׂיתָ.
בּוֹ בַזְּמַן דַּעְתִּי תָּשׁוּב עָלַי, וְלִיקַר מַלְכוּתִי הֲדָרִי, וְזִיוִי יָשׁוּב אֵלַי, וְאוֹתִי – יוֹעֲצַי וְשָׂרַי יְבַקְּשׁוּ, וְעַל מַלְכוּתִי הוּכַנְתִּי, וּגְדֻלָּה יְתֵרָה נוֹסְפָה לִי.
כָּעֵת אֲנִי נְבוּכַדְנֶצַּר, מְשַׁבֵּחַ וּמְרוֹמֵם וּמְפָאֵר אֶת מֶלֶךְ הַשָּׁמַיִם, אֲשֶׁר כָּל מַעֲשָׂיו אֱמֶת וְאָרְחוֹתָיו מִשְׁפָּט, וַאֲשֶׁר מְהַלְּכִים בְּגַאֲוָה יָכוֹל לְהַשְׁפִּיל.


Chapter 5

בֵּלְשַׁאצַּר הַמֶּלֶךְ עָשָׂה מִשְׁתֶּה גָדוֹל לְאֶלֶף שָׂרָיו, וּלְעֻמַּת הָאֶלֶף יַיִן שָׁתָה.
בֵּלְשַׁאצַּר צִוָּה בְּטַעַם הַיַּיִן, לְהָבִיא כְלֵי הַזָּהָב וְהַכֶּסֶף, אֲשֶׁר הוֹצִיא נְבוּכַדְנֶצַּר אָבִיו מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם, וְיִשְׁתּוּ בָהֶם הַמֶּלֶךְ וְשָׂרָיו, נָשָׁיו וּפִילַגְשָׁיו.
אֲזַי הֱבִיאוֹ אֶת כְּלֵי הַזָּהָב אֲשֶׁר הוֹצִיאוּ מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם, וְשָׁתוּ בָהֶם הַמֶּלֶךְ וְשָׂרָיו, נָשָׁיו וּפִילַגְשָׁיו.
שָׁתוּ אֶת הַיַּיִן, וְשִׁבְּחוּ אֶת אֱלֹהֵי הַזָּהָב וְהַכֶּסֶף, הַנְּחֹשֶׁת, הַבַּרְזֶל, הָעֵץ וְהָאֶבֶן.
בָּהּ בַּשָּׁעָה, יָצְאוּ אֶצְבְּעוֹת שֶׁל יַד אָדָם וְכָתְבוּ לְעֻמַּת הַנִּבְרֶשֶׁת, עַל הַשִּׂיד שֶׁל כֹּתֶל הַהֵיכָל שֶׁל הַמֶּלֶךְ, וְהַמֶּלֶךְ רוֹאֶה פַּס יָד אֲשֶׁר כּוֹתֶבֶת.
אֲזַי הַמֶּלֶךְ, זִיווֹ נִשְׁתַּנֶּה לוֹ וְרַעְיוֹנָיו יְבַהֲלוּהוּ, וְקִשְׁרֵי חֲלָצָיו נִתָּרִים, וּבִרְכָּיו זוֹ לְזוֹ נוֹקְשׁוֹת.
קוֹרֵא הַמֶּלֶךְ בְּחַיִל, לְהַכְנִיס אֶת הָאַשָּׁפִים, הַכַּשְׂדִּים וְהַקּוֹסְמִים; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְחַכְמֵי בָבֶל, שֶׁכָּל אִישׁ שֶׁיִּקְרָא כְתָב זֶה וּפִתְרוֹנוֹ יוֹדִיעֵנִי, אַרְגָּמָן יִלְבַּשׁ, וְרָבִיד שֶׁל זָהָב עַל צַוָּארוֹ, וּשְׁלִישִׁי בַּמַּלְכוּת יְשְׁלָט.
אֲזַי נִכְנָסִים כָּל חַכְמֵי הַמֶּלֶךְ, וְאֵינָם יְכוֹלִים אֶת הַכְּתָב לִקְרֹא וְאֶת פִּתְרוֹנוֹ לְהוֹדִיעַ לַמֶּלֶךְ.
אֲזַי הַמֶּלֶךְ בֵּלְשַׁאצַּר מְאֹד נִבְהַל, וְזִיווֹ נִשְׁתַּנָּה לוֹ, וְשָׂרָיו נְבוֹכִים.
הַמַּלְכָּה, לְעֻמַּת דִּבְרֵי הַמֶּלֶךְ וְשָׂרָיו, לְבֵית הַמִּשְׁתֶּה נִכְנָסָה; עָנְתָה הַמַּלְכָּה וְאָמְרָה: הַמֶּלֶךְ לְעוֹלָמִים חֱיֵה, אַל יְבַהֲלוּךָ רַעְיוֹנֶיךָ וְזֶיוְךָ אַל יִשְׁתַּנֶּה.
יֵשׁ אִישׁ בְּמַלְכוּתֶךָ, אֲשֶׁר רוּחַ אֱלֹהִים קְדוֹשִׁים בּוֹ, וּבִימֵי אָבִיךָ הֶאָרָה וְהַשְׂכֵּל וְחָכְמָה כְּחָכְמַת אֱלֹהִים נִמְצָאָה בוֹ; וְהַמֶּלֶךְ נְבוּכַדְנֶצַּר אָבִיךָ, לְרַב חַרְטֻמִּים, אַשָּׁפִים, כַּשְׂדִּים, קוֹסְמִים הֱקִימוֹ אָבִיךָ הַמֶּלֶךְ.
כָּל־עֻמַּת שֶׁרוּחַ יְתֵרָה וּמַדָּע וְהַשְׂכֵּל לִפְתֹּר חֲלוֹמוֹת, לְהַגָּדַת חִידוֹת וּלְהַתִּיר קְשָׁרִים נִמְצְאָה בוֹ בְּדָנִיֵּאל, אֲשֶׁר הַמֶּלֶךְ שָׂם שְׁמוֹ בֵּלְטְשַׁאצַּר, עַתָּה דָּנִיֵּאל יִקָּרֵא וְאֶת הַפִּתְרוֹן יַגִּיד.
אֲזַי דָּנִיֵּאל הָכְנַס לִפְנֵי הַמֶּלֶךְ; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל: אַתָּה הוּא דָּנִיֵּאל, אֲשֶׁר מִבְּנֵי גָּלוּת יְהוּדָה, אֲשֶׁר הֵבִיא הַמֶּלֶךְ אָבִי מִיְהוּדָה?
וְשָׁמַעְתִּי עָלֶיךָ, שֶׁרוּחַ אֱלֹהִים בָּךְ, וְהֶאָרָה וְהַשְׂכֵּל וְחָכְמָה יְתֵרָה נִמְצְאָה בָּךְ.
וְכָעֵת הָכְנְסוּ לְפָנַי הַחֲכָמִים, הָאַשָּׁפִים, שֶׁכְּתָב זֶה יִקְרְאוּ וּפִשְׁרוֹ לְהוֹדִיעֵנִי, וְאֵינָם יְכוֹלִים פֵּשֶׁר דָּבָר לְהַגִּיד.
וַאֲנִי שָׁמַעְתִּי עָלֶיךָ, שֶׁתּוּכַל פִּתְרוֹנִים לִפְתֹּר וּקְשָׁרִים לְהַתִּיר; כָּעֵת, אִם תּוּכַל אֶת הַכְּתָב לִקְרֹא וּפִתְרוֹנוֹ לְהוֹדִיעֵנִי, אַרְגָּמָן תִּלְבַּשׁ, וְרָבִיד שֶׁל זָהָב עַל צַוָּארֶךָ, וּשְׁלִישִׁי בַּמַּלְכוּת תִּשְׁלֹט.
אֲזַי עוֹנֶה דָּנִיֵּאל וְאוֹמֵר לִפְנֵי הַמֶּלֶךְ: מַתְּנוֹתֶיךָ לְךָ תִּהְיֶינָה וּתְשׁוּרוֹתֶיךָ לְאַחֵר תֵּן, אוּלָם אֶת הַכְּתָב אֶקְרָא לַמֶּלֶךְ וְאֶת הַפִּתְרוֹן אוֹדִיעֶנּוּ.
אַתָּה הַמֶּלֶךְ, הָאֵל הָעֶלְיוֹן נָתַן מַלְכוּת וּגְדֻלָּה וִיקָר וְהָדָר לִנְבֻכַדְנֶצַּר אָבִיךָ.
וּמִן הַגְּדֻלָּה אֲשֶׁר נָתַן לוֹ, כָּל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת הָיוּ זָעִים וִירֵאִים מִפָּנָיו; אֲשֶׁר הָיָה חָפֵץ – הָיָה מֵמִית, וַאֲשֶׁר הָיָה חָפֵץ – הָיָה מְחַיֶּה, וַאֲשֶׁר הָיָה חָפֵץ – הָיָה מֵרִים, וַאֲשֶׁר הָיָה חָפֵץ – הָיָה מַשְׁפִּיל.
וְכַאֲשֶׁר רָם לְבָבוֹ וְרוּחוֹ חָזְקָה לְהָזִיד, הוּרַד מִכִּסֵּא מַלְכוּתוֹ וִיקָרוֹ הֶעֱבִירוּ מִמֶּנּוּ.
וּמִבְּנֵי אָדָם גֹּרַשׁ, וְלִבּוֹ עִם הַחַיָּה שִׁוָּה, וְעִם עֲרוֹדִים מְדוֹרוֹ, עֵשֶׂב כִּשְׁוָרִים יַאֲכִילוּהוּ, וּמִטַּל הַשָּׂמִים גּוּפוֹ יִטָּבֵל, עַד אֲשֶׁר יָדַע כִּי שׁוֹלֵט הָאֵל הָעֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וְאֶת אֲשֶׁר יַחְפֹץ יָקִים עָלֶיהָ.
וְאַתָּה בְּנוֹ בֵּלְשַׁאצַּר, לֹא הִשְׁפַּלְתָּ לְבָבְךָ כָּל־עֻמַּת אֲשֶׁר כָּל זֶה יָדַעְתָּ.
וְעַל אֲדוֹנֵי הַשָּׁמַיִם הִתְרוֹמַמְתָּ, וְאֶת הַכֵּלִים שֶׁל בֵּיתוֹ הֵבִיאוּ לְפָנֶיךָ, וְאַתָּה וְשָׂרֶיךָ, נָשֶׁיךָ וּפִילַגְשֶׁיךָ יַיִן שׁוֹתִים בָּהֶם, וֵאלֹהֵי כֶסֶף וְזָהָב, נְחֹשֶׁת, בַּרְזֶל, עֵץ וָאֶבֶן, אֲשֶׁר אֵינָם רוֹאִים וְאֵינָם שׁוֹמְעִים וְאֵינָם יוֹדְעִים – שִׁבַּחְתָּ; וְאֶת הָאֱלֹהִים, אֲשֶׁר נִשְׁמָתְךָ בְיָדוֹ וְכָל דְּרָכֶיךָ לוֹ – לֹא פֵאַרְתָּ.
אֲזַי מִפָּנָיו נִשְׁלַח פַּס שֶׁל יָד, וּכְתָב זֶה רָשׁוּם.
וְזֶה הַכְּתָב אֲשֶׁר רָשׁוּם: מְנֵה מְנֵה תְּקֵל וּפַרְסִין.
זֶה פִּתְרוֹן הַדָּבָר: מְנֵה, מָנָה אֱלֹהִים מַלְכוּתְךָ וְהִסְגִּירָהּ.
תְּקֵל, נִשְׁקַלְתָּ בְּמֹאזְנַיִם וְנִמְצֵאתָ חָסֵר.
פְּרֵס, נִפְרְסָה מַלְכוּתְךָ וְנִתְּנָה לְמָדַי וּפָרָס.
אֲזַי צִוָּה בֵּלְשַׁאצַּר וְהִלְבִּישׁוּ אֶת דָּנִיֵּאל אַרְגָּמָן וְרָבִיד שֶׁל זָהָב עַל צַוָּארוֹ, וְהִכְרִיזוּ עָלָיו, שֶׁיִּהְיֶה שׁוֹלֵט שְׁלִישִׁי בַּמַּלְכוּת.
בּוֹ בַּלַּיְלָה הוּמַת בֵּלְשַׁאצַּר הַמֶּלֶךְ הַכַּשְׂדִּי.


Chapter 6

וְדָרְיָוֶשׁ הַמָּדִי קִבֵּל אֶת הַמַּלְכוּת כְּבֶן שָׁנִים שִׁשִּׁים וּשְׁתַּיִם.
טוֹב הָיָה לִפְנֵי דָרְיָוֶשׁ, וְהֵקִים עַל הַמַּלְכוּת אֶת הָאֲחַשְׁדַּרְפְּנִים מֵאָה וְעֶשְׂרִים, אֲשֶׁר יִהְיוּ בְּכָל הַמַּלְכוּת.
וּלְמַעְלָה מֵהֶם וְזִירִים שְׁלֹשָׁה, אֲשֶׁר דָּנִיֵּאל אֶחָד מֵהֶם, אֲשֶׁר יִהְיוּ הָאֲחַשְׁדַּרְפְּנִים הָאֵלֶּה נוֹתְנִים לָהֶם עֵצָה, וְהַמֶּלֶךְ לֹא יִהְיֶה נִזָּק.
אֲזַי דָּנִיֵּאל זֶה הָיָה גּוֹבֵר עַל הַוְּזִירִים וְהָאֲחַשְׁדַּרְפְּנִים, כָּל־עֻמַּת שֶׁרוּחַ יְתֵרָה בּוֹ, וְהַמֶּלֶךְ חָשַׁב לַהֲקִימוֹ עַל כָּל הַמַּלְכוּת.
אֲזַי הַוְּזִירִים וְהָאֲחַשְׁדַּרְפְּנִים הָיוּ מְבַקְּשִׁים עֲלִילָה לִמְצֹא לְדָנִיֵּאל מִצַּד הַמַּלְכוּת, וְכָל עֲלִילָה וּמָשְׁחָת אֵינָם יְכוֹלִים לִמְצֹא, כָּל־עֻמַּת שֶׁנֶּאֱמָן הוּא וְכָל מִשְׁגֶּה וּמָשְׁחָת לֹא נִמְצָא אֶצְלוֹ.
אֲזַי הָאֲנָשִׁים הָאֵלֶּה אוֹמְרִים, שֶׁלֹּא נִמְצָא לְדָנִיֵּאל זֶה כָּל עֲלִילָה, אֶלָּא אִם מָצָאנוּ אֶצְלוֹ בְּדָת אֱלֹהָיו.
אֲזַי הַוְּזִירִים וְהָאֲחַשְׁדַּרְפְּנִים הָאֵלֶּה רָגְשׁוּ אֶל הַמֶּלֶךְ, וְכֵן אוֹמְרִים לוֹ: דָּרְיָוֶשׁ הַמֶּלֶךְ, לְעוֹלָמִים חֱיֵה.
הִתְיָעֲצוּ כָּל וְזִירֵי הַמַּלְכוּת, הַסְּגָנִים וְהָאֲחַשְׁדַּרְפְּנִים, הַיּוֹעֲצִים וְהַפַּחוֹת לְהָקִים קִיּוּם הַמֶּלֶךְ וְלָתֵת תֹּקֶף לְאִסָּר, שֶׁכָּל אֲשֶׁר יְבַקֵּשׁ בַּקָּשָׁה מִכָּל אֱלוֹהַּ וְאָדָם עַד יָמִים שְׁלשִׁים חוּץ מִמְּךָ הַמֶּלֶךְ, יָשְלַךְ לְגֹב הָאֲרָיוֹת.
כָּעֵת הַמֶּלֶךְ תָּקִים אֶת הָאִסָּר וְתִרְשֹׁם אֶת הַכְּתָב, שֶׁלֹּא לְשַׁנּוֹת, כְּדָת מָדַי וּפָרַס אֲשֶׁר לֹא תַעֲבֹר.
כָּל־עֻמַּת זֶה הַמֶּלֶךְ דָּרְיָוֶשׁ רָשַׁם הַכְּתָב וְהָאִסָּר.
וְדָנִיֵּאל, כַּאֲשֶׁר יָדַע שֶׁנִּרְשַׁם הַכְּתָב, נִכְנַס לְבֵיתוֹ, וְחַלּוֹנוֹת פְּתוּחִים לוֹ בַּעֲלִיָּתוֹ נֶגֶד יְרוּשָׁלַיִם, וּזְמַנִּים שְׁלשָׁה בַּיּוֹם הוּא כוֹרֵעַ עַל בִּרְכָּיו, וּמִתְפַּלֵּל וּמוֹדֶה לִפְנֵי אֱלֹהָיו, כָּל־עֻמַּת שֶׁהָיָה עוֹשֶׂה מִלִּפְנֵי זֶה.
אֲזַי הָאֲנָשִׁים הָאֵלֶּה רָגְשׁוּ וּמָצְאוּ אֶת דָּנִיֵּאל, מְבַקֵּשׁ וּמִתְחַנֵּן לִפְנֵי אֱלֹהָיו.
אֲזַי קָרְבוּ וְאוֹמְרִים לִפְנֵי הַמֶּלֶךְ עַל אִסַּר הַמֶּלֶךְ: הֲלֹא אִסָּר רָשַׁמְתָּ, אֲשֶׁר כָּל אִישׁ שֶׁיְּבַקֵּשׁ מִכָּל אֱלוֹהַּ וְאָדָם עַד יָמִים שְׁלשִׁים, חוּץ מִמְּךָ הַמֶּלֶךְ, יָשְׁלָךְ לְגֹב הָאֲרָיוֹת; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: נָכוֹן הַדָּבָר כְּדָת מָדַי וּפָרַס אֲשֶׁר לֹא תַעֲבֹר.
אֲזַי עָנוּ וְאוֹמְרִים לִפְנֵי הַמֶּלֶךְ, שֶׁדָּנִיֵּאל אֲשֶׁר מִבְּנֵי גָּלוּת יְהוּדָה, לֹא שָׂם עָלֶיךָ הַמֶּלֶךְ לֵב וְעַל הָאִסָּר אֲשֶׁר רָשַׁמְתָּ, וּזְמַנִּים שְׁלשָׁה בַּיּוֹם מְבַקֵּשׁ בַּקָּשָׁתוֹ.
אֲזַי הַמֶּלֶךְ, כַּאֲשֶׁר אֶת הַדָּבָר שָׁמַע, מְאֹד רַע עָלָיו, וְעַל דָּנִיֵּאל שָׂם דַּעַת לְהַצִּילוֹ, וְעַד מְבוֹאֵי הַשֶּׁמֶשׁ הָיָה מִתְאַמֵּץ לְהַצִּילוֹ.
אֲזַי הָאֲנָשִׁים הָאֵלֶּה רָגְשׁוּ אֶל הַמֶּלֶךְ וְאוֹמְרִים לַמֶּלֶךְ: דַּע, הַמֶּלֶךְ, שֶׁדָּת לְמָדַי וּפָרַס, שֶׁכָּל אסר וְקִיּוּם שֶׁהַמֶּלֶךְ יָקִים אֵין לְשַׁנּוֹת.
אֲזַי הַמֶּלֶךְ צִוָּה וְהֵבִיאוּ אֶת דָּנִיֵּאל, וְהִשְׁלִיכוּ לְגֹב הָאֲרָיוֹת; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל: אֱלֹהֶיךָ אֲשֶׁר אַתָּה עוֹבֵד אוֹתוֹ תָּמִיד, הוּא יַצִּילֶךָ.
וְהוּבְאָה אֶבֶן אַחַת וְהוּשְׂמָה עַל פִּי הַגֹּב, וַחֲתָמָהּ הַמֶּלֶךְ בְּטַבַּעְתּוֹ וּבְטַבַּעַת שָׂרָיו, שֶׁלֹּא יְשַׁנֶּה דָּבָר בְּדָנִיֵּאל.
אֲזַי הָלַךְ הַמֶּלֶךְ לְהֵיכָלוֹ וְלָן בְּצוֹם, וְסַמִּים לֹא הִכְנִיס לְפָנָיו, וּשְׁנָתוֹ נָדְדָה עָלָיו.
אֲזַי הַמֶּלֶךְ עִם שַׁחַר יָקוּם בַּבֹּקֶר, וּבְהִתְבַּהֲלוּת אֶל גֹּב הָאֲרָיוֹת הָלַךְ.
וּבְקָרְבוֹ לַגֹּב אֶל דָּנִיֵּאל, בְּקוֹל עָצוּב זָעַק; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל: דָּנִיֵּאל עֶבֶד אֵל חַי, אֱלֹהֶיךָ אֲשֶׁר אַתָּה עוֹבֵד אוֹתוֹ תָּמִיד, הֲיָכוֹל לְהַצִּילְךָ מִן הָאֲרָיוֹת?
אֲזַי דָּנִיֵּאל עִם הַמֶּלֶךְ דִּבֵּר: הַמֶּלֶךְ לְעוֹלָמִים חֱיֵה!
אֱלֹהַי שָׁלַח מַלְאָכוֹ וְסָגַר אֶת פִּי הָאֲרָיוֹת וְלֹא חִבְּלוּנִי, כָּל־עֻמַּת שֶׁלְּפָנָיו זְכוּת נִמְצְאָה לִי, וְאַף לְפָנֶיךָ הַמֶּלֶךְ חִבּוּל לֹא עָשִׂיתִי.
אֲזַי הַמֶּלֶךְ מְאֹד יָטַב עָלָיו, וְאֶת דָּנִיֵּאל צִוָּה לְהַעֲלוֹת מִן הַגֹּב, וְהָעֳלָה דָנִיֵּאל מִן הַגֹּב, וְכָל חַבָּלָה לֹא נִמְצְאָה בוֹ, שֶׁהֶאֱמִין בֵּאלֹהָיו.
וְצִוָּה הַמֶּלֶךְ, וְהֵבִיאוּ אֶת הָאֲנָשִׁים הָאֵלֶּה, אֲשֶׁר הִלְשִׁינוּ עַל דָּנִיֵּאל, וּלְגֹב הָאֲרָיוֹת הִשְׁלִיכוּ אוֹתָם, אֶת בְּנֵיהֶם וּנְשֵׁיהֶם, וְלֹא הִגִּיעוּ לְקַרְקֵעַ הַגֹּב, עַד אֲשֶׁר שָׁלְטוּ בָהֶם הָאֲרָיוֹת וְאֶת כָּל עַצְמוֹתֵיהֶם הֵדֵקּוּ.
אֲזַי דָּרְיָוֶשׁ הַמֶּלֶךְ כָּתַב לְכָל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת שֶׁדָּרִים בְּכָל הָאָרֶץ: שְׁלוֹמְכֶם יִרְבֶּה!
מִלְּפָנַי הוּשַׂם צַו, אֲשֶׁר בְּכָל שִׁלְטוֹן מַלְכוּתִי יִהְיוּ זָעִים וִירֵאִים מִלִּפְנֵי אֱלֹהָיו שֶׁל דָּנִיֵּאל, שֶׁהוּא אֵל חַי וְקַיָּם לְעוֹלָמִים, וּמַלְכוּתוֹ לֹא תִּשָּׁחֵת, וְשִׁלְטוֹנוֹ עַד הַסּוֹף.
מוֹשִׁיעַ וּמַצִּיל וְעוֹשֶׂה אוֹתוֹת וְנִפְלָאוֹת בַּשָּׁמַיִם וּבָאָרֶץ, אֲשֶׁר הִצִּיל אֶת דָּנִיֵּאל מִיַּד הָאֲרָיוֹת.
וְדָנִיֵּאל זֶה הִצְלִיחַ בְּמַלְכוּת דָּרְיָוֶשׁ וּבְמַלְכוּת כּוֹרֶשׁ הַפַּרְסִי.


Chapter 7

בִּשְׁנַת אַחַת לְבֵלְאשַׁצַּר מֶלֶךְ בָּבֶל, דָּנִיֵּאל חֲלוֹם רָאָה וְחֶזְיוֹנוֹת רֹאשׁוֹ עַל מִשְׁכָּבוֹ; אֲזַי אֶת הַחֲלוֹם כָּתַב, רֹאשׁ מִלִּים אָמַר.
עוֹנֶה דָּנִיֵּאל וְאוֹמֵר: רוֹאֶה הָיִיתִי בַּחֲזוֹנִי עִם לַיְלָה, וְהִנֵּה אַרְבַּע רוּחוֹת הַשָּׁמַיִם מְגִיחוֹת לַיָּם הַגָּדוֹל.
וְאַרְבַּע חַיּוֹת גְּדוֹלוֹת עוֹלוֹת מִן הַיָּם, שׁוֹנוֹת זוֹ מִזּוֹ.
הָרִאשׁוֹנָה כְּאַרְיֵה וּכְנָפַיִם שֶׁל נֶשֶׁר לָהּ, רוֹאֶה הָיִיתִי עַד אֲשֶׁר נִמְרְטוּ כְנָפֶיהָ וְנִשְּׂאָה מִן הָאָרֶץ, וְעַל רַגְלַיִם כְּאָדָם הוּקָמָה, וּלְבַב אָדָם נִתַּן לָהּ.
וְהִנֵּה חַיָּה אַחֶרֶת שְׁנִיָּה, דּוֹמָה לְדֹב, וּלְצַד אֶחָד הָעָמְדָה, וְשָׁלֹשׁ צְלָעוֹת בְּפִיהָ בֵּין שִׁנֶּיהָ, וְכֵן אוֹמְרִים לָהּ: קוּמִי אִכְלִי בָּשָׂר רָב.
אַחֲרֵי זֶה רוֹאֶה הָיִיתִי, וְהִנֵּה אַחֶרֶת כְּנָמֵר, וְלָהּ כְּנָפַיִם אַרְבַּע שֶׁל עוֹף עַל גַּבָּהּ, וְאַרְבָּעָה רָאשִׁים לַחַיָּה, וְשִׁלְטוֹן נִתַּן לָהּ.
אַחֲרֵי זֶה רוֹאֶה הָיִיתִי בְּחֶזְיוֹנוֹת לַיְלָה, וְהִנֵּה חַיָּה רְבִיעִית נוֹרָאָה וַאֲיֻמָּה וַחֲזָקָה יוֹתֵר, וְשִׁנַּיִם שֶׁל בַּרְזֶל לָהּ גְּדוֹלוֹת, אוֹכֶלֶת וּמְדִקָּה, וְהַשְּׁאָר בְּרַגְלָהּ רוֹפֶסֶת, וְהִיא מְשֻׁנָּה מִכָּל הַחַיּוֹת אֲשֶׁר לְפָנֶיהָ, וְקַרְנַיִם עֶשֶׂר לָהּ.
מִסְתַּכֵּל הָיִיתִי בַּקַּרְנַיִם, וְהִנֵּה קֶרֶן אַחֶרֶת זְעִירָה עָלְתָה בֵינֵיהֶן, וְשָׁלֹשׁ מִן הַקַּרְנַיִם הָרִאשׁוֹנוֹת נֶעֶקְרוּ מִפָּנֶיהָ, וְהִנֵּה עֵינַיִם כְּעֵינֵי אָדָם בְּקֶרֶן זוֹ, וּפֶה מְדַבֵּר גְּדוֹלוֹת.
רוֹאֶה הָיִיתִי עַד אֲשֶׁר כִּסְאוֹת הֻנְּחוּ וְעַתִּיק יָמִים יָשַׁב, לְבוּשׁוֹ כְּשֶׁלֶג לָבָן וּשְׂעַר רֹאשׁוֹ כְּצֶמֶר נָקִי, כִּסְאוֹ שְׁבִיבִים שֶׁל אֵשׁ, גַּלְגַּלָּיו אֵשׁ דּוֹלֶקֶת.
נָהָר שֶׁל אֵשׁ נִמְשָׁךְ וְיוֹצֵא מִלְּפָנָיו, אֶלֶף אֲלָפִים יְשַׁמְּשׁוּהוּ וְרִבּוֹ רְבָבוֹת לְפָנָיו יַעֲמֹדוּ, הַדִּין יָשַׁב וּסְפָרִים נִפְתָּחוּ.
רוֹאֶה הָיִיתִי, אֲזַי מִקּוֹל הַדְּבָרִים הַגְּדוֹלִים שֶׁל הַקֶּרֶן הַמְּדַבֶּרֶת, רוֹאֶה הָיִיתִי, עַד אֲשֶׁר הוּמְתָה הַחַיָּה וְהָאֳבַד גּוּפָהּ וְנִתְּנָה לִיקוֹד אֵשׁ.
וּשְׁאָר הַחַיּוֹת הָעָבְרוּ מִשִּׁלְטוֹנָן, וְאַרְכָּה בַחַיִּים נִתְּנָה לָהֶן עַד זְמָן וּמוֹעֵד.
רוֹאֶה הָיִיתִי בְּחֶזְיוֹנוֹת הַלַּיְלָה וְהִנֵּה עִם עַנְנֵי הַשָּׁמַיִם כְּבֶן אָדָם בָּא הָיָה, וְעַד עַתִּיק הַיָּמִים הִגִּיעַ, וּלְפָנָיו הִקְרִיבוּהוּ.
וְלוֹ נִתַּן שִׁלְטוֹן וִיקָר וּמַלְכוּת, וְכָל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת אוֹתוֹ יַעֲבֹדוּ, שִׁלְטוֹנוֹ שִׁלְטוֹן עוֹלָם אֲשֶׁר לֹא יַעֲבֹר וּמַלְכוּתוֹ – אֲשֶׁר לֹא תִּשָּׁחֵת.
נִפְעֲמָה רוּחִי, אֲנִי דָּנִיֵּאל, בְּתוֹךְ זֶה, וְחֶזְיוֹנוֹת רֹאשִׁי יְבַהֲלוּנִי.
קָרַבְתִּי אֶל אֶחָד הָעוֹמְדִים, וְאֶת הַנָּכוֹן אֲבַקֵּשׁ מִמֶּנּוּ עַל כָּל זֶה, וְאָמַר לִי, וּפִתְרוֹן הַדְּבָרִים יוֹדִיעֵנִי.
אֵלֶּה הַחַיּוֹת הַגְּדוֹלוֹת שֶׁהֵן אַרְבַּע, אַרְבָּעָה מְלָכִים יָקוּמוּ מִן הָאָרֶץ.
וִיקַבְּלוּ אֶת הַמַּלְכוּת קְדוֹשֵׁי עֶלְיוֹנִים, וְיִירְשׁוּ אֶת הַמַּלְכוּת עַד עוֹלָם וְעַד עוֹלָם עוֹלָמִים.
אֲזַי חָפַצְתִּי וַדָּאוּת עַל הַחַיָּה הָרְבִיעִית, שֶׁהָיְתָה שׁוֹנָה מִכֻּלָּן, נוֹרָאָה יוֹתֵר, שִׁנֶּיהָ שֶׁל בַּרְזֶל וְצִפָּרְנֶיהָ שֶׁל נְחֹשֶׁת, אוֹכֶלֶת מְדִקָּה וְהַשְּׁאָר בְּרַגְלָהּ רוֹפֶסֶת.
וְעַל עֶשֶׂר הַקַּרְנַיִם אֲשֶׁר בְּרֹאשָׁהּ, וְאַחֶרֶת אֲשֶׁר עָלְתָה, וְנָפְלוּ מִפָּנֶיהָ שָׁלֹשׁ, וְהַקֶּרֶן הַהִיא וְעֵינַיִם לָהּ, וּפֶה מְדַבֵּר גְּדוֹלוֹת, וּמַרְאֶהָ גָּדוֹל מֵחַבְרוֹתֶיהָ.
רוֹאֶה הָיִיתִי, וְהַקֶּרֶן הַהִיא עוֹשָׂה קְרָב עִם קְדוֹשִׁים, וִיכוֹלָה לָהֶם.
עַד אֲשֶׁר בָּא עַתִּיק הַיָּמִים, וְדִין נִתַּן לִקְדוֹשֵׁי עֶלְיוֹנִים, וְהַזְּמָן הִגִּיעַ, וְאֶת הַמַּלְכוּת יָרְשׁוּ קְדוֹשִׁים.
כֵּן אָמַר: הַחַיָּה הָרְבִיעִית – מַלְכוּת רְבִיעִית תִּהְיֶה בָּאָרֶץ, אֲשֶׁר תִּשְׁנֶה מִכָּל הַמַּלְכֻיּוֹת, וְתֹאכַל אֶת כָּל הָאָרֶץ, וּתְדוּשֶׁנָּה וּתְדִקֶּנָּה.
וְעֶשֶׂר הַקַּרְנַיִם מִמֶּנָּה, הַמַּלְכוּת, עֲשָׂרָה מְלָכִים יָקוּמוּ, וְאַחֵר יָקוּם אַחֲרֵיהֶם, וְהוּא יִשְׁנֶה מִן הָרִאשׁוֹנִים, וּשְׁלֹשֶׁת מְלָכִים יַשְׁפִּיל.
וּדְבָרִים לְצַד הָעֶלְיוֹן יְדַבֵּר, וְאֶת קְדוֹשֵׁי הָעֶלְיוֹנִים יְעַנֶּה וְיַחְשֹׁב לְשַׁנּוֹת זְמַנִּים וְדָת, וְיִנָּתְנוּ בְיָדוֹ עַד מוֹעֵד וּמוֹעֲדִים וַחֲצִי מוֹעֵד.
וְהַדִּין יֵשֵׁב, וְשִׁלְטוֹנוֹ יַעֲבִירוּ, לְהַשְׁמִיד וּלְהַאֲבִיד עַד הַסּוֹף.
וְהַמַּלְכוּת וְהַשִּׁלְטוֹן וְהַגְּדֻלָּה שֶׁל הַמַּלְכֻיּוֹת תַּחַת כָּל הַשָּׁמַיִם, נִתְּנוּ לְעַם קְדוֹשֵׁי עֶלְיוֹנִים, מַלְכוּתוֹ מַלְכוּת עוֹלָם, וְכָל הַשִּׁלְטוֹנוֹת לוֹ יַעַבְדוּ וְיִשָּׁמֵעוּ.
עַד כָּאן סוֹפוֹ שֶׁל הַדָּבָר; אֲנִי דָּנִיֵּאל, מְאֹד רַעְיוֹנַי יְבַהֲלוּנִי וְזִיוִי יִשְׁתַּנֶּה עָלַי, וְאֶת הַדָּבָר בְּלִבִּי שָׁמַרְתִּי.
