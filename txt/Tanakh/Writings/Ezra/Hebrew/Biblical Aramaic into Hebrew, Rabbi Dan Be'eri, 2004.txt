רְחוּם הַקָּצִין וְשִׁמְשַׁי הַסּוֹפֵר, כָּתְבוּ אִגֶּרֶת אַחַת עַל יְרוּשָׁלַיִם, לְאַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ כָּךְ.
אָז רְחוּם הַקָּצִין וְשִׁמְשַׁי הַסּוֹפֵר, וּשְׁאָר חַבְרֵיהֶם הַשּׁוֹפְטִים וְהַפְּקִידִים, הַטַּרְפַּלִיִּים הַפַּרְסִים הָאַרְכִּיִּים הַבַּבְלִים, הַשּׁוּשַׁנִּיִּים הַדֵּהַיִים הָעֵלָמִים.
וּשְׁאָר הָאֻמּוֹת אֲשֶׁר הֶגְלָה אָסְנַפַּר הַגָּדוֹל וְהַנִּכְבָּד, וְהוֹשִׁיב אוֹתָם בְּעִיר שֶׁל שׁוֹמְרוֹן וּשְׁאָר עֵבֶר הַנָּהָר, וְעַתָּה,
זֶה תַּקְצִיר הָאִגֶּרֶת, אֲשֶׁר שָׁלְחוּ אֵלָיו אֶל אַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ: עֲבָדֶיךָ אַנְשֵׁי עֵבֶר הַנָּהָר, וְעַתָּה,
יָדוּעַ יִהְיֶה לַמֶּלֶךְ, שֶׁהַיְּהוּדִים אֲשֶׁר עָלוּ מֵאֶצְלְךָ אֵלֵינוּ בָּאוּ, אֶת ירוּשָׁלַיִם, הָעִיר הַמּוֹרֶדֶת וְהָרָעָה, בּוֹנִים, וְהַחוֹמוֹת הִשְׁלִימוּ, וְהַיְּסוֹדוֹת יְחַבְּרוּ.
עַתָּה, יָדוּעַ יִהְיֶה לַמֶּלֶךְ, שֶׁאִם הָעִיר הַהִיא תִּבָּנֶה, וְהַחוֹמוֹת יֻשְׁלְמוּ, מִנְדָּה בְלוֹ וַהֲלָךְ לֹא יִתְּנוּ וְאוֹצַר מְלָכִים יִנָּזֵק.
עַתָּה, מִפְּנֵי שֶׁמֶּלַח הַהֵיכָל מָלַחְנוּ, וְעַרְוַת הַמֶּלֶךְ לֹא הָגוּן לָנוּ לִרְאוֹת, עַל זֶה שָׁלַחְנוּ וְהוֹדַעְנוּ לַמֶּלֶךְ,
אֲשֶׁר יִבדֹּק בְּסֵפֶר הַזִּכְרוֹנוֹת שֶׁל אֲבוֹתֶיךָ, וְתִמְצָא בְּסֵפֶר הַזִּכְרוֹנוֹת, וְתֵּדַע שֶׁהָעִיר הַהִיא עִיר מוֹרֶדֶת, וּמַזֶּקֶת מְלָכִים וּמְדִינוֹת, וּמְרִידוֹת עוֹשִׂים בְּתוֹכָהּ מִימוֹת הָעוֹלָם, עַל זֶה הָעִיר הַהִיא הָחְרְבָה.
מוֹדִיעִים אֲנַחְנוּ לַמֶּלֶךְ, שֶׁאִם הָעִיר הַהִיא תִּבָּנֶה, וְהַחוֹמוֹת יֻשְׁלְמוּ, מִפְּנֵי זֶה, חֵלֶק בְּעֵבֶר הַנָּהָר אֵין לְךָ.
הַדָּבָר שָׁלַח הַמֶּלֶךְ: אֶל רְחוּם הַקָּצִין וְשִׁמְשַׁי הַסּוֹפֵר וּשְׁאָר חַבְרֵיהֶם אֲשֶׁר יוֹשְׁבִים בְּשׁוֹמְרוֹן וּשְׁאָר עֵבֶר הַנָּהָר, שָׁלוֹם! וְעַתָּה,
הַמִּכְתָּב אֲשֶׁר שְׁלַחְתֶּם אֲלֵינוּ בִּשְׁלֵמוּת נִקְרָא לְפָנַי.
וּמִמֶּנִּי נִתַּן צַו וּבָדְקוּ וּמָצְאוּ, שֶׁהָעִיר הַהִיא מִימוֹת הָעוֹלָם עַל מְלָכִים מִתְנַשֵּׂאת, וּמֶרֶד וּמְרִידוֹת נֶעֱשֶׂה בָהּ.
וּמְלָכִים חֲזָקִים הָיוּ עַל יְרוּשָׁלַיִם, וּמוֹשְׁלִים בְּכֹל עֵבֶר הַנָּהָר, וּמִדָּה בְלוֹ וַהֲלָךְ נִתָּן לָהֶם.
עַתָּה, תְּנוּ צַו לְבַטֵּל הָאֲנָשִׁים הָהֵם, וְהָעִיר הַהִיא לֹא תִּבָּנֶה, עַד שֶׁמִּמֶּנִּי צַו יִנָּתֵן.
וּזְהִירִים הֱיוּ שְׁגָגָה לַעֲשׂוֹת עַל זֶה, לָמָּה יִגְדַּל הַנֶּזֶק לְהַזִּיק מְלָכִים!
אָז מִשֶּׁתַּקְצִיר הַמִּכְתָּב שֶׁל אַרְתַּחְשַׁשְׂתָּא הַמֶּלֶךְ נִקְרָא לִפְנֵי רְחוּם וְשִׁמְשַׁי הַסּוֹפֵר וְחַבְרֵיהֶם, הָלְכוּ בִּזְרִיזוּת לִירוּשָׁלַיִם אֶל הַיְּהוּדִים, וּבִטְּלוּ אוֹתָם בִּזְרוֹעַ וְכֹחַ.
אָז בָּטְלָה מְלֶאכֶת בֵּית הָאֵל אֲשֶׁר בִּירוּשָׁלַיִם, וְהָיְתָה בְּטֵלָה עַד שְׁנַת שְׁתַּיִם לְמַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ פָּרָס.

Chapter 5

וְנִבָּא חַגַּי הַנָּבִיא וּזְכַרְיָה בֶּן עִדּוֹא הַנְּבִיאִים עַל הַיְּהוּדִים אֲשֶׁר בִּיהוּדָה וּבִירוּשָׁלַיִם, בְּשֵׁם אֱלֹהֵי יִשְׂרָאֵל עֲלֵיהֶם.
אָז קָמוּ זְרֻבָּבֶל בֶּן שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בֶּן יוֹצָדָק, וְהִתְחִילוּ לִבְנוֹת בֵּית הָאֵל אֲשֶׁר בִּירוּשָׁלַיִם, וְעִמָּם הַנְּבִיאִים שֶׁל הָאֵל עוֹזְרִים לָהֶם.
בּוֹ בַזְּמָן, בָּא אֲלֵיהֶם תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, וּשְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, וְכָךְ אוֹמְרִים לָהֶם: מִי נָתַן לָכֶם צַו, הַבַּיִת הַזֶּה לְהִבָּנוֹת וְתָכְנִיתוֹ זוֹ לְהַשְׁלִים?
אָז כָּךְ אָמַרְנוּ לָהֶם: מִי הֵם שְׁמוֹת הָאֲנָשִׁים אֲשֶׁר זֶה הַבִּנְיָן בּוֹנִים?
וְעֵין אֱלֹהֵיהֶם הָיְתָה עַל זִקְנֵי הַיְּהוּדִים, וְלֹא בִטְּלוּ אוֹתָם, עַד [אֲשֶׁר] הַדִּין-וְחֶשׁבּוֹן לְדָרְיָוֶשׁ יֵלֵךְ, וְאָז יָשִׁיבוּ הַמִּכְתָּב עַל זֶה.
תַּקְצִיר הָאִגֶּרֶת אֲשֶׁר שָׁלַח תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, וּשְׁתַר בּוֹזְנַי וַחֲבֵרָיו הַפַּרְסִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, אֶל דָּרְיָוֶשׁ הַמֶּלֶךְ.
הַדָּבָר שָׁלְחוּ אֵלָיו וְכָךְ כָּתוּב בְּתוֹכוֹ: לְדָרְיָוֶשׁ הַמֶּלֶךְ, הַשָּׁלוֹם הַכֹּל!
יָדוּעַ יִהְיֶה לַמֶּלֶךְ שֶׁהָלַכְנוּ לִיהוּדָה הַמְּדִינָה, לְבֵית הָאֵל הַגָּדוֹל, וְהוּא נִבְנֶה אֶבֶן גְּלָל, וְעֵץ מוּשָׂם בַּקִּירוֹת, וְהַמְּלָאכָה הַהִיא מָהֵר נֶעֱשֵׂית, וּמַצְלִיחַ בִּידֵיהֶם.
אָז שָׁאַלְנוּ אֶת הַזְּקֵנִים הָהֵם, וְכַזֹּאת אָמַרְנוּ לָהֶם: מִי שָׂם לָכֶם צַו, הַבַּיִת הַזֶּה לִבְנוֹת, וְהַתָּכְנִית הַזֹּאת לְהַשְׁלִים?
וְאַף שְׁמוֹתֵיהֶם שָׁאַלְנוּ אוֹתָם לְהוֹדִיעֲךָ, אֲשֶׁר נִכְתֹּב שֵׁם הָאֲנָשִׁים אֲשֶׁר בְּרֹאשָׁם.
וְכַזֵּה הַדָּבָר הֵשִׁיבוּנוּ לֵאמֹר: אֲנַחְנוּ הֵם עֲבָדָיו שֶׁל אֱלוֹהַּ הַשָּׁמַיִם וְהָאָרֶץ, וּבוֹנִים הַבַּיִת אֲשֶׁר הָיָה בָּנוּי מִלִּפנֵי זֶה שָׁנִים רַבּוֹת, וּמֶלֶךְ לְיִשְׂרָאֵל גָּדוֹל בָּנָהוּ וְהִשׁלִימוֹ.
אֵלָּא, מִשֶּׁהִרְגִּיזוּ אֲבוֹתֵינוּ אֶת אֱלוֹהַּ הַשָּׁמַיִם, נָתַן אוֹתָם בְּיַד נְבוּכַדְנֶצַּר מֶלֶךְ בָּבֶל הַכַּשְׂדִּי, וְהַבַּיִת הַזֶּה הֶחֱרִיבוֹ, וְהָעָם הֶגְלָה לְבָבֶל.
אֲבָל בִּשְׁנַת אַחַת לְכוֹרֶשׁ, הַמֶּלֶךְ שֶׁל בָּבֶל, כּוֹרֶשׁ הַמֶּלֶךְ נָתַן צַו, בֵּית הָאֵל הַזֶּה לְהִבָּנוֹת.
וְאַף הַכֵּלִים שֶׁל בֵּית הָאֵל, שֶׁל הַזָּהָב וְהַכֶּסֶף, שֶׁנְּבוּכַדְנֶצַּר הוֹצִיא מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם וְהוֹבִיל אוֹתָם לַהֵיכָל שֶׁל בָּבֶל, הוֹצִיא אוֹתָם כּוֹרֶשׁ הַמֶּלֶךְ מִן הַהֵיכָל שֶׁל בָּבֶל, וּנְתָנָם לְשֵׁשְׁבַּצַּר שְׁמוֹ אֲשֶׁר פֶּחָה שָׂמוֹ,
וְאָמַר לוֹ: אֵלֶּה הַכֵּלִים שָׂא, לֵךְ הוֹרֵד אוֹתָם בַּהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם, וּבֵית הָאֵל יִבָּנֶה עַל מְקוֹמוֹ.
אָז שֵׁשְׁבַּצַּר זֶה, בָּא נָתַן אֶת הַיְּסוֹדוֹת שֶׁל בֵּית הָאֵל אֲשֶׁר בִּירוּשָׁלַיִם, וּמֵאָז וְעַד עַתָּה נִבְנֶה וְלֹא נִשְׁלָם.
וְעַתָּה אִם עַל הַמֶּלֶךְ טוֹב, יִבָּדֵק בְּבֵית הַגְּנָזִים שֶׁל הַמֶּלֶךְ שָׁם אֲשֶׁר בְּבָבֶל, אִם יֵשׁ שֶׁמִּכּוֹרֶשׁ הַמֶּלֶךְ נִתַּן צַו, לִבְנוֹת בֵּית הָאֵל הַזֶּה בִּירוּשָׁלַיִם, וּרְצוֹן הַמֶּלֶךְ עַל זֶה יִשְׁלַח אֵלֵינוּ.

Chapter 6

אָז דָּרְיָוֶשׁ הַמֶּלֶךְ נָתַן צַו, וּבָדְקוּ בְּבֵית הַסְּפָרִים שֶׁהַגְּנָזִים מֻנָּחִים שָׁם בְּבָבֶל.
וְנִמְצָא בְּאַחְמְתָא בַּבִּירָה אֲשֶׁר בְּמָדַי הַמְּדִינָה, מְגִלָּה אַחַת וְכָךְ כָּתוּב בְּתוֹכָהּ הַזִּכָּרוֹן:
בִּשְׁנַת אַחַת לְכוֹרֶשׁ הַמֶּלֶךְ, כּוֹרֶשׁ הַמֶּלֶךְ נָתַן צַו: בֵּית הָאֵל בִּירוּשָׁלַיִם, הַבַּיִת יִבָּנֶה, מָקוֹם אֲשֶׁר זוֹבְחִים זְבָחִים, וִיסוֹדוֹתָיו נִשָּׂאִים, גָּבְהוֹ אַמּוֹת שִׁשִּׁים, רָחְבּוֹ אַמּוֹת שִׁשִּׁים.
נִדְבָּכִים שֶׁל אֶבֶן גְּלָל שְׁלוֹשָׁה וְנִדְבְּךָ שֶׁל עֵץ אֶחָד, וְהַהוֹצָאָה מִבֵּית הַמֶּלֶךְ תִּנָּתֵן.
וְאַף כְּלֵי בֵית הָאֵל, שֶׁל הַזָּהָב וְהַכֶּסֶף, שֶׁנְּבוּכַדְנֶצַּר הוֹצִיא מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם וְהוֹבִיל לְבָבֶל, יָשִׁיבוּ וְיֵלֵךְ לַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם לִמְקוֹמוֹ, וְיוּנַח בְּבֵית הָאֵל.
עַתָּה, תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, שְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם הַפַּרְסִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, רְחוֹקִים הֱיוּ מִשָּׁם.
הַנִּיחוּ לִמְלֶאכֶת בֵּית הָאֵל הַזֶּה, פַּחַת הַיְּהוּדִים וּלְזִקְנֵי הַיְּהוּדִים, בֵּית הָאֵל הַזֶּה יִבְנוּ עַל מְקוֹמוֹ.
וּמִמֶּנִּי נִתַּן צַו, לְמַה שֶּׁתַּעֲשׂוּ עִם זִקְנֵי הַיְּהוּדִים הָהֵם, לִבְנוֹת בֵּית הָאֵל הַזֶּה, וּמִנִּכְסֵי הַמֶּלֶךְ אֲשֶׁר מִמַּס עֵבֶר הַנָּהָר, מָהֵר הַהוֹצָאָה תִּהְיֶה נִתֶּנֶת לָאֲנָשִׁים הָהֵם, שֶׁלֹּא לְבַטֵּל.
וּמַה שֶּׁצְּרִיכִים, וּבְנֵי שְׁוָרִים וְאֵילִים וּכְבָשִׂים לְעוֹלוֹת לֶאֱלֹהַּ הַשָּׁמַיִם, חִטִּים מֶלַח יַיִן וְשֶׁמֶן כְּמַאֲמַר הַכֹּהֲנִים אֲשֶׁר בִּירוּשָׁלַיִם, יִהְיֶה נִתָּן לָהֶם יוֹם בְּיוֹם שֶׁלֹּא בְּרַשְׁלָנוּת.
שֶׁיִּהְיוּ מַקְרִיבִים נִיחוֹחִים לֶאֱלֹהַּ הַשָּׁמַיִם, וּמִתְפַּלְּלִים לְחַיֵּי הַמֶּלֶךְ וּבָנָיו.
וּמִמֶּנִּי נִתַּן צַו, שֶׁכָּל אִישׁ אֲשֶׁר יְשַׁנֶּה הַדָּבָר הַזֶּה, יֵעָקֵר עֵץ מִבֵּיתוֹ וְזָקוּף יִתָּלֶה עָלָיו, וּבֵיתוֹ חָרְבָּה יֵעָשֶׂה עַל זֶה.
וְהָאֵל אֲשֶׁר שִׁכֵּן שְׁמוֹ שָׁם, יְמַגֵּר כָּל מֶלֶךְ וְעַם אֲשֶׁר יִשְׁלַח יָדוֹ לְשַׁנּוֹת, לְהַחֲרִיב בֵּית הָאֵל הַהוּא אֲשֶׁר בִּירוּשָׁלַיִם, אֲנִי דָרְיָוֶשׁ שַׂמְתִּי צַו, מָהֵר יֵעָשֶׂה.
אָז תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר שְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, מִכֵּיוָן שֶׁשָּׁלַח דָּרְיָוֶשׁ הַמֶּלֶךְ כַּזֹּאת, מָהֵר עָשׂוּ.
וְזִקְנֵי הַיְּהוּדִים בּוֹנִים וּמַצְלִיחִים, בִּנְבוּאַת חַגַּי הַנָּבִיא וּזְכַרְיָה בֶּן עִדּוֹא, וּבָנוּ וְהִשְׁלִימוּ מִצַּו אֱלוֹהַּ יִשְׂרָאֵל, וּמִצַּו כּוֹרֶשׁ וְדָרְיָוֶשׁ וְאַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרָס.
וְהִשְׁלִים הַבַּיִת הַזֶּה עַד יוֹם שְׁלֹשָׁה לְחֹדֶשׁ אֲדָר, אֲשֶׁר הִיא שְׁנַת שֵׁשׁ לְמַלְכוּת דָּרְיָוֶשׁ הַמֶּלֶךְ.
וְעָשׂוּ בְנֵי יִשְׂרָאֵל הַכֹּהֲנִים וְהַלְּוִיִּים וּשְׁאָר בְּנֵי הַגּוֹלָה, חֲנֻכַּת בֵּית הָאֵל הַזֶּה בְּחֶדְוָה.
וְהִקְרִיבוּ לַחֲנֻכַּת בֵּית הָאֵל הַזֶּה, שְׁוָרִים מְאָה, אֵילִים מָאתַיִם, כְּבָשִׂים אַרְבַּע מְאָה, וּצְפִירֵי עִזִּים לְחַטָּאוֹת עַל כָּל יִשְׂרָאֵל, שְׁנֵים עָשָׂר, לְמִנְיָן שִׁבְטֵי יִשְׂרָאֵל.
וְהֶעֱמִידוּ הַכֹּהֲנִים בְּמִשְׁמְרוֹתֵיהֶם וְהַלְּוִיִּים בְּמַחְלְקוֹתֵיהֶם, עַל מְלֶאכֶת הָאֵל אֲשֶׁר בִּירוּשָׁלַיִם כִּכְתַב סֵפֶר מֹשֶׁה.

Chapter 7











וְזֶה תַּקְצִיר הָאִגֶּרֶת, אֲשֶׁר נָתַן הַמֶּלֶךְ אַרְתַּחְשַׁסְתְּא לְעֶזְרָא הַכֹּהֵן הַסֹּפֵר, סֹפֵר דִּבְרֵי מִצְווֹת יי וְחֻקָּיו עַל יִשְׂרָאֵל.
אַרְתַּחְשַׁסְתְּא, מֶלֶךְ הַמְּלָכִים, לְעֶזְרָא הַכֹּהֵן סֹפֵר הַחֹק שֶׁל אֱלוֹהַּ הַשָּׁמַיִם וְגוֹמֵר, וְעַתָּה.
מִמֶּנִּי נִתַּן צַו אֲשֶׁר כָּל מִתְנַדֵּב בְּמַלְכוּתִי, מֵעַם יִשְׂרָאֵל וְכֹהֲנָיו וְהַלְּוִיִּים, לָלֶכֶת לִירוּשָׁלַיִם עִמְּךָ, יֵלֵךְ.
מִפְּנֵי שֶׁמִלִּפְנֵי הַמֶּלֶךְ וְשִׁבְעַת יוֹעֲצָיו, שָׁלוּחַ לְפַקֵּחַ עַל יְהוּדָה וְעַל יְרוּשְׁלֶם, בְּחֹק אֱלֹהֶיךָ אֲשֶׁר בְיָדְךָ.
וּלְהוֹבִיל כֶּסֶף וְזָהָב, אֲשֶׁר הַמֶּלֶךְ וְיוֹעֲצָיו הִתְנַדְּבוּ לֵאלֹהֵי יִשְׂרָאֵל, אֲשֶׁר בִּירוּשָׁלַיִם מִשְׁכָּנוֹ.
וְכָל כֶּסֶף וְזָהָב אֲשֶׁר תִּמְצָא בְּכֹל מְדִינַת בָּבֶל, עִם הִתְנַדְּבוּת הָעָם וְהַכֹּהֲנִים, מִתְנַדְּבִים לְבֵית אֱלֹהֵיהֶם אֲשֶׁר בִּירוּשָׁלַיִם.
מִשּׁוּם כָּךְ, מָהֵר תִּקְנֶה בַּכֶּסֶף הַזֶּה שְׁוָרִים, אֵילִים, כְּבָשִׂים וּמִנְחוֹתֵיהֶם וְנִסְכֵּיהֶם, וְתַקְרִיב אוֹתָם עַל הַמִזְבֵּחַ שֶׁל בֵּית אֱלֹהֵיכֶם, אֲשֶׁר בִּירוּשָׁלַיִם.
וּמַה שֶּׁעָלֶיךָ וְעַל אַחֶיךָ יִיטַב בִּשְׁאָר הַכֶּסֶף וְהַזָּהָב לַעֲשׂוֹת, כִּרְצוֹן אֱלֹהֵיכֶם תַּעֲשׂוּ.
וְהַכֵּלִים אֲשֶׁר נִתָּנִים לְךָ לַעֲבוֹדַת בֵּית אֱלֹהֶיךָ, הַשְׁלֵם לִפְנֵי אֱלוֹהַּ יְרוּשָׁלַיִם. 
וּשְׁאָר צָרְכֵי בֵּית אֱלֹהֶיךָ אֲשֶׁר יִפֹּל לְךָ לָתֵת, תִּתֵּן מִבֵּית גִּנְזֵי הַמֶּלֶךְ.
וּמִמֶּנִּי אֲנִי אַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ נִתַּן צַו, לְכֹל הַגִּזְבָּרִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, שֶׁכָּל אֲשֶׁר יִשְׁאָלְכֶם עֶזְרָא הַכֹּהֵן, סוֹפֵר הַחֹק שֶׁל אֱלוֹהַּ הַשָּׁמַיִם, מָהֵר יֵעָשֶׂה.
עַד כֶּסֶף כִּכָּרִים מֵאָה, וְעַד חִטִּים כֹּרִים מֵאָה, וְעַד יַיִן בַּתִּים מֵאָה, וְעַד בַּתִּים שֶׁמֶן מֵאָה, וּמֶלַח, שֶׁלֹּא כָּתוּב.
כָּל אֲשֶׁר מִצַּו אֱלוֹהַּ הַשָּׁמַיִם, יֵעָשֶׂה בִּזְרִיזוּת לְבֵית אֱלוֹהַּ הַשָּׁמַיִם, שֶׁלָּמָּה יִהְיֶה קֶצֶף עַל מַלְכוּת הַמֶּלֶךְ וּבָנָיו?
וְלָכֶם מוֹדִיעִים, שֶׁכָּל הַכֹּהֲנִים וְהַלְּוִיִּים, הַמְּשׁוֹרְרִים הַשּׁוֹעֲרִים הַנְּתִינִים וְעוֹבְדֵי בֵּית הָאֵל הַזֶּה, מִנְדָּה בְלוֹ וַהֲלָךְ אֵין סַמְכוּת לְהַטִּיל עֲלֵיהֶם.
וְאַתָּה עֶזְרָא, כְּחָכְמַת אֱלֹהֶיךָ אֲשֶׁר בְּיָדְךָ, מַנֵּה שׁוֹפְטִים וְדַיָּנִים, שֶׁיִּהְיוּ דָּנִים אֶת כָּל הָעָם אֲשֶׁר בְּעֵבֶר הַנָּהָר, לְכָל יוֹדְעֵי חֻקֵּי אֱלֹהֶיךָ; וַאֲשֶׁר לֹא יוֹדֵעַ, תּוֹדִיעוּ.
וְכָל שֶׁלֹּא יִהְיֶה עוֹשֶׂה הַחֹק שֶׁל אֱלֹהֶיךָ וְהַדִּין שֶׁל הַמֶּלֶךְ, מָהֵר הַדִּין יִהְיֶה נֶעֱשֶׂה מִמֶּנּוּ, אִם לְמָוֶת, אִם לְחֵרֶם, אִם לְעֹנֶשׁ נְכָסִים, וְלַאֲסוּרִים.
