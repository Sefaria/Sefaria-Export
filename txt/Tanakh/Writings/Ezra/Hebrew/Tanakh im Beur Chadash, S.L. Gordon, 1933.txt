רְחוּם בַּעַל־פְּקֻדָּה וְשִׁמְשַׁי הַסּוֹפֵר, כָּתְבוּ אִגֶּרֶת אַחַת עַל יְרוּשָׁלַיִם, לְאַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ, כֵּן.
אֲזַי רְחוּם בַּעַל־פְּקֻדָּה וְשִׁמְשַׁי הַסּוֹפֵר וּשְׁאָר חַבְרֵיהֶם, הַדִּינִים וְהַפַּרְסִים, הַנִּטְפָּלִים, הַפַּרְסִים, הָאָרִיִּים, הַבַּבְלִים, הַשּׁוּשַׁנִּים, שֶׁהוּא הָעֵלָמִים.
וּשְׁאָר הָאֻמּוֹת, אֲשֶׁר הֶגְלָה אָסְנַפַּר הַגָּדוֹל וְהַנִּכְבָּד, וְהוֹשִׁיב אוֹתָם בְּקִרְיָה שֶׁל שֹׁמְרוֹן, וּשְׁאָר עֵבֶר הַנָּהָר, וְכָעֵת.
זֶה הֶעְתֵּק הָאִגֶּרֶת, אֲשֶׁר שָׁלְחוּ אֵלָיו אֶל אַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ: עֲבָדֶיךָ אִישׁ עֵבֶר הַנָּהָר, וְכָעֵת.
יָדוּעַ יְהִי לַמֶּלֶךְ, שֶׁהַיְּהוּדִים אֲשֶׁר עָלוּ מֵאִתְּךָ אֵלֵינוּ, בָּאוּ לִירוּשָׁלַיִם, הַקִּרְיָה הַמּוֹרֶדֶת וְהָרָעָה בּוֹנִים, וְהַחוֹמוֹת הִשְׁלִימוּ וְהַיְּסוֹדוֹת יְחַזֵּקוּ.
כָּעֵת, יָדוּעַ יְהִי לַמֶּלֶךְ, שֶׁאִם קִרְיָה זוֹ תִּבָּנֶה וְהַחוֹמוֹת יֻשְׁלְמוּ, מַתָּת, מִנְחָה וּמַס לֹא יִתְּנוּ, וְלִבְסוֹף מְלָכִים תַּזִּיק.
כָּעֵת, כָּל־עֻמַּת שֶׁמֶּלַח הַהֵיכָל מָלַחְנוּ וְעֶרְוַת הַמֶּלֶךְ לֹא יָאֶה לָנוּ לִרְאוֹת, עַל זֹאת שָׁלַחְנוּ וְהוֹדַעְנוּ לַמֶּלֶךְ.
שֶׁיְּבַקֵּר בְּסֵפֶר הַזִּכְרוֹנוֹת שֶׁל אֲבוֹתֶיךָ, וְתִמְצָא בְּסֵפֶר הַזִּכְרוֹנוֹת וְתֵדַע, שֶׁקִּרְיָה זוֹ קִרְיָה מוֹרֶדֶת, וּמַזֶּקֶת מְלָכִים וּמְדִינוֹת, וְאַלִּימוּת עוֹשִׂים בְּתוֹכָהּ מִימוֹת עוֹלָם; עַל זֶה קִרְיָה זוֹ הָחֳרָבָה.
מוֹדִיעִים אֲנַחְנוּ לַמֶּלֶךְ, שֶׁאִם קִרְיָה זוֹ תִּבָּנֶה וְהַחוֹמוֹת יֻשְׁלְמוּ, לְעֻמַּת זֶה חֵלֶק בְּעֵבֶר הַנָּהָר אֵין לָךְ.
אֶת הַצַּו שָׁלַח הַמֶּלֶךְ אֶל רְחוּם בַּעַל־פְּקֻדָּה וְשִׁמְשַׁי הַסּוֹפֵר וּשְׁאָר חַבְרֵיהֶם, שֶׁיּוֹשְׁבִים בְּשֹׁמְרוֹן וּשְׁאָר עֵבֶר הַנָּהָר: שָׁלוֹם, וְכָעֵת.
הָאִגֶּרֶת אֲשֶׁר שְׁלַחְתֶּם אֵלֵינוּ, מְפֹרָשׁ נִקְרְאָה לְפָנַי.
וּמִמֶּנִּי הוּשַׂם צַו, וּבִקְּרוּ וּמָצְאוּ, שֶׁקִּרְיָה זוֹ מִימוֹת עוֹלָם עַל מְלָכִים מִתְנַשֵּׂאת, וּמֶרֶד וְאַלִּימוּת נַעֲשֶׂה בָּהּ.
וּמְלָכִים חֲזָקִים הָיוּ עַל יְרוּשָׁלַיִם, וְשׁוֹלְטִים בְּכָל עֵבֶר הַנָּהָר, וּמַתָּת, מִנְחָה וּמַס נִתָּן לָהֶם.
כָּעֵת, שִׂימוּ צַו לְבַטֵּל הָאֲנָשִׁים הָאֵלֶּה, וְקִרְיָה זוֹ לֹא תִּבָּנֶה, עַד מִמֶּנִּי צַו יוּשָׂם.
וּזְהִירִים הֱיוּ מֵעֲשׂוֹת מִשְׁגֶּה עַל זֶה, שֶׁמָּא יִרְבֶּה הַהֶפְסֵד לְהַזָּקַת מְלָכִים.
אֲזַי מִשֶּׁהֶעְתֵּק הָאִגֶּרֶת שֶׁל אַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ נִקְרָא לִפְנֵי רְחוּם וְשִׁמְשַׁי הַסּוֹפֵר וְחַבְרֵיהֶם, הָלְכוּ בִבְהִילוּת לִירוּשָׁלַיִם אֶל הַיְּהוּדִים, וּבִטְּלוּ אוֹתָם בִּזְרוֹעַ וָחָיִל.
אֲזַי בָּטְלָה עֲבוֹדַת בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, וְהָיְתָה בְּטֵלָה עַד שְׁנַת שְׁתַּיִם לְמַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ פָּרָס.


Chapter 5

וְנִבָּא חַגַּי הַנָּבִיא וּזְכַרְיָה בֶּן עִדּוֹא הַנְּבִיאִים, עַל הַיְּהוּדִים אֲשֶׁר בִּיהוּדָה וּבִירוּשָׁלַיִם, בְּשֵׁם אֱלֹהֵי יִשְׂרָאֵל אֲלֵיהֶם.
אֲזַי קָמוּ זְרֻבָּבֶל בֶּן שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בֶּן יְהוֹצָדָק, וְהֵחֵלּוּ לִבְנוֹת אֶת בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, וְעִמָּהֶם נְבִיאֵי הָאֱלֹהִים תּוֹמְכִים אוֹתָם.
בּוֹ בַּזְּמַן, בָּא אֲלֵיהֶם תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, וּשְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, וְכֵן אוֹמְרִים לָהֶם: מִי שָׂם לָכֶם צַו, בַּיִת זֶה לִבְנוֹת וְחוֹמָה זוֹ לְהַשְׁלִים?
אֲזַי כֵּן אָמַרְנוּ לָהֶם, מִי הֵם שְׁמוֹת אֲנָשִׁים, שֶׁבּוֹנִים בִּנְיָן זֶה.
וְעֵין אֱלֹהֵיהֶם הָיְתָה עַל זִקְנֵי הַיְּהוּדִים, וְלֹא בִּטְּלוּ אוֹתָם עַד שֶׁהַפְּקֻדָּה אֶל דָּרְיָוֶשׁ תֵּלֵךְ, וַאֲזַי יָשִׁיבוּ אִגֶּרֶת עַל זֶה.
הֶעְתֵּק הָאִגֶּרֶת, אֲשֶׁר שָׁלַח תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר וּשְׁתַר בּוֹזְנַי וַחֲבֵרָיו, הַפַּרְסִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, אֶל דָּרְיָוֶשׁ הַמֶּלֶךְ.
הַדָּבָר שָׁלְחוּ אֵלָיו, וְכָזֹאת כָּתוּב בְּתוֹכוֹ: לְדָרְיָוֶשׁ הַמֶּלֶךְ כָּל הַשָּׁלוֹם.
יָדוּעַ יְהִי לַמֶּלֶךְ, שֶׁהָלַכְנוּ לִיהוּדָה הַמְּדִינָה, לְבֵית הָאֵל הַגָּדוֹל, וְהוּא נִבְנֶה אֶבֶן גָּלָל וְעֵץ מוּשָׂם בַּכְּתָלִים, וַעֲבוֹדָה זוֹ בְּמֶרֶץ נַעֲשֵׂית וּמַצְלִיחָה בְּיָדָם.
אֲזַי שָׁאַלְנוּ לַזְּקֵנִים אֵלֶּה, כֵּן אָמַרְנוּ לָהֶם: מִי שָׂם לָכֶם צַו, בַּיִת זֶה לִבְנוֹת וְחוֹמָה זוֹ לְהַשְׁלִים?
וְאַף שְׁמוֹתֵיהֶם שָׁאַלְנוּ אוֹתָם לְהוֹדִיעֲךָ, שֶׁנִּכְתֹּב שֵׁם הָאֲנָשִׁים אֲשֶׁר בְּרָאשֵׁיהֶם.
וְכֵן הַדָּבָר הֱשִׁיבוּנוּ לֵאמֹר: אֲנַחְנוּ הֵם עֲבָדָיו שֶׁל אֱלֹהֵי הַשָּׁמַיִם וְהָאָרֶץ, וּבוֹנִים בַּיִת אֲשֶׁר הָיָה בָנוּי מִלִּפְנֵי זֶה שָׁנִים רַבּוֹת, וּמֶלֶךְ גָּדוֹל לְיִשְׂרָאֵל בָּנָהוּ וְהִשְׁלִימוֹ.
אֶלָּא מִשֶּׁהִכְעִיסוּ אֲבוֹתֵינוּ אֶת אֱלֹהֵי הַשָּׁמַיִם, נָתַן אוֹתָם בְּיַד נבוכדנצר מֶלֶךְ בָּבֶל הַכַּשְׂדִּי, וּבַיִת זֶה הֶחֱרִיבוֹ וְאֶת הָעָם הֶגְלָה לְבָבֶל.
אוּלָם בִּשְׁנַת אַחַת לְכוֹרֶשׁ מֶלֶךְ בָּבֶל, כּוֹרֶשׁ הַמֶּלֶךְ שָׂם צַו, בֵּית אֱלֹהִים זֶה לִבְנוֹת.
וְאַף הַכֵּלִים שֶׁל בֵּית הָאֱלֹהִים שֶׁל זָהָב וָכֶסֶף, אֲשֶׁר נְבוּכַדְנֶצַּר הוֹצִיא מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם וְהוֹבִיל אוֹתָם לַהֵיכָל שֶׁל בָּבֶל, הוֹצִיא אוֹתָם כּוֹרֶשׁ הַמֶּלֶךְ מִן הַהֵיכָל שֶׁל בָּבֶל, וְנִתְּנוּ לְשֵׁשְׁבַּצַּר שְׁמוֹ, אֲשֶׁר פֶּחָה שָׁמוֹ.
וְאָמַר לוֹ: אֵלֶּה הַכֵּלִים שָׂא, לֵךְ הוֹרֵד אוֹתָם בַּהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם, וּבֵית הָאֱלֹהִים יִבָּנֶה עַל מְקוֹמוֹ.
אֲזַי שֵׁשְׁבַּצַּר זֶה בָּא, נָתַן יְסוֹדוֹת שֶׁל בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, וּמֵאָז וְעַד כָּעֵת הוּא נִבְנֶה וְלֹא נִשְׁלָם.
וְכָעֵת, אִם עַל הַמֶּלֶךְ טוֹב, יְבֻקַּר בְּבֵית הַגְּנָזִים שֶׁל הַמֶּלֶךְ שָׁם אֲשֶׁר בְּבָבֶל, אִם יֵשׁ שֶׁמִּכּוֹרֶשׁ הַמֶּלֶךְ הוּשַׂם צַו לִבְנוֹת בֵּית אֱלֹהִים זֶה בִּירוּשָׁלַיִם, וּרְצוֹן הַמֶּלֶךְ עַל זֶה יִשְׁלַח אֵלֵינוּ.


Chapter 6

אֲזַי דָּרְיָוֶשׁ הַמֶּלֶךְ שָׂם צַו, וּבִקְּרוּ בְּבֵית הַסְּפָרִים, אֲשֶׁר אֶת הַגְּנָזִים מוֹרִידִים שָׁמָּה בְּבָבֶל.
וְנִמְצָא בְאַחְמְתָא, בַּבִּירָה אֲשֶׁר בְּמָדַי הַמְּדִינָה, מְגִלָּה אַחַת, וְכֵן כָּתוּב בְּתוֹכָהּ: הַזִּכָּרוֹן.
בִּשְׁנַת אַחַת לְכוֹרֶשׁ הַמֶּלֶךְ, כּוֹרֶשׁ הַמֶּלֶךְ שָׂם צַו: בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם הַבַּיִת יִבָּנֶה, מָקוֹם שֶׁזּוֹבְחִים זְבָחִים וִיסוֹדָיו מוּקָמִים, רוּמוֹ אַמּוֹת שִׁשִּׁים, רָחְבּוֹ אַמּוֹת שִׁשִּׁים.
נִדְבָּכִים שֶׁל אֶבֶן גָּלָל שְׁלשָׁה, וְנִדְבָּךְ שֶׁל עֵץ אֶחָד, וְהַהוֹצָאָה מִבֵּית הַמֶּלֶךְ תִּנָּתֵן.
וְאַף כְּלֵי בֵּית הָאֱלֹהִים שֶׁל זָהָב וָכֶסֶף, אֲשֶׁר נְבוּכַדְנֶצַּר הוֹצִיא מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם וְהוֹבִיל לְבָבֶל, יָשִׁיבוּ, וְיֵלֵךְ לַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם לִמְקוֹמוֹ, וְתוֹרִיד בְּבֵית הָאֱלֹהִים.
כָּעֵת, תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, שְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, הַפַּרְסִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, מְרֻחָקִים הֱיוּ מִשָּׁם.
הַנִּיחוּ לַעֲבוֹדַת בֵּית הָאֱלֹהִים הַזֶּה, פַּחַת הַיְּהוּדִים וּלְזִקְנֵי הַיְּהוּדִים, בֵּית אֱלֹהִים זֶה יִבְנוּ עַל מְקוֹמוֹ.
וּמִמֶּנִּי הוּשַׂם צַו, לְמַה שֶׁתַּעֲשׂוּ עִם זִקְנֵי הַיְּהוּדִים הָאֵלֶּה, לִבְנוֹת אֶת בֵּית הָאֱלֹהִים הַזֶּה, וּמִנִּכְסֵי הַמֶּלֶךְ שֶׁמַּתַּת עֵבֶר הַנָּהָר, בִּמְהֵרָה הַהוֹצָאָה תִּהְיֶה נִתֶּנֶת לָאֲנָשִׁים הָאֵלֶּה, שֶׁלֹּא לְבַטֵּל.
וּמַה שֶׁצָּרִיךְ, וּבְנֵי בָקָר וְאֵילִים וּכְבָשִׁים לְעוֹלוֹת לֵאלֹהֵי הַשָּׁמַיִם, חִטִּים, מֶלַח, יַיִן וָשֶׁמֶן, כְּמַאֲמַר הַכֹּהֲנִים אֲשֶׁר בִּירוּשָׁלַיִם, יְהִי נִתָּן לָהֶם יוֹם בְּיוֹם בְּלֹא מִשְׁגֶּה.
אֲשֶׁר יִהְיוּ מַקְרִיבִים נִיחוֹחִים לֵאלֹהֵי הַשָּׁמַיִם, וּמִתְפַּלְּלִים לְחַיֵּי הַמֶּלֶךְ וּבָנָיו.
וּמִמֶּנִּי הוּשַׂם צַו, שֶׁכָּל אִישׁ אֲשֶׁר יְשַׁנֶּה דָּבָר זֶה, יֻסַּח עֵץ מִבֵּיתוֹ, וְזָקוּף יֻכֶּה עָלָיו, וּבֵיתוֹ אַשְׁפָּה יֵעָשֶׂה עַל זֶה.
וְהָאֱלֹהִים, אֲשֶׁר שִׁכֵּן שְׁמוֹ שָׁם, יְמַגֵּר כָּל מֶלֶךְ וְעַם, אֲשֶׁר יִשְׁלַח יָדוֹ לְשַׁנּוֹת, לְחַבֵּל אֶת בֵּית הָאֱלֹהִים הַזֶּה אֲשֶׁר בִּירוּשָׁלַיִם; אֲנִי דָרְיָוֶשׁ נָתַתִּי צַו, מְהֵרָה יֵעָשֶׂה!
אֲזַי תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, שְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, לְעֻמַּת אֲשֶׁר שָׁלַח דָּרְיָוֶשׁ הַמֶּלֶךְ, כֵּן מְהֵרָה עָשׂוּ.
וְזִקְנֵי הַיְּהוּדִים בּוֹנִים וּמַצְלִיחִים, בִּנְבוּאַת חַגַּי הַנָּבִיא וּזְכַרְיָה בֶּן עִדּוֹא, וּבָנוּ וְתִקְּנוּ, בְּמִצְוַת אֱלֹהֵי יִשְׂרָאֵל, וּבְמִצְוַת כּוֹרֶשׁ וְדָרְיָוֶשׁ וְאַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרַס.
וְנִשְׁלַם בַּיִת זֶה עַד יוֹם שלשה לְחֹדֶשׁ אֲדָר, שֶׁהִיא שְׁנַת שֵׁשׁ לְמַלְכוּת דָּרְיָוֶשׁ הַמֶּלֶךְ.
וְעָשׂוּ בְנֵי יִשְׂרָאֵל, הַכֹּהֲנִים וְהַלְּוִיִּים וּשְׁאָר בְּנֵי הַגּוֹלָה, חֲנֻכַּת בֵּית הָאֱלֹהִים הַזֶּה בְּחֶדְוָה.
וְהִקְרִיבוּ לַחֲנֻכַּת בֵּית הָאֱלֹהִים הַזֶּה, פָּרִים מֵאָה, אֵילִים מָאתַיִם, כְּבָשִׁים אַרְבַּע מֵאוֹת, וּצְפִירֵי עִזִּים, לְחַטֵּא עַל כָּל יִשְׂרָאֵל, שְׁנֵים עָשָׂר, לְמִנְיַן שִׁבְטֵי יִשְׂרָאֵל.
וְהֵקִימוּ הַכֹּהֲנִים בִּפְלֻגּוֹתֵיהֶם, וְהַלְּוִיִּים בְּמַחְלְקוֹתֵיהֶם, עַל עֲבוֹדַת הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, כִּכְתָב סֵפֶר מֹשֶׁה.

Chapter 7












אַרְתַּחְשַׁסְתְּא מֶלֶךְ הַמְּלָכִים, לְעֶזְרָא הַכֹּהֵן סוֹפֵר הַדָּת שֶׁל אֱלֹהֵי הַשָּׁמַיִם, (שָׁלוֹם) גָּמוּר, וְכָעֵת.
מִמֶּנִּי הוּשַׂם צַו, שֶׁכָּל מִתְנַדֵּב בְּמַלְכוּתִי מֵעַם יִשְׂרָאֵל וְכֹהֲנָיו וְהַלְּוִיִּים לָלֶכֶת לִירוּשָׁלַיִם עִמְּךָ, יֵלֵךְ.
כָּל־עֻמַּת שֶׁמִּלִּפְנֵי הַמֶּלֶךְ וְשִׁבְעַת יוֹעֲצָיו נִשְׁלַח, לְבַקֵּר אֶל יְהוּדָה וְלִירוּשָׁלַיִם, בְּדָת אֱלֹהֶיךָ אֲשֶׁר בְּיָדֶךָ.
וּלְהוֹבִיל כֶּסֶף וְזָהָב, אֲשֶׁר הַמֶּלֶךְ וְיוֹעֲצָיו הִתְנַדְּבוּ לֶאֱלֹהֵי יִשְׂרָאֵל, אֲשֶׁר בִּירוּשָׁלַיִם מִשְׁכָּנוֹ.
וְכָל כֶּסֶף וְזָהָב, אֲשֶׁר תִּמְצָא בְּכָל מְדִינַת בָּבֶל, עִם הִתְנַדְּבוּת הָעָם וְהַכֹּהֲנִים, מִתְנַדְּבִים לְבֵית אֱלֹהֵיהֶם אֲשֶׁר בִּירוּשָׁלַיִם.
כָּל־עֻמַּת זֶה בְּמֶרֶץ תִּקְנֶה בְּכֶסֶף זֶה פָּרִים, אֵילִים, כְּבָשִׂים וּמִנְחוֹתֵיהֶם וְנִסְכֵּיהֶם, וְתַקְרִיב אוֹתָם עַל הַמִּזְבֵּחַ שֶׁל בֵּית אֱלֹהֵיכֶם שֶׁבִּירוּשָׁלַיִם.
וּמַה שֶׁעָלֶיךָ וְעַל אַחֶיךָ יִיטַב, בִּשְׁאָר הַכֶּסֶף וְהַזָּהָב לַעֲשׂוֹת, כִּרְצוֹן אֱלֹהֵיכֶם תַּעֲשׂוּ.
וְהַכֵּלִים, אֲשֶׁר נִתָּנִים לְךָ לַעֲבוֹדַת בֵּית אֱלֹהֶיךָ, מְסֹר לִפְנֵי אֱלֹהֵי יְרוּשָׁלַיִם.
וּשְׁאָר צֹרֶךְ בֵּית אֱלֹהֶיךָ, אֲשֶׁר יוּטַל עָלֶיךָ לָתֵת, תִּתֵּן מִבֵּית גִּנְזֵי הַמֶּלֶךְ.
וּמִמֶּנִּי אֲנִי אַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ, הוּשַׂם צַו לְכָל הַגִּזְבָּרִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, שֶׁכָּל אֲשֶׁר יִשְׁאַלְכֶם עֶזְרָא הַכֹּהֵן, סוֹפֵר הַדָּת שֶׁל אֱלֹהֵי הַשָּׁמַיִם, בְּמֶרֶץ יֵעָשֶׂה.
עַד כֶּסֶף כִּכָּרִים מֵאָה, וְעַד חִטִּים כּוֹרִים מֵאָה, וְעַד יַיִן בַּתִּים מֵאָה, וְעַד בַּתִּים שֶׁמֶן מֵאָה, וּמֶלַח לְלֹא כְתָב.
כָּל אֲשֶׁר מִמִּצְוַת אֱלֹהֵי הַשָּׁמַיִם, יֵעָשֶׂה אֶל נָכוֹן לְבֵית אֱלֹהֵי הַשָּׁמַיִם, שַׁלָּמָה יִהְיֶה קֶצֶף עַל מַלְכוּת הַמֶּלֶךְ וּבָנָיו.
וְלָכֶם מוֹדִיעִים, אֲשֶׁר כָּל הַכֹּהֲנִים וְהַלְּוִיִּים, הַמְּשׁוֹרְרִים, הַשּׁוֹעֲרִים, הַנְּתִינִים וְעוֹבְדֵי בֵּית הָאֱלֹהִים הַזֶּה, מַתָּת מִנְחָה וּמַס אֵין רַשַּׁאי לְהַטִּיל עֲלֵיהֶם.
וְאַתָּה עֶזְרָא, כְּחָכְמַת אֱלֹהֶיךָ אֲשֶׁר בְּיָדְךָ מַנֵּה שׁוֹפְטִים וְדַיָּנִים, אֲשֶׁר יִהְיוּ דָנִים אֶת כָּל הָעָם אֲשֶׁר בְּעֵבֶר הַנָּהָר, אֶת כָּל יוֹדְעֵי דָת אֱלֹהֶיךָ, וַאֲשֶׁר אֵינוֹ יוֹדֵעַ – תּוֹדִיעוֹ.
וְכָל אֲשֶׁר לֹא יִהְיֶה עוֹשֶׂה דָת אֱלֹהֶיךָ וְדָת הַמֶּלֶךְ, בְּמֶרֶץ הַדִּין יִהְיֶה נַעֲשֶׂה בּוֹ, אִם לְמָוֶת אִם לְגָלוּת, אִם לְעֹנֶשׁ כְּסָפִים וְלֶאֱסוּרִים.
