ספרא

ברייתא דרבי ישמעאל

א. רַבִּי יִשְׁמָעֵאל אוֹמֵר:
בִּשְׁלֹשׁ עֶשְׂרֵה מִדּוֹת הַתּוֹרָה נִדְרֶשֶׁת <br>(א) מִקַּל וְחֹמֶר <br>(ב) מִגְּזֵרָה שָׁוָה <br>(ג) מִבִּנְיַן אָב מִכָּתוּב אֶחָד, מִבִּנְיַן אָב מִשְּׁנֵי כְּתוּבִים <br>(ד) מִכְּלָל וּפְרָט <br>(ה) מִפְּרָט וּכְלָל <br>(ו) מִכְּלָל וּפְרָט וּכְלָל - אִי אַתָּה דָן אֶלָּא כְּעֵין הַפְּרָט <br>(ז) מִכְּלָל שֶׁהוּא צָרִיךְ לִפְרָט וּמִפְּרָט שֶׁהוּא צָרִיךְ לִכְלָל <br>ב. (ח) כָּל דָּבָר שֶׁהָיָה בַּכְּלָל וְיָצָא מִן הַכְּלָל לְלַמֵּד, לֹא לְלַמֵּד עַל עַצְמוֹ יָצָא, אֶלָּא לְלַמֵּד עַל הַכְּלָל כֻּלּוֹ יָצָא. <br>(ט) כָּל דָּבָר שֶׁהָיָה בַּכְּלָל, וְיָצָא מִן הַכְּלָל לִטָּעֵן טָעַן אַחֵר <b>שֶׁהוּא כְּעִנְיָנוֹ</b> - יָצָא לְהָקֵל וְלֹא לְהַחְמִיר. <br>(י) כָּל דָּבָר שֶׁהָיָה בַּכְּלָל, וְיָצָא מִן הַכְּלָל לִטָּעֵן טָעַן אַחֵר <b>שלא</b> כְּעִנְיָנוֹ - יָצָא לְהָקֵל וּלְהַחְמִיר. <br>(יא) כָּל דָּבָר שֶׁהָיָה בַּכְּלָל, וְיָצָא מִן הַכְּלָל לִדּוֹן בְּדָבָר חָדָשׁ, אִי אַתָּה יָכוֹל לְהַחֲזִירוֹ לִכְלָלוֹ עַד שֶׁיַּחְזִירֶנּוּ הַכָּתוּב לִכְלָלוֹ בְּפֵרוּשׁ. <br>(יב) דָּבָר הַלָּמֵד מֵעִנְיָנוֹ, וְדָבָר הַלָּמֵד מִסּוֹפוֹ. <br>(יג) וְכֵן שְׁנֵי כְּתוּבִין הַמַּכְחִישִׁין זֶה אֶת זֶה - עַד שֶׁיָּבֹא הַכָּתוּב הַשְּׁלִישִׁי וְיַכְרִיעַ בֵּינֵיהֶן.
ג. <b>מִקַּל וָחֹמֶר</b> – כֵּיצַד? <br>וַיֹּאמֶר יְיָ אֶל מֹשֶׁה: וְאָבִיהָ יָרֹק יָרַק בְּפָנֶיהָ הֲלֹא תִכָּלֵם שִׁבְעַת יָמִים? תִּסָּגֵר...! (בַּמִּדְבָּר יב יד) <br>'שִׁבְעַת יָמִים' - קַל וָחֹמֶר לַשְּׁכִינָה אַרְבָּעָה עָשָׂר יוֹם? <br>אֶלָּא: דַּיּוֹ לַבָּא מִן הַדִּין לִהְיוֹת כַּנִּדּוֹן - <br>תִּסָּגֵר מֵרִים <b>שְׂבֵעַת יָמִים</b> מִחוּץ לַמַּחֲנֶה - וְאַחַר תֵּאָסֵף.
ד. <b>מִגְּזֵרָה שָׁוָה</b> כֵּיצַד? <br>נֶאֱמַר בְּשׁוֹמֵר חִנָּם: "אִם לֹא שָׁלַח יָדוֹ בִּמְלֶאכֶת רֵעֵהוּ" (שְׁמוֹת כב י) <br>וְנֶאֱמַר בְּשׁוֹמֵר שָׂכָר: "אִם לֹא שָׁלַח יָדוֹ בִּמְלֶאכֶת רֵעֵהוּ." (שְׁמוֹת כב ז) <br>מָה בְּשׁוֹמֵר חִנָּם שֶׁנֶּאֱמַר בּוֹ "אִם לֹא שָׁלַח יָדוֹ" - פָּטַר אֶת הַיּוֹרְשִׁין <br>אַף בְּשׁוֹמֵר שָׂכָר שֶׁנֶּאֱמַר בּוֹ "אִם לֹא שָׁלַח יָדוֹ" - יִפְטֹר בּוֹ אֶת הַיּוֹרְשִׁים.
ה. <b>מִבִּנְיַן אָב מִכָּתוּב אֶחָד</b> - כֵּיצַד? <br>לֹא הַמִּשְׁכָּב כַּהֲרֵי הַמּוֹשָׁב - וְלֹא הַמּוֹשָׁב כַּהֲרֵי הַמִּשְׁכָּב <br>הַצַּד הַשָּׁוֶה שֶׁבָּהֶן: שֶׁהֵן כֵּלִים, עֲשׂוּיִין לָנוּחַ אָדָם לְבַדּוֹ, <br>וְזָב - מְטַמֵּא אוֹתוֹ בְּרֻבּוֹ: לְטַמֵּא אָדָם בְּמַגָּע וּכְמַשָּׂא, וּלְטַמֵּא בְּגָדִים <br>אַף כֵּלִים - שֶׁהֵן עֲשׂוּיִין לָנוּחַ אָדָם לְבַדּוֹ - <br>יְהֵא הַזָּב מְטַמֵּא אוֹתוֹ בְּרֻבּוֹ: לְטַמֵּא אָדָם בְּמַגָּע וּבְמַשָּׂא, וּלְטַמֵּא כֵּלִים! <br>יָצָא הַמֶּרְכָּב - שֶׁהוּא עָשׂוּי לְסִבְלוֹן אַחֵר.
ו. <b>וּמִבִּנְיַן אָב מִשְּׁנֵי כְּתוּבִים</b> – כֵּיצַד? <br>לֹא פָּרָשַׁת הַנֵּרוֹת כַּהֲרֵי פָּרָשַׁת שִׁלּוּחַ טְמֵאִים <br>וְלָא פָּרָשַׁת שִׁלּוּחַ טְמֵאִים כַּהֲרֵי פָּרָשַׁת הַנֵּרוֹת <br>הַצַּד הַשָּׁוֶה שֶׁבָּהֶן שֶׁהֵם ב"צו" - מִיָּד וּלְדוֹרוֹת. <br>אַף כָּל דָּבָר שֶׁהוּא ב"צו" - יְהֵא מִיָּד וּלְדוֹרוֹת.
ז. <b>מִכְּלָל וּפְרָט</b> – כֵּיצַד? <br>"מִן הַבְּהֵמָה" (וְיִקְרָא א ב) - כְּלָל <br>"מִן הַבָּקָר וּמִן הַצֹּאן" - פְּרָט. <br>'כְּלָל וּפְרָט - אֵין בַּכְּלָל אֶלָּא מַה שֶׁבַּפְּרָט!
ח. <b>מִפְּרָט וּכְלָל</b> - כֵּיצַד? <br>"כִּי יִתֵּן אִישׁ אֶל רֵעֵהוּ - חֲמוֹר אוֹ שׁוֹר אוֹ שֶׂה" (שְׁמוֹת כב ט) - פְּרָט, <br>"...וְכָל בְּהֵמָה לִשְׁמֹר" - כְּלָל. <br>'פְּרָט וּכְלָל' נַעֲשָׂה כְּלָל - מוּסָף עַל הַפְּרָט.
ט. <b>מִכְּלָל וּפְרָט וּכְלָל</b> - כֵּיצַד? <br>"וְנָתַתָּ הַכֶּסֶף בְּכֹל אֲשֶׁר תְּאַוֶּה נַפְשְׁךָ" (דְּבָרִים יד כו) - כְּלָל <br>"בַּבָּקָר וּבַצֹּאן בְּיַיִן וּבַשֵּׁכָר" - פְּרָט <br>"וּבְכֹל אֲשֶׁר תִּשְׁאָלְךָ נַפְשֶׁךָ" - חָזַר וְכָלַל. <br>'כְּלָל וּפְרָט וּכְלָל' - אִי אַתָּה דָן אֶלָּא כְּעֵין הַפְּרָט: <br>לוֹמַר לְךָ - מֶה הַפְּרָט מְפֹרָשׁ: דָּבָר שֶׁהוּא וָלָד - וְלָדוֹת הָאָרֶץ וְגִדּוּלֵי קַרְקַע, <br>אַף אֵין לִי אֶלָּא 'כָּל דָּבָר', שֶׁהוּא וָלָד - וְלָדוֹת הָאָרֶץ וְגִדּוּלֵי קַרְקַע. <br>יָצְאוּ כְּמֵהִין וּפִטְרִיּוֹת.
י. <b>מִכְּלָל שֶׁהוּא צָרִיךְ לִפְרָט, וּמִפְּרָט שֶׁהוּא צָרִיךְ לִכְלָל</b> - כֵּיצַד? <br>"קַדֶּשׁ לִי כָל בְּכוֹר" (שְׁמוֹת יג ב) - יָכוֹל אַף נְקֵבָה בַּמַּשְׁמָע? ת"ל "זָכָר" (דְּבָרִים טו יט) <br>אִי 'זָכָר' - יָכוֹל אֲפִלּוּ יָצָאת לְפָנָיו? תַּלְמוּד לוֹמַר "פֶּטֶר רֶחֶם" (שְׁמוֹת לד יט). <br>אִי 'פֶּטֶר רֶחֶם' - יָכוֹל אֲפִלּוּ לְאַחַר יוֹצֵא דֹּפֶן? תַּלְמוּד לוֹמַר "בְּכוֹר". <br>-זֶהוּ כְּלָל הַצָּרִיךְ לַפְּרָט, וּפְרָט שֶׁהוּא צָרִיךְ לִכְלָל.
מִשְׁנָה א <br>א. <b>כָּל דָּבָר שֶׁהָיָה בַּכְּלָל וְיָצָא מִן הַכְּלָל לְלַמֵּד,</b> <br><b>לֹא לְלַמֵּד עַל עַצְמוֹ יָצָא - אֶלָּא לְלַמֵּד עַל הַכְּלָל כֻּלּוֹ יָצָא</b> - כֵּיצַד? <br>"וְהַנֶּפֶשׁ אֲשֶׁר תֹּאכַל בָּשָׂר מִזֶּבַח הַשְּׁלָמִים אֲשֶׁר לַיְיָ - וְטִמְּאַתּוּ עָלָיו, וְנִכְרְתָה הַנֶּפֶשׁ הַהִיא..." - (וַיִּקְרָא ז כ) <br>וַהֲלֹא שְׁלָמִים בִּכְלַל כָּל הַקֳּדָשִׁים הָיוּ! <br>דִּכְתִיב "זֹאת הַתּוֹרָה לְעוֹלָה וּלְמִנְחָה, לְחַטָּאת וּלְאָשָׁם, וּלְמִלּוּאִים וּלְזֶבַח הַשְּׁלָמִים". (וַיִּקְרָא ז לז) <br>וּכְשֶׁיָּצְאוּ מִן הַכְּלָל לְלַמֵּד - לֹא לְלַמֵּד עַל עַצְמָן יָצְאוּ, <br>אֶלָּא לְלַמֵּד עַל הַכְּלָל כֻּלּוֹ יָצְאוּ <br>לוֹמ' לְךָ: מֶה שְׁלָמִים מְיֻחָדִין - קָדָשִׁים שֶׁקְּדֻשָּׁתָן קְדֻשַּׁת מִזְבֵּחַ <br>אַף אֵין לִי אֶלָּא <b>כָּל דָּבָר</b> - שֶׁקְּדֻשָּׁתוֹ קְדֻשַּׁת מִזְבֵּחַ... <br>יָצְאוּ קָדְשֵׁי בֶּדֶק הַבַּיִת.
מִשְׁנָה ב <br>ב. <b>כָּל דָּבָר שֶׁהָיָה בַּכְּלָל, וְיָצָא מִן הַכְּלָל - לִטְעוֹן טָעַן אַחֵר שֶׁהוּא כְּעִנְיָנוֹ <br>יָצָא לְהָקֵל וְלֹא לְהַחְמִיר</b> - כֵּיצַד? <br>"וּבָשָׂר כִּי יִהְיֶה בוֹ בְעֹרוֹ שְׁחִין וְנִרְפָּא" (וַיִּקְרָא יג יח) <br>וּכְתִיב "אוֹ בָשָׂר כִּי יִהְיֶה בְעֹרוֹ מִכְוַת אֵשׁ..." (וְיִקְרָא יג כד) <br>וַהֲלֹא הַשְּׁחִין וְהַמִּכְוָה בִּכְלַל כָּל הַנְּגָעִים הָיוּ? <br>וּכְשֶׁיֵּצְאוּ מִן הַכְּלָל: לִטְעֹן טָעַן אֶחָד שֶׁהוּא כְּעִנְיָנוֹ - <br>יָצְאוּ לְהָקֵל וְלֹא לְהַחְמִיר <br>לְהָקֵל עֲלֵיהֶן - שֶׁלֹּא יָדוּנוּ בְּמִחְיָה, וְשֶׁלֹּא יָדוּנוּ אֶלָּא בַּשָּׁבוּעַ אֶחָד.
מִשְׁנָה ג <br>ג. <b>כָּל דָּבָר שֶׁהָיָה בַּכְּלָל - וְיָצָא מִן הַכְּלָל, לִטְעֹן טָעַן אֶחָד שֶׁלֹּא כְּעִנְיָנוֹ</b> <br><b>יָצָא לְהָקֵל וּלְהַחְמִיר</b> - כֵּיצַד? <br>"וְאִישׁ אוֹ אִשָּׁה כִּי יִהְיֶה בוֹ נָגַע בְּרֹאשׁ אוֹ בְזָקָן"... (וַיִּקְרָא יג כט) <br>וַהֲלֹא 'הָרֹאשׁ וְהַזָּקֵן' - בִּכְלַל 'עוֹר וּבָשָׂר' הָיוּ - <br>וּכְשֶׁיָּצְאוּ מִן הַכְּלָל: לִטָּעֵן טָעַן אֶחָד שֶׁלֹּא כְּעִנְיָנוֹ - <br>יָצְאוּ לְהָקֵל וּלְהַחְמִיר: <br>לְהָקֵל עֲלֵיהֶן: שֶׁלֹּא יָדוּנוּ כְּשֵׂעָר לָבָן <br>וּלְהַחֲמִיר עֲלֵיהֶן: שֶׁיָּדוּנוּ כְּשֵׂעָר צָהֹב.
מִשְׁנָה ד <br>ד. <b>כָּל דָּבָר שֶׁהָיָה בַּכְּלָל, וְיָצָא מִן הַכְּלָל לִדּוֹן בְּדָבָר הֶחָדָשׁ </b>- <br><b>אִי אַתָּה יָכוֹל לְהַחֲזִירוֹ לִכְלָלוֹ, עַד שֶׁיַּחְזִירֶנּוּ הַכָּתוּב לִכְלָלוֹ בְּפֵרוּשׁ</b> – כֵּיצַד? <br>"וְשָׁחַט אֶת הַכֶּבֶשׂ בִּמְקוֹם אֲשֶׁר יִשְׁחַט אֶת הַחַטָּאת וְאֶת הָעֹלָה בִּמְקוֹם הַקֹּדֶשׁ..." (וַיִּקְרָא יד יג) <br>שֶׁאֵין תַּלְמוּד לוֹמַר "כִּי כַּחַטָּאת הָאָשָׁם הוּא לַכֹּהֵן"? <br>אֶלָּא - לְפִי שֶׁיָּצָא לִדּוֹן בְּדָבָר חָדָשׁ - בְּבֹהֶן יָד וּבְבֹהֶן רֶגֶל, וּבְאֹזֶן יְמָנִית - <br>יָכוֹל לֹא יְהֵא טָעוּן מַתַּן דָּמִים לַמִּזְבֵּחַ? ת"ל: "כִּי כַּחַטָּאת - הָאָשָׁם הוּא לַכֹּהֵן"... <br>הֲרֵי הֶחֱזִירוֹ הַכָּתוּב לִכְלָלוֹ בְּפֵרוּשׁ. <br>לוֹמַר לְךָ: מָה חַטָּאת - טְעוּנָה מַתַּן מִזְבֵּחַ, אַף אָשָׁם - טָעוּן מַתַּן מִזְבֵּחַ.
מִשְׁנָה ה <br>ה. <b>דָּבָר הַלָּמֵד מֵעִנְיָנוֹ</b> - כֵּיצַד? <br>"וְאִישׁ כִּי יִמָּרֵט רֹאשׁוֹ - קֶרַח הוּא: טָהוֹר הוּא" (וְיִקְרָא יג מ) <br>יכָל - יְהֵא טָהוֹר 'מִכָּל טֻמְאָה'? תַּלְמוּד לוֹמֵד: "וְכִי יִהְיֶה בַקָּרַחַת אוֹ בַגַּבַּחַת נֶגַע לָבָן אֲדַמְדָּם"... <br>דָּבָר לָמֵד מֵעִנְיָנוֹ - שֶׁאֵינוֹ טָהוֹר מִכָּל טֻמְאָה, אֶלָּא מִטֻּמְאַת נְתָקִים בִּלְבַד.
מִשְׁנָה ו <br>ו. <b>וְדָבָר לָמֵד מִסּוֹפוֹ</b> - כֵּיצַד? <br>"וְנָתַתִּי נֶגַע צָרַעַת בְּבֵית אֶרֶץ אֲחֻזַּתְכֶם" (וַיִּקְרָא יד לד) <br>מַשְׁמַע – בַּיִת שֶׁיֵּשׁ בּוֹ אֲבָנִים וְעֵצִים וְעָפָר - מְטַמֵּא. <br>יָכֹל - אַף בַּיִת שֶׁאֵין אֲבָנִים וְעֵצִים וְעָפָר, שֶׁמְּטַמֵּא? <br>תַּלְמוּד לוֹמַר: "וְנָתַץ אֶת הַבַּיִת - אֶת אֲבָנָיו וְאֶת עֵצָיו וְאֵת כָּל עֲפַר הַבָּיִת" (וַיִּקְרָא יד מה) <br>דָּבָר לָמַד מִסּוֹפוֹ - שֶׁאֵין הַבַּיִת מְטַמֵּא, עַד שֶׁיְּהֵא בּוֹ אֲבָנִים וְעֵצִים וְעָפָר.
מִשְׁנָה ז <br>ז. <b>שְׁנֵי כְּתוּבִים הַמַּכְחִישִׁים זֶה אֶת זֶה, עַד שֶׁיָּבֹא הַשְּׁלִישִׁי וְיַכְרִיעַ בֵּינֵיהֶם</b> - כֵּיצַד? <br>כָּתוּב אֶחָד אוֹמֵר: "וַיֵּרֶד יְיָ עַל הַר סִינַי אֶל רֹאשׁ הָהָר" (שְׁמוֹת יט כ) <br>וְכָתוּב אֶחָד אוֹמֵר: "כִּי מִן הַשָּׁמַיִם דִּבַּרְתִּי עִמָּכֶם" (שְׁמוֹת כ יט) <br>הִכְרִיעַ הַשְּׁלִישִׁי: "מִן הַשָּׁמַיִם הִשְׁמִיעֲךָ אֶת קֹלוֹ לְיַסְּרֶךָּ <br>וְעַל הָאָרֶץ הֶרְאֲךָ אֶת אִשּׁוֹ הַגְּדוֹלָה" (דְּבָרִים ד לוֹ) <br>מְלַמֵּד שֶׁהִרְכִּין הַקָּדוֹשׁ בָּרוּךְ הוּא שְׁמֵי שָׁמַיִם הָעֶלְיוֹנִים עַל הַר סִינַי - וְדִבֵּר עִמָּהֶם. <br>וְכֵן אָמַר דָּוִד בְּסֵפֶר תִּלִּים: " וַיַּט שָׁמַיִם וְיָרַד, וַעֲרָפֶל תַּחַת רַגְלָיו". (תְּהִלִּים יח י) <br>הִלֵּל הַזָּקֵן דָּרַשׁ שֶׁבַע מִדּוֹת לִפְנֵי זִקְנֵי בְּתִירָה <br>• (א) קַל וָחֹמֶר * (ב) וּגְזֵרָה שָׁוָה * (ג) וּבִנְיַן אָב * (ד) וּשְׁנֵי כְּתוּבִים * (ה) וּכְלָל וּפְרָט * <br>* (ו) וּכְיוֹצֵא בוֹ בְּמָקוֹם אַחֵר * (ז) וְדָבָר לָמֵד מֵעִנְיָנוֹ * <br>אֵלּוּ שֶׁבַע מִדּוֹת שֶׁדָּרַשׁ הִלֵּל הַזָּקֵן לִפְנֵי זִקְנֵי בְּתִירָה.
מִשְׁנָה ח <br>ח. כָּתוּב אֶחָד אוֹמֵר "וּבְבֹא מֹשֶׁה אֶל אֹהֶל מוֹעֵד לְדַבֵּר אִתּוֹ", (בַּמִּדְבָּר ז פט) <br>וְכָתוּב אֶחָד אוֹמֵר "וְלֹא יָכוֹל מֹשֶׁה לָבֹא אֶל אֹהֶל מוֹעֵד"! (שֵׁמוֹת מ לה) <br>הִכְרִיעַ "כִּי שָׁכַן עָלָיו הֶעָנָן". <br>אֱמֹר מֵעַתָּה - כָּל זְמַן שֶׁהָיָה הֶעָנָן שָׁם, לֹא הָיָה מֹשֶׁה נִכְנָס לְשָׁם; <br>נִסְתַּלֵּק הֶעָנָן - הָיָה נִכְנָס וּמְדַבֵּר עִמּוֹ.... <br>רַבִּי יוֹסֵי הַגְּלִילִי אוֹמֵר: <br>הֲרֵי הוּא אוֹמֵר: "וְלֹא יָכְלוּ הַכֹּהֲנִים לַעֲמֹד לְשָׁרֵת מִפְּנֵי הֶעָנָן, כִּי מָלֵא כְבוֹד יְיָ אֶת בֵּית יְיָ"; (מְלָכִים א ח יא) <br>מְלַמֵּד שֶׁנָּתְנָה רְשׁוּת לַמַּלְאָכִים לְחַבֵּל. <br>וְכָךְ הַפָּסוּק אוֹמֵר "וְשַׂכֹּתִי כַפִּי עָלֶיךָ עַד עָבְרִי" (שְׁמוֹת לג כב) - מְלַמֵּד שֶׁנָּתְנָה רְשׁוּת לַמַּלְאָכִים לְחַבֵּל. <br>וְכָךְ הַפָּסוּק אוֹמֵר "אֲשֶׁר נִשְׁבַּעְתִּי בְאַפִּי אִם יְבֹאוּן אֶל מְנוּחָתִי" (תְּהִלִּים עה יא) <br>כְּשֶׁיָּשׁוּב אַפִּי - יְבוֹאוּן אֶל מְנוּחָתִי.
