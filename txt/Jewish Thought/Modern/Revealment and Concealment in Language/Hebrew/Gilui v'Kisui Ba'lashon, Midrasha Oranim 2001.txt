גילוי וכיסוי בלשון

בְּנֵי־אָדָם זוֹרִים בְּכָל יוֹם לָרוּחַ, בְּכַוָּנָה וּלְפִי תֻּמָם, מִלִּים חֳמָרִים חֳמָרִים,
אוֹתָן וְאֶת צֵרוּפֵיהֶן הַשּׁוֹנִים, וְרַק מוּעָטִים מֵהֶם יוֹדְעִים אוֹ מַעֲלִים עַל לֵב,
מֶה הָיוּ הַמִּלִּים הָהֵן בִּימֵי גְּבוּרָתָן.
כַּמָּה מֵאוֹתָן הַמִּלִּים לֹא בָּאוּ לָעוֹלָם אֶלָּא אַחֲרֵי חֶבְלֵי לֵידָה קָשִׁים וּמְמֻשָּׁכִים
שֶׁל דּוֹרוֹת הַרְבֵּה; כַּמָּה מֵהֶן הִבְהִיקוּ כַּבְּרָקִים פִּתְאֹם וּבְטִיסָה אַחַת הֵאִירוּ
עוֹלָם מָלֵא; דֶּרֶךְ כַּמָּה מֵהֶן נִמְשְׁכוּ וְעָבְרוּ מַחֲנוֹת מַחֲנוֹת שֶׁל נְשָׁמוֹת חַיּוֹת,
נְשָׁמָה הוֹלֶכֶת, נְשָׁמָה בָּאָה, וְכָל אַחַת הִנִּיחָה אַחֲרֶיהָ צֵל וָרֵיחַ; וְכַמָּה מֵהֶן
שִׁמְּשׁוּ נַרְתִּיקִין לְמֶכָנִיסְמוּס דַּק וּמֻרְכָּב מְאֹד שֶׁל מַחֲשָׁבוֹת עֲמֻקּוֹת וְהַרְגָּשׁוֹת
נַעֲלוֹת בְּצֵרוּפֵיהֶן וּבְצֵרוּפֵי־צֵרוּפֵיהֶן הַנִּפְלָאִים בְּיוֹתֵר.
יֵשׁ מִלִּים – הַרְרֵי אֵל, וּמִלִּים – תְּהוֹם רַבָּה. יֵשׁ שֶׁבְּמִלָּה קְטַנָּה אַחַת נִגְנְזָה
כָּל תַּמְצִית חַיֶּיהָ, כָּל הַשְׁאָרַת נַפְשָׁהּ שֶׁל שִׁיטָה פִילוֹסוֹפִית עֲמֻקָּה, סַךְ־
הַכֹּל שֶׁל חֶשְׁבּוֹן עוֹלָם שָׁלֵם. יֵשׁ מִלָּה שֶׁהִכְרִיעָה בִּשְׁעָתָהּ עַמִּים וַאֲרָצוֹת,
מְלָכִים הֵקִימָה מִכִּסְאוֹתָם, מוֹסְדוֹת אֶרֶץ וְשָׁמַיִם הִרְגִּיזָה. וְהִנֵּה בָּא יוֹם
וְהַמִּלִּים הַהֵן יָרְדוּ מִגְּדֻלָּתָן וְהֻשְׁלְכוּ לַשּׁוּק, וְעַתָּה בְּנֵי־אָדָם מְגַלְגְּלִים בָּהֶן
מִתּוֹךְ שִׁיחָה קַלָּה כְּמִי שֶׁמְּגַלְּגֵל בָּעֲדָשִׁים.
הֲיֵשׁ לִתְמֹהַּ עַל כָּךְ? אֵין מְהַרְהֲרִים אַחֲרֵי מִדּוֹת הַטֶּבַע. כָּךְ מִנְהָגוֹ שֶׁל
עוֹלָם: מִלִּים עוֹלוֹת לִגְדֻלָּה וּמִלִּים יוֹרְדוֹת וְנַעֲשׂוֹת חֻלִּין. וּבְעִקָּר הַדָּבָר,
הֲרֵי אֵין לְךָ בַּלָּשׁוֹן מִלָּה קַלָּה שֶׁלֹּא הָיְתָה בִּשְׁעַת לֵדָתָהּ גִּלּוּי נַפְשִׁי עָצוּם
וְנוֹרָא, נִצָּחוֹן גָּדוֹל וְנִשְׂגָּב שֶׁל הָרוּחַ. בְּשָׁעָה שֶׁנִּדְהָם, לְמָשָׁל, אָדָם הָרִאשׁוֹן
מִקּוֹל הָרַעַם – "קוֹל ה' בַּכֹּחַ, קוֹל ה' בֶּהָדָר" – וּבְנָפְלוֹ עַל פָּנָיו, מֻכֵּה
תִּמָּהוֹן וַאֲחוּז חֶרְדַּת אֱלֹהִים, פָּרְצָה אָז מִפִּיו מֵאֵלֶיהָ – נֶאֱמַר, בְּחִקּוּי לְקוֹל
הַטֶּבַע – מִין הֲבָרָה פִּרְאִית, מֵעֵין שַׁאֲגַת חַיָּה, קְרוֹבָה לְקוֹל "רְר" – זֶה
הַנִּמְצָא בִּשְׁמוֹת הָרַעַם שֶׁל הַרְבֵּה לְשׁוֹנוֹת – הַאִם לֹא הֵבִיא קְרִיאָה
פִּרְאִית זוֹ גְּאֻלָּה גְּדוֹלָה לְנַפְשׁוֹ הַנִּדְהֶמֶת? הַאִם בִּקְרִיאָה זוֹ, הֵד נֶפֶשׁ
שֶׁנִּזְדַּעְזְעָה לְכָל מְעַמַקֵי מְצוּלוֹתֶיהָ, נִתְגַּלָּה מִגְּבוּרַת הַיְּצִירָה, מֵחֶרְדָּתָהּ
וּמִצַּהֲלַת נִצְחוֹנָהּ בְּשִׁעוּר פָּחוֹת מֵאֵלֶּה הַמִּתְגַּלּוֹת בְּ"דִבּוּר מֻצְלָח" בַּעַל
תֹּכֶן נִשְׂגָּב, שֶׁעָלָה בְּיַד אַחַד הַחוֹזִים הַגְּדוֹלִים בִּשְׁעַת עֲלִיַּת הַנְשָׁמָה? הַאִם
בְּאוֹתָהּ הֲבָרָה קַלָּה, גַּרְגֵּר הַמִּלָּה הָעֲתִידָה, לֹא נִכְרְכָה מַסֶּכֶת פְּלָאִים שֶׁל
הַרְגָּשׁוֹת <b>בְּרֵאשִׁית</b>, הַרְגָּשׁוֹת עַזּוֹת בְּחִדּוּשָׁן וּכַבִּירוֹת בִּפְרָאוּתָן, מֵעֵין חֲרָדָה,
פַּחַד, הִשְׁתּוֹמְמוּת, הַכְנָעָה, הִתְפַּעֲלוּת, הִתְעוֹרְרוּת לַעֲמֹד עַל נַפְשׁוֹ וְעוֹד
וְעוֹד כָּאֵלֶּה? וְאִם כֵּן הוּא, הַאִם לֹא הָיָה אָדָם הָרִאשׁוֹן בְּרֶגַע זֶה אַף הוּא
צַיָּר וְחוֹזֶה נִשְׂגָּב, בּוֹרֵא עַל־פִּי אִינְטוּאִיצְיָה נִיב שְׂפָתַיִם, וְנִיב נֶאֱמָן מְאֹד
– עַל־כָּל־פָּנִים לְגַבֵּי עַצְמוֹ – לְזַעְזוּעִים נַפְשִׁיִּים עֲמֻקִּים וּמְסֻבָּכִים? וְכַמָּה
מִן הַפִילוֹסוֹפְיָה הָעֲמֻקָּה, מִן הַגִּלּוּי הָאֱלֹהִי – כְּמוֹ שֶׁכְּבָר הֵעִיר עַל זֶה חָכָם
אֶחָד – הָיָה בַּדִּבּוּר הַקַּל "אֲנִי" שֶׁהוֹצִיא אָדָם הָרִאשׁוֹן מִפִּיו?
וְאַף־עַל־פִּי־כֵן, הִנֵּה עֵינֵינוּ הָרוֹאוֹת שֶׁמִּלִּים אֵלֶּה, וְרַבּוֹת אֲחֵרוֹת כַּיּוֹצֵא
בָּהֶן, עוֹמְדוֹת מֻבְלָעוֹת בַּלָּשׁוֹן – וְלֹא כְלוּם. הַנֶּפֶשׁ כִּמְעַט שֶׁאֵינָהּ נִפְגַּעַת
מֵהֶן. תּוֹכָן נֶאֱכַל וְכֹחָן הַנַּפְשִׁי פָּג אוֹ <b>נִגְנָז</b> וְרַק קְלִפּוֹתֵיהֶן, שֶׁהֻשְׁלְכוּ מֵרְשׁוּת
הַיָּחִיד לִרְשׁוּת הָרַבִּים, עֲדַיִן עוֹמְדוֹת בַּלָּשׁוֹן וּמְשַׁמְּשׁוֹת אוֹתָנוּ שִׁמּוּשׁ
רֶפְלֶקְטִיבִי וּמְרֻשָּׁל בִּגְבוּלוֹת מְצֻמְצָמִים שֶׁל הִגָּיוֹן וּמַשָּׁא־וּמַתָּן חֶבְרָתִי בְּתוֹר
סַמָּנִים <b>חִיצוֹנִים</b> וְהַפְשָׁטוֹת לַדְּבָרִים וְלַמַּרְאוֹת. הַדָּבָר הִגִּיעַ לִידֵי כָּךְ שֶׁלְּשׁוֹן
הָאָדָם נֶעֶשְׂתָה כִּשְׁתֵּי לְשׁוֹנוֹת, שֶׁנִּבְנוֹת זוֹ מְחֻרְבָּן זוֹ: לָשׁוֹן פְּנִימִית, לְשׁוֹן
<b>הַיִּחוּד</b> וְהַנֶּפֶשׁ, שֶׁהָעִקָּר בָּהּ, כְּמוֹ בְּמוּסִיקָה, הַ"אֵיךְ", לְחוּד – בִּמְדוֹר
הַשִּׁירָה, וּלְשׁוֹן חִיצוֹנִית, לְשׁוֹן הַהַפְשָׁטָה וְהַהַכְלָלָה, שֶׁהָעִקַּר בָּהּ, כְּמוֹ
בְּמָתֵמָטִיקָה, הַ"מַּה", לְחוּד – בִּמְחִיצַת הַהִגָּיוֹן. וּמִי יוֹדֵעַ, אוּלַי כָּךְ יָפֶה
לָאָדָם, שֶׁיְּהֵא יוֹרֵשׁ אֶת קְלִפַּת הַמִּלָּה בְּלֹא תּוֹכָהּ, כְּדֵי שֶׁיְּהֵא מְמַלְּאָהּ, אוֹ
מוֹסִיף בָּהּ, כָּל פַּעַם מִכֹּחוֹ שֶׁלּוֹ וּמַזְרִיחַ בָּהּ מְאוֹר נַפְשׁוֹ הוּא. רוֹצֶה אָדָם
בְּקַּב שֶׁלּוֹ, וְאִלּוּ עָמְדָה הַמִּלָּה הַדִּבּוּרִית כָּל יָמֶיהָ בִּמְלוֹא כּוֹחָהּ וְזָהֳרָה
הָרִאשׁוֹן, אִלּוּ הָיְתָה מְלַוָּתָהּ תָּמִיד כָּל אוֹתָהּ הַפָּמַלְיָא שֶׁל הַרְגָּשׁוֹת
וּמַחֲשָׁבוֹת שֶׁנִּקְשְׁרוּ בָּהּ בִּימֵי גְּבוּרָתָהּ, אֶפְשָׁר שֶׁלֹּא הָיָה שׁוּם "חַי מְדַבֵּר"
מַגִּיעַ לְעוֹלָם לִידֵי גִּלּוּי עַצְמוּתוֹ וְאוֹר נַפְשׁוֹ שֶׁלּוֹ. סוֹף־סוֹף כְּלִי רֵיקָן מַחֲזִיק
וּמָלֵא אֵינוֹ מַחֲזִיק, וּמָה אִם הַמִּלָּה הָרֵיקָנִית מְשַׁעְבֶּדֶת, הַמְּלֵאָה לֹא כָּל־
שֶׁכֵּן.
עַל מָה יֵשׁ לִתְמֹהַּ? עַל אוֹתוֹ רֶגֶשׁ הַבִּטָּחוֹן וְעַל אוֹתָהּ קוֹרַת־הָרוּחַ שֶׁמְּלַוִּים
אֶת הָאָדָם בְּדִבּוּרוֹ, כְּאִלּוּ הוּא מַעֲבִיר בֶּאֱמֶת אֶת מַחֲשַׁבְתּוֹ אוֹ אֶת הַרְגָּשָׁתוֹ
הַמֻּבַּעַת עַל מֵי מְנוּחוֹת וְדֶרֶךְ גֶּשֶׁר שֶׁל בַּרְזֶל, וְהוּא אֵינוֹ מְשַׁעֵר כְּלָל, עַד
כַּמָּה מְרוֹפַף אוֹתוֹ הַגֶּשֶׁר שֶׁל מִלִּים, עַד כַּמָּה עֲמֻקָּה וַאֲפֵלָה הַתְּהוֹם
הַפְּתוּחָה תַּחְתָּיו וְעַד כַּמָּה יֵשׁ מִמַּעֲשֵׂה הַנֵּס בְּכָל פְּסִיעָה בְּשָׁלוֹם.
הֲרֵי הַדָּבָר בָּרוּר, שֶׁהַלָּשׁוֹן לְכָל צֵרוּפֶיהָ אֵינָהּ מַכְנִיסָה אוֹתָנוּ כְּלָל לִמְחִיצָתָם
הַפְּנִימִית, לְמַהוּתָם הַגְּמוּרָה שֶׁל דְּבָרִים, אֶלָּא אַדְּרַבָּה, הִיא עַצְמָהּ חוֹצֶצֶת
בִּפְנֵיהֶם. מִחוּץ לִמְחִיצַת הַלָּשׁוֹן, מֵאֲחוֹרֵי הַפַּרְגּוֹד שֶׁלָּהּ, רוּחוֹ שֶׁל הָאָדָם
הַמְעֻרְטֶלֶת מִקְּלִפָּתָהּ הַדִּבּוּרִית, אֵינָהּ אֶלָּא תּוֹהָה וְתוֹהָה תָּמִיד. אֵין
אֹמֶר וְאֵין דְּבָרִים, אֶלָּא תְּהִיָּה עוֹלָמִית; "מָה" נִצְחִי קָפוּא עַל הַשְּׂפָתַיִם.
בֶּאֱמֶת אֵין מָקוֹם אֲפִלּוּ לְאוֹתוֹ "מָה", שֶּׁיֵּשׁ בְּמַשְׁמָעוֹ כְּבָר תִּקְוָה
לַתְּשׁוּבָה, אֲבָל מַה יֵּשׁ שָׁם? – "<b>בְּלִימָה</b> – בְּלוּם פֶּה מִלְּדַבֵּר". אִם בְּכָל
זֹאת הִגִּיעַ הָאָדָם לִידֵי דִּבּוּר וְדַעְתּוֹ מִתְקָרֶרֶת בּוֹ, אֵין זֶה אֶלָּא מִגֹּדֶל
פַּחְדוֹ לְהִשָּׁאֵר רֶגַע אֶחָד עִם אוֹתוֹ הַ"תֹּהוּ" הָאָפֵל, עִם אוֹתָהּ הַ"בְּלִימָה"
פָּנִים אֶל פָּנִים בְּלִי חֲצִיצָה. "כִּי לֹא יִרְאַנִי הָאָדָם וָחַי" – אוֹמֵר הַתֹּהוּ,
וְכָל דִּבּוּר, כָּל נִדְנוּד שֶׁל דִּבּוּר, הוּא כִּסּוּי מַשֶׁהוּ עַל קֻרְטוֹב שֶׁל "בְּלִימָה",
קְלִפָּה סוֹגֶרֶת בְּתוֹכָהּ טִפָּה אֲפֵלָה שֶׁל "תֵּיקוּ נִצְחִי". <b>שׁוּם מִלָּה אֵין בָּה</b>
<b>בִּטּוּלָהּ הַגָּמוּר שֶׁל שׁוּם שְׁאֵלָה</b>, אֲבָל מָה יֵּשׁ בָּהּ? – <b>כִּסּוּיָהּ</b>. אֵין נַפְקָא
מִינָּה <b>מָה הִיא</b> אוֹתָהּ הַמִּלָּה, הַחְלִיפוּהָ בְּאַחֶרֶת, וּבִלְבַד שֶׁיְּהֵא בָּהּ לְפִי
שָׁעָה כְּדֵי כִּסּוּי וַחֲצִיצָה. שְׁתֵּי אַחְיוֹת צוֹרְרוֹת, מִשְׁתֵּי קְצָווֹת מַקְבִּילוֹת,
הַמּוּסִיקָה הָאִלֶּמֶת וְהַמָּתֵמָטִיקָה הַמְּסַמֶּנֶת מְעִידוֹת כְּאַחַת עַל הַמִּלָּה,
שֶׁהִיא "לַאו דַּוְקָא", שֶׁאֵינָהּ אֶלָּא מִין רִקְמַת תֹּהוּ; אֶלָּא כְּשֵׁם שֶׁהַגּוּפִים
נַעֲשִׂים מֻרְגָּשִׁים לָעַיִן וּמְסֻיָּמִים בִּמְצָרֵיהֶם עַל־יְדֵי שֶׁהֵם חוֹצְצִים בִּמְקוֹמָם
בִּפְנֵי הָאוֹר, כָּךְ הַמִּלָּה מְקַבֶּלֶת יֵשׁוּתָהּ דַּוְקָא עַל־יְדֵי כָּךְ, שֶׁהִיא סוֹתֶמֶת
כְּנֶגְדָּהּ נֶקֶב קָטָן מֵאוֹתוֹ הַתֹּהוּ וְחוֹצֶצֶת בִּפְנֵי אֲפֵלָתוֹ, שֶׁלֹּא תְּהֵא זוֹ מְפַעְפַּעַת
וְיוֹצֵאת עָלֶיהָ לְטַשְׁטֵשׁ אֶת תְּחוּמֶיהָ. הַיּוֹשֵׁב יְחִידִי בֶּאֱשׁוּן חֹשֶׁךְ וַאֲפֵלָה
וּמְרַתֵּת, מַשְׁמִיעַ קוֹלוֹ לְאָזְנָיו: קוֹרֵא אֶת "שְׁמַע" אוֹ מְצַפְצֵף בִּשְׂפָתָיו.
לָמָּה? <b>"סְגֻלָּה" הִיא לְהַסִּיחַ דַּעְתּוֹ וּלְהָפִיג פַּחְדּוֹ.</b> וְכָךְ הַמִּלָּה הַדִּבּוּרִית –
אוֹ שִׁטָּה שְׁלֵמָה שֶׁל מִלִּים – כֹּחָהּ אַף הִיא לֹא בְּתָכְנָהּ הַמְפֹרָשׁ – אִם בִּכְלָל
יֵשׁ כָּזֶה – אֶלָּא בְּהֶסַּח־הַדַּעַת הַכָּרוּךְ בָּהּ. הַעְלָמַת הָעַיִן הִיא סוֹף־סוֹף
הַמִּקְלָט הַקַּל וְהַנּוּחַ בַּיּוֹתֵר, אַף אִם מְדֻמֶּה, מִפְּנֵי הַסַּכָּנָה, וּבְמָקוֹם שֶׁפְּקִיחַת
הָעֵינַיִם הִיא גּוּפָהּ הַסַּכָּנָה, הֲרֵי אֵין לְךָ גַּם מִקְלָט בָּטוּחַ הֵימֶנָּה, וְ"יָפֶה
עָשָׂה מֹשֶׁה שֶׁהִסְתִּיר פָּנָיו". וּמִי יוֹדֵעַ, אוּלַי בֶּאֱמֶת הַדִּבּוּר שֶׁמִּבְּרֵאשִׁית
נִזְרַק לֹא מִבֵּין שְׁנַיִם, בֵּין אָדָם לַחֲבֵרוֹ, בְּתוֹר כְּלֵי שָׁרֵת חֶבְרָתִי, מֵעֵין דִּבּוּר
שֶׁלֹּא לִשְׁמוֹ, אֶלָּא דַוְקָא מִפִּי הָאָדָם הַיּוֹשֵׁב יְחִידִי, בֵּין אָדָם לְעַצְמוֹ, בְּתוֹר
צוֹרֶך נַפְשִׁי, מֵעֵין דִּבּוּר לִשְׁמוֹ גְּרֵידָא, בְּחִינַת "בְּתוֹכִי יִשְׁתּוֹמֵם רוּחִי וְעִם
לְבָבִי אָשִׂיחָה"... לֹא נִתְקָרְרָה דַּעְתּוֹ שֶׁל אָדָם הָרִאשׁוֹן עַד שֶׁהִשְׁמִיעַ אֶת
<b>עַצְמוֹ</b> לְאָזְנָיו. וְאוּלָם אוֹתוֹ הַקּוֹל שֶׁמִּבְּרֵאשִׁית שֶׁהֶעְלָה אֶת הַכָּרַת הָאָדָם
מִמְּצוּלַת הַתֹּהוּ – הוּא עַצְמוֹ עָמַד פִּתְאֹם כְּקִיר מַבְדִּיל בֵּין הָאָדָם וּבֵין מַה
שֶׁ"מֵעֵבֶר הַלָּז", כְּמִי שֶׁאוֹמֵר: מֵעַתָּה, אָדָם, יִהְיוּ פָּנֶיךָ מְכֻוָּנִים רַק כְּלַפֵּי
מַה שֶׁ"מִכָּאן וְאֵילָךְ". <b>אַחֲרֶיךָ</b> אַל תַּבִּיט ובַ"מֻּפְלָא" אַל תָּצִיץ, וְאִם תָּצִיץ
– לֹא תּוֹעִיל, כִּי לֹא יִרְאֶה הָאָדָם אֶת הַ"תֹּהוּ" פָּנִים אֶל פָּנִים וָחַי. חֲלוֹם
שֶׁנִּשְׁכַּח – לֹא יִזָּכֵר עוֹד. אֶל הַ"תֹּהוּ" תְּשׁוּקָתְךָ, וְהַדִּבּוּר יִמְשֹׁל בְּךָ.
וּבֶאֱמֶת אֵין "הַדֵּעָה וְהַדִּבּוּר" שׁוֹלְטִים אֶלָּא בְּמַה שֶׁ"מִכָּאן וְאֵילָךְ" בְּד'
אַמּוֹת שֶׁל מָקוֹם וּזְמַן. אַךְ בְּצֶלֶם יִתְהַלֵּךְ אִישׁ, וּכְכָל אֲשֶׁר יִתְקָרֵב אֶל
הָאוֹר הַמְדֻמָּהּ <b>שֶׁלְּפָנָיו</b>, כֵּן יִגְדַּל הַצֵּל <b>מֵאַחֲרָיו</b> וְכֵן לֹא יִמְעַט הָאֹפֶל מִכָּל
סְבִיבוֹתָיו. "מִכָּאן וְאֵילָךְ" אוּלַי יֵשׁ לְיַשֵּׁב אֶת הַכֹּל: בְּדֹחַק אוֹ בִּרְוָחָה,
אֲבָל יֵשׁ לְיַשֵּׁב. הָעִקָּר, שֶׁלֹּא יִשָּׁאֵר חֲלַל הַ"דֵּעָה" שֶׁל הָאָדָם רֶגַע אֶחָד
פָּנוּי בְּלֹא מִלִּים צְפוּפוֹת וּרְצוּפוֹת זוֹ אֵצֶל זוֹ כְּקַשְׁקַשֵּׁי הַשִּׁרְיוֹן, לְלֹא רֶוַח
בֵּינְתַיִם כְּחוּט הַשַּׂעֲרָה. נֵר הַדֵּעָה וְהַדִּבּוּר – הַגַּחֶלֶת וְהַשַּׁלְהֶבֶת – נֵר תָּמִיד
לֹא יִכְבֶּה. וְאוּלָם אוֹתוֹ הַשֶּׁטַח שֶׁל "מִכָּאן וְאֵילֵךְ", הַשָּׁרוּי בִּתְחוּם הָאוֹר
הַמְדֻמֶּה – סוֹף־סוֹף בַּמֶּה נֶחְשָׁב הוּא לְעֻמַּת יָם הָאֵין־סוֹף שֶׁל אֹפֶל הָעוֹלָמִים
הַנִּשְׁאָר עֲדַיִן, וְשֶׁיִּשָּׁאֵר תָּמִיד, <b>מִבַּחוּץ</b>. וְסוֹף־סוֹף הֲלֹא דַּוְקָא אוֹתָהּ הָאֲפֵלָה
הַנִּצְחִית, זוֹ הַמַּטִּילָה אֵימָה כָּל־כָּךְ – הֲלֹא הִיא הִיא לְבַדָּהּ שֶׁמּוֹשֶׁכֶת
אֵלֶיהָ כָּל יְמֵי עוֹלָם אֶת לֵב הָאָדָם בַּסֵּתֶר וּמְעוֹרֶרֶת בּוֹ גַּעְגּוּעִים כְּמוּסִים
לְהָצִיץ בָּהּ רֶגַע קָטָן. הַכֹּל יְרֵאִים מִמֶּנָּה וְהַכֹּל נִמְשָׁכִים לָהּ. בְּמוֹ פִינוּ נִבְנֶה
עָלֶיהָ מְחִיצוֹת שֶׁל מִלִּים מִלִּים וְשִׁיטוֹת שִׁיטוֹת לְהַעֲלִימָהּ מִן הָעַיִן, וּמִיָּד הַצִּפָּרְנַיִם
חוֹטְטוֹת וְחוֹטְטוֹת בְּאוֹתָן הַמְּחִיצוֹת, כְּדֵי לִפְתֹּחַ בָּהֶן אֶשְׁנָב קָטָן, סֶדֶק
כָּל־שֶׁהוּא, וּלְהָצִיץ דֶּרֶךְ שָׁם רֶגַע אֶחָד אֶל מַה שֶׁ"מֵּעֵבֶר הַלָּז". אוּלָם
אֲהָהּ, שָׁוְא יְגִיעַת אָדָם! בּוֹ בְּרֶגַע שֶׁהַסֶּדֶק נִפְתַּח לִכְאוֹרָה – וְהִנֵּה חֲצִיצָה
אַחֶרֶת, בִּדְמוּת מִלָּה אוֹ שִׁיטָה חֲדָשָׁה, עוֹמֶדֶת פִּתְאוֹם בִּמְקוֹם הַיְּשָׁנָה
וְחוֹסֶמֶת שׁוּב אֶת הָעַיִן.
וְכָךְ אֵין לְדַבֵּר סוֹף עַד הָעוֹלָם. מִלָּה הוֹלֶכֶת וּמִלָּה בָּאָה, שִׁיטָה עוֹלָה
וְשִׁיטָה יוֹרֶדֶת – וְהַ"תֵּיקוּ" הַיָּשָׁן וְהַנִּצְחִי בְּעֵינוֹ עוֹמֵד, לֹא יִשְׁתַּנֶּה וְלֹא
יִתְמַעֵט. נְתִינַת שְׁטָר עַל חוֹב, אוֹ רְשִׁימָתוֹ בַּפִּנְקָס, אֵינֶנָּה עוֹד פִּרְעוֹן
הַחוֹב; פּוֹרֶקֶת הִיא רַק <b>לְפִי שָׁעָה</b> אֶת מַשָּׁאוֹ מֵעַל הַזִכָּרוֹן וְלֹא יוֹתֵר. וְהוּא
הַדִּין בַּדִבּוּר הַהֶחְלֵטִי, כְּלוֹמַר, בִּקְרִיאַת שֵׁם וּקְבִיעַת סְדָרִים וּסְיָגִים
לַמַרְאוֹת וּלְצֵרוּפֵיהֶם. שׁוּם תְּשׁוּבָה לִשְׁאֵלַת הַמַּהוּת אֵינָהּ מֻבְלַעַת וְעוֹמֶדֶת
מִמֵּילָא בְּדִבּוּר סְתָם. אֲפִלּוּ דִּבּוּר שֶׁל תְּשׁוּבָה בְּהֶדְיָא אַף הוּא אֵינוֹ בֶּאֱמֶת אֶלָּא
נֻסָּח אַחֵר שֶׁל הַשְּׁאֵלָה עַצְמָהּ, "בִּתְמִּיָה" מְתֻרְגָּם "בְּנִיחוּתָא", סִגְנוֹן
שֶׁל כִּסּוּי בִּמְקוֹם הַגִּלוּי. אִלּוּ בָּאנוּ לְהַפְשִׁיט אֶת הַ"תּוֹךְ" הַגָּמוּר, הָאַחֲרוֹן,
שֶׁל כָּל הַמִּילִים וְהַשִּׁיטוֹת, סוֹף־סוֹף, <b>אַחֲרֵי מִצּוּי אַחֲרוֹן</b>, לֹא יַעֲלֶה בְּיָדֵנוּ
אֶלָּא שָׁם אֶחָד הַכּוֹלֵל אֶת הַכֹּל, שֵׁם בֶּן ב' אוֹתִיּוֹת.
וְאֵיזֶהוּ? עוֹד הַפַּעַם אוֹתוֹ "מָה" הַנּוֹרָא, שֶׁמֵּאֲחוֹרָיו עוֹמֵד אֵיזֶה "אִיקְס"
עוֹד נוֹרָא מִמֶּנּוּ – הַ"בְּלִימָה", אֲבָל בּוֹחֵר הָאָדָם לְפוֹרֵר אֶת חוֹבוֹ לְפֵרוּרִים
קְטַנִּים, בְּתִקְווֹת שָׁוְא לְהָקֵל בְּכָךְ אֶת הַפֵּרָעוֹן, וּכְשֶׁאֵין הַתִּקְוָה מִתְקַיֶּמֶת,
הוּא מַחֲלִיף מִלָּה בְּמִלָּה, שִׁיטָּה בְּשִׁיטָּה, כְּלוֹמַר, נוֹתֵן שְׁטָר חָדָשׁ תַּחַת
הַיָּשָׁן וְדוֹחֶה אוֹ מַרְחִיב לְעַצְמוֹ אֶת זְמַן הַפֵּרָעוֹן. בֵּין כָּךְ וּבֵין כָּךְ – וְהַחוֹב
אֵינוֹ בָּא לִידֵי גּוּבַיְנָא לְעוֹלָם.
מִלָּה אוֹ שִׁיטָה יוֹרֶדֶת אֵפוֹא מִגְּדֻלָתָהּ וּמְפַנָּה מְקוֹמָהּ לַאֲחֶרֶת לֹא מִפְּנֵי
שֶׁהוּרַע כֹּחָהּ לְגַלּוֹת, לְהָאִיר, לְבַטֵּל אֶת הַ"תֵּיקוּ", מִקְצָתוֹ אוֹ כֻּלּוֹ, אֶלָּא
דַּוְקָא לְהֶפֶךְ: לְפִי שֶׁנִּשְׁחֲקָה הַמִּלָּה אוֹ הַשִּׁיטָה מֵרוֹב מִעוּךְ וּמִשְׁמוּשׁ וְחִטּוּט
וְאֵין בָּהּ עוֹד כְּדֵי כִּסּוּי וְהָעֲלָמָה יָפָה, וּמִמֵּילָא אֵין בָּהּ עוֹד כְּדֵי הֶסַּח הַדַּעַת
לְפִי־שָׁעָה. הָאָדָם מֵצִיץ רֶגַע בַּסֶּדֶק שֶׁנִּפְתַּח, וּלְפַחְדּוֹ הַגָּדוֹל מוֹצֵא וְהִנֵּה
שׁוּב הַ"תֹּהוּ" הָאָיֹם לְפָנָיו, וְהוּא קוֹפֵץ וְסוֹתֵם לְפִי־שָׁעָה אֶת הַסֶּדֶק –
בְּמִלָּה חֲדָשָׁה כְּלוֹמַר, אוֹחֵז בַּ"סְּגֻלָה" הַבְּדוּקָה לוֹ מִכְּבָר, מֵאֲחוֹתָהּ
הַקּוֹדֶמֶת, כְּמַסִּיחָה אֶת הַדַּעַת לְפִי שָׁעָה – וְהוּא נִצּוֹל מִן הַפַּחַד. וְאַל
תִּתְמַהּ! הַ"סְּגֻלָה" מוֹעֶלֶת לְמַאֲמִינִים בָּהּ, שֶׁהֲרֵי הָאֱמוּנָה עַצְמָהּ אַף הִיא
אֵינָהּ אֶלָּא הֶסַּח הַדַּעַת. הָאֵין דֻּגְמָא קְרוֹבָה לָזֶה בֵּין הַ"מְדַבְּרִים" עַצְמָם?
אָדָם, כָּל זְמַן שֶׁהוּא חַי וְשׁוֹאֵף וּמִתְנוֹעֵעַ וּפוֹעֵל – מְמַלֵּא חָלָל, וְהַכֹּל מוּבָן,
לִכְאוֹרָה; "הַכֹּל אַתֵי שַׁפִּיר". כָּל שֶׁטֶף הַחַיִּים, כָּל תָּכְנָם – אֵינָהּ אֶלָּא
הִשְׁתַּדְּלוּת רְצוּפָה, יְגִיעָה בִּלְתִּי פּוֹסֶקֶת שֶׁל הֶסַּח הַדַּעַת. כָּל רֶגַע שֶׁל
רְדִיפָה אַחֲרֵי... הוּא בְּאוֹתוֹ רֶגַע גַּם "פְּלֵיטָה מִן..." – וְזֶה, רַק זֶה, כָּל
שְׂכָרוֹ. שְׂכַר רְדִיפָה – פְּלֵיטָה. הָרוֹדֵף מוֹצֵא כָּל רֶגַע אֶת סִפּוּקוֹ לְפִי שָׁעָה
לֹא בְּמַה <b>שֶׁהִשִּׂיג</b> אֶלָּא בְּמַה <b>שֶׁנִּמְלַט</b>, וְהוּא הַנּוֹתֵן לוֹ צֵל שֶׁל קוֹרַת־רוּחַ
וּבִטָּחוֹן לְפִי־שָׁעָה. "כִּי מִי אֲשֶׁר יְחֻבַּר אֶל כׇּל הַחַיִּים יֵשׁ בִּטָּחוֹן". אַךְ הִנֵּה
מֵת הָאָדָם – וְהֶחָלָל נִתְרוֹקֵן. אֵין מַה שֶׁיַּסִּיחַ אֶת הַדַּעַת – וְהַמְּחִיצָה
נִסְתַּלְּקָה. הַכֹּל נַעֲשָׂה פִּתְאוֹם בִּלְתִּי מוּבָן. הַ"אִיקְס" הַנֶּעְלָם מִשְׁתַּרְבֵּב
לְפָנֵינוּ בְּכָל מוֹרָא קוֹמָתוֹ – וְאָנוּ יוֹשְׁבִים לְפָנָיו רֶגַע אֶחָד עַל הָאָרֶץ בָּאֲפֵלָה
אֲבֵלִים וְדוּמָם כָּאֶבֶן. אֲבָל רַק רֶגַע אֶחָד, לְפִי שֶׁשַּׂר שֶׁל חַיִּים מַקְדִּים
סְתִימָה לַפְּתִיחָה, וְהוּא מַזְמִין לָנוּ מִיָּד "סְגֻלָּה" חֲדָשָׁה לְהַסִּיחַ בָּהּ אֶת
הַדַּעַת וּלְהָפִיג אֶת הַפַּחַד, וְעַד שֶׁלֹּא נִסְתַּם הַגּוֹלֵל עַל הַמֵּת – וּכְבָר נִמְצָא
אוֹתוֹ הֶחָלָל שֶׁנִּתְרוֹקֵן סָתוּם שׁוּב בְּאַחַת מִן הַמִּלּוֹת, תְּהֵא זוֹ שֶׁל הֶסְפֵּד,
שֶׁל תַּנְחוּמִין, שֶׁל פִילוֹסוֹפְיָה אוֹ שֶׁל אֱמוּנָה בְּהַשְׁאָרַת הַנֶּפֶשׁ וְכַדּוֹמֶה.
הָרֶגַע הַמְסֻכָּן בְּיוֹתֵר – גַּם בַּדִּבּוּר וְגַם בַּחַיִּים – אֵינוֹ אֵפוֹא אֶלָּא זֶה שֶׁבֵּין
כִּסּוּי לְכִסּוּי, כְּשֶׁהַתֹּהוּ מְנַצְנֵץ, אֲבָל רְגָעִים כָּאֵלֶּה מְעַטִּים מְאֹד גַּם בְּשִׁגְרַת
הַלָּשׁוֹן וְגַם בְּשִׁגְרַת הַחַיִּים, וְעַל־פִּי רֹב בְּנֵי אָדָם מְדַלְּגִים עֲלֵיהֶם וְאֵינָם
מַרְגִּישִׁים. שׁוֹמֵר פְּתָאִים ה'.
וּמִכָּל הָאָמוּר יוֹצֵא הֶבְדֵּל גָּדוֹל שֶׁבֵּין לְשׁוֹן בַּעֲלֵי פְּרוֹזָה לִלְשׁוֹן בַּעֲלֵי שִׁירָה.
הַלָּלוּ, בַּעֲלֵי הַפְּשָׁט, סוֹמְכִים עַצְמָם עַל הַ"צַד הַשָּׁוֶה" וְעַל הַמְשֻׁתָּף
שֶׁבַּמַּרְאוֹת וּבַמִּלִּים, עַל הַקָּבוּעַ וְעוֹמֵד בַּלָשׁוֹן, עַל הַנֻּסַּח הַמְּקֻבָּל – וּלְפִיכָךְ
הֵם עוֹבְרִים אֶת דַּרְכָּם בַּלָּשׁוֹן בֶּטַח. לְמָה הֵם דּוֹמִים? לְמִי שֶׁעוֹבֵר אֶת
הַנָּהָר עַל־פְּנֵי קֶרַח מוּצָק, עָשׂוּי מִקְשָׁה אַחַת. רַשַּׁאי וְיָכוֹל הוּא זֶה לְהַסִּיחַ
אֶת דַּעְתּוֹ לְגַמְרֵי מִן הַמְּצוּלָה הַמְּכֻסָּה, הַשּׁוֹטֶפֶת תַּחַת רַגְלָיו. וְאֵלֶּה
שֶׁכְּנֶגְדָּם, בַּעֲלֵי הָרֶמֶז, הַדְּרָשׁ וְהַסּוֹד, רְדוּפִים כָּל יְמֵיהֶם אַחֲרֵי "הַצַּד
הַמְּיַחֵד" שֶׁבַּדְּבָרִים, אַחֲרֵי אוֹתוֹ הַמַּשֶּׁהוּ הַבּוֹדֵד, אַחֲרֵי אוֹתָהּ הַנְּקֻדָּה,
שֶׁעוֹשָׂה אֶת הַמַּרְאוֹת – וְצֵרוּפֵי הַלָּשׁוֹן הַמְּכֻוָּנִים לָהֶן – כַּחֲטִיבָה אַחַת
בָּעוֹלָם, אַחֲרֵי הָרֶגַע בֶּן־הַחֲלוֹף שֶׁאֵינוֹ נִשְׁנָה עוֹד לְעוֹלָמִים, אַחֲרֵי נִשְׁמָתָם
הַיְּחִידָה וּסְגֻלָּתָם הָעַצְמִית שֶׁל הַדְּבָרִים, כְּפִי שֶׁנִּקְלְטוּ אֵלּוּ בְּרֶגַע מְסֻיָּם
בְּנֶפֶשׁ רוֹאֵיהֶם; וּלְפִיכָךְ מֻכְרָחִים הַלָּלוּ לִבְרֹחַ מִן הַקָּבוּעַ וְהַדּוֹמֵם בַּלָּשׁוֹן,
הַמִּתְנַגֵּד לְמַטְּרָתָם, אֶל הַחַי וְהַמִּתְנוֹעֵעַ שֶׁבָּהּ. אַדְּרַבָּה, הֵם עַצְמָם מְחֻיָּבִים
לְהַכְנִיס בָּהּ כָּל רֶגַע – עַל־פִּי מַפְתְּחוֹת מְסוּרִים בִּידֵיהֶם – תְּנוּעָה בִּלְתִּי
פּוֹסֶקֶת, הַרְכָּבוֹת וְצֵרוּפִים חֲדָשִׁים. הַמִּלִּים מְפַרְפְּרוֹת תַּחַת יְדֵיהֶם: כָּבוֹת
וְנִדְלָקוֹת, שׁוֹקְעוֹת וְזוֹרְחוֹת כְּפִתּוּחֵי הַחוֹתָם בְּאַבְנֵי הַחֹשֶׁן, מִתְרוֹקְנוֹת
וּמִתְמַלְּאוֹת, פּוֹשְׁטוֹת נְשָׁמָה וְלוֹבְשׁוֹת נְשָׁמָה. בְּחֹמֶר הַלָּשׁוֹן בָּא עַל־יְדֵי
כָּךְ חִלּוּפֵי מִשְׁמָרוֹת וְהֶעְתֵּק מְקוֹמוֹת. תָּג אֶחָד, קוֹצוֹ שֶׁל יוֹ"ד – וְהַמִּלָּה
הַיְּשָׁנָה זוֹרַחַת בְּאוֹר חָדָשׁ. הַחוֹל מִתְקַדֵּשׁ וְהַקּוֹדֶשׁ מִתְחַלֵּל. הַמִּלִּים
הַקְּבוּעוֹת כְּאִלּוּ נֶחֱלָצוֹת רֶגַע רֶגַע מִמִּשְׁבְּצוֹתֵיהֶן וּמַחֲלִיפוֹת מָקוֹם זוֹ עִם
זוֹ. וּבֵינְתַיִם, בֵּין כִּסּוּי לְכִסּוּי, מְהַבְהֶבֶת הַתְּהוֹם. וְזֶהוּ סוֹד הַשְׁפָּעָתָהּ
הַגְּדוֹלָה שֶׁל לְשׁוֹן הַשִּׁירָה. יֵשׁ בָּהּ מִגְּרִיַּת הַיֵּצֶר שֶׁל הַרְגָּשַׁת הָאַחְרָיוּת,
מִן הָאֵימָה הַמְּתוּקָה שֶׁל הָעֲמִידָה בַּנִסָּיוֹן. וּלְמָה הַלָּלוּ דּוֹמִים? לְמִי
שֶׁעוֹבֵר אֶת הַנָּהָר בִּשְׁעַת הַפְשָׁרָה עַל־פְּנֵי גְלִידִין מִתְנַדְנְדִים וְצָפִים.
חָלִילָה לוֹ לְהַשְׁהוֹת אֶת הָרֶגֶל עַל־גַּבֵּי גְּלִיד אֶחָד יוֹתֵר מֵהֶרֶף־עַיִן, יוֹתֵר
מִכְּדֵי קְפִיצַת הָרֶגֶל מִגְּלִיד לַחֲבֵרוֹ הַסָּמוּךְ וּמֵחֲבֵרוֹ לַחֲבֵרוֹ. בֵּין הַפְּרָצִים
מְהַבְהֶבֶת הַתְּהוֹם, הָרֶגֶל מִתְמוֹטֶטֶת, הַסַּכָּנָה קְרוֹבָה –––
וְאַף־עַל־פִּי־כֵן גַּם מֵאֵלֶּה יֵשׁ שֶׁנִּכְנָסִים בְּשָׁלוֹם מֵחוֹף זֶה וְיוֹצְאִים בְּשָּׁלוֹם
אֶל הַחוֹף הַשֵּׁנִי, לֹא אֶת הַפְּתָאִים בִּלְבַד שׁוֹמֵר ה'.
עַד כָּאן עַל לְשׁוֹן הַמִּלִּים. אֲבָל מִלְּבַד זֶה "עוֹד לֶאֱלוֹהַּ" לְשׁוֹנוֹת בְּלֹא
מִלִּים: הַנְּגִינָה, הַבְּכִיָּה וְהַשְּׂחוֹק. וּבְכֻלָם זָכָה הַ"חַי הַמְּדַבֵּר". הַלָּלוּ
מַתְחִילִים מִמָּקוֹם שֶׁהַמִּלִּים כָּלָוֹת, וְלֹא לִסְתֹּם בָּאִים אֶלָּא לִפְתֹּחַ.
מְבַעְבְּעִים וְעוֹלִים הֵם מִן הַתְּהוֹם. הֵם הֵם עֲלִיַּת הַתְּהוֹם עַצְמוֹ. וּלְפִיכָךְ
יֵשׁ שֶׁהֵם מְצִיפִים וְגוֹרְפִים אוֹתָנוּ בַּהֲמוֹן גַּלֵּיהֶם וְאֵין עוֹמֵד כְּנֶגְדָּם; וּלְפִיכָךְ
יֵשׁ שֶׁהֵם מוֹצִיאִים אֶת הָאָדָם מִן הַדַּעַת אוֹ גַּם מִן הָעוֹלָם; כָּל יְצִירַת
רוּחַ שֶׁאֵין בָּהּ מֵהֵד אֶחָד מִשְּׁלֹשָׁה אֵלֶּה, אֵין חַיֶּיהָ חַיִּים וְרָתוּי לָהּ שֶׁלֹּא
בָּאָה לָעוֹלָם.
3 באוקטובר 1915
