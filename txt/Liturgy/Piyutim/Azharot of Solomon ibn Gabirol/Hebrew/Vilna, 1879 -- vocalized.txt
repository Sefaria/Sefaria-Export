אזהרות לרבי שלמה אבן גבירול

רשות

אָמוֹן יוֹם זֶה. נָחֲלוּ עָם זֶה. עַל יַד חוֹזֶה. אִישׁ הָאֱלֹהִים:
נוֹטֶה עֲלִיָּה. וְיוֹסֵד נְשִׁיָּה. לְךָ דוּמִיָּה. תְּהִלָּה אֱלֹהִים:
יוֹם נִגְלֵיתָ. וְדָת הוֹרֵיתָ. מְאֹד נַעֲלֵיתָ. עַל כָּל אֱלֹהִים:
דְּבַר אֵל שָׁמְעוּ. עָם נוֹשָׁעוּ. וְחָלוּ וְזָעוּ. מִפְּנֵי אֱלֹהִים:
וְהִפְלִיא עֵצָה. גְּדוֹל הָעֵצָה. לְעָם חֵן מָצָא. בְּעֵינֵי אֱלֹהִים:
יְקָרָה נְעִימָה. תּוֹרָה תְּמִימָה. לְהַנְחִיל אֲיוּמָּה. בָּא הָאֱלֹהִים:
דָּת הוֹדִיעָם. וְקוֹל הִשְׁמִיעָם. רְאוּ הֲשָׁמַע עָם. קוֹל אֱלֹהִים:
בְּאָזְנֵי הֲמוֹנַי. דִּבֵּר בְּסִינַי. הֲלֹא אֲנִי יְיָ. וְאֵין עוֹד אֱלֹהִים:
נְטוֹשׁ זָר וּפִסְלוֹ. וְצֵא תֹאמַר לוֹ. הֲיַעֲשֶׂה לּוֹ. אָדָם אֱלֹהִים:
אֶת שֵׁם הָאֵל. לַשָּׁוְא אַל תּוֹאֵל. כִּי עַל כָּל אֵל. גָּדוֹל אֱלֹהִים:
לְכַבֵּד שַׁבָּת. רוּץ נָא בְחִבַּת. כִּי בוֹ שָׁבַת. וַיְכַל אֱלֹהִים:
עַנֵּג אוֹמְנִים. בְּטוּב מַעֲדַנִּים. וְתַאֲרִיךְ שָׁנִים. לִפְנֵי אֱלֹהִים:
זְרֵה שׁוֹפְכֵי דָם. וְאַל תְּהִי בְסוֹדָם. כִּי הָאָדָם. בְּצֶלֶם אֱלֹהִים:
רַחֵק זִמָּה וְחֵמָה. וּמֹאזְנֵי מִרְמָה. פֶּן בְּאַף וְחֵימָה. יִפְקֹד אֱלֹהִים:
בְּרֵעֲךָ לֹא תַעֲנֶה. שָׁוְא כְּמִתְאַנֶּה. כִּי שֵׁשׁ שָׂנֵא. יְיָ אֱלֹהִים:
קִנְיַן אָח לֹא. תַחְמוֹד גַּם לֹא. תִּתְאַו אֲשֶׁר לוֹ. נָתַן אֱלֹהִים:
וְהָעָם חָרְדוּ. וְתָמְהוּ וּפָחֲדוּ. רָאוּ וַיַּגִּידוּ. פּוֹעַל אֱלֹהִים:
דִּבְּרוּ בְכוֹשֶׁר. נְדִיבֵי יוֹשֶׁר. נַעֲשֶׂה כָל אֲשֶׁר. צִוָּה אֱלֹהִים:
הֲמוֹנֵי עַם זוּ. בְּדַת אֵל שִׂמְחוּ עִלְזוּ. וְאִמְרוּ לְכוּ חֲזוּ. מִפְעֲלוֹת אֱלֹהִים:
חַי גּוֹאֲלֵנוּ. זְקוֹף יָדֵינוּ. וְחִישׁ הַרְאֵנוּ. בְּיֶשַׁע אֱלֹהִים:

מצוות עשה

<b>שְׁ</b>מוֹר <b>לִ</b>בִּי <b>מַ</b>עֲנֶה. <b>הֱ</b>יֵה <b>בִּ</b>מְאֹד <b>נֶ</b>עֱנֶה. <b>יְ</b>רָא <b>הָ</b>אֵל <b>וּ</b>מְנֵה. <b>דְּ</b>בָרָיו <b>הַ</b>יְשָׁרִים:
וְהוּא יִסְלַח אַשְׁמָה. וְהוּא יַרְבֶּה עָצְמָה. וְהוּא יִתֵּן חָכְמָה. לְהָבִין נִמְהָרִים:
אֲסַפֵּר תּוּשִׁיּוֹת. מְתוּקוֹת לַפִּיוֹת. וְאַצִּיב תַּלְפִּיּוֹת. לְיַשֵּׁר הָעוֹבְרִים:
וְאַזְכִּיר מִצְוֹת עֲשֵׂה. בְּדַת מָעוֹז וּמַחְסֶה. וְעַל פְּשָׁעַי יְכַסֶּה. מְגַלֶּה נִסְתָּרִים:
שְׁמֹנָה וְאַרְבָּעִים. וּמָאתַיִם נְטוּעִים. כְּמוֹ מַסְמְרוֹת תְּקוּעִים. כְּמִסְפַּר אֵבָרִים:
בְּסִינַי נוֹדָעוּ. וּמֵרוֹם נִשְׁמָעוּ. וְיַחְדָּו הָטְבָּעוּ. בְּתוֹךְ עֲשֶׂרֶת דְּבָרִים:
אֲשֶׁר הָתְווּ הַתְווֹת. בְּתֵיבוֹת נִשְׁתָּווֹת. בְּמִסְפַּר הַמִּצְוֹת. וּבָהֶם נֶאֱמָרִים:
וְהַמּוֹשִׁיעֲךָ. וְהַמּוֹדִיעֲךָ. עַד אֲשֶׁר לְרֵעֲךָ. נְתָנָם נֶחְקָרִים:
וְהִרְעִים אֵל דֵּעוֹת. בְּקוֹל הִפְלִיא זְוָעוֹת. מְקַפֵּץ עַל גְּבָעוֹת. מְדַלֵּג עַל הָרִים:
וְעֵת קָרָא אוֹתָם. רְעָדָה אֲחָזָתָם. וְיָצְאָה נִשְׁמָתָם. בְּרֶתֶת וּשְׁבָרִים:
וְהוֹרִיד טַל אֵדָיו. לְעַמּוֹ וַעֲבָדָיו. וְהֵשִׁיב בַּחֲסָדָיו. נְשָׁמוֹת לִפְגָרִים:
אֲנִי הוֹצֵאתִיךָ. אֲנִי הִזְהַרְתִּיךָ. אֲנִי הִדְרַכְתִּיךָ. בְּדַרְכֵי מֵישָׁרִים:
לְיַחֵד אֵל אָיוֹם. שְׁתֵּי פְעָמִים בַּיּוֹם. לְהִתְפַּלֵּל יוֹם יוֹם. עֲרָבִים וּשְׁחָרִים:
לְעָבְדוֹ וּלְאַהֲבוֹ בְּלֵב. וּלְדָבְקָה בוֹ. וְלִתְמוֹךְ בִּנְתִיבוֹ. פְּעָמִים וַאֲשׁוּרִים:
לְקַדֵּשׁ תַּעֲצוּמוֹ. לְיִרְאָה מִזַּעְמוֹ. לְהִשָּׁבַע בִּשְׁמוֹ. בְּלִי שָׁוְא וּשְׁקָרִים:
לְצַדֵּק דִּינֵהוּ. וְלִרְדּוֹף צִדְקֵהוּ. וְלִשְׁמֹר דָּתֵהוּ וְלַעֲשׂוֹת מַאֲמָרִים:
דְּבָרָיו שִׂים מַרְפֵּא. עֲלֵי לֵב גַּם עַל פֶּה. וְכָתְבֵם עַל סִפֵּי. מְזוּזוֹת וּשְׁעָרִים:
וְשִׁנַּנְתָּם תָּמִיד. לְבִנְךָ וּלְתַלְמִיד. וּמִבְּרָכוֹת תַּצְמִיד. מְאַת הַנִּגְמָרִים:
בְּכוֹר בָּנִים תִּפְדֶּה. וְטוֹטָפוֹת תַּעְדֶּה. וּבַכְּנַעֲנִי תִרְדֶּה. וְתִמּוֹל בִּבְשָׂרִים:
וְתִפְדֶּה פֶּטֶר חֲמוֹר. וְהַשַּׁבָּת תִּשְׁמוֹר. וְהַהַלֵּל תִּגְמוֹר. בְּיָמִים נִסְפָּרִים:
וְעַל כַּנְפֵי בִגְדָךְ. גְּדִילִים תַּעֲשֶׂה לָּךְ. וְתַלְוֶה דַּל וָדָךְ. תְּפַיֵּיס בִּדְבָרִים:
קְבַע בִּרְכַּת מָזוֹן. בְּאַט וּבְחִפָּזוֹן. וְלֹא תִמְצָא רָזוֹן. בְּעִנּוּי כִּפּוּרִים:
עֲבוֹט עָנִי הָשֵׁב. וְאִתָּךְ אַל יֵשֵׁב. וְהַגְּזֵלָה תָשֵׁב. וְעוֹשֶׁק מִסְחָרִים:
הֱיֵה לַשָֹּב עָנָיו. לְשִׂכְלוֹ וּלְשָׁנָיו. וְתָקוּם מִפָּנָיו. וּפָנָיו נֶהְדָּרִים:
וְתִלְמוֹד וּתְלַמֵּד. וְהוֹרֶיךָ תְכַבֵּד. וְתָשִׁיב כְּלִי אוֹבֵד. וְתַקְדִּישׁ מְבֻכָּרִים:
וְהַדַּלִּים תְּרַחֵם. וְהַחוֹלִים תְּשִׂיחֵם. וְהָאֲבֵלִים תְּנַחֵם. וְתִקְבּוֹר נִגְזָרִים:
וְלִפְתֹּחַ וְלָתֵת. לְאֶבְיוֹן יֵחָתֵת. וְלַשְׁמִיד וּלְמוֹתֵת. עֲמָלֵק רֹאשׁ צָרִים:
לְבַעֵר מַחְמֶצֶת. בְּכָלָה נֶחְרֶצֶת. וְלִשְׂרוֹף מִפְלֶצֶת. אֱלִילִים וַאֲשֵׁרִים:
לְעַנֵּג יוֹם מְנוּחָה. בְּהַשְׁקֵט וּבְבִטְחָה. וְלָחוֹג וּלְשִׂמְחָה. וְלֶאֱהוֹב הַגֵּרִים:
וְהַצְּדָקָה לַעֲשׂוֹת. כְּפִי יָד וּכְמִסּוֹת. וּמַשָֹּאוֹת וּמַעֲמָסוֹת. לְהָקִים וּלְהָרִים:
וְתַלְמִיד שַׁעֲשׁוּעִים. לְהַדֵּר וּלְהַנְעִים. וְתוֹכַחַת רֵעִים. וְאַהֲבַת הַחֲבֵרִים:
וְתֻמָּה וַעֲנָוָה. וְיֵין קִדּוּשׁ וְחֶדְוָה. וְיִרְאַת מִקְדָּשׁ. וַחֲכָמִים עִם הוֹרִים:
וְלִמּוּד נֶאֱמָנָה. לְבָנִים בִּרְנָנָה. לְיָתוֹם וְאַלְמָנָה. וְלֵוִי עִם גֵּרִים:
שְׁמוֹר מַעְשֵׂר עָנִי. וְגַם מַעְשֵׂר שֵׁנִי. וְרִאשׁוֹן וּשְׁמִינִי. תְּקַדֵּשׁ בַּהֲדָרִים:
וּמַעֲשֵׂר מִמַּעֲשֵׂר. וְגַם וִדּוּי מַעֲשֵׂר. וּמִבְּהֵמָה תְּעַשֵֹּר. וְרֹאשׁ גֵּז מֵעֲדָרִים:
וְקָדְשֵׁי מַתָּנִי. וְגַם מַעֲשֵׂר שֵׁנִי. תֹּאכְלֵם בִּמְעוֹנִי. וְהַחַלָּה תָּרִים:
וְחַזֵּק מָךְ יִמּוֹט. וְלָעַד לֹא תִמּוֹט. וּבַשְּׁבִיעִית תִּשְׁמוֹט. סְפִיחִים וּנְזִירִים:
וְתִשְׁמוֹט מַשָּׁאוֹת. וְשִׁכְחָה עִם פֵּאוֹת. בְּשָׂדֶה וּתְבוּאוֹת. וְעוֹלְלוֹת מִבְּצִירִים:
וְלֶקֶט הַקָּצִיר. וּפֶרֶט הַבָּצִיר. עֲזוֹב לִבְלִי תָצִיר. לְדַלִּים נֵאָרִים:
וְאֵלֵימוֹ הַבֵּט. בְּהַעְנֵק וּבְהַעֲבֵט. וְתֶן פֶּן תִּלָּבֵט. נְדָבוֹת וּנְדָרִים:
שְׂמַח וֶאֱמוֹר הֶאָח. בְּהַחְיוֹתְךָ הָאָח. וְהַנוֹתָר בָּאָח. יְהוּשְׁלַךְ עַל אוּרִים:
וּפִיגוּל כִּי יַבְאֵשׁ. יִשָֹּרֵף בָּאֵשׁ. וּפֶסַח בִּצְלִי אֵשׁ. וּמַצּוֹת וּמְרוֹרִים:
וְרִאשׁוֹן וּשְׁבִיעִי. תְּקַדֵּשׁ בְּחַג יִשְׁעִי. וּמִקְרָא שָׁבוּעִי. בְּבִכּוּרֵי קְצִירִים:
וְיוֹם זִכְרוֹן תְּרוּעָה. וְיוֹם כִּפּוּר תּוֹעָה. וְסוּכָּה וְאַרְבָּעָה. צְמָחִים נִבְחָרִים:
עֲרָבִים כִּסְמָדָר. וְעֵץ עָבוֹת נֶהְדָּר. וּפְרִי עֵץ הָדָר. וְכַפּוֹת תְּמָרִים:
וְכֹהֵן הוּקַם עַל. בְּאַלְמָנָה יִגְעַל. וּבִבְתוּלָה יִבְעַל. וְיָדֵהוּ יָרִים:
וְיֵדַע הָעִתִּים. לְטַהֵר סוּת וּבָתִּים. וּפָתוֹת הַפִּתִּים. וְהַגֵּשׁ מֻקְטָרִים:
וְנִשְׁלַח נִשְׁלוֹחַ. לְהַקְטִיר וְלִמְלוֹחַ. לְכַפֵּר וְלִסְלוֹחַ. פְּשָׁעִים נֶעְתָּרִים:
וְלִיצוֹק וּלְבַלְבֵּל. וְלִשְׁחוֹט פַּר וְיוֹבֵל. וְלִמְלוֹק וּלְקַבֵּל. וְלִקְמוֹץ אַזְכָּרִים:
וְכֵן יִלְמַד עִם זֶה. לְהָנִיף וּלְהַזֶּה. וְהַשּׁוֹק וְהֶחָזֶה. לְחֻקּוֹ נֶעְצָרִים:
וְיָכִין לִמְלֶאכֶת שְׂרָד. וּבְעֵת לֶכֶת. לְבֵית הַפָּרוֹכֶת. בְּגָדִים מָשְׁזָרִים:
וְתַבְנִית הַמִּשְׁכָּן. וְלֶחֶם הַנִּתְכָּן. וְשֶׁמֶן הַמְּתֻכָּן. לְמָשְׁחָה וְלִמְאוֹרִים:
חֲנוּכָּה וּמְלֶאכֶת. קְטוֹרֶת נִסֶּכֶת. וְשֵׁשׁ הַמַּעֲרֶכֶת. וּשְׁתֵּי כִכָּרִים:
וּשְׁלָמִים וּמָנוֹת. וְעוֹלוֹת וְקָרְבָּנוֹת. וְשָׁלֹשׁ מַתָּנוֹת. בְּמוֹעֲדִים נִזְכָּרִים:
וְהָעוֹמֶר לְנַהֵל. וְנֵר שַׁבָּת יָהֵל. וּפָרָשַׁת הַקְהֵל. וְשׁוֹפְטִים עִם שׁוֹטְרִים:
לְמַחְקוֹרֶת רִיבוֹת. נְפָשׁוֹת נֶעֱרָבוֹת. וְדִין אַרְבַּע אָבוֹת. נְזִיקִּים נִשְׁמָרִים:
וְקֶצֶב וּתְבוּנָה. לְחָדְשֵׁי הַשָּׁנָה. וּמַתְּנוֹת כְּהוּנָּה. אַרְבָּעָה וְעֶשְׂרִים:
וְהַתְּקוּפוֹת לְמַלֹּאת. וְנֵר חֲנוּכָּה לְהַעֲלוֹת. וְלֵרָאוֹת וְלַעֲלוֹת. בִּשְׁנֵי מַעֲשָׂרִים:
לְהַנְעִים חוֹמָשִׁים. וְגַם לִהְיוֹת קְדוֹשִׁים. וְלִטְבוֹל נִפְרָשִׁים. בְּעִתִּים נִגְזָרִים:
וְשֶׁבֶת בִּבְהֵמָה. וְעֶבֶד עִם אָמָה. וּבִכּוּרֵי אֲדָמָה. וּמִקְרָא בִכּוּרִים:
וְזֵירוּז מָתְנַיִם. וְצֶדֶק מֹאזְנַיִם. וּמַתַּת פִּי שְׁנַיִם. לְבָנִים הַבְּכוֹרִים:
וְחוֹק אֵיפוֹתַיִם. וְחוֹק מִדּוֹתַיִם. וּמוֹצָא שְׂפָתַיִם. וּפִדְיוֹן נִמְכָּרִים:
וְחוֹפֶשׁ הָאָמָה. בְּסִימָן נִכְתָּמָה. וְדִין עָרֵי חוֹמָה. וּבָתֵּי הַחֲצֵרִים:
וְאָדָם לֹא יָדוֹן. בְּנִמְכַּר עַל זָדוֹן. בְּיוֹבֵל וּמוֹת אָדוֹן. יְשׁוּלַּח לַדְרוֹרִים:
וְצֵאת עִבְרִית חָפְשִׁית. בְּיוֹבֵל וְצֵאת שִׁשִּׁית. וְעוֹרוֹת וַחֲמִישִׁית. וְהַבְדֵּל שֵׁשׁ עָרִים:
וְדִין שִׁפְחָה חֲרוּפָה. וְסוֹד עֶגְלָה עֲרוּפָה. וּפֶסַח לִתְרוּפָה. לְלֹא הֵמָּה טְהוֹרִים:
וְטַהֲרַת הַדָּווֹת. בְּמֵימֵי הַמִּקְווֹת. וּפַר בָּא עַל מִצְוֹת. וּפַר הַכִּפּוּרִים:
וְטַהֲרַת הַטּוּמְאָה. בְּקָרְבָּן הַנִּרְאָה. וְקָרְבַּן חַטָּאָה. וּמוּסָפִין עֶשְׂרִים:
וְיוֹרֵד גַּם עוֹלֶה. וּוַדַּאי גַם נִתְלֶה. וְחֵרֶם הַמּוּפְלֶא. וְעֶרֶךְ הַנּוֹדְרִים:
וְיָד יָתֵד לַחְפּוֹר. וְהָעוֹמֶר לִסְפּוֹר. וְאֵם קַן הַצִּפּוֹר. לְשַׁלַּח בַּיְּעָרִים:
וְכָל טוּמְאָה לְהַפְלוֹת. בְּשֶׁקֶץ וּנְבֵלוֹת. וְאָדָם עִם מִכְלוֹת. וּמַשְׁקִים נִגָּרִים:
וַהֶסֵּט וְשׁוֹכֶבֶת. וּמִקְרֶה מַחְשֶׁבֶת. וְיִבּוּם נֶעֱזֶבֶת. וְאִגֶּרֶת פּוּרִים:
וְחִלּוּץ וָרוֹקֶת. לְאִשָּׁה נִתֶּקֶת. וְקוֹץ כַּף מַחְזֶקֶת. בִּמְבוּשֵׁי גְבָרִים:
וְדִין אֵשֶׁת שִׁבְיָה. וְחוֹק פִּרְיָה וְרִבְיָה. וְתִלְבּוֹשֶׁת עֶרְיָה. וְנִסּוּךְ מֵי קָרִים:
וְעָרְלָה וְהִלּוּלִים. וּמִשְׁנֶה לַמּוֹשְׁלִים. וּבְיוֹמָם לְהַשְׁלִים. שְׂכַר הַנִּשְׂכָּרִים:
בְּרָכוֹת וּקְלָלוֹת. וּמֶכֶס וּנְחָלוֹת. וְשִׂיד אֲבָנִים גְּדוֹלוֹת. וְיוֹבֵל לַדְרוֹרִים:
וַחֲצוֹצְרוֹת עֵדָה. וּמוּרָם מִתּוֹדָה. וְטִיבּוּל יוֹלֵדָה. מִדְּמֵי הַמְּקוֹרִים:
וּפָרָה לִשְׂרֵיפָה. וַחֲזֵה לִתְנוּפָה. וּמַעְשַׂר הָאֵיפָה. וְהַתְּרוּמָה לְהָרִים:
וְעִיר יַד שׁוֹלַחַת. לְמַלֵּט מִשַּׁחַת. וְעַם עִיר נִדַּחַת. לְהַחְרִיב וּלְהַחְרִים:
וְעִיר כִּי תִסָּגֵר. אֲבָנֶיהָ תַּגֵּר. וְתֵת נְבֵלָה לַגֵּר. אוֹ מָכוֹר לְנָכְרִים:
דְּמֵי כוֹפֶר תִּשְׁקוֹל. וְהַמַּסִּית תִּסְקוֹל. בְּיַעַן הֵרִים קוֹל. לְעָבְדָה לַאֲחֵרִים:
וְחוֹלֵם וְעֵד זוֹמֵם. כְּאֶחָד הַחֲרִימֵם. לְבִלְתִּי יֵרוֹמֵם. לְהִנָּבֵא שְׁקָרִים:
וּבִפְרָעוֹת תִּפְרַע. הַמּוֹצִיא שֵׁם רָע. וְאִישׁ זָב וּמְצוֹרָע. בְּהַזּוֹת נִטְהָרִים:
וּבָזֶה יְרַפֵּא. טְמֵא מֵת אוֹ נִסְפֶּה. וּמַדִּיחַ תִּסְפֶּה. וְיִירְאוּ נִשְׁאָרִים:
וְחוֹבֵל בַּחֲמִשָּׁה. תְּכַסֵּהוּ בוּשָׁה. וְדָם עַל יַבָּשָׁה. תְּכַסֶּה בַּעֲפָרִים:
וְהַסּוֹטָה תַּשְׁקֶה. לְמַעַן תִּנָּקֶה. וְעָשִׂיתָ מַעֲקֶה. לְהָסִיר מַכְמוֹרִים:
וְחָתָן כָּל שְׁנָתוֹ. יְהִי נָקִי לְבֵיתוֹ. וִיאַחֵר מִצֵּאתוֹ. וְלֹא יַעֲבוֹר דְּבָרִים:
וְיָנִיף יַד רָמָה. מְשׁוּחַ מִלְחָמָה. לְבַל תִּפּוֹל אֵימָה. בְּלֵב הַנֶּחְגָּרִים:
אֵלֶּה מִצְוֹתַי. וְאֵלֶּה חֻקּוֹתַי. וְאֵלֶּה תוֹרוֹתַי. תְּמִימִים וִישָׁרִים:
תְּחַיֶּינָה מָהִיר. וְתִסְפֶּינָה יָהִיר. וְהַמַזְהִיר יַזְהִיר. כְּזוֹהַר הַמְּאוֹרִים:
וְלַנִּזְהָר יִקְרַב. בְּשָׁמְרָם עֵקֶב רָב. וּבָאוֹר הַנֶּעֱרָב. יְשָׁרִים מַכְתִּירִים:

מצוות לא תעשה

בְּצֵל שַׁדַּי אֶחֱסֶה. וְצִדְקוֹ לֹא אֲכַסֶּה. בְּמִצְוֹת לֹא תַעֲשֶׂה. וְאַגִּיד מֵישָׁרִים:
כְּתוּבִים בֶּאֱמוּנָה. בְּעֵדוּת נֶאֱמָנָה. וְכִימֵי הַשָּׁנָה. בְּמִסְפָּר נֶחְקָרִים:
יְקָרִים מִפְּנִינִים. עֲתִידִים וּצְפוּנִים. לְבַת הָאֵיתָנִים. בְּעֶדְיֵי קִשּׁוּרִים:
צְאִי נָא לִקְרָאתִי. אֲחוֹתִי רַעְיָתִי. וְשִׁמְעִי תוֹרָתִי. וְלִקְחִי מוּסָרִים:
חֲשַׁקְתִּיךְ וָאוֹהַב. פְּדִיתִיךְ מִלַּהַב. כְּמוֹ תוֹרֵי זָהָב. צְרוּפִים וּטְהוֹרִים:
בְּצֵאתִי נִצָּבָה. בְּאֶלֶף וּרְבָבָה. וְקוֹל דּוֹדִי זֶה בָא. מְדַלֵּג עַל הָרִים:
אָנֹכִי יְיָ. קְרָאתִיךָ בְסִינַי. וְלֹא יִהְיֶה עַל פָּנַי. לְךָ אֱלֹהִים אֲחֵרִים:
וְלֹא תַעֲשֶׂה פֶסֶל. בְּרֶשַׁע וּבְכֶסֶל. וְלֹא תָשִׂים פֶּסֶל. לְהַקְנִיאוֹ בְּזָרִים:
וְגוּר מִלֶּהָבָה. לְהָבִיא תוֹעֵבָה. וְלִבְנוֹת מַצֵּבָה. וְלָטַעַת אֲשֵׁרִים:
וְסוּר מִשֵּׁמַע שָׁוְא. וְהַאֲמֵן בַּשָּׁוְא. וְלֹא תִשָֹּא לַשָּׁוְא. שְׁמוֹתַי הַיְקָרִים:
וּמִצְוַת לֹא תִנְאַף. שְׁמוֹר פֶּן יֶחֱרֶה אַף. וְלִגְנוֹב לֹא תִשְׁאַף. וְלֹא תַחְמוֹד חֲבֵרִים:
וְנֶפֶשׁ לֹא תְנַקֵּשׁ. אֲשֶׁר דָּמָהּ אֲבַקֵּשׁ. וְלֹא תִתֵּן מוֹקֵשׁ. לְהַכְשִׁיל הַעִוְרִים:
וְשֶׁקֶר לֹא תַעֲנֶה. וְיָתוֹם לֹא תְעַנֶּה. וְאַרְצְךָ לֹא תִזְנֶה. כְּאַרְצוֹת הַזָּרִים:
וְהַגֵּר לֹא תוֹנֶה. וְגָזִית לֹא תִבְנֶה. וְעַל רִיב לֹא תַעֲנֶה. לְחַרְחֵר מִתְגָּרִים:
אֱלֹהִים לֹא תְקַלֵּל. וְהַשֵּׁם לֹא תְחַלֵּל. וְכַרְמְךָ לֹא תְעוֹלֵל. וּפֶרֶט הַבְּצִירִים:
וְלֹא תְקַלֵּל חֵרֵשׁ. חֲמָסוֹ יִדָּרֵשׁ. וְלֹא יַצְמִיד חוֹרֵשׁ. שְׁוָרִים וַחֲמוֹרִים:
וְלֹא תְאַחֵר בְּבִצְעֲךָ. מְלֵאָתְךָ וְדִמְעֲךָ. וְלֹא תַעֲשׁוֹק רֵעֲךָ. וְלֹא תְאַחֵר נְדָרִים:
וְלֹא תַעְשׁוֹק שְׂכִירְךָ. וְלֹא תְלַקֵּט קְצִירְךָ. וְלֹא תַחְסוֹם שׁוֹרְךָ. בְּדִישׁוֹ הָעֳמָרִים:
וְלֹא תְפַאֵר זֵיתְךָ. וְלֹא תְכַלֶּה פְּאָתְךָ. וְלֹא תוֹנֶה עֲמִיתְךָ. בְּמֶקַח וּדְבָרִים:
וְלֹא תִשָׂא חֶטְאוֹ. אֲשֶׁר לִבּוֹ מְלָאוֹ. וְלֹא תַחְשׁוֹב לְשוֹנְאוֹ. כְּמַשְׂטֵמַת עָרִים:
וְלֹא תִקּוֹם עַמְּךָ. לְבַל אֶקּוֹם מִמְּךָ. וְלֹא תָלִין עִמְּךָ. פְּעוּלַת נִשְׂכָּרִים:
וְלֹא תִטּוֹר אֵיבוֹת. וְלֹא תִשְׁאַל אוֹבוֹת. וְלֹא יוּמְתוּ אָבוֹת. וּבָנִים עַל הוֹרִים:
וְלֹא תִשְׁמַע דִּבְרֵי. נְבִיא שֶׁקֶר מוֹרֶה. וְלֹא תִהְיֶה אַחֲרֵי. מְסִיתִים נִבְעָרִים:
וְעֶבֶד לֹא תַסְגִּיר. וּפָנִים לֹא תַכִּיר. וְעַל פֶּה לֹא תַזְכִּיר. שֵׁם אֱלֹהִים אֲחֵרִים:
וְכַסְפָּם לֹא תַחְמוֹד. וְעַל דָּם לֹא תַעֲמוֹד. וְהֵחָדֵל לִלְמוֹד. שְׁבוּעוֹת לִשְׁקָרִים:
נְצוֹר חוּקַּי עַל דַּל. וְלֹא תִשָֹּא פְנֵי דַל. וּמִנֶּשֶׁךְ תֶּחְדַּל. לְרַבּוֹת בִּמְחִירִים:
וּמִצְוַת לֹא תַטֶּה. שְׁמוֹר פֶּן בָּהּ תִּשְׂטֶה. וְדִין יָתוֹם מוּטֶּה. וְאַלְמָנָה וְגֵרִים:
וְלֹא תִקַּח שׁוֹחַד. וְלֹא יָעִיד אֶחָד. וְלֹא תֹאכַל יַחַד. נְפָשׁוֹת וּבְשָׂרִים:
וְלֹא תַחְמוֹד גְּזֵלָה. אֲשֶׁר תּוּקַּח אֲפֵלָה. וְלֹא תֹאכַל נְבֵלָה. וְזִבְחֵי הַשְֹּעִירִים:
וְלֹא תֹאכַל טְרֵפָה. וְתוֹעֵבָה הֲדוּפָה. וְלֹא תַקְטִין אֵיפָה. לְהַפְקִיעַ שְׁעָרִים:
וְלֹא תַעֲשֶׂה רָעוֹת. בְּמִדַּת קַרְקָעוֹת. וּמִשְׁקַל הַמָּעוֹת. וְגַם לֹא בַּמְּשׂוּרִים:
וְלֹא תִתֵּן אָכְלְךָ. בְּמַרְבִּית פֶּן אֲכֶלְךָ. וְלֹא תָרוּץ רַגְלְךָ. מְלָשְׁנִי בַסְּתָרִים:
וְלֹא תָשִׁית יַד עִם. רְשָׁעִים וּמְרֵעִים. וְלֹא תִגּוֹשׂ רֵעִים. וְאַחִים כַּנָּכְרִים:
עֲזוֹב עוֹמֶר שְׁכַחְתּוֹ. וְלֹא תָשׁוּב לְקַחְתּוֹ. לְאַלְמָנָה נְתַתּוֹ. וְלִיתוֹמִים וְגֵרִים:
שְׁנַת שֶׁבַע תִּצּוֹר. וּבַיּוֹבֵל תַּעְצוֹר. סְפִיחִים לֹא תִקְצוֹר. וְלֹא תִבְצוֹר נְזִירִים:
וְלֹא תִזְרַע תַּלְמְךָ. לְהוֹצִיא בוֹ לַחְמְךָ. וְלֹא תִזְמוֹר כַּרְמְךָ. וְלֹא יַעֲלֶה סִירִים:
וְלֹא תֹאכַל קָלִי. וְלֶחֶם וְכַרְמֵלִי. בְּטֶרֶם תָּבִיא לִי. רֵאשִׁית הַקְּצִירִים:
וְלֹא תֹאכַל עֲרֵלִים. בְּטֶרֶם הִלּוּלִים. וְלֹא תֶהְדַּר גְּדוֹלִים. וְדַלִּים נֶחְפָּרִים:
וְלֹא תִהְיֶה כְנוֹשֶׁה. לְאִישׁ עָנִי נִקְשֶׁה. וְלֹא תִתְּנוּ אִשֶּׁה. בְּעִוְרִים וּשְׁבוּרִים:
וְלֹא יַעֲלֶה מְשִׁיחִי. בְּמַעֲלוֹת עַל מִזְבְּחִי. וְלֹא תִזְבַּח פִּסְחִי. בְּאַחַד הַשְּׁעָרִים:
וְהַכֵּלִים תְּמַשֵּׁשׁ. וְהֶחָמֵץ תְּרוֹשֵׁשׁ. וְתִשְׂרְפֶנּוּ בַּשֵּׁשׁ. עַד אֶחָד וְעֶשְׂרִים:
וְלֹא תוֹצִיא לַחוּץ. בְּשַׂר פֶּסַח נָחוּץ. וְזָר וְעָרֵל מִחוּץ. וְתוֹשָׁב וּשְׂכִירִים:
וְזִבְחִי לֹא תְשַׁמֵּץ. בְּזָבְחוֹ עַל חָמֵץ. וְלִבָּךְ לֹא תְאַמֵּץ. לְתִתֵּן מַחְסוֹרִים:
וְנָשִׂיא לֹא תָאוֹר. וְלֹא יֵרָאֶה שְׂאוֹר. וְלֹא יָלִין עַד אוֹר. בְּשַׂר לֵיל שִׁמּוּרִים:
וְזִבְחֵי שַׁלְמֵיכֶם. וְחֶלְבֵּי זִבְחֵיכֶם. וְחֶלְבֵּי חַגֵּיכֶם. לְהוֹתִיר נֶאֱסָרִים:
וְכִי יִגַּע בָּשָׂר. בְּכָל טָמֵא נֶאֱסָר. וְלֹא תֹאכַל מַעְשַׂר. דְּגָנְךָ בִּשְׁעָרִים:
וְיֵינְךָ וְיִצְהָרְךָ. וּבְכוֹרוֹת בְּקָרְךָ. וּתְרוּמַת יָדְךָ. וּמִבְחַר הַנְּדָרִים:
וְלֹא תֹאכַל עַל דָּם. תְּמִידִי הַנִּקְדָּם. וְהַחֵלֶב וְהַדָּם. שְׁנֵי אֵלֶּה אֲסוּרִים:
וְדָג וְחַיָּה שְׁנֵיהֶם. וְעוֹפוֹת לְמִינֵיהֶם. בְּלֹא סִימָנֵיהֶם. אֲשֶׁר בָּם נִכָּרִים:
וּמוּם בּוֹ לֹא יִקְרַב. לְהַקְרִיב שַׁי נֶעֱרָב. בְּגִבֵּן אוֹ גָרָב. וּפִסְחִים אוֹ עִוְרִים:
וּמִיַּד בֶּן נֵכָר. לֹא תַקְרִיב אֶשְׁכָּר. וְלֹא יַעֲבוֹד אִכָּר. בִּבְכוֹרֵי שְׁוָרִים:
וְלֹא יַעֲלֶה לְרָצוֹן. בְּכָל מָקוֹם חִיצוֹן. וְלֹא תָגוֹז בְּכוֹר צֹאן. בְּמַשְׁקֶה הָעֲדָרִים:
וְלֹא תָמִיר בְּהֵמָה. מְקוּדֶּשֶׁת חֲתוּמָה. וְלֹא תֹאכְלוּ תְּרוּמָה. טְמֵאִים עִם זָרִים:
וְלֹא תָבִיא לְמָנָה. מְחִיר כֶּלֶב וְזוֹנָה. וְלֹא תִתֵּן לְבוֹנָה. בְּמִנְחַת הַשְֹּעוֹרִים:
וְלֹא תִקְרַב לְמָשְׁחָה. בְּשֶׁמֶן הַמִּשְׁחָה. וְלֹא תָשִׂים קָרְחָה. וְשֶׂרֶט בִּבְשָׂרִים:
קְטוֹרֶת לֹא תִרְקַח. וְעֵינֶיךָ תִּפְקַח. וְכוֹפֶר לֹא תִקַּח. לְרָצְחֵי מִסְתָּרִים:
שְׁמוֹר פֶּן תִּנָּשֶׁה. דְּבַר גִּיד הַנָּשֶׁה. וּמִצְוַת לֹא יִקְשֶׁה. בְּעֵינֶיךָ תָּרִים:
וְחוֹק אֹתוֹ וְאֶת בְּנוֹ. בִּלְבָבְךָ תִּצְפְּנוֹ. וּבִנְךָ לֹא תִתְּנוֹ. לְמוֹלֶךְ אַכְזָרִים:
וְכָל מְלָאכָה לְהַקְצוֹת. בְּשַׁבְּתוֹתַי לְרַצּוֹת. וּמִקְרָאֵי מַצּוֹת. וּמִקְרָא בִכּוּרִים:
וּמִקְרָא זִכָּרוֹן. לְצֶדֶק וּלְכִשְׁרוֹן. וּמִקְרָאֵי הָרוֹן. וּמִקְרָא כִפּוּרִים:
וְעֶצֶם לֹא תִשְׁבּוֹר. בְּלֵיל צֵאתְךָ מִבּוֹר. וְתַעַר לֹא יַעֲבוֹר. עַל רָאשֵׁי נְזִירִים:
וְלֹא יֵצֵא מִפִּיו. לְהִטַּמֵּא לְאָבִיו. וְלַאֲחוֹתוֹ וּלְאָחִיו. וְיֶתֶר הַשְּׁאֵרִים:
וְלֹא יִשְׁתֶּה יַיִן. לְחַכְלִילוּת עַיִן. וְלֹא חוֹמֶץ יַיִן. וְיֶתֶר הַדְּבָרִים:
וּכְתוֹבֶת קַעֲקַע. בְּגֵוְךָ לֹא תִרְקַע. וְהַמְקַלֵּל יוּקַע. וְיוּבַל לִקְבָרִים:
וּמִקְרֶה הַנִּרְאֶה. בְּמַחֲנֶיךָ הַפְאֶה. וְצוּרְךָ לֹא יִרְאֶה. בְּךָ עֶרְוַת דְּבָרִים:
וְנֶשֶׁךְ לֹא תְתַכֵּן. לְהָשִׂים עַל מִסְכֵּן. וְחוֹק לֹא תַעֲשׂוּ כֵן. לְאַדִּיר אַדִּירִים:
וְעֵינְךָ לֹא תַעְלִים. בְּנִדָּחִים וְנוֹפְלִים. וְלֹא תַעֲשֶׂה אֱלִילִים. כְּהַבְלֵי הַנָּכְרִים:
וְצֶמֶר לֹא תִלְבַּשׁ. בְּמוֹ פִשְׁתִּים נֶחְבָּשׁ. וְלֹא תַקְטִירוּ דְבַשׁ. תְּמָרִים וּשְׂאוֹרִים:
וְלֹא תַעֲלוּ לַחְמִי. וְלֹא תִסְּכוּ לִשְׁמִי. בְּמִזְבֵּחַ פְּנִימִי. אֲשֶׁר לִפְנֵי עֳפָרִים:
בְּצֵל נוֹף לֹא תֶחֱסוּ. וְכָהֶם לֹא תַעֲשׂוּ. וְהָאֵל לֹא תְנַסּוּ. כְּמַסַּת הַמּוֹרִים:
וְחוֹק לֹא יֵרָאוּ. דְּבָרָיו תִּירָאוּ. וְחַיִּים תִּמְצָאוּ. וְשָׁלוֹם כַּנְהָרִים:
וּבִדְבַר לֹא תְשַׁקְּצוּ. מְאֹד חִזְקוּ וְאִמְצוּ. וְהַגֵּר לֹא תִלְחֲצוּ. כִּי הֱיִיתֶם גֵּרִים:
וּבִדְבַר לֹא תְכַחֲשׁוּ. מְאֹד חִקְרוּ וְדִרְשׁוּ. וּבִדְבַר לֹא תְנַחֲשׁוּ. כְּנֶאֱלָחִים סָרִים:
וּבִדְבַר לֹא תְבַשֵּׁל. חֲדַל מֵהִתְרַשֵּׁל. וְלֹא תֹאכַל בָּשֵׁל. וְנָא לֵיל שִׁמּוּרִים:
וּבִדְבַר לֹא תְעוֹנֵן. אֲשׁוּרֶיךָ כּוֹנֵן. וְהָעֲבָדִים תְּחוֹנֵן. בִּצֵאתָם לַדְּרוֹרִים:
וְלֹא תִגְרַע חוּקָּם. לְמַלֹּאת אֶת סִפְקָם. וְלֹא תְשַׁלְּחֵם רֵיקָם. מְרוּדִים וּנְעוּרִים:
וְלֹא תַעֲבוֹד בָּהֶם. עֲבוֹדַת עַבְדֵּיהֶם. בְּיַעַן כִּי לִי הֵם. עֲבָדִים נִמְכָּרִים:
וְלֹא תִקְפּוֹץ כַּפְּךָ. וְלֹא תַקְשֶׁה עָרְפְּךָ. וְלֹא תִתֵּן כַּסְפְּךָ. בְּנֶשֶׁךְ כַּיְּהִירִים:
וְלֹא יֵרַע לְבָבְךָ. בְּתִתְּךָ לְתוֹשָׁבְךָ. וְלֹא יִמָּצֵא בְךָ. קְסָמִים וַחֲבָרִים:
וְלֹא תַעֲבִיר בִּנְךָ. וְלֹא תַשְׁחִית זְקָנְךָ. וְלֹא תָחוֹס עֵינְךָ. לְכַלּוֹת הַצּוֹרְרִים:
וְלֹא תִכְרוֹת לָהֶם. בְּרִית וְלֵאלֹהֵיהֶם. וְלֹא תַעֲשֶׂה כָהֶם. בּוֹרוֹת נִשְׁבָּרִים:
וְלֹא תֵלֵךְ אִתָּם. בְּחוּקּוֹת מִשְׁמַרְתָּם. וְלֹא תְחַיֶּה אוֹתָם. וְלֹא תָחוֹן צָרִים:
בְּאַרְצְכֶם לֹא יִשְׁכְּנוּ. וּבָם לֹא תִתְחַתְּנוּ. וּמַשְׂכִּית לֹא תִתְּנוּ. כְּהַבְלֵי שָׁוְא שׁוֹמְרִים:
וְלֹא תַשְׁאִיר עוֹנֶה. בְּעִיר עָלַי תִּזְנֶה. וְהִיא לֹא תִבָּנֶה. אֲבָל הַחֲרֵם תַּחֲרִים:
וְחֵרֶם עַם נָבָק. בְּיָדְךָ לֹא יִדְבָּק. וְהָיְתָה תֵּל אָבָק. כְּמַהְפֵּכַת זָרִים:
וְעַמּוֹנִי תְפַזֵּר. וּמוֹאָבִי תְבַזֵּר. וְלֹא יָבֹא מַמְזֵר. בְּסוֹד עַם נִבְחָרִים:
וְלֹא תִדְרוֹשׁ שְׁלוֹמִים. לְאַכְזָרִים וְקָמִים. וְלֹא תָשִׂים דָּמִים. לְהַצִּיב מִכְמוֹרִים:
וְלֹא תַהֲרוֹג נְקִיִּים. וְצַדִּיקִים עֲנִיִּים. וְלֹא תַעֲזוֹב לְוִיִּם. וְלֹא יִנְחֲלוּ עָרִים:
וְלֹא תִדְרוֹשׁ אֶל מֵת. אֲשֶׁר אֵין עִמּוֹ אֱמֶת. וְהַמְּכַשֵּׁף הַצְמֵת. אֲשֶׁר עָרוֹם יַעְרִים:
וְלֹא תַסִּיג כָּל גְּבוּל. כְּמִצְוַת שׁוֹכֵן זְבוּל. וְלֹא תַשְׁחִית עֵץ יְבוּל. בְּעֵת תִּבְנֶה מְצָרִים:
וְלֹא תְחַלֵּל בִּתְּךָ. לְהַזְנוֹת מִדַּעְתְּךָ. וְלֹא תָשִׁיב אִשְׁתְּךָ. בְּלֶכְתָּהּ לַאֲחֵרִים:
וְלֹא תִשְׁכַּב אוֹתָהּ. בְּנִדַּת טוּמְאָתָהּ. וְאִשָּׁה אֶל אֲחוֹתָהּ. לְגַלּוֹת הַמְּקוֹרִים:
וְלֹא תִהְיֶה לְאִישׁ זָר. בְּעוּלַת אָח נִגְזָר. וְלֹא יֹאכַל מוּזָר. קֳדָשִׁים מוּזָרִים:
וְלֹא תִהְיֶה קְדֵשָׁה. בְּסוֹד עֵדָה קְדוֹשָׁה. וְלֹא יִהְיֶה עַל אִשָּׁה. לְבוּשֵׁי הַגְּבָרִים:
וְלֹא יִלְבְּשׁוּ אֲנָשִׁים. לְבוּשֵׁי הַנָּשִׁים. וְלֹא יִהְיוּ קְדֵשִׁים. בְּבָנִים נִכְשָׁרִים:
וְלֹא תִרְדֶּה בָּחֳרִי. בְּאָחִיךָ הָעִבְרִי. וְלֹא תַמְלִיךְ נָכְרִי. כִּי אִם בֶּן חוֹרִים:
וְלֹא יַרְבֶּה נָשִׁים. וְסוּסִים וּפָרָשִׁים. וְלֹא יָשִׁיב חָפְשִׁים. לְאֶרֶץ הַמִּצְרִים:
וְלֹא יָסוּר לְבָבוֹ. בְּסוּסֵי מֶרְכָּבוֹ. וְכַסְפּוֹ וּזְהָבוֹ. וְחֶמְדַּת הָאוֹצָרִים:
וְלֹא תִמְכּוֹר עִבְרִי. כְּמִמְכֶּרֶת נָכְרִי. וְלֹא תְתַעֵב מִצְרִי. בְּבוֹאוֹ לַמְּגוּרִים:
וְלֹא תְתַעֵב אֲדוֹמִי. בְּבוֹאוֹ לַהֲדוֹמִי. וְלֹא יָבוֹא בְעַמִּי. פְּצוּעִים וּשְׁבוּרִים:
וְלֹא תֵצֵא עִבְרִית. אֲשֶׁר תֻּקַּח בִּבְרִית. כְּצֵאת אָמָה נָכְרִית. בְּרָאשֵׁי אֵבָרִים:
וְאִם יִבְגּוֹד גְּבִירָהּ. וְהֶפְדָּהּ בִּמְחִירָהּ. וְלֹא יִמְשׁוֹל לְמָכְרָהּ. כְּיֶתֶר מִמְכָּרִים:
וְלֹא יִגְרַע אוֹתָהּ. שְׁאֵרָהּ וּכְסוּתָהּ. וּמִשְׁכַּב עוֹנָתָהּ. בְּקַחְתּוֹ בַּת אֲחֵרִים:
וְאַרְבַּע נָשִׁים הֵן. אֲסוּרוֹת לַכֹּהֵן. וְהָאַחַת מֵהֶן. כְּשֵׁרָה לַאֲחֵרִים:
וְלֹא יְגַדֵּל פֶּרַע. וְלֹא יִפְרוֹם קֶרַע. וְלֹא יְחַלֵּל זֶרַע. וּמִקְדַּשׁ מִבְצָרִים:
וְלֹא יְטַמֵּא עַצְמוֹ. לְאָבִיו וּלְאִמּוֹ. אֲבָל עַל מֵת עַמּוֹ. אֲשֶׁר אֵין לוֹ קוֹבְרִים:
וּבִדְבַר לֹא יֵצֵא. לְפָנָיו יֵרָצֶה. וְיָגִיל עֵת יֵצֵא. כְּחָתָן מֵחֲדָרִים:
וּבִדְבַר לֹא יִכְבֶּה. גְּדוּלָּתוֹ תִּרְבֶּה. וְנֵרוֹ לֹא יִכְבֶּה. וְזָרַח כִּמְאוֹרִים:
וְלֹא תֹאכַל פִּתּוֹ. בְּקַחַת זָר בִּתּוֹ. אֲבָל תֹּאכַל אִתּוֹ. בְּשׁוּבָהּ כַּנְּעוּרִים:
וְגַם לֹא יֹאכַל בּוֹ. שְׂכִירוֹ וְתוֹשָׁבוֹ. וְעַל מֵת לֹא יָבוֹא. בְּאָהֳלֵי תַמְרוּרִים:
וְלֹא יֹאכַל פְּנִימִית. בְּזָדוֹן וּבְתַרְמִית. וְלֹא יַבְדִּיל לְהַצְמִית. שְׁסוּעֵי הַתּוֹרִים:
וְלֹא יֹאכַל מִנְחַת. כְּלִילוֹ פֶּן יֵחַת. וְלֹא תִזְבַּח מָשְׁחַת. בְּאֵילִים עִם כָּרִים:
וְזָר לֹא יִתְגַּלַּע. רְאוֹת קֹדֶשׁ כְּבַלַּע. וְטָמֵא פֶּן יְבֻלַּע. בְּשָׁאוֹן וּשְׁבָרִים:
וְלֹא יְעַבֵּט אוֹרַח. וְיִהְיֶה לְטוֹרַח. וְלֹא יִהְיֶה כְקוֹרַח. וְרֵעָיו הַמַּמְרִים:
וְלֵוִי הַזָּקֵן. עֲבוֹדָה לֹא יְתַקֵּן. וְלֹא תִקַּח אֵם מִקֵּן. עַל פְּנֵי הַצִּפֳּרִים:
וְיָמֶיךָ אַמְשִׁיךְ. בְּמִצְוַת לֹא תַשִּׁיךְ. וְאוֹרְךָ לֹא יַחְשִׁיךְ. בְּאֶרֶץ לֹא סְדָרִים:
וְלֹא תִמּוֹט רַבַּת. בְּמִצְוַת גְּבוּל שַׁבָּת. וְחוֹק מִנְחַת מַחֲבַת. וּמַאֲפֵה תַנּוּרִים:
וּבֵיתְךָ לֹא יִסַּח. בְּמִצְוַת לֹא יִזַּח. וּמִצְוַת פֶּן תִּשְׁכַּח. שְׁתַּיִם נֶאֱמָרִים:
וּמִצְוַת לֹא תְבַעֲרוּ. וּמִצְוַת לֹא תְשַׁקְּרוּ. וּמִצְוַת לֹא תִמְכְּרוּ. שְׂדֵה מִגְרַשׁ עָרִים:
וּמִצְוַת כִּלְאַיִם. וְרֶכֶב וְרֵחַיִם. וְאַבְנֵי מֹאזְנַיִם. בְּכִיסֵי מִסְחָרִים:
וּמִצְוַת לֹא תַחְבּוֹל. עֲלֵי כָתֵף תִּסְבּוֹל. וְחֵילְךָ לֹא יִבּוֹל. וְתִפְרַח כַּתְמָרִים:
וְחוֹק לֹא תוֹסִיפוּ. וְחוֹק לֹא תַחֲנִיפוּ. וְחוֹק לֹא תַקִּיפוּ. פְּאַת רֹאשׁ כַּכּוּמָרִים:
וּמִצְוַת לֹא תִגְרַע. לְעוֹלָם לֹא תִפְרַע. וְחוֹק מוֹצִיא שֵׁם רָע. וְשָׂם עֲלִילוֹת דְּבָרִים:
וּמִצְוַת לֹא תָנִיף. לְעוֹלָם לֹא תַחֲנִיף. וּמִצְוַת פֶּן יוֹסִיף. לְהַכּוֹת נוֹסָרִים:
וְחוֹק לֹא תָתוּרוּ. וְחוֹק לֹא תָגוּרוּ. וְחוֹק לֹא תָסוּרוּ. וְחוֹק שְׁבוּעַת אִסָּרִים:
וְחוֹק לֹא אָכַלְתִּי. וְחוֹק לֹא בִעַרְתִּי. וְחוֹק לֹא נָתַתִּי. לְמֵתִים נִגְזָרִים:
וּבִתְעוּדַת הַעֲבוֹט. מֵישָׁרִים תִּשְׁפּוֹט. וּפָרָשַׁת הֶדְיוֹט. וּפָרָשַׁת שׁוֹטְרִים:
וְכָל פָּרָשִׁיּוֹת. מְפִיקֵי תוּשִׁיּוֹת. אֲשֶׁר הֵם בַּצִּיּוֹת. מְקוֹם נְהָרִים יְאוֹרִים:
וְאֶחָד וְשִׁבְעִים הֵם. חֲרוּצִים בְּמַעֲשֵׂיהֶם. וְאֵלֶּה פְקוּדֵיהֶם. בְּמִכְתָּב נִזְכָּרִים:
אֲשֶׁר יִתֵּן זַּרְעוֹ. לַמּוֹלֶךְ בְּרֹב רִשְׁעוֹ. וְהַמְּגַדֵּף יִשְׁעוֹ. אֲשֶׁר יָדוֹ יָרִים:
וּמֵסִית וּמְחַלֵּל. וּבֶן סוֹבֵא וְזוֹלֵל. וּמַכֵּה וּמְקַלֵּל. וּמַדִּיחַ עָרִים:
וְהַדּוֹבֵר סָרָה. בְּשֵׁם עֲבוֹדָה זָרָה. וּבַעַל אוֹב לְזָרָא. וְיִדְּעוֹנִי שְׁקָרִים:
וְנֶפֶשׁ רוֹצַחַת. לְפַחַד וָפַחַת. וְעַם עִיר נִדַּחַת. בְּחֶרֶב נִדְקָרִים:
וְנָבִיא מִדַּעַת. תְּכַסֶּה תּוֹלַעַת. וְאִשָּׁה נִרְבַּעַת. לִבְהֵמַת הַיְּעָרִים:
וְהַבָּא עַל אֲרוּסָה. וְעַל כַּלָּה כְנוּסָה. וְעַל בְּהֵמָה אֲבוּסָה. וְהַבָּא עַל זְכָרִים:
וּבָא עַל אֵשֶׁת אִישׁ. מְאֹד יַחְפִּיר וְיַבְאִישׁ. וְדוֹדָתוֹ מַבְאִישׁ. שְׁנֵי אֵלֶּה אֲסוּרִים:
וּבָא עַל יוֹלַדְתּוֹ. וּבִתּוֹ וּבַת אִשְׁתּוֹ. וּבַת בְּנוֹ וּבַת בִּתּוֹ. לְפִי שְׁאוֹל נִפְזָרִים:
וְהַבָּא עַל אֲחוֹתוֹ. וְהַבָּא עַל חֲמוֹתוֹ. וּבָא עַל אֲחוֹת אִשְׁתּוֹ. וּבָא עַל אֲחוֹת הוֹרִים:
וְעַל אִשָּׁה וּבִתָּהּ. וּבַת בְּנָהּ וּבַת בִּתָּהּ. וְעַל נִדָּה בְּשִׁבְתָּהּ. לְמַלֹּאת מִשְׁמָרִים:
וְעַל אֵשֶׁת אָבִיו. וְעַל אֵשֶׁת אָחִיו. וְאֵם חֲמוֹתוֹ וְחָמִיו. כָּל אֵלֶּה אֲרוּרִים:
וּמִשְׁפַּט בְּנוֹת כֹּהֵן. וּמִשְׁפַּט זוֹמְמֵיהֶן. וּמִשְׁפַּט בּוֹעֲלֵיהֶן. לְהַשְׁמִיד וּלְהַחְרִים:
וְהַבּוֹעֵל אֲרַמִּית. בְּרָעָתוֹ לְהַצְמִית. וְהַגּוֹנֵב בְּתַרְמִית. קְסָווֹת נִבְחָרִים:
וְהַגּוֹנֵב אָדָם. בַּשַּׁחַת יֵרָדַם. וְאוֹכֵל חֵלֶב וָדָם. יְהִי כִּיקַר כָּרִים:
וְאָרוּר כָּל נוֹכֵל. אֲשֶׁר פִּגּוּל יֹאכֵל. וְהוּא רַק יֵאָכֵל. בְּאִשֵּׁי הַכִּיּוֹרִים:
וְשִׁבְרוֹן מָתְנַיִם. לְאוֹחֵז עֵינַיִם. וְנִקְיוֹן שִׁנַּיִם. לְאוֹכֵל נוֹתָרִים:
וְעוֹבֵד מִפְלֶצֶת. לְכָלָה וְנֶחֱרֶצֶת. וְאוֹכֵל מַחְמֶצֶת. בְּיָמִים נִשְׁמָרִים:
וּפֶסַח לֹא יַעֲשֶׂה. בַּחֹשֶׁךְ שְׁמוֹ יְכוּסֶּה. וְאִישׁ אוֹכֵל וְעוֹשֶׂה. בְּיוֹם הַכִּפּוּרִים:
וְאִישׁ סָךְ וּמְפַטֵּם. בְּעֶבְרָתִי אֶשְׁפְּטֵם. וְאִישׁ אָזְנוֹ יַאְטֵם. לְמִשְׁפַּט הַמּוֹרִים:
וְהַשּׁוֹחֵט מִחוּץ. וְהַמַּעֲלֶה בַּחוּץ. וּשְׁאֵינוֹ רָחוּץ. בְּמֵימֵי כִיּוֹרִים:
וְעוֹבֵר עַל דַּם בְּרִית. תְּהִי נַפְשׁוֹ לְהַכְרִית. וְהִיא תִקְוָה וְאַחֲרִית. לְשַׁלֵּח הָאֲסִירִים:
וְהַטָּמֵא יוּדָּשׁ. בְּבוֹאוֹ לַמִּקְדָּשׁ. וְהָאוֹכֵל נִקְדָּשׁ. בְּטוּמְאַת הַבְּשָׂרִים:
וְהְָאוֹכֵל טֶבֶל. יְבוֹאֵהוּ חֶבֶל. וְיִיגַע לַהֶבֶל. וְיֹאחֲזוּהוּ צִירִים:
וְזָר אוֹכֵל תְּרוּמָה. תְּבוֹאֵהוּ מְהוּמָה. וְכֹהֵן כִּי יִטְמָא. בְּעָבְרוֹ מַאֲמָרִים:
וּמִתֵּבֵל יוּסָר. טְבוּל יוֹם הַנִּמְסָר. לְשַׁמֵּשׁ וּמְחוּסָּר. בְּגָדִים נִסְפָּרִים:
וְהַזָּר הָעוֹבֵד. בְּמוֹתוֹ אֶכָּבֵד. וּמֵאֶרֶץ יֹאבֵד. מְחוּסַּר כִּפּוּרִים:
וְהַמָּוֶת רָתוּי. לְפָרוּעַ וְשָׁתוּי. וְטָמֵא הָאָתוּי. לְשַׁמֵּשׁ בַּחֲדָרִים:

חתימה

כָּלוּ מִצְוֹת עֲשֵׂה. וּמִצְוֹת לֹא תַעֲשֶׂה. בְּעֶזְרַת צוּר מַחְסֶה. וּמָגֵן לַיְשָׁרִים:
אֵלֶּה הַתּוֹרוֹת. יְסוֹדוֹת נִבְחָרוֹת. וְלָהֵנָּה פֹּארוֹת. כְּסַנְסִנֵּי תְמָרִים:
יְמַהֵר אֵל עֶלְיוֹן. לְקַבֵּץ עַם אֶבְיוֹן. וְיִבְנֶה הַר צִיּוֹן. וְעֵמֶק הַפְּגָרִים:
וְיָקִים אֶת סוּכּוֹ. וְיַגִּיהַּ חֶשְׁכּוֹ. וְיִתֵּן עוֹז לְמַלְכּוֹ. וְקַרְנֵנוּ יָרִים:
וְיִרְאוּ מְתֵי מִסְפָּר. בְּקוּם יְשֵׁנֵי עָפָר. וְכִתְקוֹעַ שׁוֹפָר. וְכִנְשׂוֹא נֵס הָרִים:
וּמֵרֹאשׁ הוֹמִיּוֹת. תְּרַנֵּנָּה פִּיוֹת. וּמִמְּעוֹנוֹת אֲרָיוֹת. וּמֵהַרְרֵי נְמֵרִים:
וְאָז כָּל עַם שׁוֹגֵג. בְּרָעָה יִתְמוֹגֵג. לְקוֹל הָמוֹן חוֹגֵג. בְּשִׂמְחָה וּבְשִׁירִים:
וְעַל הַר הַגָּלִיל. נְהַלֵּךְ בֶּחָלִיל. וְאָז תַּחְפֹּץ כָּלִיל. וְאָז יַעֲלוּ פָרִים:
