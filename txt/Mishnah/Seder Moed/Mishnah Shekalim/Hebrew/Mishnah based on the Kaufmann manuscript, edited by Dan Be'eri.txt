משנה שקלים



Chapter 1

<small>א</small>
בְּאֶחָד בְּאָדָר, 
מְשַׁמְּעִים עַל הַשְּׁקָלִים וְעַל הַכִּלְאַיִם. 
וּבַחֲמִשָּׁה עָשָׂר בּוֹ, 
קוֹרִין אֶת הַמְּגִלָּה בַכְּרַכִּים, 
וּמְתַקְּנִים אֶת הַדְּרָכִים וְאֶת הָרְחוֹבוֹת, 
וְאֶת מִקְווֹת הַמַּיִם, 
וְעוֹשִׂין כָּל צָרְכֵי הָרַבִּים, 
וּמְצַיְּנִים אֶת הַקְּבָרוֹת, 
וְיוֹצְאִין אַף עַל הַכִּלְאַיִם.

<small>ב</small>
אָמַר רְבִּי יְהוּדָה: 
בָּרִאשׁוֹנָה, 
הָיוּ עוֹקְרִים וּמַשְׁלִיכִים לִפְנֵיהֶם. 
מִשֶּׁרַבּוּ עוֹבְרֵי עֲבֵרָה, 
הָיוּ מַשְׁלִיכִים עַל הַדְּרָכִים; 
הִתְקִינוּ שֶׁיְּהוּ מַבְקִירִים אֶת כָּל הַשָּׂדֶה.

<small>ג</small>
בַּחֲמִשָּׁה עָשָׂר בּוֹ, 
שֻׁלְחָנוֹת יָשְׁבוּ בַמְּדִינָה. 
בְּעֶשְׂרִים וַחֲמִשָּׁה, יָשְׁבוּ בַמִּקְדָּשׁ; 
מִשֶּׁיָּשְׁבוּ בַמִּקְדָּשׁ, הִתְחִילוּ לְמַשְׁכֵּן. 
וְאֶת מִי מְמַשְׁכְּנִים? 
לְוִיִּם וְיִשְׂרְאֵלִים, 
גֵּרִים וַעֲבָדִים מְשֻׁחְרָרִים, 
אֲבָל לֹא נָשִׁים וְלֹא עֲבָדִים וְלֹא קְטַנִּים. 
וְכָל קָטָן שֶׁהִתְחִיל אָבִיו לִשְׁקוֹל עַל יָדוֹ, 
אֵינוּ פוֹסֵק. 
וְאֵין מְמַשְׁכְּנִין אֶת הַכֹּהֲנִים, 
מִפְּנֵי דַרְכֵי שָׁלוֹם.

<small>ד</small>
אָמַר רְבִּי יְהוּדָה: 
הֵעִיד בֶּן כּוּבָרִי בְיַבְנֶה, 
שֶׁכָּל כֹּהֵן שֶׁהוּא שׁוֹקֵל אֵינוּ חוֹטֵא. 
אָמַר לוֹ רַבָּן יוֹחָנָן בֶּן זַכַּי: 
לֹא כִי, 
אֶלָּא שֶׁכָּל כֹּהֵן שֶׁאֵינוּ שׁוֹקֵל, חוֹטֵא, 
אֶלָּא שֶׁהַכֹּהֲנִים דּוֹרְשִׁים מִקְרָא זֶה לְעַצְמָן: 
"וְכָל מִנְחַת כֹּהֵן כָּלִיל תִּהְיֶה לֹא תֵאָכֵל" (ויקרא ו,טז) 
הוֹאִיל וְהָעֹמֶר וּשְׁתֵּי הַלֶּחֶם וְלֶחֶם הַפָּנִים שֶׁלָּנוּ, 
הֵיאָךְ הֵן נֶאֱכָלִים? 

<small>ה</small>
אַף עַל פִּי שֶׁאָמָרוּ: 
אֵין מְמַשְׁכְּנִים נָשִׁים וַעֲבָדִים וּקְטַנִּים, 
אֲבָל אִם שָׁקָלוּ, מְקַבְּלִים מִיָּדָם. 
נָכְרִי וְכוּתִי שֶׁשָּׁקָלוּ, אֵין מְקַבְּלִים מִיָּדָם. 
אֵין מְקַבְּלִים מִיָּדָם קִנֵּי זָבִין וְקִנֵּי זָבוֹת וְקִנֵּי יוֹלְדוֹת; 
חַטָּאוֹת וַאֲשָׁמוֹת מְקַבְּלִים מִיָּדָם. 
זֶה הַכְּלָל: 
כָּל שֶׁהוּא נִדָּר וְנִדָּב, 
מְקַבְּלִים מִיָּדָם; 
וְכָל שֶׁאֵינוּ לֹא נִדָּר וְלֹא נִדָּב, 
אֵין מְקַבְּלִים מִיָּדָם. 
וְכֵן הוּא מְפָרֵשׁ עַל יְדֵי עֶזְרָה: (עזרא ד,ג) 
"לֹא לָכֶם וָלָנוּ לִבְנוֹת בַּיִת לֵאלֹהֵינוּ".

<small>ו</small>
וְאֵלּוּ חַיָּבִין בַּקָּלְבּוֹן: 
לְוִיִּם וְיִשְׂרְאֵל גֵּרִים וַעֲבָדִים מְשֻׁחְרָרִים, 
אֲבָל לֹא נָשִׁים וְלֹא עֲבָדִים וְלֹא קְטַנִּים. 
הַשּׁוֹקֵל עַל יַד עֶבֶד, וְעַל יַד אִשָּׁה, 
וְעַל יַד כֹּהֵן, וְעַל יַד הַקָּטָן, פָּטוּר. 
עַל יָדוֹ וְעַל יַד חֲבֵרוֹ, חַיָּב קָלְבּוֹן אֶחָד. 
רְבִּי מֵאִיר אוֹמֵר: 
שְׁנֵי קָלְבּוֹנוֹת. 
נָתָן סֶלַע לִטֹּל שֶׁקֶל, 
חַיָּב שְׁנֵי קָלְבּוֹנוֹת.

<small>ז</small>
הַשּׁוֹקֵל עַל יַד עָנִי וְעַל יַד שְׁכֵנוֹ 
וְעַל יַד בֶּן עִירוֹ, פָּטוּר. 
וְאִם הִלְוָן, חַיָּב. 
הָאַחִים הַשּׁוּתָפִין שֶׁחַיָּבִים בַּקָּלְבּוֹן, 
פְּטוּרִים מִמַּעְשַׂר בְּהֵמָה; 
וְשֶׁחַיָּבִים בְּמַעְשַׂר בְּהֵמָה, 
פְּטוּרִים מִן הַקָּלְבּוֹן. 
וְכַמָּה הוּא קָלְבּוֹן? 
מָעָה כֶסֶף. 
דִּבְרֵי רְבִּי מֵאִיר. 
וַחֲכָמִים אוֹמְרִים: 
חֵצִי.





Chapter 2

<small>א</small>
מְצָרְפִין שְׁקָלִים דְּרָכוֹנוֹת מִפְּנֵי מַסּוּי הַדֶּרֶךְ. 
כַּשֵּׁם שֶׁהָיוּ שׁוֹפָרוֹת בַּמִּקְדָּשׁ, 
כָּךְ הָיוּ בַמְּדִינָה. 
בְּנֵי הָעִיר שֶׁשִּׁלְּחוּ אֶת שִׁקְלֵיהֶם,
וְנִגְנָבוּ אוֹ שֶׁאָבָדוּ, 
אִם נִתְרְמָה הַתְּרוּמָה, נִשְׁבָּעִים לַגִּזְבָּרִים, 
וְאִם לָאו, נִשְׁבָּעִים לִבְנֵי הָעִיר, 
וּבְנֵי הָעִיר שׁוֹקְלִים תַּחְתֵּיהֶם. 
נִמְצָאוּ, אוֹ שֶׁהֶחְזִירוּם הַגַּנָּבִים, 
אֵלּוּ וָאֵלּוּ שְׁקָלִים, 
וְאֵין עוֹלִים לָהֶם לְשָׁנָה הַבָּאָה.

<small>ב</small>
הַנּוֹתֵן שִׁקְלוֹ לַחֲבֵרוֹ לִשְׁקוֹל עַל יָדוֹ, 
וּשְׁקָלוֹ עַל יְדֵי עַצְמוֹ, 
אִם נִתְרְמָה הַתְּרוּמָה, 
מָעַל. 
הַשּׁוֹקֵל שִׁקְלוֹ מִן הַהֶקְדֵּשׁ, 
אִם נִתְרְמָה הַתְּרוּמָה, 
וְאִם קָרְבָה הַבְּהֵמָה, 
מָעַל. 
מִמַּעֲשֵׂר שֵׁנִי, וּמִדְּמֵי שְׁבִיעִית, 
יֹאכַל כְּנֶגְדָּן.

<small>ג</small>
הַמְכַנֵּס מָעוֹת וְאָמַר: 
"הֲרֵי אֵלּוּ שִׁקְלִי", 
בֵּית שַׁמַּי אוֹמְרִים: 
מוֹתָרָן נְדָבָה. 
וּבֵית הֶלֵּל אוֹמְרִים: 
מוֹתָרָן חֻלִּים. 
"שֶׁאָבִיא מֵהֶן שִׁקְלִי", 
שָׁוִים שֶׁהַמּוֹתָר חֻלִּין. 
"אֵלּוּ לְחַטָּאתִי", 
שָׁוִים שֶׁהַמּוֹתָר נְדָבָה; 
"שֶׁאָבִיא מֵהֶן חַטָּאתִי", 
שָׁוִים שֶׁהַמּוֹתָר חֻלִּים.

<small>ד</small>
אָמַר רְבִּי שִׁמְעוֹן: 
מַה בֵּין שְׁקָלִים לַחַטָּאת? 
אֶלָּא שֶׁהַשְּׁקָלִים, יֵשׁ לָהֶם קִצְבָה, 
וְלַחַטָּאת אֵין לָהּ קִצְבָה. 
רְבִּי יְהוּדָה אוֹמֵר: 
אַף לִשְׁקָלִים אֵין לָהֶם קִצְבָה; 

<small>ה</small>
שֶׁכְּשֶׁעָלוּ יִשְׂרָאֵל מִן הַגּוֹלָה, 
הָיוּ שׁוֹקְלִים דְּרָכוֹנוֹת, 
חָזְרוּ לִשְׁקוֹל סְלָעִים, 
חָזְרוּ לִשְׁקוֹל טְבָעִים, 
בִּקְשׁוּ לִשְׁקוֹל דִּינָרִים. 
אָמַר רְבִּי שִׁמְעוֹן: 
אַף עַל פִּי כֵן, יַד כֻּלָּם שָׁוָה, 
אֲבָל חַטָּאת, 
זֶה מֵבִיא בְסֶלַע, וְזֶה מֵבִיא בִשְׁתַּיִם, 
וְזֶה מֵבִיא בְשָׁלוֹשׁ.

<small>ו</small>
מוֹתַר שְׁקָלִים, חֻלִּים. 
מוֹתַר עֲשִׂירִית הָאֵפָה, 
מוֹתַר קִנֵּי זָבִים, קִנֵּי זָבוֹת, קִנֵּי יוֹלְדוֹת, 
חַטָּאוֹת וַאֲשָׁמוֹת, 
מוֹתָרָן נְדָבָה. 
זֶה הַכְּלָל: 
כָּל שֶׁהוּא בָא מִשּׁוּם חֵטְא וּמִשּׁוּם אַשְׁמָה, 
מוֹתָרָן נְדָבָה. 

<small>ז</small>
מוֹתַר עוֹלָה לָעוֹלָה. 
מוֹתַר מִנְחָה לַמִּנְחָה. 
מוֹתַר שְׁלָמִים לִשְׁלָמִים. 
מוֹתַר הַפֶּסַח לִשְׁלָמִים. 
מוֹתַר נְזִירִים לִנְזִירִים. 
מוֹתַר נָזִיר לִנְדָבָה. 
מוֹתַר שְׁבוּיִים לִשְׁבוּיִים. 
מוֹתַר שָׁבוּי לְאוֹתוֹ שָׁבוּי. 
מוֹתַר עֲנִיִּים לַעֲנִיִּים. 
מוֹתַר עָנִי לְאוֹתוֹ עָנִי. 
מוֹתַר הַמֵּתִים לַמֵּתִים. 
מוֹתַר הַמֵּת לְיוֹרְשָׁיו. 
רְבִּי מֵאִיר אוֹמֵר: 
מוֹתַר הַמֵּת יְהֵא מֻנָּח עַד שֶׁיָּבֹא אֵלִיָּהוּ. 
רְבִּי נָתָן אוֹמֵר: 
מוֹתַר הַמֵּת, בּוֹנִים לוֹ נְפָשׁ עַל קִבְרוֹ.





Chapter 3

<small>א</small>
בִּשְׁלֹשָׁה פְרָקִים בַּשָּׁנָה 
תּוֹרְמִין אֶת הַלִּשְׁכָּה: 
בִּפְרַס הַפֶּסַח, 
בִּפְרַס עֲצֶרֶת, 
בִּפְרַס הֶחָג, 
וְהֵן גֳּרָנוֹת שֶׁלְּמַעְשַׂר בְּהֵמָה. 
דִּבְרֵי רְבִּי עֲקִיבָה. 
בֶּן עַזַּי אוֹמֵר: 
בְּעֶשְׂרִים וְתִשְׁעָה בַאֲדָר, 
בְּאֶחָד בְּסִיוָן, 
בְּעֶשְׂרִים וְתִשְׁעָה בֶאֱלוּל. 
רְבִּי לְעָזָר וּרְבִּי שִׁמְעוֹן אוֹמְרִים: 
בְּאֶחָד בְּנִיסָן, 
בְּאֶחָד בְּסִיוָן, 
בְּאֶחָד בְּתִשְׁרִי. 
וְלָמָּה אָמְרוּ בְעֶשְׂרִים וְתִשְׁעָה בֶאֱלוּל, 
וְלֹא אָמְרוּ בְאֶחָד בְּתִשְׁרִי? 
מִפְּנֵי שֶׁהוּא יוֹם טוֹב, 
וְאֵי אֶפְשָׁר לְעַשֵּׂר בְּיוֹם טוֹב, 
לְפִיכָךְ הִקְדִּימוּהוּ בְּעֶשְׂרִים וְתִשְׁעָה בֶאֱלוּל.

<small>ב</small>
בְּשָׁלוֹשׁ קֻפּוֹת שֶׁלְּשָׁלוֹשׁ שָׁלוֹשׁ סְאִים 
תּוֹרְמִין אֶת הַלִּשְׁכָּה, 
וְכָתוּב עֲלֵיהֶם אָלֶ"ף בֵּי"ת גִּימֶ"ל. 
רְבִּי יִשְׁמָעֵאל אוֹמֵר: 
יְוָנִית כָּתוּב עֲלֵיהֶם אַלְפָ"א בִּיטָ"א גַמָּ"א. 
אֵין הַתּוֹרֵם נִכְנָס לֹא בְפֵרְגוּד חָפוּת, 
וְלֹא בְמַנְעָל, וְלֹא בְסַנְדָּל, 
וְלֹא בִתְפִלָּה, וְלֹא בְקָמִיעַ; 
שֶׁמֵּא יַעְנִי, וְיֹאמְרוּ: 
מֵעֲוֹן הַלִּשְׁכָּה הֶעְנִי, 
אוֹ שֶׁמֵּא יַעְשִׁיר וְיֹאמְרוּ: 
מִתְּרוּמַת הַלִּשְׁכָּה הֶעְשִׁיר; 
שֶׁאָדָם צָרִיךְ לָצֵאת יְדֵי הַבְּרִיּוֹת 
כְּדֶרֶךְ שֶׁהוּא צָרִיךְ לָצֵאת יְדֵי הַמָּקוֹם, 
שֶׁנֶּאֱמַר: (במדבר לב,כב) 
"וִהְיִיתֶם נְקִיִּם מֵיי וּמִיִּשְׂרָאֵל", 
וְאוֹמֵר: (משלי ג,ד) 
"וּמְצָא חֵן וְשֵׂכֶל טוֹב בְּעֵינֵי אֱלהִים וְאָדָם".

<small>ג</small>
שֶׁלְּבֵית רַבָּן גַּמְלִיאֵל 
הָיָה נִכְנָס וְשִׁקְלוֹ בֵין אֶצְבְּעוֹתָיו, 
וְזוֹרְקוֹ לִפְנֵי הַתּוֹרֵם, 
וְהַתּוֹרֵם מִתְכַּוֵּן וְדוֹחֲפוֹ לַקֻּפָּה. 
אֵין הַתּוֹרֵם תּוֹרֵם עַד שֶׁיֹּאמַר לָהֶם: 
"אֶתְרֹם", 
וְהֵן אוֹמְרִים לוֹ: 
"תְּרֹם, תְּרֹם, תְּרֹם!" 
שָׁלוֹשׁ פְּעָמִים.

<small>ד</small>
תָּרַם אֶת הָרִאשׁוֹנָה, וְחִפָּה בִּקְטַבְלָיוֹת, 
שְׁנִיָּה, וְחִפָּה בִּקְטַבְלָיוֹת, 
שְׁלִישִׁית לֹא הָיָה מְחַפָּהּ. 
וְלָמָּה לֹא הָיָה מְחַפָּהּ?
שֶׁמֵּא יִשְׁכַּח וְיִתְרֹם מִן הַדָּבָר הַתָּרוּם. 
תָּרַם אֶת הָרִאשׁוֹנָה לְשֵׁם אֶרֶץ יִשְׂרָאֵל, 
שְׁנִיָּה לְשֵׁם הַכְּרַכִּים הַמֻּקָּפִים לָהּ, 
שְׁלִישִׁית לְשֵׁם בָּבֶל וּלְשֵׁם מָדַי וּלְשֵׁם מְדִינוֹת הָרְחוֹקוֹת.





Chapter 4

<small>א</small>
הַתְּרוּמָה מָה הָיוּ עוֹשִׂין בָּהּ? 
לוֹקְחִין בָּהּ תְּמִידִים וּמוּסָפִים וְנִסְכֵּיהֶם, 
וְהָעֹמֶר וּשְׁתֵּי הַלֶּחֶם וְלֶחֶם הַפָּנִים, 
וְכָל קָרְבְּנוֹת הַצִּבּוּר. 
שׁוֹמְרֵי סְפָחִים בַּשְּׁבִיעִית, 
נוֹטְלִים שְׂכָרָן מִתְּרוּמַת הַלִּשְׁכָּה. 
רְבִּי יוֹסֵה אוֹמֵר: 
אַף הָרוֹצֶה מִתְנַדֵּב שׁוֹמֵר חִנָּם. 
אָמְרוּ לוֹ: 
אַף אַתָּה אוֹמֵר שֶׁאֵינָן בָּאִים אֶלָּא מִשֶּׁלַּצִּבּוּר.

<small>ב</small>
פָּרָה וְשָׂעִיר הַמִּשְׁתַּלֵּחַ וְלָשׁוֹן שֶׁלִּזְהוֹרִית 
בָּאִים מִתְּרוּמַת הַלִּשְׁכָּה. 
כֶּבֶשׁ פָּרָה וְכֶבֶשׁ שָׂעִיר הַמִּשְׁתַּלֵּחַ וְלָשׁוֹן שֶׁבֵּין קְרָנָיו, 
וְאַמַּת הַמַּיִם וְחוֹמַת הָעִיר וּמִגְדְּלוֹתֶיהָ, 
וְכָל צָרְכֵי הָעִיר 
בָּאִים מִשִּׁירֵי הַלִּשְׁכָּה. 
אַבָּא שָׁאוּל אוֹמֵר: 
כֶּבֶשׁ פָּרָה, 
כֹּהֲנִים גְּדוֹלִים עוֹשִׂין אוֹתוֹ מִשֶּׁלְּעַצְמָן.

<small>ג</small>
מוֹתַר שִׁירֵי לִשְׁכָּה מָה הָיוּ עוֹשִׂין בָּהֶן? 
לוֹקְחִין בָּהֶן יֵינוֹת וּשְׁמָנִים וּסְלָתוֹת, 
וְהַשָּׂכָר לַהֶקְדֵּשׁ. 
דִּבְרֵי רְבִּי יִשְׁמָעֵאל. 
רְבִּי עֲקִיבָה אוֹמֵר: 
אֵין מִשְׂתַּכְּרִין בְּשֶׁלַּהֶקְדֵּשׁ וְלֹא מִשֶּׁלָּעֲנִיִּים.

<small>ד</small>
מוֹתַר הַתְּרוּמָה מָה הָיוּ עוֹשִׂין בָּהּ? 
רִקּוּעֵי זָהָב צִפּוּי לְבֵית קֹדֶשׁ הַקֳּדָשִׁים. 
רְבִּי יִשְׁמָעֵאל אוֹמֵר: 
מוֹתַר הַפֵּרוֹת לְקַיִץ הַמִּזְבֵּחַ, 
וּמוֹתַר תְּרוּמָה לִכְלֵי שָׁרֵת. 
רְבִּי עֲקִיבָה אוֹמֵר: 
מוֹתַר תְּרוּמָה לְקַיִץ הַמִּזְבֵּחַ, 
וּמוֹתַר נְסָכִים לִכְלֵי שָׁרֵת. 
רְבִּי חֲנַנְיָה סְגַן הַכֹּהֲנִים אוֹמֵר: 
מוֹתַר נְסָכִים לְקַיִץ הַמִּזְבֵּחַ, 
וּמוֹתַר תְּרוּמָה לִכְלֵי שָׁרֵת. 
וְזֶה וָזֶה לֹא הָיוּ מוֹדִים בַּפֵּרוֹת.

<small>ה</small>
מוֹתַר הַקְּטֹרֶת מָה הָיוּ עוֹשִׂין בָּהּ? 
מַפְרִישִׁין מִמֶּנָּה שְׂכַר הָאֻמָּנִים, 
וּמְחַלְּלִים אוֹתָהּ עַל מָעוֹת הָאֻמָּנִים, 
וְנוֹתְנִים אוֹתָהּ לָאֻמָּנִים בִּשְׂכָרָן, 
וְחוֹזְרִים וְלוֹקְחִין אוֹתָהּ מִתְּרוּמָה חֲדָשָׁה. 
אִם בָּא חָדָשׁ בִּזְמַנּוֹ, 
לוֹקְחִין אוֹתָהּ מִתְּרוּמָה חֲדָשָׁה, 
וְאִם לָאו, מִן הַיְשָׁנָה.

<small>ו</small>
הַמַּקְדִּישׁ נְכָסָיו, 
וְהָיוּ בָהֶן דְּבָרִים רְאוּיִין לְקָרְבְּנוֹת הַצִּבּוּר, 
יִנָּתְנוּ לָאֻמָּנִים בִּשְׂכָרָן. 
דִּבְרֵי רְבִּי עֲקִיבָה. 
אָמַר לוֹ בֶן עַזַּי: 
אֵינָה הִיא הַמִּדָּה, 
אֶלָּא מַפְרִישִׁין מֵהֶן שְׂכַר הָאֻמָּנִים, 
וּמְחַלְּלִין אוֹתָן עַל מָעוֹת הָאֻמָּנִים, 
וְנוֹתְנִין אוֹתָם לָאֻמָּנִין בִּשְׂכָרָן, 
וְחוֹזְרִים וְלוֹקְחִים אוֹתָם מִתְּרוּמָה חֲדָשָׁה.

<small>ז</small>
הַמַּקְדִּישׁ נְכָסָיו, 
וְהָיְתָה בָהֶן בְּהֵמָה רְאוּיָה עַל גַּבֵּי הַמִּזְבֵּחַ, 
זְכָרִים וּנְקֵבוֹת, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
זְכָרִים יִמָּכְרוּ לִצְרִיכֵי עוֹלוֹת, 
וּנְקֵבוֹת יִמָּכֵרוּ לִצְרִיכֵי זִבְחֵי שְׁלָמִים, 
וְיָבִיא בִדְמֵיהֶם עוֹלוֹת, 
וּדְמֵיהֶם יִפְּלוּ עִם שְׁאָר נְכָסִים לְבֶדֶק הַבַּיִת. 

<small>ח</small>
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
זְכָרִים עַצְמָן יִקְרְבוּ עוֹלוֹת, 
וּנְקֵבוֹת יִמָּכְרוּ לִצְרִיכֵי זִבְחֵי שְׁלָמִים, 
וְיָבִיא בִדְמֵיהֶם עוֹלוֹת, 
וּשְׁאָר נְכָסִים יִפְּלוּ לְבֶדֶק הַבַּיִת. 

<small>ט</small>
אָמַר רְבִּי עֲקִיבָה: 
רוֹאֶה אֲנִי אֶת דִּבְרֵי רְבִּי אֱלִיעֶזֶר 
מִדִּבְרֵי רְבִּי יְהוֹשֻׁעַ, 
שֶׁרֶבִּי אֱלִיעֶזֶר הִשְׁוָה אֶת מִדָּתוֹ, 
וּרְבִּי יְהוֹשֻׁעַ חָלַק. 
אָמַר רְבִּי פַפְּיַס: 
שָׁמַעְתִּי דִבְרֵי שְׁנֵיהֶם, 
הַמַּקְדִּישׁ בְּפֵרוּשׁ כְּדִבְרֵי רְבִּי אֱלִיעֶזֶר, 
וְהַמַּקְדִּישׁ סְתָם כְּדִבְרֵי רְבִּי יְהוֹשֻׁעַ.

<small>י</small>
הַמַּקְדִּישׁ נְכָסָיו, 
וְהָיוּ בָהֶם דְּבָרִים רְאוּיִם עַל גַּבֵּי הַמִּזְבֵּחַ, 
יֵינוֹת וּשְׁמָנִים וְעוֹפוֹת, 
רְבִּי אֶלְעָזָר אוֹמֵר: 
יִמָּכְרוּ לִצְרִיכֵי אוֹתוֹ הַמִּין, 
וְיָבִיא בִדְמֵיהֶם עוֹלוֹת, 
וּשְׁאָר נְכָסִים יִפְּלוּ לְבֶדֶק הַבַּיִת.

<small>יא</small>
אַחַת לִשְׁלֹשִׁים יוֹם, 
מְשַׁעֲרִים אֶת הַלִּשְׁכָּה. 
כָּל הַמְקַבֵּל עָלָיו לִסְפֵק סְלָתוֹת מֵאַרְבַּע, 
עָמְדוּ מִשָּׁלֹשׁ, יְסַפֵּק מֵאַרְבַּע; 
מִשָּׁלֹשׁ, וְעָמְדוּ מֵאַרְבַּע, יְסַפֵּק מֵאַרְבַּע; 
שֶׁהַיַּד הֶקְדֵּשׁ לָעֶלְיוֹנָה. 
אִם הִתְלִיעָה סֹלֶת, הִתְלִיעָה לוֹ, 
וְאִם הֶחְמִיץ יַיִן, הֶחְמִיץ לוֹ. 
וְאֵינוּ מְקַבֵּל אֶת מָעוֹתָיו, 
עַד שֶׁיְּהֵא הַמִּזְבֵּחַ מְרַצֶּה.





Chapter 5

<small>א</small>
וְאֵלּוּ הֵן הַמְמֻנִּים שֶׁהָיוּ בַמִּקְדָּשׁ: 
יוֹחָנָן בֶּן פִּינְחָס עַל הַחוֹתָמוֹת, 
אֲחִיָּה עַל הַנְּסָכִים, 
מַתְּיָה בֶן שְׁמוּאֵל עַל הַפְּיָסוֹת, 
פְּתַחְיָה עַל הַקִּנִּים; 
פְּתַחְיָה זֶהוּא מָרְדְּכַי. 
וְלָמָּה נִקְרָא שְׁמוֹ פְתַחְיָה? 
שֶׁהוּא פוֹתֵחַ דְּבָרִים וְדוֹרְשָׁן, 
וְיוֹדֵעַ בְּשִׁבְעִים לָשׁוֹן. 

<small>ב</small>
בֶּן אֲחִיָּה עַל חֳלִי מֵעַיִם, &lt;חֲלִי&gt;
נְחוֹנְיָה חוֹפֵר שִׁיחִים, 
גְּבִנֵּי כָרוֹז, 
בֶּן גִּבֵּר עַל נְעִילַת שְׁעָרִים, 
בֶּן בָּבִי עַל הַפָּקִיעַ, 
בֶּן אַרְזָה עַל הַצַּלְצֵל, 
הָגְרַס בֶּן לֵוִי עַל הַשִּׁיר, 
בֵּית גַּרְמוּ עַל מַעֲשֵׂה לֶחֶם הַפָּנִים, 
בֵּית אַבְטִינָס עַל מַעֲשֵׂה הַקְּטֹרֶת, 
וְאֶלְעָזָר עַל הַפָּרְכוֹת, 
וּפִינְחָס עַל הַמַּלְבּוּשׁ. &lt;המלביש&gt;

<small>ג</small>
אֵין פּוֹחֲתִים מִשְּׁלֹשָׁה גִזְבָּרִים, 
וּמִשִּׁבְעָה מַרְכוּלִים, 
וְאֵין עוֹשִׂין סְרָרָה עַל הַצִּבּוּר בְּמָמוֹן פָּחוּת מִשְּׁנַיִם, 
חוּץ מִבֶּן אֲחִיָּה שֶׁעַל חֳלִי מֵעַיִם, 
וְאֶלְעָזָר שֶׁעַל הַפָּרְכוֹת, 
שֶׁאוֹתָם קִבְּלוּ צִבּוּר עֲלֵיהֶם.

<small>ד</small>
אַרְבָּעָה חוֹתָמוֹת הָיוּ בַמִּקְדָּשׁ, 
וְכָתוּב עֲלֵיהֶם: 
עֵגֶל, זָכָר, גְּדִי, וְחוֹטֵא. 
בֶּן עַזַּי אוֹמֵר: 
חֲמִשָּׁה הָיוּ, וַאֲרָמִית כָּתוּב עֲלֵיהֶם: 
עֵגַל, דְּכַר, גְּדִי, חוֹטֵא דַל וְחוֹטֵא עָשִׁיר. 
עֵגַל מְשַׁמֵּשׁ עִם נִסְכֵּי בָקָר, 
גְּדוֹלִים וּקְטַנִּים, זְכָרִים וּנְקֵבוֹת; 
גְּדִי מְשַׁמֵּשׁ עִם נִסְכֵּי צֹאן, 
גְּדוֹלִים וּקְטַנִּים, זְכָרִים וּנְקֵבוֹת, 
חוּץ מִשֶּׁלָּאֵילִים; 
אַיִל מְשַׁמֵּשׁ עִם נִסְכֵּי אֵילִים בִּלְבַד; 
חוֹטֵא מְשַׁמֵּשׁ עִם נִסְכֵּי שָׁלוֹשׁ בְּהֵמוֹת שֶׁלִּמְצֹרָע.

<small>ה</small>
מִי שֶׁהוּא מְבַקֵּשׁ נְסָכִים, 
הוֹלֵךְ לוֹ אֵצֶל יוֹחָנָן, 
שֶׁהוּא מְמֻנֶּה עַל הַחוֹתָמוֹת, 
וְנוֹתֵן לוֹ מָעוֹת וּמְקַבֵּל מִמֶּנּוּ חוֹתָם. 
בָּא לוֹ אֵצֶל אֲחִיָּה, 
שֶׁהוּא מְמֻנֶּה עַל הַנְּסָכִים, 
וְנוֹתֵן לוֹ חוֹתָם, 
וּמְקַבֵּל מִמֶּנּוּ נְסָכִים. 
וְלָעֶרֶב בָּאִין זֶה אֵצֶל זֶה, 
וַאֲחִיָּה מוֹצִיא אֶת הַחוֹתָמוֹת, 
וּמְקַבֵּל כְּנֶגְדָּן מָעוֹת; 
אִם הוֹתִירוּ, הוֹתִירוּ לַהֶקְדֵּשׁ, 
וְאִם פָּחֲתוּ, יְשַׁלֵּם יוֹחָנָן מִבֵּיתוֹ, 
שֶׁהַיַּד הֶקְדֵּשׁ לָעֶלְיוֹנָה.

<small>ו</small>
מִי שֶׁאָבַד חוֹתָמוֹ, 
מַמְתִּינִים לוֹ עַד הָעֶרֶב, 
אִם מָצְאוּ לוֹ כְּדֵי חוֹתָמוֹ, נוֹתְנִין לוֹ; 
וְאִם לָאו, לֹא הָיָה לוֹ; 
וְשֵׁם כָּל הַיּוֹם כָּתוּב עֲלֵיהֶן מִפְּנֵי הָרַמָּאִים.

<small>ז</small>
שְׁתֵּי לְשָׁכוֹת הָיוּ בַמִּקְדָּשׁ: 
אַחַת, לִשְׁכַּת חֲשָׁיִים, 
וְאַחַת, לִשְׁכַּת הַכֵּלִים. 
לִשְׁכַּת חֲשָׁיִים, 
יִרְאֵי חֵט נוֹתְנִין לְתוֹכָהּ בַּחְשַׁי, 
וַעֲנִיִּים בְּנֵי טוֹבִים מִתְפַּרְנְסִין מִתּוֹכָהּ בַּחְשַׁי. 
לִשְׁכַּת הַכֵּלִים, 
כָּל מִי שֶׁהוּא מִתְנַדֵּב כֶּלִי, 
זוֹרְקוֹ לְתוֹכָהּ, 
וְאַחַת לִשְׁלֹשִׁים יוֹם, 
הַגִּזְבָּרִין פּוֹתְחִים אוֹתָהּ, 
כֶּלִי שֶׁמָּצְאוּ בוֹ צֹרֶךְ לְבֶדֶק הַבַּיִת, 
מַנִּיחִים אוֹתוֹ, 
וְהַשְּׁאָר נִמְכָּרִים, &lt;וְהַשַּׁאַר&gt;
וּדְמֵיהֶם נוֹפְלִים לְלִשְׁכַּת בֶּדֶק הַבַּיִת.





Chapter 6

<small>א</small>
שְׁלֹשָׁה עָשָׂר שׁוֹפָרוֹת, 
שְׁלֹשָׁה עָשָׂר שֻׁלְחָנוֹת, 
וּשְׁלֹשׁ עֶשְׂרֵה הַשְׁתַּחֲוָיוֹת הָיוּ בַמִּקְדָּשׁ. 
שֶׁלְּבֵית רַבָּן גַּמְלִיאֵל 
וְשֶׁלְּבֵית רְבִּי חֲנַנְיָה סְגַן הַכֹּהֲנִים, 
הָיוּ מִשְׁתַּחֲוִין בְּאַרְבַּע עֶשְׂרֵה. 
וְאֵיכָן הָיְתָה יְתֵרָה? 
כְּנֶגֶד דֵּיר הָעֵצִים, 
שֶׁכֵּן מָסֹרֶת בְּיָדָם מֵאֲבוֹתֵיהֶם, 
שֶׁשָּׁם הָאָרוֹן גָּנוּז.

<small>ב</small>
מַעֲשֶׂה בְכֹהֵן אֶחָד שֶׁהָיָה מִתְעַסֵּק, 
וְרָאָה אֶת הָרִצְפָּה שֶׁהִיא מְשֻׁנָּה מֵחֲבֵרוֹתֶיהָ. 
בָּא וְאָמַר לַחֲבֵרוֹ, 
וְלֹא הִסְפִּיק לִגְמוֹר אֶת הַדָּבָר, 
עַד שֶׁיָּצְתָה נִשְׁמָתוֹ, 
וְיָדְעוּ בְיֵחוּד שֶׁשָּׁם הָאָרוֹן גָּנוּז.

<small>ג</small>
וְאֵיכָן הָיוּ הַשְׁתַּחֲוָיוֹת הָאֵלּוּ? 
אַרְבַּע בַּצָּפוֹן, וְאַרְבַּע בַּדָּרוֹם, 
שָׁלוֹשׁ בַּמִּזְרָח, וּשְׁתַּיִם בַּמַּעֲרָב, 
כְּנֶגֶד הַשְּׁעָרִים. 
שְׁעָרִים דְּרוֹמִיִּים סְמוּכִים לַמַּעֲרָב: 
שַׁעַר הָעֶלְיוֹן, שַׁעַר הַדֶּלֶק, 
שַׁעַר הַבְּכוֹרוֹת, שַׁעַר הַמַּיִם. 
וְלָמָּה נִקְרָא שְׁמוֹ שַׁעַר הַמַּיִם? 
שֶׁבּוֹ מַכְנִיסִים צְלוֹחִית שֶׁלַּמַּיִם שֶׁלַּנִּסּוּךְ בֶּחָג. 
רְבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר: &lt;ליעזר&gt;
בּוֹ הַמַּיִם מְפַכִּים וַעֲתִידִים לִהְיוֹת יוֹצְאִין 
מִתַּחַת מִפְתַּן הַבַּיִת. 

<small>ד</small>
לְעֻמָּתָן בַּצָּפוֹן סְמוּכִים לַמַּעֲרָב: 
שַׁעַר יְכָנְיָה, שַׁעַר הַקָרְבָּן, 
שַׁעַר הַנָּשִׁים, שַׁעַר הַשִּׁיר. 
וְלָמָּה נִקְרָא שְׁמוֹ שַׁעַר יְכָנְיָה? 
שֶׁבּוֹ יָצָא יְכָנְיָה בְגָלוּתוֹ. &lt;וגלותו&gt;
וְשֶׁבַּמִּזְרָח שַׁעַר נִיקָנוֹר, 
וּשְׁנֵי פְשֶׁפְשִׁים הָיוּ לוֹ, 
אֶחָד בִּימִינוֹ וְאֶחָד בִּשְׂמאלוֹ. 
וּשְׁנַיִם בַּמַּעֲרָב שֶׁלֹּא הָיָה לָהֶן שֵׁם.

<small>ה</small>
שְׁלֹשָׁה עָשָׂר שֻׁלְחָנוֹת הָיוּ בַמִּקְדָּשׁ: 
שְׁמוֹנָה שֶׁלַּשַּׁיִשׁ בְּבֵית הַמִּטְבְּחַיִם, 
שֶׁעֲלֵיהֶן מְדִיחִין אֶת הַקְּרָבַיִם; 
וּשְׁנַיִם בְּמַעֲרַב הַכֶּבֶשׁ, 
אֶחָד שֶׁלַּשַּׁיִשׁ וְאֶחָד שֶׁלַּכֶּסֶף. 
עַל שֶׁלַּשַּׁיִשׁ נוֹתְנִים אֶת הָאֵבָרִים, 
וְעַל שֶׁלַּכֶּסֶף נוֹתְנִים כְּלֵי שָׁרֵת; 
וּשְׁנַיִם בָּאוּלָם מִבִּפְנִים עַל פֶּתַח הַבַּיִת, 
אֶחָד שֶׁלַּשַּׁיִשׁ וְאֶחָד שֶׁלַּזָּהָב. 
עַל שֶׁלַּשַּׁיִשׁ נוֹתְנִין אֶת לֶחֶם הַפָּנִים בִּכְנֵיסָתוֹ, 
וְעַל שֶׁלַּזָּהָב בִּיצִיאָתוֹ, 
שֶׁמַּעֲלִים בַּקֹּדֶשׁ וְלֹא מוֹרִידִים; 
וְאֶחָד שֶׁלַּזָּהָב מִבִּפְנִים, 
שֶׁעָלָיו לֶחֶם הַפָּנִים תָּמִיד.

<small>ו</small>
שְׁלֹשָׁה עָשָׂר שׁוֹפָרוֹת הָיוּ בַמִּקְדָּשׁ, 
וְכָתוּב עֲלֵיהֶם: 
תִּקְלִין חַדַתִּין וְתִקְלִין עַתִּיקִין, 
קִנִּים וְגּוֹזָלֵי עוֹלָה, 
עֵצִים וּלְבוֹנָה, 
זָהָב לַכַּפֹּרֶת, 
שִׁשָּׁה לִנְדָבָה. 
תִּקְלִין חַדַתִּין, שֶׁבְּכָל שָׁנָה וְשָׁנָה, 
וְעַתִּיקִין, מִי שֶׁלֹּא שָׁקַל אֶשְׁתְּקַד, 
שׁוֹקֵל לְשָׁנָה הַבָּאָה; 

<small>ז</small>
קִנִּים, הֵן תּוֹרִין; 
וְגּוֹזָלֵי עוֹלָה, הֵן בְּנֵי יוֹנָה, 
כֻּלָּם עוֹלוֹת. 
דִּבְרֵי רְבִּי יְהוּדָה. 
וַחֲכָמִים אוֹמְרִים: 
קִנִּים, אֶחָד חַטָּאת וְאֶחָד עוֹלָה, 
וְגּוֹזָלֵי עוֹלָה, כֻּלָּם עוֹלוֹת.

<small>ח</small>
הָאוֹמֵר: 
"הֲרֵי עָלַי עֵצִים", 
לֹא יִפְחֹת מִשְּׁנֵי גִיזֵרִין; 
"לְבוֹנָה", 
לֹא יִפְחֹת מִן הַקֹּמֶץ; 
"זָהָב", 
לֹא יִפְחֹת מִדִּינַר זָהָב. 
שִׁשָּׁה לִנְדָבָה. 
"נְדָבָה", 
מָה הָיוּ עוֹשִׂין בָּהּ? 
לוֹקְחִין בָּהּ עוֹלוֹת, 
הַבָּשָׂר לַשֵּׁם וְהָעוֹרוֹת לַכֹּהֲנִים. 
זֶה מִדְרָשׁ דָּרַשׁ יְהוֹיָדָע כֹּהֵן גָּדוֹל: 
"אָשָׁם הוּא, אָשֹׁם אָשַׁם לַיי" (ויקרא ה,יט), 
זֶה הַכְּלָל: 
כָּל שֶׁהוּא בָא מִשּׁוּם חֵט וּמִשּׁוּם אַשְׁמָה, 
יִלָּקְחוּ בּוֹ עוֹלוֹת, 
הַבָּשָׂר לַשֵּׁם, וְהָעוֹרוֹת לַכֹּהֲנִים. 
נִמְצְאוּ שְׁנֵי כְתוּבִים קַיָּמִים: 
אָשָׁם לַיי וְאָשָׁם לַכֹּהֵן, 
וְאוֹמֵר: (מלכים ב יב,יז) 
"כֶּסֶף אָשָׁם וְכֶסֶף חַטָּאוֹת לֹא יוּבָא בֵית יי, 
לַכֹּהֲנִים יִהְיוּ".





Chapter 7

<small>א</small>
מָעוֹת שֶׁנִּמְצְאוּ בֵין שְׁקָלִים לִנְדָבָה: 
קָרוֹב לִשְׁקָלִים, יִפְּלוּ לִשְׁקָלִים, 
לִנְדָבָה, יִפְּלוּ לִנְדָבָה; 
מַחְצָה לְמַחְצָה, יִפְּלוּ לִנְדָבָה. 
בֵּין עֵצִים לִלְבוֹנָה: 
קָרוֹב לָעֵצִים, יִפְּלוּ לָעֵצִים, 
לִלְבוֹנָה, יִפְּלוּ לִלְבוֹנָה, 
מַחְצָה לְמַחְצָה, יִפְּלוּ לִלְבוֹנָה. 
בֵּין קִנִּים לְגוֹזָלֵי עוֹלָה: 
קָרוֹב לַקִּנִּים, יִפְּלוּ לַקִּנִּים, 
לְגוֹזָלֵי עוֹלָה, יִפְּלוּ לְגוֹזָלֵי עוֹלָה, 
מַחְצָה לְמַחְצָה, יִפְּלוּ לְגוֹזָלֵי עוֹלָה. 
בֵּין חֻלִּין לְמַעֲשֵׂר שֵׁנִי: 
קָרוֹב לַחֻלִּין, יִפְּלוּ לַחֻלִּין, 
לְמַעֲשֵׂר שֵׁנִי, יִפְּלוּ לְמַעֲשֵׂר שֵׁנִי, 
מַחְצָה לְמַחְצָה, יִפְּלוּ לְמַעֲשֵׂר שֵׁנִי. 
זֶה הַכְּלָל: 
הוֹלְכִים אַחַר הַקָּרוֹב; 
מַחְצָה לְמַחְצָה, לְהַחְמִיר.

<small>ב</small>
מָעוֹת שֶׁנִּמְצְאוּ לִפְנֵי סוֹחֲרֵי בְהֵמָה, 
לְעוֹלָם מַעֲשֵׂר. 
וּבְהַר הַבַּיִת, חֻלִּין. 
וּבִירוּשָׁלַיִם, 
בִּשְׁאָר כָּל יְמוֹת הַשָּׁנָה, חֻלִּין, 
וּבְשָׁעַת הָרֶגֶל, הַכֹּל מַעֲשֵׂר.

<small>ג</small>
בָּשָׂר שֶׁנִּמְצָא בָעֲזָרָה, 
אֵבָרִים, עוֹלוֹת, 
וַחֲתִיכוֹת, חַטָּאוֹת, 
וּבִירוּשָׁלַיִם, זִבְחֵי שְׁלָמִים. 
זֶה וָזֶה, 
תְּעַבֵּר צוּרָתוֹ וְיֵצֵא לְבֵית הַשְּׂרִיפָה. 

<small>ד</small>
נִמְצָא בַגְּבוּלִים, 
אֵבָרִים, נְבֵלוֹת, 
וַחֲתִיכוֹת, מֻתָּרוֹת. 
וּבְשָׁעַת הָרֶגֶל, שֶׁהַבָּשָׂר מְרֻבֶּה, 
אַף אֵבָרִים מֻתָּרִים.

<small>ה</small>
בְּהֵמָה שֶׁנִּמְצֵאת מִירוּשָׁלַיִם וְעַד מִגְדַּל עֵדֶר, 
כְּמִדָּתָהּ לְכָל רוּחַ: 
זְכָרִים, עוֹלוֹת וּנְקֵבוֹת, זִבְחֵי שְׁלָמִים. 
רְבִּי יְהוּדָה אוֹמֵר: &lt;יודה&gt;
אַף הָרָאוּי לִפְסָחִים, פְּסָחִים, 
קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם.

<small>ו</small>
בָּרִאשׁוֹנָה הָיוּ מְמַשְׁכְּנִים אֶת מוֹצְאֶיהָ, 
עַד שֶׁהוּא מֵבִיא עִמָּהּ נְסָכֶיהָ. 
חָזְרוּ לִהְיוֹת מַנִּיחִים אוֹתָהּ וּבוֹרְחִים. 
הִתְקִינוּ בֵית דִּין 
שֶׁיְּהוּ נְסָכֶיהָ בָאִים מִשֶּׁלַּצִּבּוּר.

<small>ז</small>
אָמַר רְבִּי שִׁמְעוֹן: 
שִׁבְעָה דְבָרִים הִתְקִינוּ בֵית דִּין, 
וְזֶה אֶחָד מֵהֶן. 
גּוֹי שֶׁשִּׁלַּח עוֹלָתוֹ מִמְּדִינַת הַיָּם, 
אִם שִׁלַּח עִמָּהּ נְסָכִים, קְרֵבִים מִשֶּׁלּוֹ, 
וְאִם לָאו, קְרֵבִים מִשֶּׁלַּצִּבּוּר. 
וְכֵן גֵּר שֶׁמֵּת וְהִנִּיחַ זְבָחִים, 
אִם יֶשׁ לוֹ נְסָכִים, קְרֵבִים מִשֶּׁלּוֹ, 
וְאִם לָאו, קְרֵבִים מִשֶּׁלַּצִּבּוּר. 
תְּנַי בֵּית דִּין הוּא עַל כֹּהֵן גָּדוֹל שֶׁמֵּת, 
שֶׁתְּהֵא מִנְחָתוֹ קְרֵבָה מִשֶּׁלַּצִּבּוּר. 
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁלַּיּוֹרְשִׁים. 
וּשְׁלֵמָה הָיְתָה קְרֵבָה.

<small>ח</small> 
עַל הַמֶּלַח וְעַל הָעֵצִים, 
שֶׁיְּהוּ הַכֹּהֲנִים נֵאוֹתִים בָּהֶם. 
וְעַל הַפָּרָה, 
שֶׁלֹּא יְהוּ מוֹעֲלִים בְּאֶפְרָהּ. &lt;מוּעֲלִים&gt;
עַל הַקִּנִּים הַפְּסוּלוֹת, 
שֶׁיְּהוּ בָאוֹת מִשֶּׁלַּצִּבּוּר; 
רְבִּי יוֹסֵה אוֹמֵר: 
הַמְסַפֵּק אֶת הַקִּנִּים, 
מְסַפֵּק אֶת הַפְּסוּלוֹת.





Chapter 8

<small>א</small>
כָּל הָרוֹקִים הַנִּמְצְאִים בִּירוּשָׁלַיִם טְהוֹרִים, 
חוּץ מִשּׁוּק הָעֶלְיוֹן. 
דִּבְרֵי רְבִּי מֵאִיר. 
רְבִּי יוֹסֵה אוֹמֵר: 
בִּשְׁאָר כָּל יְמוֹת הַשָּׁנָה, 
שֶׁבָּאֶמְצַע טְמֵאִים, 
וְשֶׁבִּצְדָדִים טְהוֹרִים; 
וּבְשָׁעַת הָרֶגֶל, 
שֶׁבָּאֶמְצַע טְהוֹרִים, 
וְשֶׁבִּצְדָדִים טְמֵאִים, 
שֶׁמִּפְּנֵי שֶׁהַמֵּעוּטִים מִסְתַּלְּקִים לִצְדָדִים.

<small>ב</small>
כָּל הַכֵּלִים הַנִּמְצְאִים בִּירוּשָׁלַיִם, 
דֶּרֶךְ יְרִידָה לְבֵית הַטְּבִילָה, טְמֵאִים, 
דֶּרֶךְ עֲלִיָּה, טְהוֹרִים, 
שֶׁלֹּא כִירִידָתָן עֲלָיָתָן. 
דִּבְרֵי רְבִּי מֵאִיר. 
רְבִּי יוֹסֵה אוֹמֵר: 
כֻּלָּם טְהוֹרִים, 
חוּץ מִן הַסַּל וְהַמַּגְרֵפָה 
וְהַמְּרִיצָה הַמְיֻחָדִים לִקְבָרוֹת.

<small>ג</small>
סְכִין שֶׁנִּמְצֵאת בְּאַרְבָּעָה עָשָׂר, 
שׁוֹחֵט בָּהּ מִיָּד; 
בִּשְׁלֹשָׁה עָשָׂר, שׁוֹנֶה וּמַטְבִּיל. 
קָפִּיס, בָּזֶה וּבָזֶה, שׁוֹנֶה וּמַטְבִּיל. 
חָל אַרְבָּעָה עָשָׂר לִהְיוֹת בַּשַּׁבָּת, 
שׁוֹחֵט בָּהּ מִיָּד. 
בַּחֲמִשָּׁה עָשָׂר, שׁוֹחֵט בָּהּ מִיָּד. 
נִמְצֵאת קְשׁוּרָה לִסְכִין, 
הֲרֵי זוֹ כִסְכִין.

<small>ד</small>
פָּרֹכֶת שֶׁנִּטַּמָּאת בִּוְלַד הַטֻּמְאָה, 
מַטְבִּילִים אוֹתָהּ בִּפְנִים, 
וּמַכְנִיסִים אוֹתָהּ מִיָּד. 
שֶׁנִּטַּמָּאת בְּאַב הַטֻּמְאָה, 
מַטְבִּילִין אוֹתָהּ בַּחוּץ, 
וְשׁוֹטְחִים אוֹתָהּ בַּחֵיל, 
מִפְּנֵי שֶׁהִיא צְרִיכָה הַעְרֵב שֶׁמֶשׁ.
אִם הָיְתָה חֲדָשָׁה, 
שׁוֹטְחִין אוֹתָהּ עַל גַּג הָאִצְטְבָא, &lt;האיסטווה&gt;
כְּדֵי שֶׁיִּרְאוּ הָעָם אֶת מְלַאכְתָּהּ שֶׁהִיא נָאָה.

<small>ה</small>
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר 
מִשּׁוּם שִׁמְעוֹן בֶּן הַסֶּגֶן: 
פָּרֹכֶת, עָבְיָהּ טֶפַח, 
עַל שִׁבְעִים וּשְׁנַיִם נִירִים נֶאֱרֶגֶת, 
עַל כָּל נִימָה וְנִימָה עֶשְׂרִים וְאַרְבָּעָה חוּטִים. 
אָרְכָּהּ אַרְבָּעִים אַמָּה, 
וְרָחְבָּהּ עֶשְׂרִים אַמָּה, 
וּבִשְׁמוֹנִים וּשְׁתַּיִם רִבּוֹא הָיְתָה נֶעְשֵׂית; &lt;ריבו&gt;
וּשְׁתַּיִם עוֹשִׂין בְּכָל שָׁנָה, 
וּשְׁלֹשׁ מֵאוֹת כֹּהֲנִים מַטְבִּילִים אוֹתָהּ.

<small>ו</small>
בְּשַׂר קָדְשֵׁי קָדָשִׁים שֶׁנִּטַּמָּא 
בֵין בְּאַב הַטֻּמְאָה 
בֵין בִּוְלַד הַטֻּמְאָה, 
בֵּין מִבִּפְנִים, בֵּין מִבַּחוּץ, 
בֵּית שַׁמַּי אוֹמְרִים: 
הַכֹּל יִשָּׂרֵף בִּפְנִים, 
חוּץ מִשֶּׁנִּטַּמָּא בְאַב הַטֻּמְאָה בַחוּץ. 
וּבֵית הֶלֵּל אוֹמְרִים: 
הַכֹּל יִשָּׂרֵף בַּחוּץ, 
חוּץ מִשֶּׁנִּטַּמָּא בִוְלַד הַטֻּמְאָה בִפְנִים.

<small>ז</small>
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אֶת שֶׁנִּטַּמָּא בְאַב הַטֻּמְאָה, 
בֵּין בִּפְנִים בֵּין בַּחוּץ, 
יִשָּׂרֵף בַּחוּץ; 
וְאֶת שֶׁנִּטַּמָּא בִוְלַד הַטֻּמְאָה, 
בֵּין בַּחוּץ בֵּין בִּפְנִים, 
יִשָּׂרֵף בִּפְנִים. 
רְבִּי עֲקִיבָה אוֹמֵר: 
מְקוֹם טֻמְאָתוֹ, שָׁם תְּהֵא שְׂרִיפָתוֹ.

<small>ח</small>
אֵבָרֵי הַתָּמִיד נִתָּנִים מֵחֲצִי כֶבֶשׁ וּלְמַטָּה בַמַּעֲרָב, 
וְשֶׁלַּמּוּסָפִין נִתָּנִין מֵחֲצִי כֶבֶשׁ לְמַטָּה בַמִּזְרָח, 
וְשֶׁלְּרָאשֵׁי חֳדָשִׁים נִתָּנִים עַל כַּרְכֹּב הַמִּזְבֵּחַ מִלְּמַעְלָה. 
הַשְּׁקָלִים וְהַבִּכּוּרִים אֵינָן נוֹהֲגִין אֶלָּא בִפְנֵי הַבַּיִת, 
אֲבָל מַעְשַׂר דָּגָן וּמַעְשַׂר בְּהֵמָה וּבְכוֹרוֹת 
נוֹהֲגִין בִּפְנֵי הַבַּיִת וְשֶׁלֹּא בִפְנֵי הַבַּיִת. 
הַמַּקְדִּישׁ שְׁקָלִים וּבִכּוּרִים, 
הֲרֵי זֶה קֹדֶשׁ. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
הָאוֹמֵר: 
"בִּכּוּרִים קֹדֶשׁ", אֵינָן קֹדֶשׁ.
