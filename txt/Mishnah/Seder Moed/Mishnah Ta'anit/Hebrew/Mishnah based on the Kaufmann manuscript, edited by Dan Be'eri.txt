משנה תענית



Chapter 1

<small>א</small>
מֵאֶמָּתַי מַזְכִּירִין גְּבוּרוֹת גְּשָׁמִים? 
רְבִּי לִיעֶזֶר אוֹמֵר: 
מִיּוֹם טוֹב הָרִאשׁוֹן שֶׁלֶּחָג.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
מִיּוֹם טוֹב הָאַחֲרוֹן.  
אָמַר לוֹ רְבִּי יְהוֹשֻׁעַ: 
הוֹאִיל וְאֵין הַגְּשָׁמִים אֶלָּא סִימָן קְלָלָה בֶחָג, 
לָמָּה מַזְכִּיר? 
אָמַר לוֹ רְבִּי אֱלִיעֶזֶר: 
אַף הוּא אֵינוּ אוֹמֵר אֶלָּא  
מַשִּׁיב הָרוּחַ וּמוֹרִיד הַגֶּשֶׁם, 
בְּעוֹנָתוֹ.  
אָמַר לוֹ: 
אִם כֵּן, 
לְעוֹלָם יְהֵא מַזְכִּיר. 

<small>ב</small>
אֵין שׁוֹאֲלִים גְּשָׁמִים אֶלָּא סָמוּךְ לִגְשָׁמִים.  
רְבִּי יְהוּדָה אוֹמֵר: 
הָעוֹבֵר לִפְנֵי הַתֵּבָה בְיוֹם טוֹב הָאַחֲרוֹן שֶׁלֶּחָג, 
הָאַחֲרוֹן מַזְכִּיר; 
וְהָרִאשׁוֹן אֵינוּ מַזְכִּיר.  
וּבְיוֹם טוֹב הָרִאשׁוֹן שֶׁלַּפֶּסַח, 
הָרִאשׁוֹן מַזְכִּיר 
וְהָאַחֲרוֹן אֵינוּ מַזְכִּיר.  

<small>ג</small>
עַד אֶמָּתַי שׁוֹאֲלִין? 
רְבִּי יְהוּדָה אוֹמֵר: 
עַד שֶׁיַּעֲבוֹר הַפֶּסַח.  
רְבִּי מֵאִיר אוֹמֵר: 
עַד שֶׁיֵּצֵא נִיסָן, 
שֶׁנֶּאֱמַר: (יואל ב,כג) 
"וַיּוֹרֶד לָכֶם גֶּשֶׁם מוֹרֶה [יורה] וּמַלְקוֹשׁ בָּרִאשׁוֹן". 

<small>ד</small>
בִּשְׁלֹשָׁה בְמַרְחֶשְׁוָן שׁוֹאֲלִין גְּשָׁמִים.  
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
בְּשִׁבְעָה בוֹ, חֲמִשָּׁה עָשָׂר יוֹם אַחַר הֶחָג, 
כְּדֵי שֶׁיַּגִּיעַ הָאַחֲרוֹן שֶׁבְּיִשְׂרָאֵל לִנְהַר פְּרָת. 

<small>ה</small>
הִגִּיעַ שִׁבְעָה עָשָׂר בְּמַרְחֶשְׁוָן וְלֹא יָרְדוּ גְשָׁמִים, 
הִתְחִילוּ הַיְחִידִים מִתְעַנִּים.  
אוֹכְלִים וְשׁוֹתִים מִשֶּׁחָשֵׁכָה, 
וּמֻתָּרִים בִּמְלָאכָה וּבִרְחִיצָה וּבִיסִיכָה, 
וּבִנְעִילַת הַסַּנְדָּל וּבְתַשְׁמִישׁ הַמִּטָּה. 

ו (תוספת בצד)
הִגִּיעַ רֹאשׁ חֹדֶשׁ כִּסְלֵו וְלֹא יָרְדוּ גְשָׁמִים, 
בֵּית דִּין גּוֹזְרִין שָׁלוֹשׁ תַּעְנִיּוֹת עַל הַצִּבּוּר; 
אוֹכְלִין וְשׁוֹתִין מִשֶּׁחָשֵׁכָה, 
וּמֻתָּרִין בִּמְלָאכָה וּבִרְחִיצָה וּבְסִיכָה, 
וּבִנְעִילַת הַסַּנְדָּל וּבְתַשְׁמִישׁ הַמִּטָּה. 

עָבְרוּ אֵלּוּ וְלֹא נֶעְנוּ, 
בֵית דִּין גּוֹזְרִים שָׁלוֹשׁ תַּעְנִיּוֹת אֲחֵרוֹת עַל הַצִּבּוּר.  
אוֹכְלִים וְשׁוֹתִים מִבְּעוֹד יוֹם, 
וַאֲסוּרִין בִּמְלָאכָה וּבִרְחִיצָה וּבִיסִיכָה, 
וּבִנְעִילַת הַסַּנְדָּל וּבְתַשְׁמִישׁ הַמִּטָּה, 
וְנוֹעֲלִין אֶת הַמֶּרְחֲצָאוֹת.  

<small>ז</small>
עָבְרוּ אֵלּוּ וְלֹא נֶעְנוּ, 
בֵית דִּין גּוֹזְרִים עוֹד שֶׁבַע, 
שֶׁהֵן שְׁלֹשׁ עֶשְׂרֵה תַעְנִיּוֹת עַל הַצִּבּוּר.  
וּמָה אֵלּוּ יְתֵרוֹת עַל הָרִאשׁוֹנוֹת? 
שֶׁבָּאֵלּוּ מַתְרִיעִים וְנוֹעֲלִים אֶת הַחֲנוּיוֹת.  
וּבַשֵּׁנִי מָטִים עִם חֲשֵׁכָה, 
וּבַחֲמִישִׁי מֻתָּרִין, מִפְּנֵי כְּבוֹד הַשַּׁבָּת. 

<small>ח</small>
עָבְרוּ אֵלּוּ וְלֹא נֶעְנוּ, 
מְמַעֲטִים בְּמַשָּׂא וּבְמַתָּן, 
בְּבִנְיָן וּבִנְטִיעָה, 
בְּאֵרוּסִים וּבְנִשּׂוּאִים, 
וּבִשְׁאֵלַת שָׁלוֹם בֵּין אָדָם לַחֲבֵרוֹ, 
כִּבְנֵי אָדָם נְזוּפִים לִפְנֵי הַמָּקוֹם.  
וְהַיְחִידִים חוֹזְרִים וּמִתְעַנִּים עַד שֶׁיֵּצֵא נִיסָן.  
יָצָא נִיסָן וּבָאוּ גְשָׁמִים, סִימַן קְלָלָה, 
שֶׁנֶּאֱמַר: (שמואל א יב,יז) 
"הֲלא קְצִיר חִטִּים הַיּוֹם, 
אֶקְרָא אֶל יי וְיִתֵּן קוֹלוֹת וּמָטָר" וְגוֹמֵר. 





Chapter 2

<small>א</small>
סֵדֶר תַּעְנִיּוֹת כֵּיצַד? 
מוֹצִיאִין אֶת הַתֵּבָה לִרְחוֹבָהּ שֶׁלָּעִיר; 
וְנוֹתְנִין אֵפֶר מַקְלָה עַל גַּבֵּי הַתֵּבָה, 
וּבְרֹאשׁ הַנָּשִׂיא וּבְרֹאשׁ אַב בֵּית דִּין; 
וְכָל אֶחָד וְאֶחָד נוֹטֵל וְנוֹתֵן בְּרֹאשׁוֹ.  
הַזָּקֵן שֶׁבָּהֶן אוֹמֵר לִפְנֵיהֶם דִּבְרֵי כִבּוּשִׁים: 
אַחֵינוּ, לֹא נֶאֱמַר בְּאַנְשֵׁי נִינְוֵה: 
"וַיַּרְא הָאֱלהִים אֶת שַׂקָּם וְאֶת תַּעֲנִיתָם", 
אֶלָּא (יונה ג,י) "וַיַּרְא הָאֱלהִים אֶת מַעֲשֵׂיהֶם, 
כִּי שָׁבוּ מִדַּרְכָּם הָרָעָה"; 
וּבַקַּבָּלָה מָהוּא אוֹמֵר? (יואל ב,יג) 
"וְקִרְעוּ לְבַבְכֶם וְאַל  בִּגְדֵיכֶם, 
וְשׁוּבוּ אֶל יי אֱלֹהֵיכֶם" וגו'. 

<small>ב</small>
עָמְדוּ בִתְפִלָּה, 
וּמוֹרִידִין לִפְנֵי הַתֵּבָה זָקֵן וְרָגִיל, 
וְיֵשׁ לוֹ בָנִים וּבֵיתוֹ רֵיקָם, 
כְּדֵי שֶׁיְּהֵא לִבּוֹ שָׁלֵם בַּתְּפִלָּה.  
וְאוֹמֵר לִפְנֵיהֶם עֶשְׂרִים וְאַרְבַּע בְּרָכוֹת: 
שְׁמוֹנֶה עֶשְׂרֵה שֶׁבְּכָל יוֹם, 
וּמוֹסִיף עֲלֵיהֶם עוֹד שֵׁשׁ,

וְאֵלּוּ הֵן: 
זִכְרוֹנוֹת וְשׁוֹפָרוֹת, 
"אֶל יי בַּצָּרָתָה לִּי", 
"אֶשָּׂא עֵינַי אֶל הֶהָרִים", 
"מִמַּעֲמַקִּים קְרָאתִיךָ יי", 
"תְּפִלָּה לְעָנִי כִי  יַעֲטֹף".  

<small>ג</small>
רְבִּי יְהוּדָה אוֹמֵר: 
לֹא הָיָה צָרִיךְ לוֹמַר זִכְרוֹנוֹת וְשׁוֹפָרוֹת, 
אֶלָּא אוֹמֵר תַּחְתֵּיהֶם: 
"רָעָב כִּי יִהְיֶה בָאָרֶץ, דֶּבֶר" וְגוֹ', 
"אֲשֶׁר הָיָה דְבַר יי אֶל יִרְמְיָהוּ 
עַל דִּבְרֵי הַבַּצָּרוֹת", 
וְאוֹמֵר חוֹתָמוֹתֵיהֶם. 

<small>ד</small>
עַל הָרִאשׁוֹנָה הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת אַבְרָהָם בְּהַר הַמּוֹרִיָּה, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי גּוֹאֵל יִשְׂרָאֵל.  

<small>ה</small>
עַל הַשְּׁנִיָּה הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת אֲבוֹתֵיכֶם עַל יַם סוּף, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי זוֹכֵר הַנִּשְׁכָּחוֹת.  

<small>ו</small>
עַל הַשְּׁלִישִׁית הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת יְהוֹשֻׁעַ בַּגִּלְגָּל, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי שׁוֹמֵעַ תְּרוּעָה.  

<small>ז</small>
עַל הָרְבִיעִית הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת שְׁמוּאֵל בַּמִּצְפָּה, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי שׁוֹמֵעַ צְעָקָה.  

<small>ח</small>
עַל הַחֲמִישִׁית הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת אֵלִיָּהוּ בְהַר הַכַּרְמֶל, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי שׁוֹמֵעַ תְּפִלָּה.  

<small>ט</small>
עַל הַשִּׁשִּׁית הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת יוֹנָה מִמְּעֵי הַדָּגָה, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי הָעוֹנֶה בְעֵת צָרָה.  

<small>י</small>
עַל הַשְּׁבִיעִית הוּא אוֹמֵר: 
מִי שֶׁעָנָה אֶת דָּוִד וְאֶת שְׁלֹמֹה בְנוֹ בִירוּשָׁלַיִם, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה; 
בָּרוּךְ אַתָּה יי הַמְרַחֵם עַל הָאָרֶץ. 

<small>יא</small>
מַעֲשֶׂה בִימֵי רְבִּי חֶלְפְּתָה 
וּבִימֵי רְבִּי חֲנַנְיָה בֶן תַּרְדְּיוֹן, 
שֶׁעָבַר אֶחָד לִפְנֵי הַתֵּבָה, 
וְגָמַר אֶת כָּל הַבְּרָכָה, 
וְעָנוּ אַחֲרָיו: 
"אָמֵן.  תִּקְעוּ, הַכֹּהֲנִים, תִּקְעוּ!" 
מִי שֶׁעָנָה אֶת אַבְרָהָם אָבִינוּ בְהַר הַמּוֹרִיָּה, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה.  
הָרִיעוּ בְנֵי אַהֲרֹן, הָרִיעוּ! 
מִי שֶׁעָנָה אֶת אֲבוֹתֵיכֶם עַל יַם סוּף, 
הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע בְּקוֹל צַעֲקַתְכֶם הַיּוֹם הַזֶּה.  
וּכְשֶׁבָּא דָבָר אֵצֶל חֲכָמִים, וְאָמְרוּ: 
לֹא הָיוּ נוֹהֲגִים כֵּן, אֶלָּא בְּשַׁעַר מִזְרָח. 

<small>יב</small>
שָׁלוֹשׁ תַּעְנִיּוֹת הָרִאשׁוֹנוֹת, 
אַנְשֵׁי מִשְׁמָר מִתְעַנִּים וְלֹא מַשְׁלִימִים, 
וְאַנְשֵׁי בֵית אָב לֹא הָיוּ מִתְעַנִּים.  
שָׁלוֹשׁ שְׁנִיּוֹת, 
אַנְשֵׁי מִשְׁמָר מִתְעַנִּים וּמַשְׁלִימִים, 
וְאַנְשֵׁי בֵית אָב מִתְעַנִּים וְלֹא מַשְׁלִימִים.  
שֶׁבַע אַחֲרוֹנוֹת, 
אֵלּוּ וָאֵלּוּ מִתְעַנִּים וּמַשְׁלִימִים.  
דִּבְרֵי רְבִּי יְהוֹשֻׁעַ.  
וַחֲכָמִים אוֹמְרִים: 
שָׁלוֹשׁ תַּעְנִיּוֹת הָרִאשׁוֹנוֹת, 
אֵלּוּ וָאֵלּוּ לֹא הָיוּ מִתְעַנִּים.  
שָׁלוֹשׁ שְׁנִיּוֹת, 
אַנְשֵׁי מִשְׁמָר מִתְעַנִּים וְלֹא מַשְׁלִימִים, 
וְאַנְשֵׁי בֵית אָב לֹא הָיוּ מִתְעַנִּים.  
וְשֶׁבַע אַחֲרוֹנוֹת, 
אַנְשֵׁי מִשְׁמָר מִתְעַנִּים וּמַשְׁלִימִים, 
וְאַנְשֵׁי בֵית אָב מִתְעַנִּים וְלֹא מַשְׁלִימִים. 

<small>יג</small>
אַנְשֵׁי מִשְׁמָר מֻתָּרִין לִשְׁתּוֹת יַיִן בַּלֵּילוֹת, 
אֲבָל לֹא בַיָּמִים; 
וְאַנְשֵׁי בֵית אָב, לֹא בַיּוֹם וְלֹא בַלַּיְלָה.  
אַנְשֵׁי מִשְׁמָר וְאַנְשֵׁי מַעֲמָד 
אֲסוּרִין מִלְּסַפֵּר וּמִלְּכַבֵּס, 
וּבַחֲמִישִׁי מֻתָּרִין, מִפְּנֵי כְבוֹד הַשַּׁבָּת. 

<small>יד</small>
כָּל הַכָּתוּב בִּמְגִלַּת תַּעְנִית "דִּילָא לְמִסְפַּד", 
לְפָנָיו אָסוּר, וּלְאַחֲרָיו מֻתָּר.  
רְבִּי יוֹסֵה אוֹמֵר: 
לְפָנָיו וּלְאַחֲרָיו אָסוּר.  
"דִּילָא לְהִתְעַנָּיָא", 
לְפָנָיו וּלְאַחֲרָיו מֻתָּר; 
רְבִּי יוֹסֵה אוֹמֵר: 
לְפָנָיו אָסוּר, וּלְאַחֲרָיו מֻתָּר. 

<small>יה</small>
אֵין גּוֹזְרִין תַּעְנִית עַל הַצִּבּוּר כַּתְּחִלָּה בַחֲמִישִׁי, 
שֶׁלֹּא לְהַבְקִיעַ הַשְּׁעָרִים; 
אֶלָּא שָׁלוֹשׁ תַּעְנִיּוֹת הָרִאשׁוֹנוֹת, 
שֵׁנִי וַחֲמִישִׁי וְשֵׁנִי, 
וְשָׁלוֹשׁ שְׁנִיּוֹת, 
חֲמִישִׁי שֵׁנִי וַחֲמִישִׁי.  
רְבִּי יוֹסֵה אוֹמֵר: 
כַּשֵּׁם שֶׁאֵין הָרִאשׁוֹנוֹת בַּחֲמִישִׁי, 
כָּךְ לֹא שְׁנִיּוֹת וְלֹא אַחֲרוֹנוֹת. 

<small>יו</small>
אֵין גּוֹזְרִין תַּעְנִית בְּרָאשֵׁי חֳדָשִׁים, 
בַּחֲנֻכָּה וּבַפּוּרִים.  
אִם הִתְחִילוּ, אֵין מַפְסִיקִין.  
דִּבְרֵי רַבָּן גַּמְלִיאֵל.  
אָמַר רְבִּי מֵאִיר: 
אַף עַל פִּי שֶׁאָמַר רַבָּן גַּמְלִיאֵל "אֵין מַפְסִיקִים", 
מוֹדֶה הָיָה שֶׁאֵין מַשְׁלִימִין.  
וְכֵן תִּשְׁעָה בְאָב שֶׁחָל לִהְיוֹת בְּעֶרֶב שַׁבָּת. 





Chapter 3

<small>א</small>
סֵדֶר תַּעְנִיּוֹת אֵלּוּ הָאָמוּר בִּרְבֵיעָה רִאשׁוֹנָה.  
אֲבָל צְמָחִים שֶׁשָּׁנוּ, מַתְרִיעִים עֲלֵיהֶם מִיָּד.  
וְכֵן שֶׁפָּסְקוּ גְשָׁמִים, 
מִגֶּשֶׁם לְגֶשֶׁם אַרְבָּעִים יוֹם, 
מַתְרִיעִין עֲלֵיהֶם מִיָּד, 
מִפְּנֵי שֶׁהִיא מַכַּת בַּצֹּרֶת. 

<small>ב</small>
יָרְדוּ לִצְמָחִין, 
אֲבָל לֹא לָאִילָן, 
לָאִילָן אֲבָל לֹא לִצְמָחִין, 
לָזֶה וְלָזֶה, 
אֲבָל לֹא לַבּוֹרוֹת וְלֹא לַשִּׁיחִים וְלֹא לַמְּעָרוֹת, 
מַתְרִיעִין עֲלֵיהֶן מִיָּד. 

<small>ג</small>
וְכֵן עִיר שֶׁלֹּא יָרְדוּ עָלֶיהָ גְשָׁמִים, 
כַּכָּתוּב: (עמוס ד,ז) 
"וְהִמְטַרְתִּי עַל עִיר אֶחָת, 
וְעַל עִיר אַחַת לֹא אַמְטִיר, 
חֶלְקָה אַחַת תִּמָּטֵר, 
וְחֶלְקָה אֲשֶׁר לֹא תַמְטִיר עָלֶיהָ תִּיבָשׁ".  
אוֹתָהּ הָעִיר מִתְעַנָּא וּמַתְרַעַת, 
וְכָל סְבִיבוֹתֶיהָ מִתְעַנּוֹת וְלֹא מַתְרִיעוֹת.  
וְרֶבִּי עֲקִיבָה אוֹמֵר: 
מַתְרִיעוֹת וְלֹא מִתְעַנּוֹת. 

<small>ד</small>
וְכֵן עִיר שֶׁיֶּשׁ בָּהּ דֶּבֶר אוֹ מַפֹּלֶת, 
אוֹתָהּ הָעִיר מִתְעַנָּה וּמַתְרַעַת; 
וְכָל סְבִיבוֹתֶיהָ מִתְעַנּוֹת וְלֹא מַתְרִיעוֹת.  
רְבִּי עֲקִיבָה אוֹמֵר: 
מַתְרִיעוֹת וְלֹא מִתְעַנּוֹת.  

<small>ה</small>
אֵי זֶה הוּא דֶבֶר? 
עִיר הַמּוֹצִיאָה חֲמֵשׁ מֵאוֹת רַגְלִי, 
יָצְאוּ מִמֶּנָּה שְׁלֹשָׁה מֵתִים 
לִשְׁלֹשָׁה יָמִים זֶה אַחַר זֶה.  

<small>ו</small>
עַל אֵלּוּ מַתְרִיעִים בְּכָל מָקוֹם: 
עַל הַשִּׁדָּפוֹן וְעַל הַיֵּרָקוֹן, 
עַל הָאַרְבֶּה וְעַל הֶחָסִיל, 
וְעַל חַיָּה רָעָה וְעַל הַחֶרֶב.  
מַתְרִיעִים עָלֶיהָ, 
מִפְּנֵי שֶׁהִיא מַכָּה מְהַלֶּכֶת. 

<small>ז</small>
מַעֲשֶׂה שֶׁיָּרְדוּ זְקֵנִים מִירוּשָׁלַיִם לְעָרֵיהֶם, 
וְגָזְרוּ תַעֲנִית 
עַל שֶׁנִּרְאָה כִמְלֹא פִי תַנּוּר שִׁדָּפוֹן בְּאַשְׁקְלוֹן.  
וְעוֹד גָּזְרוּ תַעֲנִית, 
עַל שֶׁאָכְלוּ זְאֵבִים שְׁנֵי תִינוֹקוֹת בְּעֵבֶר הַיַּרְדֵּן.  
רְבִּי יוֹסֵה אוֹמֵר: 
לֹא עַל שֶׁאָכְלוּ אֶלָּא עַל שֶׁנִּרְאוּ. 

<small>ח</small>
וְעַל אֵלּוּ מַתְרִיעִים בַּשַּׁבָּת: 
עַל עִיר שֶׁהִקִּיפוּהָ גוֹיִם אוֹ נָהָר, 
וְעַל הַסְּפִינָה הַמִּטָּרֶפֶת בַּיָּם.  
רְבִּי יוֹסֵה אוֹמֵר: 
לְעֶזְרָה אֲבָל לֹא לִצְעָקָה.  
שִׁמְעוֹן הַתִּמְנִי אוֹמֵר: 
אַף עַל הַדֶּבֶר.  
וְלֹא הוֹדוּ לוֹ חֲכָמִים. 

<small>ט</small>
עַל כָּל צָרָה שֶׁתָּבֹא עַל הַצִּבּוּר, 
מַתְרִיעִים עָלֶיהָ, חוּץ מֵרֹב גְּשָׁמִים.  
מַעֲשֶׂה שֶׁאָמְרוּ לְחוּנִי הַמַּעְגָּל: 
הִתְפַּלֵּל שֶׁיֵּרְדוּ גְשָׁמִים.  
אָמַר לָהֶם: 
צְאוּ וְהַכְנִיסוּ תַנּוּרֵי פְסָחִים, 
בִּשְׁבִיל שֶׁלֹּא יִמַּקּוּ.  
הִתְפַּלַּל, וְלֹא יָרְדוּ גְשָׁמִים.  

<small>י</small>
עָג עוּגָה וְעָמַד בְּתוֹכָהּ, 
וְאָמַר לְפָנָיו: 
רַבּוּנוֹ שֶׁלָּעוֹלָם, 
בָּנֶיךָ שָׂמוּ פְנֵיהֶם עָלַי, 
שֶׁאֲנִי כְבֶן בַּיִת לְפָנֶיךָ.  
נִשְׁבָּע אֲנִי בְשִׁמְךָ הַגָּדוֹל, 
שֶׁאֵינִי זָז מִכָּן, 
עַד שֶׁתְּרַחֵם עַל בָּנֶיךָ.  

<small>יא</small>
הִתְחִילוּ הַגְּשָׁמִים מְנַטְּפִים.  
אָמַר: 
לֹא כָךְ שָׁאַלְתִּי, 
אֶלָּא גִשְׁמֵי בוֹרוֹת שִׁיחִים וּמְעָרוֹת.  
יָרְדוּ בְזַעַף.  
אָמַר: 
לֹא כָךְ שָׁאַלְתִּי, 
אֶלָּא גִשְׁמֵי רָצוֹן, בְּרָכָה וּנְדָבָה.  
יָרְדוּ כְתִקְנָן, 
עַד שֶׁעָלוּ יִשְׂרָאֵל מִירוּשָׁלַיִם לְהַר הַבַּיִת 
מִפְּנֵי הַגְּשָׁמִים.  
בָּאוּ וְאָמְרוּ לוֹ: 
כַּשֵּׁם שֶׁהִתְפַּלַּלְתָּ עֲלֵיהֶם שֶׁיֵּרֵדוּ, 
כָּךְ הִתְפַּלֵּל שֶׁיֵּלְכוּ לָהֶם.  
אָמַר לָהֶם: 
צְאוּ וּרְאוּ אִם נִמְחָת אֶבֶן הַטּוֹעִים.  

<small>יב</small>
שָׁלַח לוֹ שִׁמְעוֹן בֶּן שָׁטָח: 
צָרִיךְ אַתָּה לִנַּדּוֹת,
אֲבָל מָה אֶעֱשֶׂה לָךְ, 
וְאַתָּה מִתְחַטֵּא לִפְנֵי הַמָּקוֹם, 
כְּבֵן שֶׁהוּא מִתְחַטֵּא לְאָבִיו וְעוֹשֶׂה לוֹ רְצוֹנוֹ.  
וְעָלֶיךָ הַכָּתוּב אוֹמֵר: (משלי כג,כה) 
"יִשְׂמַח אָבִיךָ וְאִמֶּךָ וְתָגֵל יוֹלַדְתֶּךָ". 

<small>יג</small>
הָיוּ מִתְעַנִּים וְיָרְדוּ לָהֶם גְּשָׁמִים, 
קֹדֶם לְהָנֵץ הַחַמָּה, לֹא יַשְׁלִימוּ; 
לְאַחַר הָנֵץ הַחַמָּה יַשְׁלִימוּ.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
קֹדֶם לַחֲצוֹת, לֹא יַשְׁלִימוּ, 
אַחַר חֲצוֹת, יַשְׁלִימוּ.  

<small>יד</small>
מַעֲשֶׂה שֶׁגָּזְרוּ תַעֲנִית בְּלוֹד, 
וְיָרְדוּ לָהֶם גְּשָׁמִים קֹדֶם לַחֲצוֹת.  
אָמַר לָהֶם רְבִּי טַרְפוֹן: 
צְאוּ וְאִכְלוּ וּשְׁתוּ וַעֲשׁוּ יוֹם טוֹב.  
וְיָצְאוּ וְאָכְלוּ וְשָׁתוּ וְעָשׁוּ יוֹם טוֹב, 
וּבָאוּ בֵין הָעַרְבַּיִם וְקָרְאוּ הַלֵּל הַגָּדוֹל. 
וְאֵי זֶה הוּא הַלֵּל הַגָּדוֹל?
הוֹדוּ לֵאלֹהֵי הָאֱלֹהִים כִּי לְעוֹלָם חַסְדּוֹ. 
הוֹדוּ לַאֲדֹנֵי הָאֲדֹנִים כִּי לְעוֹלָם חַסְדּוֹ. 




Chapter 4

<small>א</small>
בִּשְׁלֹשָׁה פְרָקִים בַּשָּׁנָה, 
הַכֹּהֲנִים נוֹשְׂאִין אֶת כַּפֵּיהֶן אַרְבַּע פְּעָמִים בַּיּוֹם, 
בַּשַּׁחְרִית וּבַמּוּסָף וּבַמִּנְחָה וּבִנְעִילַת שְׁעָרִים: 
בַּתַּעְנִיּוֹת וּבְמַעֲמָדוֹת וּבְיוֹם הַכִּפּוּרִים. 

<small>ב</small>
אֵלּוּ הֵן הַמַּעֲמָדוֹת? 
לְפִי שֶׁנֶּאֱמַר: (במדבר כח,ב) 
"צַו אֶת בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵיהֶם, 
אֶת קָרְבָּנִי לַחְמִי לְאִשַּׁי"; 
וְכִי הֵיאָךְ קָרְבָּנוֹ שֶׁלָּאָדָם קָרֵב, 
וְהוּא אֵינוּ עוֹמֵד עַל גַּבָּיו? 
אֶלָּא שֶׁהִתְקִינוּ הַנְּבִיאִים הָרִאשׁוֹנִים 
עֶשְׂרִים וְאַרְבָּעָה מִשְׁמָרוֹת; 
עַל כָּל מִשְׁמָר וּמִשְׁמָר הָיָה עַמּוּד בִּירוּשָׁלַיִם, 
שֶׁלַּכֹּהֲנִים, וְשֶׁלַּלְּוִיִּם וְשֶׁלְּיִשְׂרָאֵל.  
הִגִּיעַ זְמַן הַמִּשְׁמָר, 
כֹּהֲנָיו וּלְוִיָּו עוֹלִים בִּירוּשָׁלַיִם, 
וְיִשְׂרָאֵל שֶׁבְּאוֹתוֹ הַמִּשְׁמָר, 
מִתְכַּנְּסִים לְעָרֵיהֶם וְקוֹרְאִים מַעֲשֵׂה בְרֵאשִׁית. 

(וְאַנְשֵׁי הַמַּעֲמָד הָיוּ מִתְעַנִּין אַרְבָּעָה יָמִים בַּשָּׁבוּעַ, 
מִיּוֹם שֵׁנִי וְעַד יוֹם חֲמִישִׁי.  
וְלֹא הָיוּ מִתְעַנִּין עֶרֶב שַׁבָּת, 
מִפְּנֵי כְבוֹד הַשַּׁבָּת; 
וְלֹא בְאֶחָד בַּשַּׁבָּת, 
כְּדֵי שֶׁלֹּא יֵצְאוּ מִמְּנוּחָה וָענֶג, 
לִיגִיעָה וְתַעֲנִית וְיָמוּתוּ. ) 

<small>ג</small>
בַּיּוֹם הָרִאשׁוֹן "בְּרֵאשִׁית", "יְהִי רָקִיעַ".  
בַּשֵּׁנִי "יְהִי רָקִיעַ", "יִקָּווּ הַמַּיִם".  
בַּשְּׁלִישִׁי "יִקָּווּ הַמַּיִם", "יְהִי מְאֹרֹת".  
בָּרְבִיעִי "יְהִי מְאֹרֹת", "יִשְׁרְצוּ הַמַּיִם".  
בַּחֲמִישִׁי "יִשְׁרְצוּ הַמַּיִם", "תּוֹצֵא הָאָרֶץ".  
בַּשִּׁשִּׁי "תּוֹצֵא הָאָרֶץ", 
"וַיְכֻלּוּ הַשָּׁמַיִם וְהָאָרֶץ וְכָל צְבָאָם".  

<small>ד</small>
פָּרָשָׁה גְדוֹלָה קוֹרִין אוֹתָהּ בִּשְׁנַיִם, 
וְהַקְּטַנָּה בַיָּחִיד; 
בַּשַּׁחְרִית וּבַמּוּסָף וּבַמִּנְחָה, 
נִכְנָסִין וְקוֹרִין עַל פִּיהֶן, 
כְּקוֹרִין אֶת שְׁמַע.  
עֶרֶב שַׁבָּת בַּמִּנְחָה לֹא הָיוּ נִכְנָסִין, 
מִפְּנֵי כְבוֹד הַשַּׁבָּת. 

<small>ה</small>
כָּל יוֹם שֶׁיֶּשׁ בּוֹ הַלֵּל, אֵין בּוֹ מַעֲמַד שַׁחְרִית; 
קָרְבַּן מוּסָף, אֵין בּוֹ נְעִילָה; 
קָרְבַּן עֵצִים, אֵין בּוֹ מִנְחָה.  
דִּבְרֵי רְבִּי עֲקִיבָה.  
אָמַר לוֹ בֶן עַזַּי: 
כָּךְ הָיָה רְבִּי יְהוֹשֻׁעַ שׁוֹנֶה: 
קָרְבַּן מוּסָף, אֵין בּוֹ מִנְחָה; 
קָרְבַּן עֵצִים, אֵין בּוֹ נְעִילָה.  
חָזַר רְבִּי עֲקִיבָה לִהְיוֹת שׁוֹנֶה כְדִבְרֵי בֶן עַזַּי. 

<small>ו</small>
זְמַן עֲצֵי כֹהֲנִים וְהָעָם בְּתִשְׁעָה.  
בְּאֶחָד בְּנִיסָן, בְּנֵי אָרַח בֶּן יְהוּדָה; 
בְּעֶשְׂרִים בְּתַמּוּז, בְּנֵי דָוִד בֶּן יְהוּדָה; 
בַּחֲמִשָּׁה בְאָב, בְּנֵי פַרְעשׁ בֶּן יְהוּדָה; 
בְּשִׁבְעָה בוֹ, בְּנֵי יוֹנָדָב בֶּן רֵכָב; 
בַּעֲשָׂרָה בוֹ, בְּנֵי סְנָאָה בֶן בִּנְיָמִין; 
בַּחֲמִשָּׁה עָשָׂר בּוֹ, בְּנֵי זַתּוּאֵל בֶּן יְהוּדָה, 

<small>ז</small>
וְעִמָּהֶם כֹּהֲנִים וּלְוִיִּם וְכָל מִי שֶׁטָּעָה שִׁבְטוֹ, 
וּבְנֵי גֹנְבֵי עֲלִי, בְּנֵי קוֹצְעֵי קְצִיעוֹת.  
בְּעֶשְׂרִים בּוֹ, בְּנֵי פַחַת מוֹאָב בֶּן יְהוּדָה; 
בְּעֶשְׂרִים בֶּאֱלוּל, בְּנֵי עַדִּין בֶּן יְהוּדָה; 
בְּאֶחָד בְּטֵבֵת שָׁבוּ בְּנֵי פַרְעוֹשׁ שְׁנִיָּה.  
בְּאֶחָד בְּטֵבֵת, לֹא הָיָה בוֹ מַעֲמָד, 
שֶׁהָיָה בוֹ הַלֵּל, קָרְבַּן מוּסָף וְקָרְבַּן עֵצִים. 

<small>ח</small>
חֲמִשָּׁה דְבָרִים אֵרְעוּ אֶת אֲבוֹתֵינוּ 
בְשִׁבְעָה עָשָׂר בְּתַמּוּז, 
וַחֲמִשָּׁה בְתִשְׁעָה בְאָב.  
בְּשִׁבְעָה עָשָׂר בְּתַמּוּז, 
נִשְׁתַּבְּרוּ הַלּוּחוֹת, 
וּבָטַל הַתָּמִיד, 
וְהָבְקָעָה הָעִיר, 
וְשָׂרַף אַפִּיסְטְמוֹס אֶת הַתּוֹרָה, 
וְהֶעֱמִיד צֶלֶם בַּהֵיכָל.  
בְּתִשְׁעָה בְאָב, 
נִגְזַר עַל אֲבוֹתֵינוּ שֶׁלֹּא יִכָּנְסוּ לָאָרֶץ, 
וְחָרַב הַבַּיִת בָּרִאשׁוֹנָה וּבַשְּׁנִיָּה, 
וְנִלְכְּדָה בֵית תֵּר, 
וְנֶחְרְשָׁה הָעִיר.  

<small>ט</small>
מִשֶּׁנִּכְנַס אָב, מְמַעֲטִים בְּשִׂמְחָה. 

שַׁבָּת שֶׁחָל תִּשְׁעָה בְאָב לִהְיוֹת בְּתוֹכָהּ, 
אָסוּרִים מִלְּסַפֵּר וּמִלְּכַבֵּס, 
וּבַחֲמִישִׁי מֻתָּרִים, מִפְּנֵי כְבוֹד הַשַּׁבָּת.  

<small>י</small>
עֶרֶב תִּשְׁעָה בְאָב, 
לֹא יֹאכַל אָדָם שְׁנֵי תַבְשִׁילִים, 
לֹא יֹאכַל בָּשָׂר וְלֹא יִשְׁתֶּה יַיִן.  
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
יְשַׁנֶּה.  
רְבִּי יְהוּדָה מְחַיֵּב בִּכְפִיַּת הַמִּטָּה; 
וְלֹא הוֹדוּ לוֹ חֲכָמִים. 

<small>יא</small>
אָמַר רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל: 
לֹא הָיוּ יָמִים טוֹבִים לְיִשְׂרָאֵל 
כַּחֲמִשָּׁה עָשָׂר בְּאָב וּכְיוֹם הַכִּפּוּרִים, 
שֶׁבָּהֶן בְּנֵי יְרוּשָׁלַיִם יוֹצְאִין בִּכְלֵי לָבָן שְׁאוּלִים, 
שֶׁלֹּא לְבַיֵּשׁ אֶת מִי שֶׁאֵין לוֹ; 
כָּל הַכֵּלִים טְעוּנִים טְבִילָה.  
וּבְנוֹת יְרוּשָׁלַיִם יוֹצְאוֹת וְחוֹלוֹת בַּכְּרָמִים.  

<small>יב</small>
וּמָה הָיוּ אוֹמְרוֹת? 
שָׂא נָא עֵינֶיךָ בָחוּר וּרְאֵה, 
מָה אַתְּ בּוֹרֵר לָךְ.  
אַל תִּתֵּן עֵינֶיךָ בַנּוֹי, 
תֵּן עֵינֶיךָ בַמִּשְׁפָּחָה.  

("שֶׁקֶר הַחֵן וְהֶבֶל הַיּפִי, 
אִשָּׁה יִרְאַת יי הִיא תִתְהַלָּל" (משלי לא,ל); 
וְאוֹמֵר: (שם שם,לא) 
"תְּנוּ לָהּ מִפְּרִי יָדֶיהָ, 
וִיהַלְלוּהָ בַשְּׁעָרִים מַעֲשֶׂיהָ". ) 

וְכֵן הוּא אוֹמֵר: (שיר השירים ג,יא) 
"צְאֶינָה וּרְאֶינָה בְנוֹת צִיּוֹן בַּמֶּלֶךְ שְׁלֹמֹה, 
בַּעֲטָרָה שֶׁעִטְּרָה לוֹ אִמּוֹ בְיוֹם חֲתֻנָּתוֹ וּבְיוֹם שִׂמְחַת לִבּוֹ"; 
"בְּיוֹם חֲתֻנָּתוֹ", זוֹ מַתַּן תּוֹרָה, 
"וּבְיוֹם שִׂמְחַת לִבּוֹ", זֶה בִנְיַן בֵּית הַמִּקְדָּשׁ, 
יְהִי רָצוֹן שֶׁיִּבָּנֶה בִּמְהֵרָה בְיָמֵינוּ.
