משנה פסחים



Chapter 1

<small>א</small>
אוֹר לְאַרְבָּעָה עָשָׂר, 
בּוֹדְקִין אֶת הֶחָמֵץ לְאוֹר הַנֵּר.  
כָּל מָקוֹם שֶׁאֵין מַכְנִיסִין בּוֹ חָמֵץ, 
אֵינוּ צָרִיךְ בְּדִיקָה.  
וְלָמָּה אָמָרוּ "שְׁתֵּי שׁוּרוֹת בַּמַּרְתֵּף"? 
בְּמָקוֹם שֶׁמַּכְנִיסִין בּוֹ חָמֵץ.  
בֵּית שַׁמַּי אוֹמְרִים: 
שְׁתֵּי שׁוּרוֹת עַל פְּנֵי כָל הַמַּרְתֵּף; 
וּבֵית הֶלֵּל אוֹמְרִים: 
שְׁתֵּי שׁוּרוֹת הַחִיצוֹנוֹת, שֶׁהֵן הָעֶלְיוֹנוֹת. 

<small>ב</small>
אֵין חוֹשְׁשִׁין שֶׁמֵּא גֵרְרָה חֻלְדָּה 
מִבַּיִת לְבַיִת וּמִמָּקוֹם לְמָקוֹם.  
אִם כֵּן, מֵחָצֵר לֶחָצֵר אוֹ מֵעִיר לָעִיר, 
אֵין לַדָּבָר סוֹף. 

<small>ג</small>
רְבִּי יְהוּדָה אוֹמֵר: 
בּוֹדְקִים אוֹר אַרְבָּעָה עָשָׂר, 
וּבְאַרְבָּעָה עָשָׂר בַּשַּׁחְרִית וּבְשָׁעַת הַבֵּעוּר; 
וַחֲכָמִים אוֹמְרִים: 
לֹא בָדַק בְּאוֹר אַרְבָּעָה עָשָׂר, 
יִבְדֹּק בְּאַרְבָּעָה עָשָׂר; 
אִם לֹא בָדַק בְּאַרְבָּעָה עָשָׂר, 
יִבְדֹּק בְּתוֹךְ הַמּוֹעֵד; 
לֹא בָדַק בְּתוֹךְ הַמּוֹעֵד, 
יִבְדֹּק לְאַחַר הַמּוֹעֵד.  
וּמַה שֶׁהוּא מְשַׁיֵּר יַנִּיחֶנוּ בְצִנְעָה, 
כְּדֵי שֶׁלֹּא יְהֵי צָרִיךְ בְּדִיקָה אַחֲרָיו. 

<small>ד</small>
רִבִּי מֵאִיר אוֹמֵר: 
אוֹכְלִים כָּל חָמֵשׁ, 
וְשׁוֹרְפִים בִּתְחִלַּת שֵׁשׁ; 
רַבִּי יְהוּדָה אוֹמֵר: 
אוֹכְלִים כָּל אַרְבַּע, 
וְתוֹלִים כָּל חָמֵשׁ, 
וְשׁוֹרְפִים בִּתְחִלַּת שֵׁשׁ. 

<small>ה</small>
וְעוֹד אָמַר רְבִּי יְהוּדָה: 
שְׁתֵּי חַלּוֹת שֶׁלַתּוֹדָה פְסוּלוֹת &lt; שֶׁלתּוֹדָה &gt;
וּמֻנָּחוֹת עַל גַּג הָאַסְטְוָה, 
כָּל זְמַן שֶׁהֵן מֻנָּחוֹת, כָּל הָעָם אוֹכְלִים; 
נִטְּלָה אַחַת מֵהֶן, 
תּוֹלִים, לֹא אוֹכְלִים וְלֹא שׁוֹרְפִים; 
נִטְּלוּ שְׁתֵּיהֶן, 
הִתְחִילוּ כָל הָעָם שׂוֹרְפִין.  

<small>ו</small>
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
חֻלִּים נֶאֱכָלִים כָּל אַרְבַּע, 
וּתְרוּמָה כָל חָמֵשׁ, 
וְשׁוֹרְפִים בִּתְחִלַּת שֵׁשׁ. 

<small>ז</small>
רְבִּי חֲנַנְיָה סְגַן הַכֹּהֲנִים אוֹמֵר: 
מִימֵיהֶם שֶׁלַּכֹּהֲנִים, 
לֹא נִמְנְעוּ מִלִּשְׂרֹף אֶת הַבָּשָׂר שֶׁנִּטַּמָּא בִוְלַד הַטֻּמְאָה, 
עִם הַבָּשָׂר שֶׁנִּטַּמָּא בְאַב הַטֻּמְאָה, 
אַף עַל פִּי שֶׁמּוֹסִיפִין לוֹ טֻמְאָה עַל טֻמְאָתוֹ.  

<small>ח</small>
הוֹסִיף רְבִּי עֲקִיבָה: 
מִימֵיהֶן שֶׁלַּכֹּהֲנִים, 
לֹא נִמְנְעוּ מִלְּהַדְלִיק אֶת הַשֶּׁמֶן שֶׁנִּפְסַל בִּטְבוּל יוֹם, 
בְּנֵר שֶׁנִּטַּמָּא בִטְמֵא מֵת, 
אַף עַל פִּי שֶׁמּוֹסִיפִין לוֹ טֻמְאָה עַל טֻמְאָתוֹ. 

<small>ט</small>
אָמַר רְבִּי מֵאִיר: 
מִדִּבְרֵיהֶם לָמַדְנוּ, 
שֶׁשּׂוֹרְפִים תְּרוּמָה טְהוֹרָה עִם הַטְּמֵאָה בַפֶּסַח.  
אָמַר לוֹ רְבִּי יוֹסֵה: 
אֵינָה הִיא הַמִּדָּה.  
מוֹדֶה רְבִּי אֱלִיעֶזֶר וּרְבִּי יְהוֹשֻׁעַ, 
שֶׁשּׂוֹרְפִין זוֹ לְעַצְמָהּ וְזוֹ לְעַצְמָהּ; 
וְעַל מָה נֶחְלָקוּ? 
עַל הַתְּלוּיָה וְעַל הַטְּמֵאָה, 
שֶׁרְבִּי אֱלִיעֶזֶר אוֹמֵר: 
תִּשָּׂרֵף זוֹ לְעַצְמָהּ וְזוֹ לְעַצְמָהּ; 
וּרְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
שְׁתֵּיהֶם כְּאַחַת. 





Chapter 2

<small>א</small>
כָּל שָׁעָה שֶׁהוּא מֻתָּר לֹאכָל, 
מַאֲכִיל לַבְּהֵמָה וְלַחַיָּה וְלָעוֹפוֹת, 
וּמוֹכְרוֹ לַנָּכְרִי, 
וּמֻתָּר בַּהֲנָיָתוֹ.  
עִבֵּר זְמַנּוֹ, אָסוּר בַּהֲנָיָתוֹ.  
לֹא יַסִּיק בּוֹ תַנּוּר וְכִירַיִם.  
רְבִּי יְהוּדָה אוֹמֵר: 
אֵין בֵּעוּר חָמֵץ אֶלָּא שְׂרֵפָה; 
וַחֲכָמִים אוֹמְרִים: 
מְפָרֵר וְזוֹרֶה לָרוּחַ אוֹ מֵטִיל לַיָּם. 

<small>ב</small>
חָמֵץ שֶׁלַּנָּכְרִי שֶׁעָבַר עָלָיו הַפֶּסַח, 
מֻתָּר בַּהֲנָיָה; 
וְשֶׁלְּיִשְׂרָאֵל אָסוּר, שֶׁנֶּאֱמַר: 
(שמות יג,ז) "לֹא יֵרָאֶה לָךְ". 

נָכְרִי שֶׁהִלְוָה אֶת יִשְׂרָאֵל עַל חֲמֵצוֹ לְאַחַר הַפֶּסַח, 
מֻתָּר בַּהֲנָיָה; 
וְיִשְׂרָאֵל שֶׁהִלְוָה אֶת הַנָּכְרִי עַל חֲמֵצוֹ לְאַחַר הַפֶּסַח, 
אָסוּר בַּהֲנָיָה.  
חָמֵץ שֶׁנָּפְלָה עָלָיו מַפֹּלֶת, 
הֲרֵי הוּא כִמְבֹעָר; 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
כָּל שֶׁאֵין הַכֶּלֶב יָכוֹל לְחַפֵּשׁ אַחֲרָיו. 

<small>ג</small>
הָאוֹכֵל תְּרוּמָה חָמֵץ בַּפֶּסַח, 
שׁוֹגֵג, מְשַׁלֵּם קֶרֶן וְחֹמֶשׁ; 
מֵזִיד, פָּטוּר מִתַּשְׁלוּמִים וּמִדְּמֵי עֵצִים. 

<small>ד</small>
וְאֵלּוּ דְבָרִים שֶׁאָדָם יוֹצֵא בָהֶן יְדֵי חוֹבָתוֹ בַפֶּסַח: 
בַּחִטִּים וּבַשְׂעוֹרִים וּבַכֻּסְּמִים וּבְשִׁבֹּלֶת שׁוּעָל וּבְשִׁיפוֹן 
וּבִדְמַי וּבְמַעֲשֵׂר רִאשׁוֹן שֶׁנִּטְּלָה תְרוּמָתוֹ, 
וּבְמַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁנִּפְדּוּ.  
הַכֹּהֲנִים בַּחַלָּה וּבַתְּרוּמָה, 
אֲבָל לֹא בַטֶּבֶל, 
וְלֹא בְמַעֲשֵׂר רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְרוּמָתוֹ, 
וְלֹא בְמַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁלֹּא נִפְדּוּ.  
חַלּוֹת תּוֹדָה וּרְקִיקֵי נָזִיר, 
עֲשָׂאָן לְעַצְמוֹ, אֵינוּ יוֹצֵא בָהֶן; 
לִמְכֹּר בַּשּׁוּק, יוֹצֵא בָהֶן. 

<small>ה</small>
וְאֵלּוּ יְרָקוֹת שֶׁאָדָם יוֹצֵא בָהֶן יְדֵי חוֹבָתוֹ בַפֶּסַח: 
בַּחַזֶּרֶת וּבָעֻלְשִׁים וּבִתְמָכָא וּבְחַרְחֲבִנָּה וּבְמַרוֹר.  
יוֹצְאִין בָּהֶן בֵּין לַחִים, בֵּין כְּמֵשִׁין, 
אֲבָל לֹא כְבוּשִׁין וְלֹא שְׁלוּקִין וְלֹא מְבֻשָּׁלִים.  
מִצְטָרְפִין בְּכַזַּיִת.  
וְיוֹצְאִין בְּקֶלַח שֶׁלָּהֶן, 
וּבַדְּמַאי, 
וּבְמַעֲשֵׂר רִאשׁוֹן שֶׁנִּטְּלָה תְרוּמָתוֹ, 
וּמַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁנִּפְדּוּ. 

<small>ו</small>
אֵין שׁוֹרִים אֶת הַמֻּרְסָן לַתַּרְנוֹגְלִים, 
אֲבָל חוֹלְטִין.  
הָאִשָּׁה לֹא תִשְׁרֵי אֶת הַמֻּרְסָן שֶׁתּוֹלֵךְ בְּיָדָהּ לַמַּרְחֵץ, 
אֲבָל שָׁפָה הִיא עַל בְּשָׂרָהּ יָבֵשׁ.  
לֹא יִלְעֹס אָדָם חִטִּין וְיִתֵּן עַל מַכָּתוֹ, 
מִפְּנֵי שֶׁהֵן מַחְמִיצוֹת. 

<small>ז</small>
אֵין נוֹתְנִין אֶת הַקֶּמַח לְתוֹךְ חֲרֹסֶת 
וְלֹא לְתוֹךְ חַרְדַּל.  
אִם נָתַן יֵאָכֵל מִיָּד; 
רַבִּי מֵאִיר אוֹסֵר.  
אֵין מְבַשְּׁלִין אֶת הַפֶּסַח 
לֹא בְמַשְׁקִים וְלֹא בְמֵי פֵרוֹת; 
אֲבָל סָכִים וּמַטְבִּילִין אוֹתוֹ בָהֶן.  
מֵי תַשְׁמִישָׁיו שֶׁלַּנַּחְתּוֹם יִשָּׁפֵכוּ, 
מִפְּנֵי שֶׁהֵן מַחְמִיצִים. 





Chapter 3

<small>א</small>
וְאֵלּוּ עוֹבְרִים בַּפֶּסַח: 
כֻּתַּח הַבַּבְלִי, 
שֵׁכַר הַמָּדִי, 
חֹמֶץ הָאֲדוֹמִי, 
זִיתוֹם הַמִּצְרִי, 
זֵמַן שֶׁלַּצַּבָּעִים, 
וַאֲמֵילָן שֶׁלַּטַּבָּחִים, 
וְקוֹלָן שֶׁלַּסּוֹפְרִים.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אַף טִפּוּלֵי נָשִׁים.  
זֶה הַכְּלָל: 
כָּל שֶׁהוּא מִין דָּגָן הֲרֵי זֶה עוֹבֵר בַּפֶּסַח.  
הֲרֵי אֵלּוּ בְאַזְהָרָה, 
וְאֵין בָּהֶן מִשֵּׁם כָּרֵת. 

<small>ב</small>
בָּצֵק שֶׁבְּסִדְקֵי עֲרֵבָה, 
אִם יֵשׁ כַּזַּיִת בְּמָקוֹם אֶחָד, 
חַיָּב לְבַעֵר, 
פָּחוּת מִכֵּן, בָּטֵל בְּמֵעוּטוֹ.  
וְכֵן לְעִנְיַן הַטֻּמְאָה: 
אִם הִקְפִּיד עָלָיו, חוֹצֵץ, 
אִם רוֹצֶה הוּא בְקִיּוּמוֹ, 
הֲרֵי הוּא כַעֲרֵבָה.  
בָּצֵק הַחֵרֵשׁ, 
אִם יֵשׁ כַּיּוֹצֵא בוֹ שֶׁהֶחְמִיץ, 
הֲרֵי זֶה אָסוּר. 

<small>ג</small>
כֵּיצַד מַפְרִישִׁין חַלַּת טֻמְאָה בְיוֹם טוֹב? 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אַל תִּקְרָא לָהּ שֵׁם עַד שֶׁתֵּאָפֶה; 
בֶּן בְּתִירָה אוֹמֵר: 
תָּטִיל לַצּוֹנִים.  
אָמַר רְבִּי יְהוֹשֻׁעַ: 
לֹא זֶה הוּא חָמֵץ שֶׁמֻּזְהָרִים עָלָיו 
בַּל יֵרָאֶה וּבַל יִמָּצֵא, 
אֶלָּא מַפְרְשַׁתָּהּ וּמַנְּחַתָּהּ עַד הָעֶרֶב, 
וְאִם הֶחְמִיצָה, הֶחְמִיצָה. 

<small>ד</small>
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
שָׁלוֹשׁ נָשִׁים לָשׁוֹת כְּאַחַת 
וְאוֹפוֹת בְּתַנּוּר אֶחָד זוֹ אַחַר זוֹ; 
וַחֲכָמִים אוֹמְרִים: 
שָׁלוֹשׁ נָשִׁים עֲסִיקוֹת בַּבָּצֵק, 
אַחַת לָשָׁה וְאַחַת עוֹרֶכֶת וְאַחַת אוֹפָה.  
רְבִּי עֲקִיבָה אוֹמֵר: 
לֹא כָל הַנָּשִׁים וְלֹא כָל הָעֵצִים וְלֹא כָל הַתַּנּוּרִים שָׁוִים.  
זֶה הַכְּלָל: 
תֹּאפָה תִלְטֹשׂ בְּצוֹנִין. 

<small>ה</small>
סֵעוּר יִשָּׂרֵף, 
וְהָאוֹכְלוֹ פָטוּר. 
סִדּוּק יִשָּׂרֵף, 
וְהָאוֹכְלוֹ חַיָּב מִיתָה.  
אֵי זֶה הוּא סֵעוּר? 
כְּקַרְנֵי חֲגָבִים.  
סִדּוּק? 
שֶׁנִּתְעָרְבוּ סְדָקָיו זֶה בָזֶה.  
דִּבְרֵי רְבִּי יְהוּדָה.  
וַחֲכָמִים אוֹמְרִים: 
זֶה וָזֶה, הָאוֹכְלוֹ חַיָּב מִיתָה; &lt;בַזֶה&gt;
אֵי זֶה הוּא סֵעוּר? 
כָּל שֶׁהִכְסִיפוּ פָנָיו, 
כְּאָדָם שֶׁעָמְדוּ שְׂעָרוֹתָיו. 

<small>ו</small>
אַרְבָּעָה עָשָׂר שֶׁחָל לִהְיוֹת בַּשַּׁבָּת, 
מְבַעֲרִים אֶת הַכֹּל מִלִּפְנֵי הַשַּׁבָּת.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
בִּזְמַנָּן.  
רְבִּי אֶלְעָזָר בִּרְבִּי צָדוֹק אוֹמֵר: 
תְּרוּמָה מִלִּפְנֵי הַשַּׁבָּת וְחֻלִּים בִּזְמַנָּן. 

<small>ז</small>
הַהוֹלֵךְ לִשְׁחוֹט אֶת פִּסְחוֹ, 
וְלָמוֹל אֶת בְּנוֹ, 
וְלֹאכַל סְעוֹדַת אֵרוּסִים בְּבֵית חָמִיו, 
וְנִזְכַּר שֶׁיֵּשׁ לוֹ חָמֵץ בְּתוֹךְ הַבַּיִת; 
אִם יָכוֹל לַחְזוֹר וּלְבָעֵר וְלַחְזוֹר בְּמִצְוָתוֹ, 
יַחְזוֹר; 
וְאִם לָאו, יְבַטֵּל בְּלִבּוֹ.  
וּלְהַצִּיל מִיַּד הַגַּיִס, 
וּמִיַּד הַנָּהָר, 
וּמִיַּד הַדְּלֵקָה, 
וּמִיַּד הַמַּפְּלוֹת, יְבַטֵּל בְּלִבּוֹ, 
וְיִשְׁבּוֹת שְׁבִיתַת הָרֵשׁוּת וְיַחְזוֹר מִיָּד. 

<small>ח</small>
וְכֵן מִי שֶׁיָּצָא מִירוּשָׁלַיִם, 
וְנִזְכַּר שֶׁיֵּשׁ בְּיָדוֹ בְשַׂר הַקֹּדֶשׁ, 
אִם עִבֵּר הַצּוֹפִים, 
שׁוֹרְפוֹ בִמְקוֹמוֹ, 
וְאִם לָאו, 
חוֹזֵר וְשׁוֹרְפוֹ לִפְנֵי הַבִּירָה מֵעֲצֵי הַמַּעֲרָכָה.  
וְעַד כַּמָּה הֵן חוֹזְרִין? 
רְבִּי מֵאִיר אוֹמֵר: 
זֶה וָזֶה כַבֵּיצָה. &lt;מחוק ע"י מגיה: כַזַּיִת&gt; 
רְבִּי יְהוּדָה אוֹמֵר: 
זֶה וָזֶה בְּכַזַּיִת.  &lt;מחוק ע"י מגיה: כַבֵּיצָה&gt;
וַחֲכָמִים אוֹמְרִים: 
בְּשַׂר הַקֹּדֶשׁ כַּזַּיִת, 
וְחָמֵץ כַּבֵּיצָה. 





Chapter 4

<small>א</small>
מָקוֹם שֶׁנָּהֲגוּ לַעֲשׁוֹת מְלָאכָה בְעַרְבֵי פְסָחִים עַד חֲצוֹת, 
עוֹשִׂין, 
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לַעֲשׁוֹת, 
אֵינָן עוֹשִׂין.  
הַהוֹלֵךְ מִמָּקוֹם שֶׁעוֹשִׂין לְמָקוֹם שֶׁאֵינָן עוֹשִׂין, 
אוֹ מִמָּקוֹם שֶׁאֵינָן עוֹשִׂין לְמָקוֹם שֶׁעוֹשִׂין, 
נוֹתְנִין עָלָיו חָמְרֵי הַמָּקוֹם שֶׁיָּצָא מִשָּׁם, 
וְחָמְרֵי הַמָּקוֹם שֶׁהָלַךְ לַשָּׁם.  
וְאַל יְשַׁנֶּה אָדָם, מִפְּנֵי הַמַּחְלְקוֹת. 

<small>ב</small>
כַּיּוֹצֵא בוֹ: 
הַמּוֹלִיךְ פֵּרוֹת שְׁבִיעִית 
מִמָּקוֹם שֶׁכָּלוּ לְמָקוֹם שֶׁלֹּא כָלוּ, 
אוֹ מִמָּקוֹם שֶׁלֹּא כָלוּ לְמָקוֹם שֶׁכָּלוּ, 
חַיָּב לְבַעֵר.  
רְבִּי יְהוּדָה אוֹמֵר: 
צֵא וְהָבֵא לָךְ אַף אַתָּה. 

<small>ג</small>
מָקוֹם שֶׁנָּהֲגוּ לִמְכּוֹר בְּהֵמָה דַקָּה לַגּוֹיִם, 
מוֹכְרִין; 
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לִמְכּוֹר, 
אֵינָן מוֹכְרִין.  
וְאַל יְשַׁנֶּה אָדָם, מִפְּנֵי הַמַּחְלְקוֹת. 
בְּכָל מָקוֹם, 
אֵין מוֹכְרִין לָהֶם בְּהֵמָה גַסָּה, 
עֲגָלִים וּסְיָחִים שְׁלֵמִים וּשְׁבוּרִים; 
רְבִּי יְהוּדָה מַתִּיר בִּשְׁבוּרָה; 
בֶּן בְּתִירָה מַתִּיר בַּסּוּס. 

<small>ד</small>
מָקוֹם שֶׁנָּהֲגוּ לֹאכַל צָלִי בְלֵילֵי פְסָחִים, 
אוֹכְלִין; 
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לֹאכַל, 
אֵינָן אוֹכְלִין.  
מָקוֹם שֶׁנָּהֲגוּ לְהַדְלִיק אֶת הַנֵּר בְּלֵילֵי יוֹם הַכִּפּוּרִים, 
מַדְלִיקִים; 
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לְהַדְלִיק, 
אֵינָן מַדְלִיקִים; 
מַדְלִיקִים בְּבָתֵּי כְנֵסָיוֹת וּבְבָתֵּי מִדְרָשׁוֹת, 
וּבְמָבוֹאוֹת אֲפֵלִים וְעַל גַּבֵּי הַחוֹלִים. 

<small>ה</small>
מָקוֹם שֶׁנָּהֲגוּ לַעֲשׁוֹת מְלָאכָה בְתִשְׁעָה בְאָב, 
עוֹשִׂין; 
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לַעֲשׁוֹת, 
אֵינָן עוֹשִׂין; 
בְּכָל מָקוֹם, 
תַּלְמִידֵי חֲכָמִים בְּטֵלִים.  
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
יַעֲשׂוּ כָל הָעָם עַצְמָן כְּתַלְמִידֵי חֲכָמִים.  

<small>ו</small>
וַחֲכָמִים אוֹמְרִים: 
בִּיהוּדָה עוֹשִׂין מְלָאכָה בְעַרְבֵי פְסָחִים עַד חֲצוֹת, 
וּבַגָּלִיל אֵינָן עוֹשִׂין כָּל עִקָּר.  
וּבַלַּיְלָה, 
בֵּית שַׁמַּי אוֹסְרִין, 
וּבֵית הֶלֵּל מַתִּירִין עַד שֶּתָּנֵץ הַחַמָּה. 

<small>ז</small>
רְבִּי מֵאִיר אוֹמֵר: 
כָּל מְלָאכָה שֶׁהִתְחִיל בָּהּ קֹדֶם לְאַרְבָּעָה עָשָׂר, 
גּוֹמְרָהּ בְּאַרְבָּעָה עָשָׂר; 
אֲבָל לֹא יַתְחִיל בָּהּ כַּתְּחִלָּה בְאַרְבָּעָה עָשָׂר, 
אַף עַל פִּי שֶׁהוּא יָכוֹל לְגָמְרָהּ.  
וַחֲכָמִים אוֹמְרִים: 
שָׁלוֹשׁ אֻמָּנִיּוֹת עוֹשִׂין מְלָאכָה בְעַרְבֵי פְסָחִים: 
הַחַיָּטִים וְהַסַּפָּרִים וְהַכּוֹבְסִים.  
רְבִּי יוֹסֵה בִרְבִּי יְהוּדָה אוֹמֵר: 
אַף הָרוֹצְעָנִים. 

<small>ח</small>
מוֹשִׁיבִים שׁוֹבַכִּים לַתַּרְנוֹגְלִים בְּאַרְבָּעָה עָשָׂר.  
תַּרְנְגֹלֶת שֶׁבָּרָחָה, 
מַחְזִירִין אוֹתָהּ לִמְקוֹמָהּ, 
וְאִם מֵתָה, 
מוֹשִׁיבִין אַחֶרֶת תַּחְתֶּיהָ.  
גּוֹרְפִין מִתַּחַת רַגְלֵי בְהֵמָה בְאַרְבָּעָה עָשָׂר, 
וּבַמּוֹעֵד, מְסַלְּקִים לִצְדָדִים.  
מוֹלִיכִים וּמְבִיאִין כֵּלִים מִבֵּית הָאֻמָּן, 
אַף עַל פִּי שֶׁאֵינָן לְצֹרֶךְ הַמּוֹעֵד. 

<small>ט</small>
שִׁשָּׁה דְבָרִים עָשׁוּ אַנְשֵׁי יְרִיחוֹ; 
עַל שְׁלֹשָׁה מֵחוּ בְיָדָן, 
וְעַל שְׁלֹשָׁה לֹא מֵחוּ בְיָדָן.  
וְאֵלּוּ שֶׁלֹּא מֵחוּ בְיָדָן: 
מַרְכִּיבִים דְּקָלִים בְּכָל יוֹם, 
וְכוֹרְכִים אֶת שְׁמַע, 
וְקוֹצְרִים וְגוֹדְשִׁים לִפְנֵי הָעֹמֶר, 
וְלֹא מֵחוּ בְיָדָן.  
וְאֵלּוּ שֶׁמֵּחוּ בְיָדָן: 
מַתִּירִין בְּגַמְזִיּוֹת שֶׁלַּהֶקְדֵּשׁ, 
וְאוֹכְלִין מִתַּחַת הַנְּשָׁרִים, 
וְנוֹתְנִין פֵּאָה לַיָּרָק; 
וּמֵחוּ בְיָדָן חֲכָמִים. 

(בכתב יד קאופמן נמצאת כהוספת מגיה בשוליים, בלי ניקוד)

שִׁשָּׁה דְבָרִים עָשָׂה חִזְקִיָּה הַמֶּלֶךְ, 
עַל שְׁלֹשָׁה הוֹדוּ לוֹ וְעַל שְׁלֹשָׁה לֹא הוֹדוּ לוֹ: 
גֵּרֵר עַצְמוֹת אָבִיו עַל מִטָּה שֶׁל חֲבָלִים, 
וְהוֹדוּ לוֹ; 
כִּתֵּת נְחַשׁ הַנְּחֹשֶׁת, 
וְהוֹדוּ לוֹ; 
גָּנַז סֵפֶר רְפוּאוֹת, 
וְהוֹדוּ לוֹ.  
עַל שְׁלֹשָׁה לֹא הוֹדוּ לוֹ: 
קִצֵּץ דְּלָתוֹת שֶׁלַּהֵיכָל וְשִׁגְּרָן לְמֶלֶךְ אַשּׁוּר, 
וְלֹא הוֹדוּ לוֹ; 
סָתַם מֵי גִיחוֹן הָעֶלְיוֹן, 
וְלֹא הוֹדוּ לוֹ; 
עִבֵּר נִיסָן בְּנִיסָן, 
וְלֹא הוֹדוּ לוֹ. 





Chapter 5

<small>א</small>
תָּמִיד נִשְׁחָט בִּשְׁמוֹנֶה וּמַחְצָה, 
וְקָרֵב בְּתֵשַׁע וּמַחְצָה.  
עֶרֶב פְּסָחִים, נִשְׁחָט בְּשֶׁבַע וּמַחְצָה, 
וְקָרֵב בִּשְׁמוֹנֶה וּמַחְצָה, 
בֵּין בַּחוֹל בֵּין בַּשַּׁבָּת.  
חָל עֶרֶב פְּסָחִים לִהְיוֹת עֶרֶב שַׁבָּת, 
נִשְׁחָט בְּשֵׁשׁ וּמַחְצָה 
וְקָרֵב בְּשֶׁבַע וּמַחְצָה, 
וְהַפֶּסַח אַחֲרָיו. 

<small>ב</small>
הַפֶּסַח שֶׁשְּׁחָטוֹ שֶׁלֹּא לִשְׁמוֹ, 
קִבֵּל וְהִלֵּךְ וְזָרַק שֶׁלֹּא לִשְׁמוֹ, 
אוֹ לִשְׁמוֹ וְשֶׁלֹּא לִשְׁמוֹ, 
אוֹ שֶׁלֹּא לִשְׁמוֹ וְלִשְׁמוֹ, 
פָּסוּל.  
כֵּיצַד לִשְׁמוֹ וְשֶׁלֹּא לִשְׁמוֹ? 
לְשֵׁם פֶּסַח וּלְשֵׁם שְׁלָמִים; 
שֶׁלֹּא לִשְׁמוֹ וְלִשְׁמוֹ? 
לְשֵׁם שְׁלָמִים וּלְשֵׁם הַפֶּסַח. 

<small>ג</small>
שְׁחָטוֹ שֶׁלֹּא לְאוֹכְלָיו וְשֶׁלֹּא לִמְנוּיָיו, 
לָעֲרֵלִים וְלַטְּמֵאִים, 
פָּסוּל.  
לְאוֹכְלָיו וְשֶׁלֹּא לְאוֹכְלָיו, 
לִמְנוּיָיו וְשֶׁלֹּא לִמְנוּיָיו, 
לַמּוּלִים וְלָעֲרֵלִים וְלַטְּמֵאִים וְלַטְּהוֹרִים, 
כָּשֵׁר.  
שְׁחָטוֹ קֹדֶם לַחֲצוֹת, 
פָּסוּל, שֶׁנֶּאֱמַר בּוֹ: 
(שְׁמוֹת יב, ו) "בֵּין הָעַרְבָּיִם".  
קֹדֶם לַתָּמִיד, 
כָּשֵׁר, וּבִלְבַד שֶׁיְּהֵא אֶחָד מְמָרֵס בְּדָמוֹ, 
עַד שֶׁיִּזָּרֵק דַּם הַתָּמִיד; 
וְאִם נִזְרַק, 
כָּשֵׁר. 

<small>ד</small>
הַשּׁוֹחֵט אֶת הַפֶּסַח עַל הֶחָמֵץ, 
עוֹבֵר בְּלֹא תַעֲשֶׂה.  
רְבִּי יְהוּדָה אוֹמֵר: 
אַף הַתָּמִיד.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
הַפֶּסַח בְּאַרְבָּעָה עָשָׂר, 
לִשְׁמוֹ, חַיָּב, 
וְשֶׁלֹּא לִשְׁמוֹ, פָּטוּר; 
וּשְׁאָר כָּל הַזְּבָחִים, 
בֵּין לִשְׁמָן וּבֵין שֶׁלֹּא לִשְׁמָן, 
פָּטוּר.  
וּבַמּוֹעֵד, 
לִשְׁמוֹ, פָּטוּר, 
שֶׁלֹּא לִשְׁמוֹ, חַיָּב; 
וְשֶׁאָר כָּל הַזְּבָחִים, 
בֵּין לִשְׁמָן, בֵּין שֶׁלֹּא לִשְׁמָן, חַיָּב, 
חוּץ מִן הַחַטָּאת שֶׁשְּׁחָטָהּ שֶׁלֹּא לִשְׁמָהּ. 

<small>ה</small>
הַפֶּסַח נִשְׁחָט בְּשָׁלוֹשׁ כִּתִּים, 
שֶׁנֶּאֱמַר: (שמות יב,ו) 
"וְשָׁחֲטוּ אתוֹ כֹל קְהַל עֲדַת יִשְׂרָאֵל", 
קָהָל וְעֵדָה וְיִשְׂרָאֵל.  
נִכְנְסָה כַת הָרִאשׁוֹנָה וְנִתְמַלַּת הָעֲזָרָה, 
נָעֲלוּ דַלְתוֹת הָעֲזָרָה, 
תָּקְעוּ וְהֵרִיעוּ וְתָקָעוּ.  
הַכֹּהֲנִים עוֹמְדִים שׁוּרוֹת שׁוּרוֹת, 
וּבִידֵיהֶם בַּזִּכֵּי כֶסֶף וּבַזִּכֵּי זָהָב; 
שׁוּרָה שֶׁכֻּלָּהּ כֶּסֶף, כֶּסֶף, 
וְשׁוּרָה שֶׁכֻּלָּהּ זָהָב, זָהָב, 
וְלֹא הָיוּ מְעֹרָבִים.  
וְלֹא הָיוּ לַבַּזִּכִּים שׁוּלַיִם, 
שֶׁמֵּא יַנִּיחוּם וְיִקְרַשׁ הַדָּם. 

<small>ו</small>
שָׁחַט יִשְׂרָאֵל וְקִבֵּל הַכֹּהֵן, 
נוֹתְנוֹ לַחֲבֵרוֹ וַחֲבֵרוֹ לַחֲבֵרוֹ, 
מְקַבֵּל אֶת הַמָּלֵא, וּמַחְזִיר אֶת הָרֵיקָם.  
כֹּהֵן הַקָּרוֹב אֵצֶל הַמִּזְבֵּחַ, 
זוֹרְקוֹ זְרִיקָה אַחַת כְּנֶגֶד הַיְסוֹד. 

<small>ז</small>
יָצָאת כַּת הָרִאשׁוֹנָה וְנִכְנְסָה שְׁנִיָּה, 
יָצָאת שְׁנִיָּה וְנִכְנְסָה שְׁלִישִׁית.  
כְּמַעֲשֵׂה הָרִאשׁוֹנָה, 
כֵּן מַעֲשֵׂה שְׁנִיָּה וּשְׁלִישִׁית.  
קָרְאוּ אֶת הַהַלֵּל.  
אִם גָּמָרוּ, שָׁנוּ, 
וְאִם שָׁנוּ, שִׁלֵּשׁוּ, 
אַף עַל פִּי שֶׁלֹּא שִׁלֵּשׁוּ מִימֵיהֶם.  
רְבִּי יְהוּדָה אוֹמֵר: 
מִיָּמֶיהָ שֶׁלְּכַת הַשְּׁלִישִׁית, 
לֹא הִגִּיעָה לְ"אָהַבְתִּי כִי יִשְׁמַע יי", 
מִפְּנֵי שֶׁעַמָּהּ מְמֻעָטִים. 

<small>ח</small>
כְּמַעֲשֵׂהוּ בַחוֹל, כֵּן מַעֲשֵׂהוּ בַשַּׁבָּת, 
אֶלָּא שֶׁהַכֹּהֲנִים מְדִיחִין אֶת הָעֲזָרָה, 
שֶׁלֹּא כִרְצוֹן חֲכָמִים.  
רְבִּי יְהוּדָה אוֹמֵר: 
כּוֹס הָיָה מְמַלֵּא מִדַּם הַתַּעֲרֹבֶת, 
זוֹרְקוֹ זְרִיקָה אַחַת עַל גַּבֵּי הַמִּזְבֵּחַ.  
וְלֹא הוֹדוּ לוֹ חֲכָמִים. 

<small>ט</small>
כֵּיצַד תּוֹלִים וּמַפְשִׁיטִים? 
אַנְקְלָיוֹת שֶׁלַּבַּרְזֶל הָיוּ קְבוּעִים בַּכּוֹתָלִים וּבָעַמּוּדִים, 
שֶׁבָּהֶן תּוֹלִים וּמַפְשִׁיטִים.  
כָּל מִי שֶׁאֵין לוֹ מָקוֹם לִתְלוֹת, 
מַקְלוֹת דַּקִּים חֲלָקִים הָיוּ שָׁם, 
וּמַנִּיחַ עַל כְּתֵפוֹ וְעַל כֶּתֶף חֲבֵרוֹ, 
וְתוֹלֶה וּמַפְשִׁיט.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אַרְבָּעָה עָשָׂר שֶׁחָל לִהְיוֹת בַּשַּׁבָּת, 
מַנִּיחַ אֶת יָדוֹ עַל כֶּתֶף חֲבֵרוֹ, 
וְיַד חֲבֵרוֹ עַל כְּתֵפוֹ, 
וְתוֹלֶה וּמַפְשִׁיט. 

<small>י</small>
קְרָעוֹ וְהוֹצִיא אֶת אֱמוּרָיו, 
נְתָנָן בַּמֶּגֶס וְהִקְטִירָן עַל גַּבֵּי הַמִּזְבֵּחַ.  
יָצָאת כַּת הָרִאשׁוֹנָה, 
וְיָשְׁבָה לָהּ לְהַר הַבַּיִת, 
שְׁנִיָּה בַחֵיל, 
וְהַשְּׁלִישִׁית בִּמְקוֹמָהּ.  
חֲשֵׁכָה, 
יָצְאוּ וְצָלוּ אֶת פִּסְחֵיהֶם. 





Chapter 6

<small>א</small>
אֵלּוּ דְבָרִים בַּפֶּסַח דּוֹחִין אֶת הַשַּׁבָּת: 
שְׁחִיטָתוֹ וּזְרִיקַת דָּמוֹ, 
וּמֵחוּי קְרָבָיו וְהֶקְטֵר חֲלָבָיו; 
אֲבָל צְלִיָּתוֹ וַהֲדָחַת קְרָבָיו, 
אֵינָן דּוֹחִים.  
הֶרְכֵּבוֹ וַהֲבָאָתוֹ מִחוּץ לִתְחוּם וַחֲתִיכַת יַבַּלְתּוֹ, 
אֵינָן דּוֹחִים.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
דּוֹחִים. 

<small>ב</small>
אָמַר רְבִּי אֱלִיעֶזֶר: 
מָה אִם שְׁחִיטָה, שֶׁהִיא מִשֵּׁם מְלָאכָה, 
דּוֹחָה אֶת הַשַּׁבָּת, 
אֵלּוּ שֶׁהֵן מִשֵּׁם שְׁבוּת, 
לֹא יִדְחוּ אֶת הַשַּׁבָּת? 
אָמַר לוֹ רְבִּי יְהוֹשֻׁעַ: 
יוֹם טוֹב יוֹכִיח, 
שֶׁהִתִּיר בּוֹ מִשֵּׁם מְלָאכָה, 
וְאָסַר בּוֹ מִשֵּׁם שְׁבוּת.  
אָמַר לוֹ רְבִּי אֱלִיעֶזֶר: 
מַה זֶּה, יְהוֹשֻׁעַ? 
וּמָה רְאָיָה רְשׁוּת לַמִּצְוָה? 

<small>ג</small>
הֵשִׁיב רְבִּי עֲקִיבָה: 
הַזָּיָה תוֹכִיחַ, 
שֶׁהִיא מִצְוָה וְהִיא מִשֵּׁם שְׁבוּת, 
אֵינָה דּוֹחָה אֶת הַשַּׁבָּת; 
אַף אַתָּה אַל תִּתְמַהּ עַל אֵלּוּ, 
שֶׁאַף עַל פִּי שֶׁהֵן מִצְוָה וְהֵן מִשֵּׁם שְׁבוּת, 
לֹא יִדְחוּ אֶת הַשַּׁבָּת.  

<small>ד</small>
אָמַר לוֹ רְבִּי אֱלִיעֶזֶר: 
וְעָלֶיהָ אֲנִי דָן; 
מָה אִם שְׁחִיטָה, שֶׁהִיא מִשֵּׁם מְלָאכָה, 
דּוֹחָה אֶת הַשַּׁבָּת, 
הַזָּיָה שֶׁהִיא מִשֵּׁם שְׁבוּת, 
אֵינָה דּוֹחָה אֶת הַשַּׁבָּת? 
אָמַר לוֹ רְבִּי עֲקִיבָה: 
חִלּוּף! 
מָה אִם הַזָּיָה, שֶׁהִיא מִשֵּׁם שְׁבוּת, 
אֵינָה דּוֹחָה אֶת הַשַּׁבָּת, 
אַף שְׁחִיטָה, שֶׁהִיא מִשֵּׁם מְלָאכָה, 
לֹא תִדְחֶה אֶת הַשַּׁבָּת? 

<small>ה</small>
אָמַר לוֹ רְבִּי אֱלִיעֶזֶר: 
"עֲקִיבָה! 
עָקַרְתָּ מַה שֶּׁכָּתוּב: (במדבר ט,ג) 
"בֵּין הָעַרְבַּיִם... בְּמֹעֲדוֹ", 
בֵּין בַּחוֹל בֵּין בַּשַּׁבָּת."  
אָמַר לוֹ: 
"הָבֵא לִי מוֹעֵד לָאֵלּוּ כְּמוֹעֵד לִשְׁחִיטָה!"  

<small>ו</small>
כְּלָל אָמַר רְבִּי עֲקִיבָה: 
כָּל מְלָאכָה שֶׁאֶפְשָׁר לָהּ לֵעָשׁוֹת מֵעֶרֶב שַׁבָּת, 
אֵינָה דּוֹחָה אֶת הַשַּׁבָּת.  
שְׁחִיטָה, 
שֶׁאֵי אֶפְשָׁר לָהּ לֵעָשׁוֹת מֵעֶרֶב שַׁבָּת, &lt;שאיפשר&gt;
דּוֹחָה אֶת הַשַּׁבָּת. 

<small>ז</small>
אֵימָתַי מֵבִיאִים עִמּוֹ חֲגִיגָה? 
בִּזְמַן שֶׁהוּא בָא בַחוֹל בְּטַהֲרָה וּבִמְמֻעָט.  
בִּזְמַן שֶׁהוּא בָא בַשַּׁבָּת בִּמְרֻבֶּה וּבְטֻמְאָה, 
אֵין מְבִיאִין עִמּוֹ חֲגִיגָה. 

<small>ח</small>
חֲגִיגָה הָיְתָה בָאָה מִן הַבָּקָר וּמִן הַצֹּאן, 
מִן הַכְּבָשִׂים וּמִן הָעִזִּים, 
מִן הַזְּכָרִים וּמִן הַנְּקֵבוֹת; 
וְנֶאֱכֶלֶת לִשְׁנֵי יָמִים. 

<small>ט</small>
הַפֶּסַח שֶׁשְּׁחָטוֹ שֶׁלֹּא לִשְׁמוֹ בַּשַּׁבָּת, 
חַיָּב עָלָיו חַטָּאת. 
וּשְׁאָר כָּל הַזְּבָחִים שֶׁשְּׁחָטָן לְשֵׁם הַפֶּסַח, 
אִם אֵינָן רְאוּיִים הֵן, 
חַיָּב; 
וְאִם רְאוּיִין, 
רְבִּי לִיעֶזֶר מְחַיֵּב חַטָּאת, 
רְבִּי יְהוֹשֻׁעַ פּוֹטֵר.  

<small>י</small>
אָמַר רְבִּי אֱלִיעֶזֶר: 
מָה אִם הַפֶּסַח, שֶׁהוּא מֻתָּר לִשְׁמוֹ, 
כְּשֶׁשִּׁנָּה אֶת שְׁמוֹ, חַיָּב, 
זְבָחִים, שֶׁהֵם אֲסוּרִים לִשְׁמָן, 
כְּשֶׁשִּׁנָּה אֶת שְׁמָן, 
אֵינוּ דִין שֶׁיְּהֵא חַיָּב? 
אָמַר לוֹ רְבִּי יְהוֹשֻׁעַ: 
לֹא, אִם אָמַרְתָּ בַפֶּסַח שֶׁשִּׁנָּהוּ בְדָבָר אָסוּר, 
תֹּאמַר בִּזְבָחִים, שֶׁשִּׁנָּן בְּדָבָר מֻתָּר? 
אָמַר לוֹ רְבִּי לִיעֶזֶר: 
אֱמוּרֵי צִבּוּר יוֹכִיחוּ, 
שֶׁהֵן מֻתָּרִין לִשְׁמָן, 
וְהַשּׁוֹחֵט לִשְׁמָן חַיָּב.  
אָמַר לוֹ רְבִּי יְהוֹשֻׁעַ: 
לֹא, אִם אָמַרְתָּ בֶאֱמוּרֵי צִבּוּר, 
שֶׁיֵּשׁ לָהֶן קִצְבָה, 
תֹּאמַר בַּפֶּסַח שֶׁאֵין לוֹ קִצְבָה? 
רְבִּי מֵאִיר אוֹמֵר: 
אַף הַשּׁוֹחֵט לְשֵׁם אֱמוּרֵי צִבּוּר פָּטוּר. 

<small>יא</small>
שְׁחָטוֹ שֶׁלֹּא לְאוֹכְלָיו וְשֶׁלֹּא לִמְנוּיָיו, 
לָעֲרֵלִים וְלַטְּמֵאִים, 
חַיָּב.  
לְאוֹכְלָיו וְשֶׁלֹּא לְאוֹכְלָיו, 
לִמְנוּיָיו וְשֶׁלֹּא לִמְנוּיָיו, 
לַמּוּלִים וְלָעֲרֵלִים, 
וְלַטְּמֵאִים וְלַטְּהוֹרִים, 
פָּטוּר.  
שְׁחָטוֹ וְנִמְצָא בַעַל מוּם, 
חַיָּב.  
שְׁחָטוֹ וְנִמְצָא טְרֵפָה בַסֵּתֶר, 
פָּטוּר.  
שְׁחָטוֹ וְנוֹדַע שֶׁמָּשְׁכוּ הַבְּעָלִים אֶת יְדֵיהֶן, 
אוֹ שֶׁמֵּתוּ, אוֹ שֶׁנִּטַּמְּאוּ, 
פָּטוּר, מִפְּנֵי שֶׁשָּׁחַט בִּרְשׁוּת. 





Chapter 7

<small>א</small>
כֵּיצַד צוֹלִין אֶת הַפֶּסַח? 
שְׁפוּד שֶׁלָּרִמּוֹן, 
תּוֹחֲבוֹ מִתּוֹךְ פִּיו עַד בֵּית נְקוּבָתוֹ, 
וְנוֹתֵן אֶת כְּרָעָיו וְאֶת בְּנֵי מֵעָיו לְתוֹכוֹ.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי עֲקִיבָה אוֹמֵר: 
כְּמִין בִּשּׁוּל הוּא זֶה! 
אֶלָּא תוֹלָן חוּצָה לוֹ. 

<small>ב</small>
אֵין צוֹלִין אֶת הַפֶּסַח 
לֹא בִשְׁפוּד וְלֹא בְאַסְכָּלָה.  
רְבִּי צָדוֹק אוֹמֵר: 
מַעֲשֶׂה בְרַבָּן גַּמְלִיאֵל שֶׁאָמַר לִטְבִי עַבְדּוֹ: 
"צֵא וּצְלֵה לָנוּ אֶת הַפֶּסַח עַל הָאַסְכָּלָה!"  
נָגַע בְּחַרְסוֹ שֶׁלַּתַּנּוּר, 
יִקְלֹף אֶת מְקוֹמוֹ.  
נָטַף מֵרֻטְבוֹ עַל הַחֶרֶס, חָזַר עָלָיו, 
יִטֹּל אֶת מְקוֹמוֹ.  
נָטַף מֵרֻטְבוֹ עַל הַסֹּלֶת, 
יִקְמֹץ אֶת מְקוֹמוֹ. 

<small>ג</small>
סָכוֹ בְשֶׁמֶן שֶׁלַּתְּרוּמָה, 
אִם חֲבוֹרַת הַכֹּהֲנִים, 
יֹאכֵלוּ; 
אִם שֶׁלְּיִשְׂרָאֵל, אִם חַי, 
יְדִיחֶנּוּ, 
וְאִם צָלִי, 
יִקְלֹף אֶת הַחִיצוֹן.  
סָכוֹ בְשֶׁמֶן שֶׁלְּמַעֲשֵׂר שֵׁנִי, 
לֹא יַעֲשֶׂנּוּ דָמִים עַל בְּנֵי חֲבוֹרָה, 
שֶׁאֵין פּוֹדִים מַעֲשֵׂר שֵׁנִי בִירוּשָׁלַיִם. 

<small>ד</small>
חֲמִשָּׁה דְבָרִים בָּאִים בְּטֻמְאָה, 
וְאֵינָן נֶאֱכָלִים בְּטֻמְאָה: 
הָעֹמֶר, 
וּשְׁתֵּי הַלֶּחֶם, 
וְלֶחֶם הַפָּנִים, 
וְזִבְחֵי שַׁלְמֵי צִבּוּר, 
וּשְׂעִירֵי רָאשֵׁי חֳדָשִׁים.  
הַפֶּסַח שֶׁבָּא בְטֻמְאָה נֶאֱכָל בְּטֻמְאָה, 
שֶׁלֹּא בָא מִתְּחִלָּתוֹ אֶלָּא לַאֲכִילָה. 

<small>ה</small>
נִטַּמָּא הַבָּשָׂר, וְהַחֵלֶב קַיָּם, 
אֵינוּ זוֹרֵק אֶת הַדָּם.  
נִטַּמָּא הַחֵלֶב, וְהַבָּשָׂר קַיָּם, 
זוֹרֵק אֶת הַדָּם.  
וּבַמֻּקְדָּשִׁים אֵינוּ כֵן, 
אַף עַל פִּי שֶׁנִּטַּמָּא הַבָּשָׂר וְהַחֵלֶב קַיָּם, 
זוֹרֵק אֶת הַדָּם. 

<small>ו</small>
נִטַּמָּא קָהָל אוֹ רֻבּוֹ, 
אוֹ שֶׁהָיוּ הַכֹּהֲנִים טְמֵאִין וְהַקָּהָל טָהוֹר, 
יֵעָשֶׂה בְטֻמְאָה.  
נִטַּמָּא מֵעוּט הַקָּהָל, 
הַטְּהוֹרִים עוֹשִׂין אֶת הָרִאשׁוֹן, 
וְהַטְּמֵאִים עוֹשִׂין אֶת הַשֵּׁנִי. 

<small>ז</small>
הַפֶּסַח שֶׁנִּזְרַק אֶת דָּמוֹ, 
וְאַחַר כָּךְ נוֹדַע שֶׁהוּא טָמֵא, 
הַצִּיץ מְרַצֶּה.  
נִטַּמָּא הַגּוּף, 
אֵין הַצִּיץ מְרַצֶּה; 
מִפְּנֵי שֶׁאָמְרוּ: 
נָזִיר וְעוֹשֵׂה פֶסַח, 
הַצִּיץ מְרַצֶּה עַל טֻמְאַת הַדָּם, 
וְאֵין הַצִּיץ מְרַצֶּה עַל טֻמְאַת הַגּוּף.  
נִטַּמָּא טֻמְאַת הַתְּהוֹם, 
הַצִּיץ מְרַצֶּה. 

<small>ח</small>
נִטַּמָּא שָׁלֵם אוֹ רֻבּוֹ, 
שׂוֹרְפִין אוֹתוֹ לִפְנֵי הַבִּירָה מֵעֲצֵי הַמַּעֲרָכָה.  
נִטַּמָּא מֵעוּטוֹ, וְהַנּוֹתָר, 
שׂוֹרְפִים אוֹתוֹ בְחַצְרוֹתֵיהֶם עַל גַּגּוֹתֵיהֶן, 
מֵעֲצֵי עַצְמָן.  
הַצִּיקָנִים שׂוֹרְפִין אוֹתוֹ לִפְנֵי הַבִּירָה, 
בִּשְׁבִיל לֵהָנוֹת מֵעֲצֵי הַמַּעֲרָכָה. 

<small>ט</small>
הַפֶּסַח שֶׁיָּצָא אוֹ שֶׁנִּטַּמָּא, 
יִשָּׂרֵף מִיָּד.  
נִטַּמּוּ הַבְּעָלִים אוֹ שֶׁמֵּתוּ, 
תְּעֻבַּר צוּרָתוֹ &lt;מחוק: וישרף בששה עשר&gt;
וְיֵצֵא לְבֵית הַשְּׂרִיפָה.  
רְבִּי יוֹחָנָן בֶּן בְּרוֹקָה אוֹמֵר: 
אַף זֶה יִשָּׂרֵף מִיָּד, 
שֶׁאֵין לוֹ אוֹכְלִים. 

<small>י</small>
הָעֲצָמוֹת וְהַגִּידִים וְהַנּוֹתָר יִשָּׂרְפוּ בְשִׁשָּׁה עָשָׂר.  
חָל שִׁשָּׁה עָשָׂר לִהְיוֹת בַּשַּׁבָּת, 
יִשָּׂרְפוּ בְשִׁבְעָה עָשָׂר, 
שֶׁאֵינָן דּוֹחִים לֹא אֶת הַשַּׁבָּת וְלֹא אֶת יוֹם טוֹב. 

<small>יא</small>
כָּל הַנֶּאֱכָל בְּשׁוֹר הַגָּדוֹל יֵאָכֵל בִּגְדִי הָרַךְ, 
וּבְרָאשֵׁי הַכְּנָפַיִם וְהַסְּחוּסִים.  
הַשּׁוֹבֵר אֶת הָעֶצֶם בַּפֶּסַח הַטָּהוֹר, 
הֲרֵי זֶה לוֹקֶה אַרְבָּעִים; 
אֲבָל הַמּוֹתִיר בַּטָּהוֹר, וְהַשּׁוֹבֵר בַּטָּמֵא, 
אֵינוּ לוֹקֶה אַרְבָּעִים. 

<small>יב</small>
אֵבֶר שֶׁיָּצָא מִקְצָתוֹ, 
חוֹתֵךְ עַד שֶׁמַּגִּיעַ לָעֶצֶם, 
וְקוֹלֵף עַד שֶׁמַּגִּיעַ לַפֶּרֶק, וְחוֹתֵךְ.  
וּבַמֻּקְדָּשִׁים, קוֹצֵץ בַּקּוֹפִיס, 
שֶׁאֵין בּוֹ שְׁבִירַת הָעֶצֶם.  
מִן הָאֶגֶף וְלִפְנִים כְּלִפְנִים, 
מִן הָאֶגֶף וְלַחוּץ כְּלַחוּץ.  
וְהַחַלּוֹנוֹת וְעֹבִי הַחוֹמָה כְלִפְנִים. 

<small>יג</small>
שְׁתֵּי חֲבוֹרוֹת שֶׁהָיוּ אוֹכְלוֹת בְּבַיִת אֶחָד, 
אֵלּוּ הוֹפְכִים אֶת פְּנֵיהֶן הֵילָךְ וְאוֹכְלִין, 
וְאֵלּוּ הוֹפְכִים אֶת פְּנֵיהֶם הֵילָךְ וְאוֹכְלִים, 
וְהַמֵּחַם בָּאֶמְצַע.  
וּכְשֶׁהַשַּׁמָּשׁ עוֹמֵד לִמְזוֹג, 
קוֹפֵּץ אֶת פִּיו וּמַחְזִיר אֶת פָּנָיו, 
עַד שֶׁהוּא מַגִּיעַ אֵצֶל חֲבוֹרָתוֹ וְאוֹכֵל.  
וְהַכַּלָּה הוֹפֶכֶת אֶת פָּנֶיהָ וְאוֹכֶלֶת. &lt;וְהַכָּלָה&gt;





Chapter 8

<small>א</small>
הָאִשָּׁה, בִּזְמַן שֶׁהִיא בְּבֵית בַּעְלָהּ, 
שָׁחַט עָלֶיהָ אָבִיהָ, שָׁחַט עָלֶיהָ בַעְלָהּ, 
תֹּאכַל מִשֶּׁלְּבַעְלָהּ.  
הָלְכָה רֶגֶל הָרִאשׁוֹן לַעֲשׁוֹת בְּבֵית אָבִיהָ, 
שָׁחַט עָלֶיהָ אָבִיהָ, שָׁחַט עָלֶיהָ בַעְלָהּ, 
תֹּאכַל מִמָּקוֹם שֶׁהִיא רוֹצָה.  
יָתוֹם שֶׁשָּׁחֲטוּ עָלָיו אַפִּטְרוֹפִּים, 
יֹאכַל מִמָּקוֹם שֶׁהוּא רוֹצֶה.  
עֶבֶד שֶׁלִּשְׁנֵי שׁוּתָפִים, 
לֹא יֹאכַל מִשֶּׁלִּשְׁנֵיהֶם.  
מִי שֶׁחֶצְיוֹ עֶבֶד וְחֶצְיוֹ בֶן חוֹרִים, 
לֹא יֹאכַל מִשֶּׁלְּרַבּוֹ. 

<small>ב</small>
הָאוֹמֵר לְעַבְדּוֹ: 
"צֵא וּשְׁחֹט עָלַי אֶת הַפֶּסַח", 
שָׁחַט גְּדִי, יֹאכַל, 
שָׁחַט טָלֶה, יֹאכַל, 
שָׁחַט גְּדִי וְטָלֶה, יֹאכַל מִן הָרִאשׁוֹן.  
שָׁכַח מָה אָמַר לוֹ רַבּוֹ, 
כֵּיצַד יַעֲשֶׂה? 
יִשְׁחֹט גְּדִי וְטָלֶה וְיֹאמַר: 
אִם גְּדִי אָמַר לִי רְבִּי, 
גְּדִי שֶׁלּוֹ וְטָלֶה שֶׁלִּי; 
וְאִם טָלֶה אָמַר לִי רְבִּי, 
טָלֶה שֶׁלּוֹ וּגְדִי שֶׁלִּי.  
שָׁכַח רַבּוֹ מָה אָמַר לוֹ, 
שְׁנֵיהֶם יֵצְאוּ לְבֵית הַשְּׂרִיפָה; 
וּפְטוּרִים מִלַּעֲשׁוֹת פֶּסַח שֵׁנִי. 

<small>ג</small>
הָאוֹמֵר לְבָנָיו: 
"הֲרֵי אֲנִי שׁוֹחֵט אֶת הַפֶּסַח 
לְמִי שֶׁיַּעֲלֶה מִכֶּם רִאשׁוֹן לִירוּשָׁלַיִם", 
כֵּיוָן שֶׁהִכְנִיס הָרִאשׁוֹן רֹאשׁוֹ וְרֻבּוֹ לָעִיר, 
זָכָה בְחֶלְקוֹ וְזִכָּה אֶת אֶחָיו עִמּוֹ.  
לְעוֹלָם נִמְנִים עָלָיו, 
עַד שֶׁיְּהֵא בוֹ כַזַּיִת לְכָל אֶחָד וְאֶחָד.  
נִמְנִים וּמוֹשְׁכִים אֶת יְדֵיהֶם מִמֶּנּוּ, 
עַד שֶׁיִּשְׁחֹט.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
עַד שֶׁיִּזָרֵק הַדָּם. 

<small>ד</small>
הַמַּמְנֶה אֲחֵרִים עִמּוֹ עַל חֶלְקוֹ, 
רַשָּׁיִים לִתֵּן לוֹ אֶת שֶׁלּוֹ, 
הוּא אוֹכֵל מִשֶּׁלּוֹ וְהֵן אוֹכְלִין מִשֶּׁלָּהֶם. 

<small>ה</small>
זָב שֶׁרָאָה שְׁתֵּי רְאִיּוֹת, 
שׁוֹחֲטִים עָלָיו בַּשְּׁבִיעִי.  
רָאָה שָׁלוֹשׁ, 
שׁוֹחֲטִין עָלָיו בַּשְּׁמִינִי.  
וְכֵן שׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם, 
שׁוֹחֲטִין עָלֶיהָ בַשֵּׁנִי.  
רָאָת שְׁנֵי יָמִים, 
שׁוֹחֲטִים עָלֶיהָ בַשְּׁלִישִׁי.  
וְהַזָּבָה, 
שׁוֹחֲטִין עָלֶיהָ בַשְּׁמִינִי. 

<small>ז</small>
הָאוֹנֵן וְהַמְפַקֵּחַ בַּגַּל, 
וְכֵן מִי שֶׁהִבְטִיחוּהוּ לְהוֹצִיאוֹ מִבֵּית הָאֲסוּרִים, 
הַחוֹלֶה וְהַזָּקֵן שֶׁהֵן יְכוּלִין לֹאכַל כַּזַּיִת, 
שׁוֹחֲטִין עֲלֵיהֶם.  
וְעַל כֻּלָּם, 
אֵין שׁוֹחֲטִים עֲלֵיהֶם בִּפְנֵי עַצְמָן, 
שֶׁלֹּא יְבִיאוּהוּ לִידֵי פָּסוּל.  
לְפִיכָךְ, אִם אֵרַע בָּהֶם פְּסוּל, 
פְּטוּרִים מִלַּעֲשׁוֹת פֶּסַח שֵׁנִי, 
חוּץ מִן הַמְפַקֵּחַ בַּגַּל, 
שֶׁהָיָה טָמֵא מִתְּחִלָּתוֹ. 

(בכתב יד קאופמן, סדורה לפני משנה ו, 
ומעתקת שוב בשוליים לפני משנה ח)
<small>ו</small>
אֵין שׁוֹחֲטִין אֶת הַפֶּסַח עַל הַיָּחִיד.  
דִּבְרֵי רְבִּי יְהוּדָה.  
רְבִּי יוֹסֵה מַתִּיר.  
אֲפִלּוּ חֲבוֹרָה שֶׁלְּמֵאָה וְאֵינָן יְכוּלִין לֹאכַל כַּזַּיִת, 
אֵין שׁוֹחֲטִים עֲלֵיהֶם,  
וְאֵין עוֹשִׂין חֲבוֹרַת נָשִׁים וַעֲבָדִים וּקְטַנִּים. 

<small>ח</small>
הָאוֹנֵן טוֹבֵל וְאוֹכֵל אֶת פִּסְחוֹ לָעֶרֶב, 
אֲבָל לֹא בַקָּדָשִׁים.  
הַשּׁוֹמֵעַ עַל מֵתוֹ, 
וְהַמְלַקֵּט עֲצָמוֹת, 
טוֹבֵל וְאוֹכֵל בַּקָּדָשִׁים.  
גֵּר שֶׁנִּתְגַּיַּר עֶרֶב פְּסָחִים, 
בֵּית שַׁמַּי אוֹמְרִים: 
טוֹבֵל וְאוֹכֵל אֶת פִּסְחוֹ לָעֶרֶב; 
וּבֵית הֶלֵּל אוֹמְרִים: 
הַפּוֹרֵשׁ מִן הָעָרְלָה כְּפוֹרֵשׁ מִן הַקֶּבֶר. 





Chapter 9

<small>א</small>
מִי שֶׁהָיָה טָמֵא אוֹ בְדֶרֶךְ רְחוֹקָה, 
וְלֹא עָשָׂה אֶת הָרִאשׁוֹן, 
יַעֲשֶׂה אֶת הַשֵּׁנִי.  
שָׁגַג אוֹ נֶאֱנַס וְלֹא עָשָׂה אֶת הָרִאשׁוֹן, 
יַעֲשֶׂה אֶת הַשֵּׁנִי.  
אִם כֵּן, לָמָּה נֶאֱמַר: 
"טָמֵא אוֹ בְדֶרֶךְ רְחוֹקָה"? 
שֶׁאֵלּוּ פְטוּרִין מִן הַהֶכָּרֵת, 
וְאֵלּוּ חַיָּבִים בַּהֶכָּרֵת. 

<small>ב</small>
וְאֵי זוֹ הִיא דֶּרֶךְ רְחוֹקָה? 
מִן הַמּוֹדִיעִית וְלַחוּץ, 
וּכְמִדָּתָהּ לְכָל רוּחַ.  
דִּבְרֵי רְבִּי עֲקִיבָה.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
מֵאִסְקֻפַּת הָעֲזָרָה וְלַחוּץ.  
אָמַר רְבִּי יוֹסֵה: 
לְפִיכָךְ נָקוּד עַל הֵ, לוֹמַר: 
לֹא מִפְּנֵי שֶׁהִיא רְחוֹקָה וַדַּי, 
אֶלָּא מֵאִסְקֻפַּת הָעֲזָרָה וְלַחוּץ. 

<small>ג</small>
מַה בֵּין הָרִאשׁוֹן לַשֵּׁנִי? 
הָרִאשׁוֹן אָסוּר בְּבַל יֵרָאֶה וּבַל יִמָּצֵא; 
הַשֵּׁנִי, חָמֵץ וּמַצָּה עִמּוֹ בַבַּיִת.  
הָרִאשׁוֹן טָעוּן הַלֵּל בַּאֲכִילָתוֹ, 
וְהַשֵּׁנִי אֵינוּ טָעוּן הַלֵּל בַּאֲכִילָתוֹ.  
וְזֶה וָזֶה טְעוּנִים הַלֵּל בַּעֲשִׂיָּתָן, 
נֶאֱכָלִים צָלִי עַל מַצָּה וּמְרוֹרִים, 
וְדוֹחִים אֶת הַשַּׁבָּת. 

<small>ד</small>
הַפֶּסַח שֶׁבָּא בְטֻמְאָה, 
לֹא יֹאכְלוּ מִמֶּנּוּ זָבִים וְזָבוֹת, נִדּוֹת וְיוֹלְדוֹת; 
אִם אָכָלוּ, פְּטוּרִים.  
רְבִּי אֱלִיעֶזֶר פּוֹטֵר אַף עַל בִּיאַת הַמִּקְדָּשׁ. 

<small>ה</small>
מַה בֵּין פֶּסַח מִצְרַיִם לְפֶסַח דּוֹרוֹת? 
פֶּסַח מִצְרַיִם מַקָּחוֹ מִבֶּעָשׁוֹר, 
וְטָעוּן הַזָּיָה וַאֲגֻדַּת אֵזוֹב 
עַל הַמַּשְׁקוֹף וְעַל שְׁתֵּי הַמְּזוּזוֹת, 
וְנֶאֱכָל בְּחִפָּזוֹן לַיְלָה אֶחָד. 
וּפֶסַח דּוֹרוֹת נוֹהֵג כָּל שִׁבְעָה. 

<small>ו</small>
אָמַר רְבִּי יְהוֹשֻׁעַ: 
שָׁמַעְתִּי שֶׁתְּמוּרַת הַפֶּסַח קְרֵבָה, 
וּתְמוּרַת הַפֶּסַח אֵינָה קְרֵבָה, 
וְאֵין לִי לְפָרֵשׁ.  
אָמַר רְבִּי עֲקִיבָה: 
אֲנִי אֲפָרֵשׁ: 
הַפֶּסַח שֶׁנִּמְצָא קֹדֶם לִשְׁחִיטַת הַפֶּסַח, 
יִרְעֶה עַד שֶׁיִּסְתָּאֵב, וְיִמָּכֵר, 
וְיָבִיא בְדָמָיו שְׁלָמִים, וְכֵן תְּמוּרָתוֹ; 
לְאַחַר הַפֶּסַח, יָבִיא שְׁלָמִים, וְכֵן תְּמוּרָתוֹ. 

<small>ז</small>
הַמַּפְרִישׁ נְקֵבָה לְפִסְחוֹ, 
אוֹ זָכָר בֶּן שְׁתֵּי שָׁנִים, 
יִרְעֶה עַד שֶׁיִּסְתָּאֵב, וְיִמָּכֵר, 
וְיִפְּלוּ דָמָיו לִנְדָבָה.  
הַמַּפְרִישׁ פִּסְחוֹ וּמֵת, 
לֹא יְבִיאֶנּוּ בְנוֹ אַחֲרָיו לְשֵׁם פֶּסַח, 
אֶלָּא לְשֵׁם שְׁלָמִים. 

<small>ח</small>
הַפֶּסַח שֶׁנִּתְעָרַב בַּזְּבָחִים, 
יִרְעוּ עַד שֶׁיִּסְתָּאֵבוּ, וְיִמָּכְרוּ, 
וְיָבִיא בִדְמֵי הַיָּפֶה שֶׁבָּהֶן מִמִּין זֶה, 
וּבִדְמֵי הַיָּפֶה שֶׁבָּהֶן מִמִּין זֶה, 
וְיַפְסִיד הַמּוֹתָר מִבֵּיתוֹ.  
נִתְעָרַב בִּבְכוֹרוֹת, 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אִם חֲבוֹרַת הַכֹּהֲנִים, 
יֹאכֵלוּ. 

<small>ט</small>
חֲבוֹרָה שֶׁאָבַד פִּסְחָהּ, 
אָמְרוּ לְאֶחָד: 
"צֵא וּבַקֵּשׁ, וּשְׁחֹט עָלֵינוּ!" 
וְהָלַךְ וּמָצָא וְשָׁחַט, 
וְהֵם לָקְחוּ וְשָׁחָטוּ; 
אִם שֶׁלּוֹ נִשְׁחַט רִאשׁוֹן, 
הוּא אוֹכֵל מִשֶּׁלּוֹ, וְהֵן אוֹכְלִין עִמּוֹ; 
וְאִם שֶׁלָּהֶם נִשְׁחַט רִאשׁוֹן, 
הוּא אוֹכֵל מִשֶּׁלּוֹ וְהֵן אוֹכְלִין מִשֶּׁלָּהֶם; 
וְאִם אֵין יָדוּעַ אֵי זֶה מֵהֶן נִשְׁחַט רִאשׁוֹן, 
אוֹ שֶׁשָּׁחֲטוּ שְׁנֵיהֶם כְּאַחַת, 
הוּא אוֹכֵל מִשֶּׁלּוֹ, וְהֵם אֵינָן אוֹכְלִים עִמּוֹ, 
וְשֶׁלָּהֶם יֵצֵא לְבֵית הַשְּׂרִיפָה, 
וּפְטוּרִים מִלַּעֲשׁוֹת פֶּסַח שֵׁנִי.  

<small>י</small>
אָמַר לָהֶם: 
"אִם אֵחַרְתִּי, שַׁחֲטוּ עָלַי!" 
הָלַךְ וּמָצָא וְשָׁחַט, 
וְהֵן לָקְחוּ וְשָׁחָטוּ; 
אִם שֶׁלָּהֶם נִשְׁחַט רִאשׁוֹן, 
הֵן אוֹכְלִין מִשֶּׁלָּהֶם, וְהוּא אוֹכֵל עִמָּהֶן; 
אִם שֶׁלּוֹ נִשְׁחַט רִאשׁוֹן, 
הוּא אוֹכֵל מִשֶּׁלּוֹ וְהֵן אוֹכְלִים מִשֶּׁלָּהֶן; 
אֵין יָדוּעַ אֵי זֶה מֵהֶן נִשְׁחַט רִאשׁוֹן, 
אוֹ שֶׁשָּׁחֲטוּ שְׁנֵיהֶם כְּאַחַת, 
הֵם אוֹכְלִים מִשֶּׁלָּהֶם, וְהוּא אֵינוּ אוֹכֵל עִמָּהֶם, 
וְשֶׁלּוֹ יֵצֵא לְבֵית הַשְּׂרִיפָה, 
וּפָטוּר מִלַּעֲשׁוֹת פֶּסַח שֵׁנִי.  

<small>יא</small>
אָמַר לָהֶם וְאָמְרוּ לוֹ, 
אוֹכְלִים מִן הָרִאשׁוֹן; 
לֹא אָמַר לָהֶם וְלֹא אָמְרוּ לוֹ, 
אֵינָן אַחְרָיִים זֶה בָזֶה. 

<small>יב</small>
שְׁתֵּי חֲבוֹרוֹת שֶׁנִּתְעָרְבוּ פִסְחֵיהֶם, 
אֵלּוּ מוֹשְׁכִים לָהֶם אֶחָד, 
וְאֵלּוּ מוֹשְׁכִים לָהֶם אֶחָד; 
אֶחָד מֵאֵלּוּ בָא לוֹ אֵצֶל אֵלּוּ, 
וְאֶחָד מֵאֵלּוּ בָא לוֹ אֵצֶל אֵלּוּ; 
וְכָךְ הֵם אוֹמְרִים: 
אִם שֶׁלָּנוּ הוּא הַפֶּסַח הַזֶּה, 
יָדֶיךָ מְשׁוּכוֹת מִשֶּׁלָּךְ וְנִמְנֵיתָ עַל שֶׁלָּנוּ; 
וְאִם שֶׁלָּךְ הוּא הַפֶּסַח הַזֶּה, 
יָדֵינוּ מְשׁוּכוֹת מִשֶּׁלָּנוּ וְנִמְנִינוּ עַל שֶׁלָּךְ.  

<small>יג</small>
וְכֵן חָמֵשׁ חֲבוֹרוֹת שֶׁלַּחֲמִשָּׁה חֲמִשָּׁה, 
וְשֶׁלַּעֲשָׂרָה עֲשָׂרָה, 
נִמְשָׁכִים לָהֶם אֶחָד מִכָּל חֲבוֹרָה וַחֲבוֹרָה, 
וְכָךְ הֵם אוֹמְרִים. 

שְׁנַיִם שֶׁנִּתְעָרְבוּ פִסְחֵיהֶם, 
זֶה מוֹשֵׁךְ לוֹ אֶחָד וְזֶה מוֹשֵׁךְ לוֹ אֶחָד; 
זֶה מַמְנֶה עִמּוֹ אֶחָד מֵהַשּׁוּק, 
וְזֶה מַמְנֶה עִמּוֹ אֶחָד מֵהַשּׁוּק; 
זֶה בָא אֵצֶל זֶה וְזֶה בָא אֵצֶל זֶה, 
וְכָךְ הוּא אוֹמֵר: 
אִם שֶׁלִּי הוּא הַפֶּסַח הַזֶּה, 
יָדֶיךָ מְשׁוּכוֹת מִשֶּׁלָּךְ, וְנִמְנֵיתָ עַל שֶׁלִּי; 
וְאִם שֶׁלָּךְ הוּא הַפֶּסַח הַזֶּה, 
יָדַי מְשׁוּכוֹת מִשֶּׁלִּי וְנִמְנֵיתִי עַל שֶׁלָּךְ. 





Chapter 10

<small>א</small>
עֶרֶב פְסָחִים סָמוּךְ לַמִּנְחָה 
לֹא יֹאכַל אָדָם עַד שֶׁתֶּחְשַׁךְ.  
אֲפִילוּ עָנִי שֶׁבְּיִשְׂרָאֵל לֹא יֹאכַל עַד שֶׁיֵּסֵב.  
לֹא יִפְחֲתוּ לוֹ מֵאַרְבָּעָה כוֹסוֹת שֶׁלַּיַּיִן, 
אֲפִלּוּ מִן הַתַּמְחוּי. 

<small>ב</small>
מָזְגוּ לוֹ כוֹס רִאשׁוֹן, 
בֵּית שַׁמַּי אוֹמְרִים: 
מְבָרֵךְ עַל הַיּוֹם, 
וְאַחַר כָּךְ מְבָרֵךְ עַל הַיַּיִן; 
וּבֵית הֶלֵּל אוֹמְרִים: 
מְבָרֵךְ עַל הַיַּיִן, 
וְאַחַר כָּךְ מְבָרֵךְ עַל הַיּוֹם. 

<small>ג</small>
הֵבִיאוּ לְפָנָיו, 
מְטַבֵּל בַּחֲזֶרֶת עַד שֶׁהוּא מַגִּיעַ לְפַרְפֶּרֶת הַפַּת.  
הֵבִיאוּ לְפָנָיו מַצָּה וַחֲזֶרֶת וַחֲרֹסֶת, 
אַף עַל פִּי שֶׁאֵין חֲרֹסֶת מִצְוָה.  
רְבִּי לְעָזָר בִּרְבִּי צָדוֹק אוֹמֵר: 
מִצְוָה.  
וּבַמִּקְדָּשׁ מְבִיאִים לְפָנָיו גּוּפוֹ שֶׁלַּפֶּסַח. 

<small>ד</small>
מָזְגוּ לוֹ כוֹס שֵׁנִי, 
וְכֵן הַבֵּן שׁוֹאֵל.  
אִם אֵין דַּעַת בַּבֵּן, 
אָבִיו מְלַמְּדוֹ: 
"מַה נִּשְׁתַּנָּה הַלַּיְלָה הַזֶּה מִכָּל הַלֵּילוֹת? 
שֶׁבְּכָל הַלֵּילוֹת אֵין אָנוּ מְטַבְּלִים אֲפִלּוּ פַעַם אַחַת, 
הַלַּיְלָה הַזֶּה שְׁתֵּי פְעָמִים;
שֶׁבְּכָל הַלֵּילוֹת אָנוּ אוֹכְלִין חָמֵץ וּמַצָּה, 
הַלַּיְלָה הַזֶּה כֻלּוֹ מַצָּה; 
שֶׁבְּכָל הַלֵּילוֹת אָנוּ אוֹכְלִים בָּשָׂר צָלִי שָׁלוּק וּמְבֻשָּׁל, 
הַלַּיְלָה הַזֶּה כֻלּוֹ צָלִי."
לְפִי דַעְתּוֹ שֶׁלַּבֵּן אָבִיו מְלַמְּדוֹ.  
מַתְחִיל בִּגְנוּת וּמְסַיֵּם בִּשְׁבָח, 
וְדוֹרֵשׁ מֵ "אֲרַמִּי אֹבֵד אָבִי" (דברים כו,ה), 
עַד שֶׁהוּא גוֹמֵר כָּל הַפָּרָשָׁה. 

<small>ה</small>
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
כָּל שֶׁלֹּא אָמַר שְׁלוֹשָׁה דְבָרִים אֵלּוּ בַפֶּסַח, 
לֹא יָצָא יְדֵי חוֹבָתוֹ: 
"פֶּסַח, מַצָּה וּמְרוֹרִים.  
פֶּסַח, 
עַל שֵׁם שֶׁפָּסַח הַמָּקוֹם עַל בָּתֵּי אֲבוֹתֵינוּ בְמִצְרַיִם. 
מְרוֹרִים, 
עַל שֵׁם שֶׁמֵּרְרוּ הַמִּצְרִיִּים אֶת חַיֵּי אֲבוֹתֵינוּ בְמִצְרַיִם.  
מַצָּה, 
עַל שֵׁם שֶׁנִּגְאָלוּ.  
לְפִיכָךְ אָנוּ חַיָּבִים לְהוֹדוֹת, 
לְהַלֵּל, לְשַׁבֵּח, לְפָאֵר, לְרוֹמֵם, לְגַדֵּל 
לְמִי שֶׁעָשָׂה לָנוּ וְלַאֲבוֹתֵינוּ אֶת כָּל הַנִּסִּים הָאֵלּוּ, 
וְהוֹצִיאָנוּ מֵעַבְדוּת לְחֵרוּת, 
וְנֹאמַר לְפָנָיו הַלְלוּיָהּ." 

<small>ו</small>
עַד אֵיכָן הוּא אוֹמֵר? 
בֵּית שַׁמַּי אוֹמְרִים: 
עַד "אֵם הַבָּנִים שְׂמֵחָה", 
וּבֵית הֶלֵּל, 
עַד "חַלָּמִישׁ לְמַעְיְנוֹ מָיִם".  
וְחוֹתֵם בִּגְאֻלָּה.  

<small>ז</small>
רְבִּי טַרְפוֹן אוֹמֵר: 
"אֲשֶׁר גְּאָלָנוּ וְגָאַל אֶת אֲבוֹתֵינוּ מִמִּצְרַיִם, 
וְהִגִּיעָנוּ הַלַּיְלָה הַזֶּה",
וְאֵינוּ חוֹתֵם.  
רְבִּי עֲקִיבָה אוֹמֵר: 
"כֵּן יי אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ, 
יַגִּיעֵנוּ לִָרְגָלִים הַבָּאִים לִקְרָאתֵנוּ לְשָׁלוֹם, 
שְׂמֵחִים בְּבִנְיַן עוֹלָם, 
לֹאכַל מִן הַפְּסָחִים וּמִן הַזְּבָחִים 
אֲשֶׁר יַגִּיעַ דָּמָם עַל קִיר מִזְבְּחָךְ לְרָצוֹן, 
וְנוֹדֶה לָךְ עַל גְּאֻלָּתֵנוּ.  
בָּרוּךְ אַתָּה יי, גָּאַל יִשְׂרָאֵל". &lt;גואל&gt;

<small>ח</small>
מָזְגוּ לוֹ כוֹס שְׁלִישִׁי, 
וּבֵרַךְ עַל מְזוֹנוֹ. 
רְבִיעִי, גּוֹמֵר אֶת הַהַלֵּל, 
וְאוֹמֵר עָלָיו בִּרְכַּת הַשִּׁיר.  
בֵּין הַכּוֹסוֹת הָאֵלּוּ, 
אִם רָצָה לִשְׁתּוֹת, יִשְׁתֶּה. 
בֵּין הַשְּׁלִישִׁי לָרְבִיעִי, לֹא יִשְׁתֶּה. 

וְאֵין מַפְטִירִים אַחַר הַפֶּסַח אֲפִיקִימוֹן.  
יָשְׁנוּ, מִקְצָתָם יֹאכֵלוּ, וְכֻלָּם לֹא יֹאכֵלוּ.  
רְבִּי יוֹסֵה אוֹמֵר: 
אִם נִתְנַמְנְמוּ, יֹאכֵלוּ, 
וְאִם נִרְדְּמוּ, לֹא יֹאכֵלוּ. 

<small>ט</small>
הַפֶּסַח אַחַר חֲצוֹת מְטַמֵּא אֶת הַיָּדַיִם.  
הַפִּגּוּל וְהַנּוֹתָר מְטַמְּאִים אֶת הַיָּדַיִם.  
בֵּרַךְ בִּרְכַּת הַפֶּסַח, פָּטַר אֶת שֶׁלָּזֶבַח, 
אֶת שֶׁלָּזֶבַח, לֹא פָטַר אֶת שֶׁלַּפֶּסַח.  
דִּבְרֵי רְבִּי יִשְׁמָעֵאל.  
רְבִּי עֲקִיבָה אוֹמֵר: 
לֹא זוֹ פּוֹטֶרֶת זוֹ, 
וְלֹא זוֹ פּוֹטֶרֶת זוֹ. 

חסל פסח
