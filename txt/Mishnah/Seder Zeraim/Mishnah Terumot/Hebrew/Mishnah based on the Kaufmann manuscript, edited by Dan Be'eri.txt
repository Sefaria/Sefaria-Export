משנה תרומות



Chapter 1

<small>א</small>
חֲמִשָּׁה לֹא יִתְרֹמוּ, 
וְאִם תָּרָמוּ, אֵין תְּרוּמָתָן תְּרוּמָה: 
הַחֵרֵשׁ וְהַשּׁוֹטֶה וְהַקָּטָן, 
וְהַתּוֹרֵם אֶת שֶׁאֵינוּ שֶׁלּוֹ,  
נָכְרִי שֶׁתָּרַם אֶת שֶׁלְּיִשְׂרָאֵל אֲפִלּוּ בִרְשׁוּת, 
אֵין תְּרוּמָתוֹ תְרוּמָה. 

<small>ב</small>
חֵרֵשׁ הַמְדַבֵּר וְאֵינוּ שׁוֹמֵעַ לֹא יִתְרֹם. 
אִם תָּרַם, תְּרוּמָתוֹ תְרוּמָה.  
חֵרֵשׁ שֶׁדִּבְּרוּ חֲכָמִים בְּכָל מָקוֹם, 
שֶׁאֵינוּ לֹא שׁוֹמֵעַ וְלֹא מְדַבֵּר. 

<small>ג</small>
קָטָן שֶׁלֹּא הֵבִיא שְׁתֵּי שְׂעָרוֹת, 
רְבִּי יְהוּדָה אוֹמֵר: 
תְּרוּמָתוֹ תְרוּמָה.  
רְבִּי יוֹסֵה אוֹמֵר: 
אִם עַד שֶׁלֹּא בָא לְעוֹנַת נְדָרִים, 
אֵין תְּרוּמָתוֹ תְרוּמָה; 
אִם מִשֶּׁבָּא לְעוֹנַת נְדָרִים, 
תְּרוּמָתוֹ תְרוּמָה. 

<small>ד</small>
אֵין תּוֹרְמִין זֵיתִין עַל שֶׁמֶן, 
וְלֹא עֲנָבִים עַל יַיִן.  
וְאִם תָּרַם, 
בֵּית שַׁמַּי אוֹמְרִים: 
תְּרוּמַת עַצְמָן בָּהֶן. 
וּבֵית הֶלֵּל אוֹמְרִים: 
אֵין תְּרוּמָתָן תְּרוּמָה. 

<small>ה</small>
אֵין תּוֹרְמִין מִן הַלֶּקֶט, מִן הַשִּׁכְחָה וּמִן הַפֵּאָה, 
וּמִן הַהֶבְקֵר, 
וְלֹא מִמַּעֲשֵׂר רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְרוּמָתוֹ, 
וְלֹא מִמַּעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁלֹּא נִפְדּוּ, 
וְלֹא מִן הַחַיָּב עַל הַפָּטוּר, 
וְלֹא מִן הַפָּטוּר עַל הַחַיָּב, 
וְלֹא מִן הַתָּלוּשׁ עַל הַמְחֻבָּר, 
וְלֹא מִן הַמְחֻבָּר עַל הַתָּלוּשׁ, 
וְלֹא מִן הֶחָדָשׁ עַל הַיָּשָׁן, 
וְלֹא מִן הַיָּשָׁן עַל הֶחָדָשׁ, 
וְלֹא מִפֵּרוֹת הָאָרֶץ עַל פֵּרוֹת חוּצָה לָאָרֶץ, 
וְלֹא מִפֵּרוֹת חוּצָה לָאָרֶץ עַל פֵּרוֹת הָאָרֶץ; 
וְאִם תָּרָמוּ, אֵין תְּרוּמָתָן תְּרוּמָה. &lt;ואם תרם, אין תרומתו&gt;

<small>ו</small>
חֲמִשָּׁה לֹא יִתְרֹמוּ, 
וְאִם תָּרְמוּ, תְּרוּמָתָן תְּרוּמָה: 
הָאִלֵּם, וְהַשִּׁכּוֹר, וְהֶעָרֹם, 
וְהַסּוֹמֶה, וּבַעַל קֶרִי, 
לֹא יִתְרֹמוּ, 
וְאִם תָּרָמוּ, תְּרוּמָתָן תְּרוּמָה. 

<small>ז</small>
אֵין תּוֹרְמִין לֹא בַמִּדָּה וְלֹא בַמִּשְׁקָל וְלֹא בַמִּנְיָן, 
אֲבָל תּוֹרֵם הוּא אֶת הַמָּדוּד וְאֶת הַשָּׁקוּל וְאֶת הַמָּנוּי.  
אֵין תּוֹרְמִין בַּסַּל וּבַקֻּפָּה שֶׁהֵן שֶׁלַמִּדָּה, 
אֲבָל תּוֹרֵם הוּא בָהֶן חֶצְיָן אוֹ שְׁלִישָׁן.  
לֹא יִתְרֹם בִּסְאָה חֶצְיָהּ, 
שֶׁחֶצְיָהּ מִדָּה. 

<small>ח</small>
אֵין תּוֹרְמִין שֶׁמֶן עַל זֵיתִים הַנִּכְתָּשִׁים, 
וְלֹא יַיִן עַל עֲנָבִים הַנִּדְרָכוֹת. 
וְאִם תָּרַם תְרוּמָה, יַחְזֹר וְיִתְרֹם.  
הָרִאשׁוֹנָה מְדַמַּעַת בִּפְנֵי עַצְמָהּ, 
וְחַיָּבִין עָלֶיהָ חֹמֶשׁ, 
אֲבָל לֹא שְׁנִיָּה. 

<small>ט</small>
וְתוֹרְמִין שֶׁמֶן עַל זֵיתִים הַנִּכְבָּשִׁין, 
וְיַיִן עַל עֲנָבִים לַעֲשׁוֹתָן צִמּוּקִין.  
מִי שֶׁתָּרַם שֶׁמֶן עַל זֵיתִים לַאֲכִילָה, 
וְזֵיתִים עַל זֵיתִים לַאֲכִילָה, 
וְיַיִן עַל עֲנָבִים לַאֲכִילָה, 
וַעֲנָבִים עַל עֲנָבִים לַאֲכִילָה, 
וְנִמְלַךְ לְדָרְכָן, 
אֵינוּ צָרִיךְ לִתְרֹם. 

<small>י</small>
אֵין תּוֹרְמִין מִדָּבָר שֶׁנִּגְמְרָה מְלַאכְתּוֹ 
עַל דָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ, 
וְלֹא מִדָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ 
עַל דָּבָר שֶׁנִּגְמְרָה מְלַאכְתּוֹ, 
וְלֹא מִדָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ 
עַל דָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ; 
וְאִם תָּרַם, 
תְּרוּמָתוֹ תְּרוּמָה. 


 


Chapter 2

<small>א</small>
אֵין תּוֹרְמִין מִן הַטָּהוֹר עַל הַטָּמֵא; 
וְאִם תָּרָמוּ, תְּרוּמָתָן תְּרוּמָה.  
בֶּאֱמֶת, הָעִגּוּל שֶׁלִּדְבֵלָה שֶׁנִּטַּמָּא מִקְצָתוֹ, 
תּוֹרֵם מִן הַטָּהוֹר שֶׁיֶּשׁ בּוֹ עַל הַטָּמֵא שֶׁיֶּשׁ בּוֹ.  
וְכֵן אֲגֻדָּה שֶׁלַּיָּרָק, וְכֵן עֲרֵמָה.  
הָיוּ שְׁנֵי עִגּוּלִין, שְׁתֵּי אֲגֻדּוֹת, שְׁתֵּי עֲרֵמוֹת, 
אַחַת טְמֵאָה וְאַחַת טְהוֹרָה, 
לֹא יִתְרֹם מִזֶּה עַל זֶה.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
תּוֹרְמִין מִן הַטָּהוֹר עַל הַטָּמֵא. 

<small>ב</small>
אֵין תּוֹרְמִין מִן הַטָּמֵא עַל הַטָּהוֹר. 
וְאִם תָּרַם, 
שׁוֹגֵג, תְּרוּמָתוֹ תְרוּמָה, 
מֵזִיד, לֹא עָשָׂה כְלוּם.  
וְכֵן בֶּן לֵוִי שֶׁהָיָה לוֹ מַעְשֵׂר טֶבֶל, 
הָיָה מַפְרִישׁ עָלָיו וְהוֹלֵךְ, 
שׁוֹגֵג, מַה שֶּׁעָשָׂה עָשׁוּי, 
מֵזִיד, לֹא עָשָׂה כְלוּם.  
רְבִּי יְהוּדָה אוֹמֵר: 
אִם הָיָה יוֹדֵעַ בּוֹ מִתְּחִלָּה, 
אַף עַל פִּי שׁוֹגֵג, 
לֹא עָשָׂה כְלוּם. 

<small>ג</small>
הַמַּטְבִּיל כֵּלִים בַּשַּׁבָּת, 
שׁוֹגֵג, יִשְׁתַּמֵּשׁ בָּהֶן, 
מֵזִיד, לֹא יִשְׁתַּמֵּשׁ בָּהֶן.  
הַמְעַשֵּׂר וְהַמְבַשֵּׁל בַּשַּׁבָּת, 
שׁוֹגֵג, יֹאכַל, 
מֵזִיד, לֹא יֹאכַל.  
הַנּוֹטֵעַ בַּשַּׁבָּת, 
שׁוֹגֵג, יְקַיֵּם, 
מֵזִיד, יְעַקֵּר.  
וּבַשְּׁבִיעִית, 
בֵּין שׁוֹגֵג בֵּין מֵזִיד, יַעֲקֹר. 

<small>ד</small>
אֵין תּוֹרְמִין מִמִּין עַל שֶׁאֵינוּ מִינוֹ. 
וְאִם תָּרַם, אֵין תְּרוּמָתוֹ תְרוּמָה.  
כָּל מִין חִטִּים אֶחָד.  
כָּל מִין תְּאֵנִים אֶחָד.  
כָּל מִין תְּאֵנִים וּגְרוֹגְרוֹת וּדְבֵלָה אֶחָד, 
וְתוֹרֵם מִזֶּה עַל זֶה.  
כָּל מָקוֹם שֶׁיֵּשׁ כֹּהֵן, 
תּוֹרֵם מִן הַיָּפֶה. 
כָּל מָקוֹם שֶׁאֵינוּ כֹהֵן, 
תּוֹרֵם מִן הַמִּתְקַיֵּם.  
רְבִּי יְהוּדָה אוֹמֵר: 
לְעוֹלָם הוּא תּוֹרֵם מִן הַיָּפֶה. 

<small>ה</small>
תּוֹרְמִים בָּצָל קָטָן שָׁלֵם, 
אֲבָל לֹא חֲצִי בָצָל גָּדוֹל.  
רְבִּי יְהוּדָה אוֹמֵר: 
לֹא כִי, אֶלָּא חֲצִי בָצָל גָּדוֹל.  
וְכֵן הָיָה רְבִּי יְהוּדָה אוֹמֵר: 
תּוֹרְמִים בְּצָלִים בְּנֵי מְדִינָה עַל הַכֻּפְרִים, 
אֲבָל לֹא כֻּפְרִים עַל בְּנֵי מְדִינָה, 
מִפְּנֵי שֶׁהוּא מַאֲכַל פּוֹלִיטִיקִין. 

<small>ו</small>
וְתוֹרְמִין זֵיתֵי שֶׁמֶן עַל זֵיתֵי כֶבֶשׁ, 
וְלֹא זֵיתֵי כֶבֶשׁ עַל זֵיתֵי שֶׁמֶן. 
וְיַיִן שֶׁאֵינוּ מְבֻשָּׁל עַל הַמְבֻשָּׁל, 
אֲבָל לֹא מְבֻשָּׁל עַל שֶׁאֵינוּ מְבֻשָּׁל.  
זֶה הַכְּלָל: 
כָּל שֶׁהוּא כִלְאַיִם בַּחֲבֵרוֹ, 
לֹא יִתְרֹם מִזֶּה עַל זֶה, 
אֲפִלּוּ מִן הַיָּפֶה עַל הָרָע, 
וְעַל שֶׁאֵינוּ כִלְאַיִם בַּחֲבֵרוֹ, 
תּוֹרֵם מִן הַיָּפֶה עַל הָרָע, 
אֲבָל לֹא מִן הָרָע עַל הַיָּפֶה.  
וְאִם תָּרַם מִן הָרָע עַל הַיָּפֶה, 
תְּרוּמָתוֹ תְרוּמָה, 
חוּץ מִן הַזּוֹנִים עַל הַחִטִּים, 
שֶׁאֵינָן אֹכֶל.  
וְהַקִּשּׁוּת וְהַמַּלְפְּפוֹנוֹת, מִין אֶחָד.  
רְבִּי יְהוּדָה אוֹמֵר: 
שְׁנֵי מִינִים. 


 


Chapter 3

<small>א</small>
הַתּוֹרֵם קִשּׁוּת וְנִמְצֵאת מָרָה, 
אֲבַטִּיחַ וְנִמְצָא סָרוּחַ, 
תְּרוּמָה, וְיַחְזֹר וְיִתְרֹם.  
הַתּוֹרֵם חָבִית שֶׁלַּיַּיִן וְנִמְצֵאת שֶׁלַּחֹמֶץ, 
אִם יָדוּעַ שֶׁהָיְתָה שֶׁלַּחֹמֶץ עַד שֶׁלֹּא תְרָמָהּ, 
אֵינָה תְּרוּמָה; 
אִם מִשֶּׁתְּרָמָהּ הֶחְמִיצָה, 
הֲרֵי זוֹ תְרוּמָה; 
אִם סָפֵק, 
תְּרוּמָה, וְיַחְזֹר וְיִתְרֹם.  
הָרִאשׁוֹנָה אֵינָה מְדַמַּעַת בִּפְנֵי עַצְמָהּ, 
וְאֵין חַיָּבִין עָלֶיהָ חֹמֶשׁ; 
וְכֵן הַשְּׁנִיָּה. 

<small>ב</small>
נָפְלָה אַחַת מֵהֶן לְתוֹךְ הַחֻלִּין, 
אֵינָה מְדַמְּעַתָּן.  
נָפְלָה שְׁנִיָּה לְמָקוֹם אַחֵר, 
אֵינָה מְדַמְּעַתָּן.  
נָפְלוּ שְׁתֵּיהֶן לְמָקוֹם אֶחָד, 
מְדַמְּעוֹת כִּקְטַנָּה שֶׁבִּשְׁתֵּיהֶם. &lt;בִּקְטַנָּה&gt;

<small>ג</small>
הַשּׁוּתָפִין שֶׁתָּרְמוּ זֶה אַחַר זֶה, 
רְבִּי עֲקִיבָה אוֹמֵר: 
תְּרוּמַת שְׁנֵיהֶם תְּרוּמָה. 
וַחֲכָמִים אוֹמְרִים: 
אֵין תְּרוּמַת הַשֵּׁנִי תְרוּמָה. 
רְבִּי יוֹסֵה אוֹמֵר: 
אִם תָּרַם הָרִאשׁוֹן כַּשֵּׁעוּר, 
אֵין תְּרוּמַת הַשֵּׁנִי תְרוּמָה, 
וְאִם לֹא תָרַם הָרִאשׁוֹן כַּשֵּׁעוּר, 
תְּרוּמַת הַשֵּׁנִי תְרוּמָה. 

<small>ד</small>
בַּמֵּי דְבָרִים אֲמוּרִים? 
בְּשֶׁלּא דִּבֵּר.  
אֲבָל הִרְשָׁה אֶת בֶּן בֵּיתוֹ, 
אוֹ אֶת עַבְדּוֹ, אוֹ אֶת שִׁפְחָתוֹ לִתְרֹם, 
תְּרוּמָתוֹ תְרוּמָה.  
בִּטֵּל, 
אִם עַד שֶׁלֹּא תָרַם בִּטֵּל, 
אֵין תְּרוּמָתוֹ תְרוּמָה, 
וְאִם מִשֶּׁתָּרַם בִּטֵּל, 
תְּרוּמָתוֹ תְרוּמָה.  
הַפּוֹעֲלִים, אֵין לָהֶם רְשׁוּת לִתְרֹם, 
חוּץ מִן הַדְּרוּכוֹת, 
שֶׁהֵן מְטַמִּין אֶת הַגַּת מִיָּד. 

<small>ה</small>
הָאוֹמֵר: 
"תְּרוּמַת הַכָּרִי זֶה בְתוֹכוֹ, 
וּמַעְשְׂרוֹתָיו בְּתוֹכוֹ, 
וּתְרוּמַת מַעֲשֵׂר זֶה בְתוֹכוֹ", 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
קָרָא שֵׁם.  
וַחֲכָמִים אוֹמְרִים: 
עַד שֶׁיֹּאמַר: 
"בִּצְפוֹנוֹ וּבִדְרוֹמוֹ".  
רְבִּי אֶלְעָזָר חֲסָמָה אוֹמֵר: 
הָאוֹמֵר: 
"תְּרוּמַת כָּרִי מִמֶּנּוּ עָלָיו", 
קָרָא שֵׁם.  
רְבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר: 
הָאוֹמֵר: 
"עִשּׂוּר מַעֲשֵׂר זֶה עָשׁוּי תְּרוּמַת מַעֲשֵׂר עָלָיו", 
קָרָא שֵׁם. 

<small>ו</small>
הַמַּקְדִּים תְּרוּמָה לַבִּכּוּרִים, 
מַעֲשֵׂר רִאשׁוֹן לַתְּרוּמָה, 
וּמַעֲשֵׂר שֵׁנִי לָרִאשׁוֹן, 
אַף עַל פִּי שֶׁהוּא עוֹבֵר בְּ'לֹא תַעֲשֶׂה', 
מַה שֶּׁעָשָׂה עָשׁוּי, שֶׁנֶּאֱמַר: (שְׁמוֹת כב,כח) 
"מְלֵאָתְךָ וְדִמְעֲךָ לֹא תְאַחֵר". 

<small>ז</small>
וּמְנַיִן שֶׁיְּקַדְּמוּ בִכּוּרִים לִתְרוּמָה? 
זֶה קָרוּי תְּרוּמָה וְרֵאשִׁית, 
וְזֶה קָרוּי תְּרוּמָה וְרֵאשִׁית!  
אֶלָּא יְקַדְּמוּ הַבִּכּוּרִים, שֶׁהֵן בִּכּוּרִים לַכֹּל, 
וּתְרוּמָה לָרִאשׁוֹן, שֶׁהִיא רֵאשִׁית, 
וּמַעֲשֵׂר רִאשׁוֹן לַשֵּׁנִי, שֶׁיֶּשׁ בּוֹ רֵאשִׁית. 

<small>ח</small>
הַמִּתְכַּוֵּן לוֹמַר "תְּרוּמָה", וְאָמַר "מַעֲשֵׂר", 
"מַעֲשֵׂר", וְאָמַר "תְּרוּמָה", 
"עוֹלָה", וְאָמַר "שְׁלָמִים", 
"שְׁלָמִים", וְאָמַר "עוֹלָה", 
"שֶׁינִי נִכְנָס לַבַּיִת הַזֶּה", וְאָמַר "לָזֶה", 
"שֶׁינִי נֶהְנֶה לָזֶה", וְאָמַר "לָזֶה", 
לֹא אָמַר כְּלוּם, 
עַד שֶׁיְּהֵא פִיו וְלִבּוֹ שָׁוִים. 

<small>ט</small>
הַנָּכְרִי וְהַכּוּתִי, 
תְּרוּמָתָן תְּרוּמָה, 
וּמַעְשְׂרוֹתֵיהֶן מַעֲשֵׂר, &lt;ומעשרותן&gt;
וְהֶקְדֵּשָׁן הֶקְדֵּשׁ.  
רְבִּי יְהוּדָה אוֹמֵר: 
אֵין לַנָּכְרִי כֶרֶם רְבָעִי. 
וַחֲכָמִים אוֹמְרִים: 
יֶשׁ לוֹ.  
תְּרוּמַת הַנָּכְרִי מְדַמַּעַת, וְחַיָּבִין עָלֶיהָ חֹמֶשׁ. 
וּרְבִּי שִׁמְעוֹן פּוֹטֵר. 


 


Chapter 4

<small>א</small>
הַמַּפְרִישׁ מִקְצָת תְּרוּמָה וּמַעַשְׂרוֹת, 
מוֹצִיא מִמֶּנּוּ תְרוּמָה עָלָיו, 
אֲבָל לֹא לְמָקוֹם אַחֵר.  
רְבִּי מֵאִיר אוֹמֵר: 
אַף מוֹצִיא הוּא מִמָּקוֹם אַחֵר תְּרוּמָה וּמַעַשְׂרוֹת. 

<small>ב</small>
מִי שֶׁהָיוּ פֵרוֹתָיו בַּמְּגוֹרָה, 
וְנָתַן סְאָה לְבֶן לֵוִי וּסְאָה לֶעָנִי, 
וּמַפְרִישׁ עוֹד שְׁמוֹנַת סְאִים וְאוֹכְלָן.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
אֵינוּ מַפְרִישׁ אֶלָּא לְפִי חֶשְׁבּוֹן. 

<small>ג</small>
שֵׁעוּר תְּרוּמָה: 
עַיִן יָפָה, [אֶחָד] מֵאַרְבָּעִים.  
בֵּית שַׁמַּי אוֹמְרִים: 
מִשְּׁלֹשִׁים.  
וְהַבֵּינוֹנִית, מֵחֲמִשִּׁים, 
וְהָרָעָה, מִשִּׁשִּׁים.  
תָּרַם וְעָלָה בְיָדוֹ אֶחָד מִשִּׁשִּׁים, 
תְּרוּמָה, אֵינוּ צָרִיךְ לִתְרֹם.  
חָזַר וְהוֹסִיף, חַיָּב בַּמַּעַשְׂרוֹת.  
עָלָה בְיָדוֹ מִשִּׁשִּׁים וְאֶחָד, 
תְּרוּמָה, יַחְזֹר וְיִתְרֹם כְּמוֹת שֶׁהוּא לָמוּד, &lt;לִמּוּד&gt;
בַּמִּדָּה וּבַמִּשְׁקָל וּבַמִּנְיָן.  
רְבִּי יְהוּדָה אוֹמֵר: &lt;יודה&gt;
שֶׁלֹּא מִן הַמֻּקָּף. 

<small>ד</small>
הָאוֹמֵר לִשְׁלוּחוֹ: 
"צֵא וּתְרֹם!" 
תּוֹרֵם בְּדַעְתּוֹ שֶׁלְּבַעַל הַבַּיִת.  
וְאִם אֵינוּ יוֹדֵעַ דַּעְתּוֹ שֶׁלְּבַעַל הַבַּיִת, 
תּוֹרֵם בַּבֵּינוֹנִית אֶחָד מֵחֲמִשִּׁים.  
פָּחַת עֲשָׂרָה אוֹ הוֹסִיף עֲשָׂרָה, 
תְּרוּמָתוֹ תְרוּמָה.  
אִם נִתְכַּוַּן לְהוֹסִיף אֲפִלּוּ אֶחָד, 
אֵין תְּרוּמָתוֹ תְרוּמָה. 

<small>ה</small>
הַמַּרְבֶּה בִתְרוּמָה, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אֶחָד מֵעֲשָׂרָה, כִּתְרוּמַת מַעֲשֵׂר. 
יָתֵר מִכֵּן, יַעֲשֶׂנָּה תְרוּמַת מַעְשֵׂר, 
אֲבָל לֹא מִמָּקוֹם אַחֵר.  
רְבִּי יִשְׁמָעֵאל אוֹמֵר: 
מַחְצָה חֻלִּין וּמַחְצָה תְרוּמָה.  
רְבִּי טַרְפוֹן וּרְבִּי עֲקִיבָה אוֹמְרִים: 
עַד שֶׁיְּשַׁיֵּר שָׁם חֻלִּין. 

<small>ו</small>
בִּשְׁלֹשָׁה פְרָקִים מְשַׁעֲרִים אֶת הַכַּלְכַּלָּה: 
בַּבַּכּוֹרוֹת וּבִסְיָפוֹת וּבְאֶמְצַע הַקַּיִץ.  
הַמּוֹנֶה מְשֻׁבָּח, 
וְהַמּוֹדֵד מְשֻׁבָּח מִמֶּנּוּ, 
וְהַשּׁוֹקֵל מְשֻׁבָּח מִשְּׁלָשְׁתָּן. 

<small>ז</small>
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
תְּרוּמָה עוֹלָה בְמֵאָה וְאֶחָד.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
בְּמֵאָה וָעוֹד. 
"וָעוֹד" זֶה, אֵין לוֹ שֵׁעוּר.  
רְבִּי יוֹסֵה בֶן מְשֻׁלָּם אוֹמֵר: 
"וָעוֹד" קַב לְמֵאָה סְאָה, 
שְׁתוּת לִמְדַמֵּעַ. 

<small>ח</small>
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
תְּאֵנִים שְׁחוֹרוֹת מַעֲלוֹת אֶת הַלְּבָנוֹת, 
וְהַלְּבָנוֹת מַעֲלוֹת אֶת הַשְּׁחוֹרוֹת. &lt;לבנות&gt;
עִגּוּלֵי דְבֵלָה, 
הַגְּדוֹלִים מַעֲלִים אֶת הַקְּטַנִּים, 
וְהַקְּטַנִּים מַעֲלִים אֶת הַגְּדוֹלִים. 
הַעִגּוּלִים מַעֲלִין אֶת הַמַּלְבְּנִים, 
וְהַמַּלְבְּנִים מַעֲלִין אֶת הַעִגּוּלִים.  
רְבִּי אֱלִיעֶזֶר אוֹסֵר.  
וּרְבִּי עֲקִיבָה אוֹמֵר: 
בַּיָּדוּעַ מַה נָּפְלָה, 
אֵין מַעֲלוֹת זוֹ אֶת זוֹ, 
וּבְשֶׁאֵינוּ יָדוּעַ מַה נָּפְלָה, 
מַעֲלוֹת זוֹ אֶת זוֹ. 

<small>ט</small>
כֵּיצַד? 
חֲמִשִּׁים תְּאֵנִים שְׁחוֹרוֹת וַחֲמִשִּׁים לְבָנוֹת: 
נָפְלָה שְׁחוֹרָה, 
שְׁחוֹרוֹת אֲסוּרוֹת וּלְבָנוֹת מֻתָּרוֹת; 
נָפְלָה לְבָנָה, 
לְבָנוֹת אֲסוּרוֹת וּשְׁחוֹרוֹת מֻתָּרוֹת; 
בְּשֶׁאֵינוּ יָדוּעַ מַה נָּפְלָה, 
מַעֲלוֹת זוֹ אֶת זוֹ.  
וּבָזוֹ רְבִּי אֱלִיעֶזֶר מַחְמִיר, 
וּרְבִּי יְהוֹשֻׁעַ מֵקֵל. 

<small>י</small>
וּבָזוֹ רְבִּי אֱלִיעֶזֶר מֵקֵל וּרְבִּי יְהוֹשֻׁעַ מַחְמִיר: 
בַּדּוֹרֵס לִטְרָא קְצִיעוֹת עַל פִּי הַכַּד, 
וְאֵין יָדוּעַ אֵי זוֹ הִיא, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
רוֹאִין אוֹתָן כְּאִלּוּ הֵן פְּרוּדוֹת, &lt;כילו&gt;
הַתַּחְתּוֹנוֹת מַעֲלוֹת אֶת הָעֶלְיוֹנוֹת.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
לֹא תַעֲלֶה עַד שֶׁיְּהֵא שָׁם מֵאָה כַד. 

<small>יא</small>
סְאָה תְרוּמָה שֶׁנָּפְלָה עַל פִּי מְגוֹרָה וּקְפָיָהּ, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אִם יֵשׁ בַּקָּפוּי, תַּעֲלֶה בְאֶחָד וּמֵאָה.  
וּרְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
לֹא תַעֲלֶה.  
סְאָה תְרוּמָה שֶׁנָּפְלָה עַל פִּי מְגוֹרָה, 
יִקְּפֶנָּה.  
וְאִם כֵּן, לָמָּה אָמְרוּ: 
תְּרוּמָה עוֹלָה בְמֵאָה וְאֶחָד? 
בְּשֶׁאֵינוּ יָדוּעַ אִם בְּלוּלוֹת הֵן, &lt;ואם אינו&gt;
אוֹ לְאַיִן נָפְלָה. 

<small>יב</small>
שְׁתֵּי קֻפּוֹת, שְׁתֵּי מְגוֹרוֹת, 
שֶׁנָּפְלָה סְאָה תְרוּמָה לְתוֹךְ אַחַת מֵהֶן, 
וְאֵין יָדוּעַ לְאֵי זוֹ מֵהֶן נָפְלָה, 
מַעֲלוֹת זוֹ אֶת זוֹ.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אֲפִלּוּ הֵן בִּשְׁתֵּי עֲיָרוֹת, 
מַעֲלוֹת זוֹ אֶת זוֹ. 

<small>יג</small>
אָמַר רְבִּי יוֹסֵה: 
מַעֲשֶׂה בָא לִפְנֵי רְבִּי עֲקִיבָה, 
בַּחֲמִשִּׁים אֲגֻדּוֹת שֶׁלַּיָּרָק, 
שֶׁנָּפְלָה אַחַת מֵהֶן לְתוֹכָן חֶצְיָהּ תְּרוּמָה, 
וְאָמַרְתִּי לְפָנָיו: 
תַּעֲלֶה, 
לֹא שֶׁהַתְּרוּמָה עוֹלָה בַּחֲמִשִּׁים וְאֶחָד, 
אֶלָּא שֶׁהָיוּ שָׁם מֵאָה וּשְׁנֵי חֲצָיִים. 


 


Chapter 5

<small>א</small>
סְאָה תְרוּמָה טְמֵאָה שֶׁנָּפְלָה לְפָחוּת מִמֵּאָה חֻלִּין, 
אוֹ לְמַעֲשֵׂר רִאשׁוֹן, אוֹ לְמַעֲשֵׂר שֵׁנִי, אוֹ לַהֶקְדֵּשׁ, 
בֵּין טְמֵאִים בֵּין טְהוֹרִים יִרְקָבוּ.  
אִם טְהוֹרָה הָיְתָה אוֹתָהּ הַסְּאָה, 
יִמָּכְרוּ לַכֹּהֲנִים בִּדְמֵי תְרוּמָה, 
חוּץ מִדְּמֵי אוֹתָהּ סְאָה.  
וְאִם לְמַעֲשֵׂר רִאשׁוֹן נָפְלָה, 
יִקְרָא שֵׁם לִתְרוּמַת מַעֲשֵׂר.  
וְאִם לְמַעֲשֵׂר שֵׁנִי וּלְהֶקְדֵּשׁ נָפְלָה, 
הֲרֵי אֵלּוּ יִפָּדוּ.  
וְאִם טְמֵאִים הָיוּ אוֹתָן הַחֻלִּין, 
יֵאָכֵלוּ נִקּוּדִים אוֹ קָלִיּוֹת, 
אוֹ יִלּוֹשׁוּ בְמֵי פֵרוֹת, 
אוֹ יִתְחַלְּקוּ לָעִסּוֹת, 
כְּדֵי שֶׁלֹּא יְהֵא בְמָקוֹם אֶחָד כַּבֵּיצָה. 

<small>ב</small>
סְאָה תְרוּמָה טְמֵאָה, 
שֶׁנָּפְלָה לְמֵאָה חֻלִּין טְהוֹרִים, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: &lt;ליעזר&gt;
תֵּרוֹם וְתִשָּׂרֵף.  
שֶׁאֲנִי אוֹמֵר: &lt;שני&gt;
סְאָה שֶׁנָּפְלָה הִיא סְאָה שֶׁעָלַת.  
וַחֲכָמִים אוֹמְרִים: 
תַּעֲלֶה וְתֵאָכֵל נִקּוּדִים אוֹ קָלִיּוֹת, 
אוֹ תִלּוֹשׂ בְּמֵי פֵרוֹת, 
אוֹ תִתְחַלֵּק לָעִסּוֹת, 
כְּדֵי שֶׁלֹּא יְהֵי בְמָקוֹם אֶחָד כַּבֵּיצָה. 

<small>ג</small>
סְאָה תְרוּמָה טְהוֹרָה שֶׁנָּפְלָה לְמֵאָה חֻלִּין טְמֵאִים, 
תַּעֲלֶה וְתֵאָכֵל נִקּוּדִים אוֹ קָלִיּוֹת, 
אוֹ תִלּוֹשׂ בְּמֵי פֵרוֹת, 
אוֹ תִתְחַלֵּק לָעִסּוֹת, 
כְּדֵי שֶׁלֹּא יְהֵא בְמָקוֹם אֶחָד כַּבֵּיצָה. 

<small>ד</small>
סְאָה תְרוּמָה טְמֵאָה שֶׁנָּפְלָה לְמֵאָה סְאָה תְרוּמָה טְהוֹרָה, 
בֵּית שַׁמַּי אוֹסְרִין, 
וּבֵית הֶלֵּל מַתִּירִין.  
אָמְרוּ בֵית הֶלֵּל לְבֵית שַׁמַּי: 
הוֹאִיל וּטְהוֹרָה אֲסוּרָה לַזָּרִים, 
וְאַף טְמֵאָה אֲסוּרָה לַכֹּהֲנִים, 
מַה טְּהוֹרָה עוֹלָה, 
אַף טְמֵאָה תַעֲלֶה!  
אָמְרוּ לָהֶם בֵּית שַׁמַּי: 
לֹא! אִם הֶעֱלוּ הַחֻלִּים הַקַּלִּים, 
הַמֻּתָּרִים לַזָּרִים, אֶת הַטְּהוֹרָה, 
תַּעֲלֶה תְרוּמָה חֲמוּרָה, 
הָאֲסוּרָה לַזָּרִים, אֶת הַטְּמֵאָה? 
לְאַחַר שֶׁהוֹדוּ, 
רֶבִּי אֱלִיעֶזֶר אוֹמֵר: 
תֵּרוֹם וְתִשָּׂרֵף.  
וַחֲכָמִים אוֹמְרִים: 
אָבְדָה בְּמִעוּטָהּ. 

<small>ה</small>
סְאָה תְרוּמָה שֶׁנָּפְלָה לְמֵאָה, 
הִגְבִּיהָהּ וְנָפְלָה לְמָקוֹם אַחֵר, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
מְדַמַּעַת כִּתְרוּמָה וַדַּי.  
וַחֲכָמִים אוֹמְרִים: 
אֵינָה מְדַמַּעַת אֶלָּא לְפִי חֶשְׁבּוֹן. 

<small>ו</small>
סְאָה תְרוּמָה שֶׁנָּפְלָה לְפָחוּת מִמֵּאָה וְנִדְמָעוּ, 
וְנָפַל מִן הַמְדֻמָּע לְמָקוֹם אַחֵר, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
מְדַמַּעַת כִּתְרוּמָה וַדַּי.  
וַחֲכָמִים אוֹמְרִים: 
אֵין מְדֻמָּע מַדְמִיעַ אֶלָּא לְפִי חֶשְׁבּוֹן,  
וְאֵין הַמְחֻמָּץ מַחֲמִיץ אֶלָּא לְפִי חֶשְׁבּוֹן, &lt;מַחָמֶץ&gt;
וְאֵין מַיִם שְׁאוּבִין פּוֹסְלִין אֶת הַמִּקְוֶה אֶלָּא לְפִי חֶשְׁבּוֹן. 

<small>ז</small>
סְאָה תְרוּמָה שֶׁנָּפְלָה לְמֵאָה, 
הִגְבִּיהָהּ וְנָפְלָה אַחֶרֶת, 
הִגְבִּיהָהּ וְנָפְלָה אַחֶרֶת, 
הֲרֵי זוֹ מֻתֶּרֶת, 
עַד שֶׁתַּרְבֶּה תְרוּמָה לַחֻלִּין. 

<small>ח</small>
סְאָה תְרוּמָה שֶׁנָּפְלָה לְמֵאָה, 
וְלֹא הִסְפִּיק לְהַגְבִּיהָהּ עַד שֶׁנָּפְלָה אַחֶרֶת, 
הֲרֵי זוֹ אֲסוּרָה.  
וּרְבִּי שִׁמְעוֹן מַתִּיר. 

<small>ט</small>
סְאָה תְרוּמָה שֶׁנָּפְלָה לְמֵאָה, 
וּטְחָנָן וּפָחֲתוּ, 
כַּשֵּׁם שֶׁפָּחֲתוּ הַחֻלִּין, 
כָּךְ פָּחֲתָה תְרוּמָה, 
מֻתָּר.  
סְאָה תְרוּמָה שֶׁנָּפְלָה לְפָחוּת מִמֵּאָה, 
וּטְחָנָן וְהוֹתִירוּ, 
כַּשֵּׁם שֶׁהוֹתִירוּ הַחֻלִּין, 
כָּךְ הוֹתִירָה תְרוּמָה, 
אָסוּר.  
אִם יָדוּעַ שֶׁחִטִּים שֶׁלַּחֻלִּין יָפוֹת מִשֶּׁלִּתְרוּמָה, 
מֻתָּר.  
סְאָה תְרוּמָה שֶׁנָּפְלָה לְפָחוּת מִמֵּאָה, 
וְאַחַר כָּךְ נָפְלוּ שָׁם חֻלִּין, 
אִם שׁוֹגֵג, מֻתָּר, 
וְאִם מֵזִיד, אָסוּר. 


 


Chapter 6

<small>א</small>
הָאוֹכֵל תְּרוּמָה שׁוֹגֵג, 
מְשַׁלֵּם קֶרֶן וְחֹמֶשׁ.  
אֶחָד הָאוֹכֵל וְאֶחָד הַשּׁוֹתֶה וְאֶחָד הַסָּךְ, 
אֶחָד תְּרוּמָה טְהוֹרָה וְאֶחָד תְּרוּמָה טְמֵאָה, 
מְשַׁלֵּם חֻמְשָׁהּ וְחֹמֶשׁ חֻמְשָׁהּ.  
אֵינוּ מְשַׁלֵּם תְּרוּמָה אֶלָּא חֻלִּים מְתֻקָּנִים, 
וְהֵם נֶעֱשִׂים תְּרוּמָה,  
וְהַתַּשְׁלוּמִין תְּרוּמָה.  
אִם רָצָה הַכֹּהֵן לִמְחֹל, 
אֵינוּ מוֹחֵל. 

<small>ב</small>
בַּת יִשְׂרָאֵל שֶׁאָכְלָה תְרוּמָה, 
וְאַחַר כָּךְ נִשֵּׂאת לַכֹּהֵן, 
אִם תְּרוּמָה שֶׁלֹּא זָכָה בָהּ כֹּהֵן אָכְלָה, 
מְשַׁלֶּמֶת קֶרֶן וְחֹמֶשׁ לְעַצְמָהּ; 
אִם תְּרוּמָה שֶׁזָּכָה בָהּ כֹּהֵן אָכְלָה, 
מְשַׁלֶּמֶת קֶרֶן לַבְּעָלִים וְחֹמֶשׁ לְעַצְמָהּ, 
מִפְּנֵי שֶׁאָמְרוּ: 
הָאוֹכֵל תְּרוּמָה שׁוֹגֵג, 
מְשַׁלֵּם קֶרֶן לַבְּעָלִים, 
וְחֹמֶשׁ לְמִי שֶׁיִּרְצֶה. 

<small>ג</small>
הַמַּאֲכִיל אֶת פּוֹעֲלָיו וְאֶת אוֹרְחָיו תְּרוּמָה, 
הוּא מְשַׁלֵּם אֶת הַקֶּרֶן, 
וְהֵן מְשַׁלְּמִין אֶת הַחֹמֶשׁ.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
הֵן מְשַׁלְּמִין קֶרֶן וְחֹמֶשׁ, 
וְהוּא מְשַׁלֵּם לָהֶן דְּמֵי סְעוּדָתָן. 

<small>ד</small>
הַגּוֹנֵב תְּרוּמָה וְלֹא אֲכָלָהּ, 
מְשַׁלֵּם תַּשְׁלוּמֵי כֶפֶל דְּמֵי תְרוּמָה.  
אֲכָלָהּ, 
מְשַׁלֵּם שְׁנֵי קְרָנִים וְחֹמֶשׁ: 
קֶרֶן וְחֹמֶשׁ מִן הַחֻלִּין, 
וְקֶרֶן דְּמֵי תְרוּמָה.  
גָּנַב תְּרוּמַת הֶקְדֵּשׁ וַאֲכָלָהּ, &lt;תְּרוּמַה&gt;
מְשַׁלֵּם שְׁנֵי חֳמָשִׁין וְקֶרֶן, 
שֶׁאֵין בַּהֶקְדֵּשׁ תַּשְׁלוּמֵי כָפֶל. 

<small>ה</small>
אֵין מְשַׁלְּמִין מִן הַלֶּקֶט, מִן הַשִּׁכְחָה, 
וּמִן הַפֵּאָה וּמִן הַהֶבְקֵר, 
וְלֹא מִמַּעֲשֵׂר רִאשׁוֹן שֶׁנִּטְּלָה תְרוּמָתוֹ, 
וְלֹא מִמַּעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁנִּפְדּוּ, 
שֶׁאֵין הַקֹּדֶשׁ פּוֹדֶה אֶת הַקֹּדֶשׁ.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים מַתִּירִין בָּאֵלּוּ. 

<small>ו</small>
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
מְשַׁלְּמִין מִמִּין עַל שֶׁאֵינוּ מִינוֹ, 
וּבִלְבַד שֶׁיְּשַׁלֵּם מִן הַיָּפֶה עַל הָרָע.  
רְבִּי עֲקִיבָה אוֹמֵר: 
אֵין מְשַׁלְּמִין אֶלָּא מִמִּין עַל מִינוֹ. 

<small>ז</small>
לְפִיכָךְ אִם אָכַל קִשּׁוּאִים שֶׁלְּעֶרֶב שְׁבִיעִית, 
יַמְתִּין לַקִּשּׁוּאִים שֶׁלְּמוֹצָאֵי שְׁבִיעִית, 
וִישַׁלֵּם מֵהֶם.  
מְקוֹם שֶׁרְבִּי אֱלִיעֶזֶר מֵקֵל, 
מִשָּׁם רְבִּי עֲקִיבָה מַחְמִיר, 
שֶׁנֶּאֱמַר: (וַיִּקְרָא כב,יד) 
"וְנָתַן לַכֹּהֵן אֶת הַקֹּדֶשׁ", 
כָּל שֶׁהוּא רָאוּי לְהֵעָשׂוֹת קֹדֶשׁ.  
דִּבְרֵי רְבִּי אֱלִיעֶזֶר.  
רְבִּי עֲקִיבָה אוֹמֵר: 
"וְנָתַן לַכֹּהֵן אֶת הַקֹּדֶשׁ", קֹדֶשׁ שֶׁאָכַל. 





Chapter 7

<small>א</small>
הָאוֹכֵל תְּרוּמָה מֵזִיד, 
מְשַׁלֵּם אֶת הַקֶּרֶן, 
אֵינוּ מְשַׁלֵּם אֶת הַחֹמֶשׁ,  
וְהַתַּשְׁלוּמִין חֻלִּין.  
אִם רָצָה הַכֹּהֵן לִמְחֹל, מוֹחֵל. 

<small>ב</small>
בַּת כֹּהֵן שֶׁנִּשֵּׂאת לְיִשְׂרָאֵל, 
וְאַחַר כָּךְ אָכְלָה תְרוּמָה, 
מְשַׁלֶּמֶת אֶת הַקֶּרֶן, 
וְאֵינָהּ מְשַׁלֶּמֶת אֶת הַחֹמֶשׁ, 
וּמִיתָתָהּ בִּשְׂרִיפָה.  
נִשֵּׂאת לְאֶחָד מִכָּל הַפְּסוּלִין, 
מְשַׁלֶּמֶת הַקֶּרֶן וְחֹמֶשׁ, 
וּמִיתָתָהּ בְּחָנֵק.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
זוֹ וָזוֹ מְשַׁלֶּמֶת אֶת הַקֶּרֶן, 
וְאֵינָה מְשַׁלֶּמֶת אֶת הַחֹמֶשׁ, 
וּמִיתָתָן בִּשְׂרִיפָה. 

<small>ג</small>
הַמַּאֲכִיל אֶת בָּנָיו קְטַנִּים, 
וְאֶת עֲבָדָיו בֵּין גְּדוֹלִים וּבֵין קְטַנִּים, 
הָאוֹכֵל תְּרוּמַת חוּצָה לָאָרֶץ, 
הָאוֹכֵל פָּחוּת מִכַּזַּיִת תְּרוּמָה, 
מְשַׁלֵּם אֶת הַקֶּרֶן, 
וְאֵינוּ מְשַׁלֵּם אֶת הַחֹמֶשׁ, 
וְהַתַּשְׁלוּמִין חֻלִּין.  
אִם רָצָה הַכֹּהֵן לִמְחֹל, מוֹחֵל. 

<small>ד</small>
זֶה הַכְּלָל: 
כָּל הַמְשַׁלֵּם קֶרֶן וְחֹמֶשׁ, 
הַתַּשְׁלוּמִין תְּרוּמָה; 
אִם רָצָה הַכֹּהֵן לִמְחֹל, אֵינוּ מוֹחֵל.  
וְכָל הַמְשַׁלֵּם אֶת הַקֶּרֶן וְאֵינוּ מְשַׁלֵּם אֶת הַחֹמֶשׁ, 
הַתַּשְׁלוּמִין חֻלִּין; 
אִם רָצָה הַכֹּהֵן לִמְחֹל, מוֹחֵל. 

<small>ה</small>
שְׁתֵּי קֻפּוֹת, 
אַחַת שֶׁלִּתְרוּמָה וְאַחַת שֶׁלַּחֻלִּין, 
שֶׁנָּפְלָה סְאָה תְרוּמָה לְתוֹךְ אַחַת מֵהֶן, 
וְאֵין יָדוּעַ לְאֵי זוֹ מֵהֶן נָפְלָה, 
הֲרֵי אֲנִי אוֹמֵר: &lt;הוא&gt;
לְתוֹךְ שֶׁלִּתְרוּמָה נָפְלָה.  
וְאֵין יָדוּעַ אֵי זוֹ הִיא שֶׁלִּתְרוּמָה וְאֵי זוֹ הִיא שֶׁלַּחֻלִּין, 

<small>ו</small>
אָכַל אַחַת מֵהֶן, 
פָּטוּר; 
וְהַשְּׁנִיָּה, 
נוֹהֵג בָּהּ כִּתְרוּמָה, וְחַיֶּבֶת בַּחַלָּה.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יוֹסֵה פוֹטֵר.  
אָכַל אַחֵר אֶת הַשְּׁנִיָּה, 
פָּטוּר.  
אָכַל אֶחָד אֶת שְׁתֵּיהֶן, 
מְשַׁלֵּם כִּקְטַנָּה שֶׁבִּשְׁתֵּיהֶם. 

<small>ז</small>
נָפְלָה אַחַת מֵהֶן לְתוֹךְ הַחֻלִּין, 
וְאֵינָה מְדַמְּעַתָּן, 
וְהַשְּׁנִיָּה, 
נוֹהֵג בָּהּ כִּתְרוּמָה, וְחַיֶּבֶת בַּחַלָּה.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יוֹסֵה פוֹטֵר.  
נָפְלָה שְׁנִיָּה לְמָקוֹם אַחֵר, 
אֵינָה מְדַמְּעַתָּהּ. &lt;מדמעתן&gt; 
נָפְלוּ שְׁתֵּיהֶן לְמָקוֹם אֶחָד, 
מְדַמְּעוֹת כִּקְטַנָּה שֶׁבִּשְׁתֵּיהֶן. 

<small>ח</small>
זָרַע אֶת אַחַת מֵהֶן, 
פָּטוּר, 
וְהַשְּׁנִיָּה, 
נוֹהֵג בָּהּ כִּתְרוּמָה, וְחַיֶּבֶת בַּחַלָּה.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יוֹסֵה פוֹטֵר.  
זָרַע אַחֵר אֶת הַשְּׁנִיָּה, 
פָּטוּר.  
זָרַע אֶחָד אֶת שְׁתֵּיהֶן, 
דָּבָר שֶׁזַּרְעוֹ כָלֶה, מֻתָּר, 
וְדָבָר שֶׁאֵין זַרְעוֹ כָלֶה, אָסוּר. 


 


Chapter 8

<small>א</small>
הָאִשָּׁה שֶׁהָיְתָה אוֹכֶלֶת בִּתְרוּמָה, &lt;שאכלה&gt;
בָּאוּ וְאָמְרוּ לָהּ "מֵת בַּעְלִיךְ", 
אוֹ "גֵּרְשֵׁךְ"; 
וְכֵן הָעֶבֶד שֶׁהָיָה אוֹכֵל בִּתְרוּמָה, 
בָּאוּ וְאָמְרוּ לוֹ "מֵת רַבָּךְ", 
אוֹ "מְכָרָךְ לְיִשְׂרָאֵל", 
אוֹ "נְתָנָךְ מַתָּנָה", 
אוֹ "עֲשָׂאָךְ בֶּן חוֹרִין"; 
וְכֵן כֹּהֵן שֶׁהָיָה אוֹכֵל בִּתְרוּמָה, 
וְנוֹדַע שֶׁהוּא בֶן גְּרוּשָׁה אוֹ בֶן חֲלוּצָה, 
רְבִּי אֱלִיעֶזֶר מְחַיֵּב קֶרֶן וְחֹמֶשׁ, 
רְבִּי יְהוֹשֻׁעַ פּוֹטֵר.  
הָיָה עוֹמֵד וּמַקְרִיב עַל גַּבֵּי הַמִּזְבֵּחַ, 
וְנוֹדַע שֶׁהוּא בֶן גְּרוּשָׁה אוֹ בֶן חֲלוּצָה, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
כָּל הַקָּרְבָּנוֹת שֶׁהִקְרִיב עַל גַּבֵּי הַמִּזְבֵּחַ, 
פְּסוּלִים. 
וּרְבִּי יְהוֹשֻׁעַ מַכְשִׁיר.  
נוֹדַע שֶׁהוּא בַּעַל מוּם, 
עֲבוֹדָתוֹ פְסוּלָה. 

<small>ב</small>
וְכֻלָּם שֶׁהָיְתָה תְרוּמָה לְתוֹךְ פִּיהֶם, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
יִבְלָעוּ. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
יִפְלֹטוּ.  
אָמְרוּ לוֹ: 
"נִטְמֵאתָ", "נִטַּמַּת תְּרוּמָה", 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
יִבְלַע. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
יִפְלֹט. &lt;יִפְלוֹטוּ&gt; 
אִם "טָמֵא הָיִיתָ", וּ"טְמֵאָה הָיְתָה תְרוּמָה", 
אוֹ נוֹדַע שֶׁהוּא טֶבֶל, 
אוֹ מַעֲשֵׂר רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְרוּמָתוֹ, 
אוֹ מַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁלֹּא נִפְדּוּ, 
אוֹ שֶׁטָּעַם טַעַם פְּשֵׁפֵשׁ לְתוֹךְ פִּיהוּ, 
הֲרֵי זֶה יִפְלֹט. 

<small>ג</small>
הָיָה אוֹכֵל בָּאֶשְׁכּוֹל, 
וְנִכְנַס מִן הַגִּנָּה לֶחָצֵר, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
יִגְמֹר. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
לֹא יִגְמֹר. 
חָשֵׁכָה לֵילֵי שַׁבָּת, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
לֹא יִגְמֹר.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
יִגְמֹר. 

<small>ד</small>
יַיִן שֶׁלִּתְרוּמָה שֶׁנִּתְגַּלָּה, 
יִשָּׁפֵךְ, 
וְאֵין צֹרֶךְ לוֹמַר שֶׁלַּחֻלִּים.  
שְׁלֹשָׁה מַשְׁקִין אֲסוּרִין מִשֵּׁם גִּלּוּי: 
הַמַּיִם וְהַיַּיִן וְהֶחָלָב. 
וּשְׁאָר כָּל הַמַּשְׁקִין מֻתָּרִים.  
כַּמָּה יִשְׁהוּ וִיהוּ אֲסוּרִין? 
כְּדֵי שֶׁיֵּצֵא הָרַחַשׁ מִמָּקוֹם קָרוֹב וְיִשְׁתֶּה. 

<small>ה</small>
שֵׁעוּר הַמַּיִם הַמְגֻלִּין: 
כְּדֵי שֶׁתֹּאבַד בָּהֶן הַמָּרָה. &lt;המידה&gt; 
רְבִּי יוֹסֵה אוֹמֵר: 
בַּכֵּלִים, כָּל שֶׁהֵן, 
וּבַקַּרְקָעוֹת, אַרְבָּעִים סְאָה. 

<small>ו</small>
נִקּוּרֵי תְאֵנִים, 
וַעֲנָבִים, 
הַקִּשּׁוּאִים, 
וְהַדִּלּוּעִים, 
וְהָאֲבַטִּיחִים, 
וְהַמַּלְפְּפוֹנוֹת, 
אֲפִלּוּ הֵן בַּכַּד, 
אֶחָד גָּדוֹל וְאֶחָד קָטָן, 
אֶחָד תָּלוּשׁ וְאֶחָד מְחֻבָּר, 
כָּל שֶׁיֵּשׁ בּוֹ לֵחָה, אָסוּר.  
וּנְשִׁיכַת הַנָּחָשׁ אֲסוּרָה, 
מִפְּנֵי סַכָּנַת נְפָשׁוֹת. 
הַמְשַׁמֶּרֶת שֶׁלַּיַּיִן אֲסוּרָה מִשּׁוּם גִּלּוּי. 
רְבִּי נְחֶמְיָה מַתִּיר. 

<small>ז</small>
חָבִית שֶׁלִּתְרוּמָה שֶׁנּוֹלַד לָהּ סְפֵק טֻמְאָה, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אִם הָיְתָה מֻנַּחַת בִּמְקוֹם הַתָּרְפָּה, 
יַנִּיחֶנָּה בִמְקוֹם הַמֻּצְנָע; 
אִם הָיְתָה מְגֻלָּה, יְכַסֶּנָּה. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
אִם הָיְתָה מֻנַּחַת בִּמְקוֹם הַמֻּצְנָע, 
יַנִּיחֶנָּה בִמְקוֹם הַתָּרְפָּה; &lt;התרופה&gt;
אִם הָיְתָה מְכֻסָּה, 
יְגַלֶּנָּה.  
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
אַל יְחַדֵּשׁ בָּהּ דָּבָר. 

<small>ח</small>
חָבִית שֶׁנִּשְׁבָּרָה בַגַּת הָעֶלְיוֹנָה, &lt;בְּגַת&gt;
הַתַּחְתּוֹנָה טְמֵאָה, 
מוֹדֶה רְבִּי אֱלִיעֶזֶר וּרְבִּי יְהוֹשֻׁעַ, 
שֶׁאִם יָכוֹל לְהַצִּיל מִמֶּנָּה רְבִיעִית בְּטַהֲרָה, 
יַצִּיל, 
וְאִם לָאו, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
תֵּרֵד וְתִטַּמָּא, 
וְאַל יְטַמְּאֶנָּה בְיָדָיו. 

<small>ט</small>
וְכֵן חָבִית שֶׁלַּשֶּׁמֶן שֶׁנִּשְׁפָּכָה, 
מוֹדֶה רְבִּי אֱלִיעֶזֶר וּרְבִּי יְהוֹשֻׁעַ, 
שֶׁאִם יָכוֹל לְהַצִּיל מִמֶּנָּה רְבִיעִית בְּטַהֲרָה, 
יַצִּיל, 
וְאִם לָאו, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
תֵּרֵד וְתִבָּלַע, 
וְאַל יְבַלְּעֶנָּה בְיָדָיו. 

<small>י</small>
עַל זוֹ וְעַל זוֹ אָמַר רְבִּי יְהוֹשֻׁעַ: 
לֹא זוֹ הִיא תְּרוּמָה שֶׁאֲנִי מֻזְהָר עָלֶיהָ מִלְּטַמְּאָהּ, &lt;שני&gt;
אֶלָּא מִלְּאָכְלָהּ.  
וּבַל תְּטַמְּאָהּ כֵּיצַד? &lt;כיצד מחוק&gt;
הָיָה עוֹבֵר מִמָּקוֹם לְמָקוֹם, 
וְכִכָּרוֹת שֶׁלִּתְרוּמָה בְיָדוֹ, 
אָמַר לוֹ נָכְרִי: 
"תֶּן לִי אַחַת מֵהֶן וַאֲטַמֵּא, 
וְאִם לָאו, הֲרֵי אֲנִי מְטַמֵּא אֶת כֻּלָּם", 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
יְטַמֵּא אֶת כֻּלָּם, 
וְאַל יִתֵּן לוֹ אַחַת מֵהֶן.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
יַנִּיחַ לְפָנָיו אַחַת עַל הַסֶּלַע. 

<small>יא</small>
וְכֵן נָשִׁים שֶׁאָמְרוּ לָהֶם גּוֹיִם: 
תְּנוּ לָנוּ אַחַת מִכֶּם וּנְטַמֵּא, 
וְאִם לָאו, הֲרֵי אָנוּ מְטַמְּאִין אֶת כֻּלְּכֶם, 
יְטַמְּאוּ אֶת כֻּלָּם, 
וְאַל יִמְסְרוּ לָהֶם נֶפֶשׁ אַחַת מִיִּשְׂרָאֵל. 


 


Chapter 9

<small>א</small>
הַזּוֹרֵעַ תְּרוּמָה, 
שׁוֹגֵג, יוֹפַךְ, 
מֵזִיד, יְקַיֵּם.  
אִם הֵבִיאָה שָׁלִישׁ, 
בֵּין שׁוֹגֵג בֵּין מֵזִיד, יְקַיֵּם.  
וּבַפִּשְׁתָּן, 
מֵזִיד, יוֹפַךְ. 

<small>ב</small>
וְחַיֶּבֶת בַּלֶּקֶט וּבַשִּׁכְחָה וּבַפֵּאָה, 
וַעֲנִיֵּי יִשְׂרָאֵל וַעֲנִיֵּי כֹהֲנִים מְלַקְּטִים, 
וַעֲנִיֵּי יִשְׂרָאֵל מוֹכְרִין אֶת שֶׁלָּהֶן לַכֹּהֲנִים 
בִּדְמֵי תְרוּמָה, וְהַדָּמִין שֶׁלָּהֶן.  
אָמַר רְבִּי טַרְפוֹן: 
לֹא יְלַקְּטוּ אֶלָּא עֲנִיֵּי כֹהֲנִים, 
שֶׁמֵּא יִשְׁכְּחוּ וְיִתְּנוּ לְתוֹךְ פִּיהֶם.  
אָמַר לוֹ רְבִּי עֲקִיבָה: 
אִם כֵּן, 
לֹא יְלַקְּטוּ אֶלָּא טְהוֹרִים. 

<small>ג</small>
וְחַיֶּבֶת בַּמַּעַשְׂרוֹת וּבְמַעְשַׂר עָנִי, &lt;וּבְמַעְשֵׂר&gt;
וַעֲנִיֵּי יִשְׂרָאֵל וַעֲנִיֵּי כֹהֲנִים נוֹטְלִים, 
וַעֲנִיֵּי יִשְׂרָאֵל מוֹכְרִין אֶת שֶׁלָּהֶן לַכֹּהֲנִים בִּדְמֵי תְרוּמָה, 
וְהַדָּמִים שֶׁלָּהֶן.  
הַחוֹבֵט מְשֻׁבָּח. 
וְהַדָּשׁ, כֵּיצַד יַעֲשֶׂה? 
תּוֹלֶה כְפֵיפוֹת בְּצַוְּארֵי בְהֵמָה, 
וְנוֹתֵן לְתוֹכָן מֵאוֹתוֹ הַמִּין. 
נִמְצָא לֹא זוֹמֵם אֶת הַבְּהֵמָה, 
וְלֹא מַאֲכִיל אֶת הַתְּרוּמָה. 

<small>ד</small>
גִּדּוּלֵי תְרוּמָה, תְּרוּמָה, 
וְגִּדּוּלֵי גִדּוּלִים, חֻלִּים.  
אֲבָל הַטֶּבֶל, 
וּמַעֲשֵׂר רִאשׁוֹן, 
וּסְפִיחֵי שְׁבִיעִית, 
וּתְרוּמַת חוּץ לָאָרֶץ, 
וְהַמְדֻמָּע, 
וְהַבִּכּוּרִים, 
גִּדּוּלֵיהֶם חֻלִּין.  
גִּדּוּלֵי הֶקְדֵּשׁ וּמַעֲשֵׂר שֵׁנִי, חֻלִּים, 
וּפוֹדֶה אוֹתָם בִּזְמַן זַרְעָן. &lt;זְרָעָן&gt;

<small>ה</small>
מֵאָה לִגְנָה שֶׁלִּתְרוּמָה וְאַחַת שֶׁלַּחֻלִּים, 
כֻּלָּם מֻתָּרִים בְּדָבָר שֶׁזַּרְעוֹ כָלֶה. 
אֲבָל בְּדָבָר שֶׁאֵין זַרְעוֹ כָלֶה, 
אֲפִלּוּ מֵאָה שֶׁלַּחֻלִּין וְאַחַת שֶׁלִּתְרוּמָה, 
כֻּלָּם אֲסוּרִין. 

<small>ו</small>
הַטֶּבֶל, 
גִּדּוּלָיו מֻתָּרִין בְּדָבָר שֶׁזַּרְעוֹ כָלֶה; 
אֲבָל בְּדָבָר שֶׁאֵין זַרְעוֹ כָלֶה, 
גִּדּוּלֵי גִדּוּלִין אֲסוּרִין.  
וְאֵי זֶה הוּא דָבָר שֶׁאֵין זַרְעוֹ כָלֶה? 
כְּגוֹן הַלּוּף, וְהַשּׁוּם וְהַבְּצָלִים.  
רְבִּי יְהוּדָה אוֹמֵר: 
הַשּׁוּם כִּשְׂעוֹרִין. 

<small>ז</small>
הַמְנַכֵּשׁ עִם הַנָּכְרִי בַחֲסִיּוֹת, 
אַף עַל פִּי שֶׁפֵּרוֹתָיו טֶבֶל, 
אוֹכֵל מֵהֶן עֲרַי.  
שִׁתְלֵי תְרוּמָה שֶׁנִּטַּמְּאוּ, 
שְׁתָלָן, טָהָרוּ מִלְּטַמֵּא. &lt;ישתלו&gt; 
וַאֲסוּרִין מִלֹּאכַל עַד שֶׁיָּגֹם הָאֹכֶל.  
רְבִּי יְהוּדָה אוֹמֵר: 
עַד שֶׁיָּגֹם וְיִשְׁנֶה. 





Chapter 10

<small>א</small>
בָּצָל שֶׁנְּתָנוֹ בְתוֹךְ עֲדָשִׁים, 
אִם שָׁלֵם, מֻתָּר; 
וְאִם חִתְּכוֹ, בְּנוֹתֵן טַעַם; 
וּשְׁאָר כָּל הַתַּבְשִׁיל, 
בֵּין שָׁלֵם, בֵּין מְחֻתָּךְ, 
בְּנוֹתֵן טַעַם.  
רְבִּי יְהוּדָה מַתִּיר בְּצַחֲנָה, 
שֶׁאֵינוּ אֶלָּא לִטּוֹל אֶת הַזַּהְמָה. 

<small>ב</small>
תַּפּוּחַ שֶׁרִסְּקוֹ וּנְתָנוֹ לְתוֹךְ עִסָּה וְחִמֵּצָהּ, 
הֲרֵי זוֹ אֲסוּרָה.  
שְׂעוֹרִים שֶׁנָּפְלוּ לְתוֹךְ הַבּוֹר שֶׁלַּמַּיִם, 
אַף עַל פִּי שֶׁהִבְאִישׁוּ מֵימָיו, 
מֵימָיו מֻתָּרִין. 

<small>ג</small>
הָרוֹדֶה פַת חַמָּה, 
וּנְתָנָהּ עַל פִּי חָבִית שֶׁלַּיַּיִן שֶׁלִּתְרוּמָה, 
רְבִּי מֵאִיר אוֹסֵר, 
וּרְבִּי יְהוּדָה מַתִּיר.  
רְבִּי יוֹסֵה מַתִּיר בְּשֶׁלַּחִטִּים, 
וְאוֹסֵר בְּשֶׁלִּשְׂעוֹרִים, 
מִפְּנֵי שֶׁהַשְּׂעוֹרִים שׁוֹאֲבוֹת. 

<small>ד</small>
תַּנּוּר שֶׁהִסִּיקוֹ בִּכְמָן שֶׁלִּתְרוּמָה, 
וְאָפָה בוֹ אֶת הַפַּת, 
הַפַּת מֻתֶּרֶת, 
שֶׁאֵינוּ טַעַם כְּמָן, אֶלָּא רֵיחַ כְּמָן. 

<small>ה</small>
תַּלְתָּן שֶׁנָּפְלָה לְתוֹךְ הַבּוֹר שֶׁלַּיַּיִן, &lt;שֶׁלַּמַּיִם&gt;
בִּתְרוּמָה וּבְמַעֲשֵׂר שֵׁנִי, 
אִם יֵשׁ בַּזֶּרַע כְּדֵי לִתֵּן טַעַם, 
אֲבָל לֹא בָעֵץ; 
וּבַשְּׁבִיעִית וּבְכִלְאֵי הַכֶּרֶם וּבַהֶקְדֵּשׁ, 
אִם יֵשׁ בַּזֶּרַע וּבָעֵץ כְּדֵי לִתֵּן טַעַם. 

<small>ו</small>
מִי שֶׁהָיוּ לוֹ חֲבִלֵּי תַלְתָּן שֶׁלְּכִלְאֵי הַכֶּרֶם, &lt;שֶׁלְכִּלְאֵי&gt;
יִדָּלֵקוּ.  
הָיוּ לוֹ חֲבִלֵּי תַלְתָּן שֶׁלַּטֶּבֶל, 
כּוֹתֵשׁ וּמְחַשֵּׁב כַּמָּה זֶרַע בָּהֶם, 
וּמַפְרִישׁ אֶת הַזֶּרַע, 
וְאֵינוּ צָרִיךְ לְהַפְרִישׁ אֶת הָעֵץ.  
אִם הִפְרִישׁ, לֹא יֹאמַר: 
"אֶכְתֹּשׁ וְאֶטֹּל אֶת הָעֵץ, וְאֶתֵּן אֶת הַזֶּרַע", 
אֶלָּא נוֹתֵן הָעֵץ עִם הַזֶּרַע. 

<small>ז</small>
זֵיתֵי חֻלִּין שֶׁכְּבָשָׁן עִם זֵיתֵי תְרוּמָה, 
פְּצוּעֵי חֻלִּין עִם פְּצוּעֵי תְרוּמָה, 
פְּצוּעֵי חֻלִּין עִם שְׁלֵמֵי תְרוּמָה, 
אוֹ בְמֵי תְרוּמָה, 
אָסוּר. 
אֲבָל שְׁלֵמֵי חֻלִּין עִם פְּצוּעֵי תְרוּמָה, 
מֻתָּר. 

<small>ח</small>
דָּג טָמֵא שֶׁכְּבָשׁוֹ עִם דָּג טָהוֹר: 
כָּל גָּרֵב שֶׁהוּא מַחֲזִיק סָאתַיִם, 
אִם יֶשׁ בּוֹ מִשְׁקַל עֲשָׂרָה זוּז בִּיהוּדָה, 
שֶׁהֵן חָמֵשׁ סְלָעִים בַּגָּלִיל, 
דָּג טָמֵא, צֵירוֹ אָסוּר.  
רְבִּי יְהוּדָה אוֹמֵר: 
רְבִיעִית בְּסָאתַיִם.  
וּרְבִּי יוֹסֵה אוֹמֵר: 
אֶחָד מִשִּׁשָּׁה עָשָׂר בּוֹ. 

<small>ט</small>
חֲגָבִים טְמֵאִים שֶׁנִּכְבָּשׁוּ עִם חֲגָבִים טְהוֹרִין, 
לֹא פָסְלוּ אֶת צֵירָן.  
הֵעִיד רְבִּי צָדוֹק עַל צֵיר חֲגָבִים טְמֵאִים, 
שֶׁהוּא טָהוֹר. 

<small>י</small>
כָּל הַנִּכְבָּשִׁין זֶה עִם זֶה מֻתָּרִים, 
אֶלָּא עִם הָחֵסִית: 
חֵסִית שֶׁלַּחֻלִּין עִם חֵסִית שֶׁלִּתְרוּמָה, 
חֵסִית שֶׁלַּחֻלִּין עִם יָרָק שֶׁלִּתְרוּמָה, 
אָסוּר. 
אֲבָל יָרָק שֶׁלַּחֻלִּין עִם חֵסִית שֶׁלִּתְרוּמָה, 
מֻתָּר. 

<small>יא</small>
רְבִּי יוֹסֵה אוֹמֵר: 
כָּל הַנִּשְׁלָקִים עִם הַתְּרָדִים אֲסוּרִין, 
מִפְּנֵי שֶׁהֵן נוֹתְנִין אֶת הַטַּעַם.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כְּרוּב שֶׁלַּשֶּׁקִי עִם כְּרוּב שֶׁלַּבַּעַל, 
אָסוּר, מִפְּנֵי שֶׁהוּא בּוֹלֵעַ.  
רְבִּי עֲקִיבָה אוֹמֵר: 
כָּל הַמִּתְבַּשְּׁלִין זֶה עִם זֶה מֻתָּרִין, 
אֶלָּא עִם הַבָּשָׂר.  
רְבִּי יוֹחָנָן בֶּן נוּרִי אוֹמֵר: 
הַכָּבֵד אוֹסֶרֶת וְאֵינָה נֶאֱסֶרֶת, 
מִפְּנֵי שֶׁהִיא פוֹלֶטֶת וְאֵינָה בוֹלַעַת. 

<small>יב</small>
בֵּיצָה שֶׁנִּתְבַּשְּׁלָה בְּתַבְלִים אֲסוּרִין, 
אֲפִלּוּ חֶלְמוֹן שֶׁלָּהּ אָסוּר, 
מִפְּנֵי שֶׁהוּא בוֹלֵעַ.  
מֵי כְבָשִׁים וּמֵי שְׁלָקוֹת שֶׁלִּתְרוּמָה, 
אֲסוּרִין לַזָּרִים. 





Chapter 11

<small>א</small>
אֵין נוֹתְנִין דְּבֵלָה וּגְרוֹגְרוֹת לְתוֹךְ הַמּוּרְיָס, 
מִפְּנֵי שֶׁהוּא מְאַבְּדָן. 
אֲבָל נוֹתְנִין אֶת הַיַּיִן לַמּוּרְיָס.  
וְאֵין מְפַטְּמִין אֶת הַשֶּׁמֶן, 
אֲבָל עוֹשִׂים אֶת הַיַּיִן יִנָמֵילִין.  
אֵין מְבַשְּׁלִים יַיִן שֶׁלִּתְרוּמָה, 
מִפְּנֵי שֶׁהוּא מַמְעִיטוֹ.  
רְבִּי יְהוּדָה מַתִּיר, 
מִפְּנֵי שֶׁהוּא מַשְׁבִּיחוֹ. 

<small>ב</small>
דְּבַשׁ תְּמָרִים וְיֵין תַּפּוּחִים וְחֹמֶץ סִתְוָנִיּוֹת, 
וּשְׁאָר כָּל מֵי פֵרוֹת שֶׁלִּתְרוּמָה, 
רְבִּי אֱלִיעֶזֶר מְחַיֵּב קֶרֶן וְחֹמֶשׁ. 
רְבִּי יְהוֹשֻׁעַ פּוֹטֵר.  
רְבִּי אֱלִיעֶזֶר מְטַמֵּא מִשֵּׁם מַשְׁקֶה.  
אָמַר רְבִּי יְהוֹשֻׁעַ: 
לֹא מָנוּ חֲכָמִים שִׁבְעָה מַשְׁקִים בְּמִינֵי פַטָּמִים, &lt;היה כתוב: כמוני&gt;
אֶלָּא אָמְרוּ: 
שִׁבְעָה מַשְׁקִין טְמֵאִין, 
וּשְׁאָר כָּל הַמַּשְׁקִין טְהוֹרִין. 

<small>ג</small>
אֵין עוֹשִׂין תְּמָרִים דְּבַשׁ, 
וְלֹא תַפּוּחִים יַיִן, 
וְלֹא סִתְוָנִיּוֹת חֹמֶץ.  
וּשְׁאָר כָּל הַפֵּרוֹת, 
אֵין מְשַׁנִּין אוֹתָן מִבִּרְיָתָן בִּתְרוּמָה וּבְמַעֲשֵׂר שֵׁנִי, 
אֶלָּא זֵיתִים וַעֲנָבִים בִּלְבַד.  
אֵין סוֹפְגִים אַרְבָּעִים מִשֵּׁם עָרְלָה, 
אֶלָּא עַל הַיּוֹצֵא מִן הַזֵּיתִים וּמִן הָעֲנָבִים.  
וְאֵין מְבִיאִין בִּכּוּרִים מַשְׁקִין, 
אֶלָּא הַיּוֹצֵא מִן הַזֵּיתִים וּמִן הָעֲנָבִים.  
וְאֵינוּ מִטַּמֵּא מִשֵּׁם מַשְׁקֶה, &lt;מְטַמֵּא&gt;
אֶלָּא הַיּוֹצֵא מִן הַזֵּיתִים וּמִן הָעֲנָבִים.  
אֵין מַקְרִיבִים עַל גַּבֵּי הַמִּזְבֵּחַ, 
אֶלָּא הַיּוֹצֵא מִן הַזֵּיתִים וּמִן הָעֲנָבִים. 

<small>ד</small>
עָקְצֵי תְאֵנִים וּגְרוֹגְרוֹת, 
וְהַכַּלֵּיסִים וְהֶחָרוּבִין שֶׁלִּתְרוּמָה, 
אֲסוּרִין לַזָּרִים. 

<small>ה</small>
גַּלְעִנֵּי תְרוּמָה, 
כָּל זְמַן שֶׁהוּא מְכַנְּסָן, אֲסוּרוֹת. 
אִם הִשְׁלִיכָן, מֻתָּרוֹת.  
וְכֵן עַצְמוֹת הַקָּדָשִׁים, 
כָּל זְמַן שֶׁהוּא מְכַנְּסָן, אֲסוּרִין, 
וְאִם הִשְׁלִיכָן, מֻתָּרִין.  
הַמֻּרְסָן מֻתָּר.  
סֻבִּים שֶׁלַּחֲדָשׁוֹת, אֲסוּרוֹת.  
וְשֶׁלַּיְּשָׁנוֹת, מֻתָּרוֹת.  
נוֹהֵג בִּתְרוּמָה כְּדֶרֶךְ שֶׁהוּא נוֹהֵג בַּחֻלִּין.  
הַמְסַלֵּת קַב אוֹ קַבַּיִם לַסְּאָה, 
לֹא יְאַבֵּד אֶת הַשְּׁאָר, 
אֶלָּא יַנִּיחֶנּוּ בִמְקוֹם מֻצְנָע. 

<small>ו</small>
מְגוֹרָה שֶׁפִּנָּה מִמֶּנָּה חִטֵּי תְרוּמָה, 
אֵין מְחַיְּבִין אוֹתוֹ לִהְיוֹת יוֹשֵׁב וּמְלַקֵּט אַחַת אַחַת, 
אֶלָּא מְכַבֵּד כְּדַרְכּוֹ וְנוֹתֵן לְתוֹכָהּ חֻלִּין. 

<small>ז</small>
וְכֵן חָבִית שֶׁלַּשֶּׁמֶן שֶׁנִּשְׁפָּכָה, 
אֵין מְחַיְּבִין אוֹתוֹ לִהְיוֹת יוֹשֵׁב וּמַטְַפִיֵחַ, 
אֶלָּא נוֹהֵג בָּהּ כְּדֶרֶךְ שֶׁהוּא נוֹהֵג בַּחֻלִּין. 

<small>ח</small>
הַמְעָרֶה מִכַּד לְכַד וְנִטֵּף שָׁלוֹשׁ טִפִּים, 
נוֹתֵן לְתוֹכָהּ חֻלִּין.  
הִרְכִּינָהּ וּמִצַּת, 
הֲרֵי זוֹ תְרוּמָה.  
וְכַמָּה תְהֵא בִתְרוּמַת מַעֲשֵׂר שֶׁלִּדְמַי, &lt;בתרומה&gt;
וְיוֹלִיכֶנָּה לַכֹּהֵן? 
אֶחָד מִשְּׁמוֹנָה לִשְׁמִינִית. 

<small>ט</small>
כַּרְשִׁנֵּי תְרוּמָה, 
מַאֲכִילִין אוֹתָן לַבְּהֵמָה, וְלַחַיָּה, וְלַתַּרְנָגְלִין.  
יִשְׂרָאֵל שֶׁסָּכַר פָּרָה מִכֹּהֵן, 
מַאֲכִילָהּ כַּרְשִׁנֵּי תְרוּמָה. 
וְכֹהֵן שֶׁסָּכַר פָּרָה מִיִּשְׂרָאֵל, 
אַף עַל פִּי שֶׁמְּזוֹנוֹתֶיהָ עָלָיו, 
לֹא יַאֲכִילָהּ כַּרְשִׁנֵּי תְרוּמָה.  
יִשְׂרָאֵל שֶׁשָּׁם פָּרָה מִכֹּהֵן, 
לֹא יַאֲכִילָהּ כַּרְשִׁנֵּי תְרוּמָה, 
וְכֹהֵן שֶׁשָּׁם פָּרָה מִיִּשְׂרָאֵל, 
מַאֲכִילָהּ כַּרְשִׁנֵּי תְרוּמָה. 

<small>י</small>
מַדְלִיקִין שֶׁמֶן שְׂרֵפָה בְבָתֵּי כְנֵסָיוֹת, 
וּבְבָתֵּי מִדְרָשׁוֹת, 
וּבְמָבוֹאוֹת אֲפֵלִים, 
וְעַל גַּבֵּי הַחוֹלִין, בִּרְשׁוּת כֹּהֵן.  
בַּת יִשְׂרָאֵל שֶׁנִּשֵּׂאת לַכֹּהֵן, 
וְהִיא לִמּוּדֵת לָבֹא אֵצֶל אָבִיהָ, 
אָבִיהָ מַדְלִיק בִּרְשׁוּתָהּ.  
מַדְלִיקִין בְּבֵית הַמִּשְׁתֶּה, 
אֲבָל לֹא בְבֵית הָאֵבֶל.  
דִּבְרֵי רְבִּי יְהוּדָה.  
רְבִּי יוֹסֵה אוֹמֵר: 
בְּבֵית הָאֵבֶל, 
אֲבָל לֹא בְבֵית הַמִּשְׁתֶּה.  
רְבִּי מֵאִיר אוֹסֵר כָּן וְכָן.  
רְבִּי שִׁמְעוֹן מַתִּיר כָּן וְכָן.
