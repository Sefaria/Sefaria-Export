משנה ברכות



Chapter 1

<small>א</small>
מֵאֵמָּתַי קוֹרִין אֶת שְׁמַע בַּעֲרָבִים? 
מִשָּׁעָה שֶׁהַכֹּהֲנִים נִכְנָסִים לֹאכַל בִּתְרוּמָתָן, 
עַד סוֹף הָאַשְׁמֹרֶת הָרִאשׁוֹנָה. 
דִּבְרֵי רְבִּי אֱלִיעֶזֶר. 
וַחֲכָמִים אוֹמְרִים: 
עַד חֲצוֹת. 
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
עַד שֶׁיַּעֲלֶה עַמּוּד הַשַּׁחַר. 

<small>ב</small>
מַעֲשֶׂה שֶׁבָּאוּ בָנָיו מִבֵּית הַמִּשְׁתֶּה; 
אָמְרוּ לוֹ: 
"לֹא קָרִינוּ אֶת שְׁמַע."
 אָמַר לָהֶם: 
"אִם לֹא עָלָה עַמּוּד הַשַּׁחַר, 
מֻתָּרִין אַתֶּם לִקְרוֹת". 

<small>ג</small>
וְלֹא זוֹ בִלְבַד, 
אֶלָּא כָל שֶׁאָמְרוּ חֲכָמִים "עַד חֲצוֹת", 
מִצְוָתָן עַד שֶׁיַּעֲלֶה עַמּוּד הַשַּׁחַר.

<small>ד</small> 
הֶקְטֵר חֲלָבִים וְאֵבָרִים, [וַאֲכִילַת פְּסָחִים] 
מִצְוָתָן עַד שֶׁיַּעֲלֶה עַמּוּד הַשַּׁחַר; 

<small>ה</small>
כָּל הַנֶּאֱכָלִים לְיוֹם אֶחָד, 
מִצְוָתָן עַד שֶׁיַּעֲלֶה עַמּוּד הַשַּׁחַר. 

<small>ו</small>
אִם כֵּן, לָמָּה אָמְרוּ חֲכָמִים "עַד חֲצוֹת"? 
אֶלָּא לְהַרְחִיק אֶת הָאָדָם מִן הָעֲבֵירָה.

<small>ז</small>
מֵאֵמָּתַי קוֹרִין אֶת שְׁמַע בַּשְּׁחָרִים? 
מִשֶּׁיַּכִּירוּ בֵין תְּכֵלֶת לַלָּבָן; 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
בֵּין תְּכֵלֶת לַכָּרַתָּן, 
עַד הָנֵץ הַחַמָּה. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
עַד שָׁלֹשׁ שָׁעוֹת, 
שֶׁכֵּן דֶּרֶךְ בְּנֵי מְלָכִין לַעֲמֹד בְּשָׁלוֹשׁ שָׁעוֹת. 
הַקּוֹרֵא מִכָּן וְאֶלָּךְ לֹא הִפְסִיד, 
כְּאָדָם שֶׁהוּא קוֹרֵא בַתּוֹרָה.

<small>ח</small>
בֵּית שַׁמַּי אוֹמְרִים: 
בָּעֶרֶב כָּל הָאָדָם יַטּוּ וְיִקְרוּ, 
וּבַבֹּקֶר יַעַמֹדוּ, שֶׁנֶּאֱמַר: 
(דְּבָרִים ו, ז) "בְּשָׁכְבְּךָ וּבְקוּמֶךָ". 
בֵּית הֶלֵּל אוֹמְרִים: 
כָּל הָאָדָם קוֹרִין כְּדַרְכָּן, שֶׁנֶּאֱמַר: 
(דְּבָרִים ו, ז) "וּבְלֶכְתְּךָ בַדֶּרֶךְ". 
אִם כֵּן, לָמָּה נֶאֱמַר "בְּשָׁכְבְּךָ וּבְקוּמֶךָ"? 
אֶלָּא, בְּשָׁעָה שֶׁדֶּרֶךְ בְּנֵי אָדָם שׁוֹכְבִים, 
וּבְשָׁעָה שֶׁדֶּרֶךְ בְּנֵי אָדָם עוֹמְדִין. 

<small>ט</small>
אָמַר רֶבִּי טַרְפוֹן: 
אֲנִי הָיִיתִי בָא בַדֶּרֶךְ, 
וְהִטֵּיתִי לִקְרוֹת, כְּדִבְרֵי בֵית שַׁמַּי, 
וְסִכַּנְתִּי בְעַצְמִי מִפְּנֵי הַלֵּסְטִים. 
אָמְרוּ לוֹ: 
כְּדַי הָיִיתָ לָחוּב בְּעַצְמָךְ, 
שֶׁעָבַרְתָּ עַל דִּבְרֵי בֵית הֶלֵּל.

<small>י</small>
בַּשַּׁחַר מְבָרֵךְ שְׁתַּיִם לְפָנֶיהָ וְאַחַת לְאַחֲרֶיהָ, 
וּבָעֶרֶב שְׁתַּיִם לְפָנֶיהָ וּשְׁתַּיִם לְאַחֲרֶיהָ, 
אַחַת אֲרֻכָּה, וְאַחַת קְצָרָה. 
מָקוֹם שֶׁאָמְרוּ לְהַאֲרִיךְ, אֵינוּ רַשַּׁי לְקַצֵּר; 
לְקַצֵּר, אֵינוּ רַשַּׁי לְהַאֲרִיךְ. 
לַחְתֹּם, אֵינוּ רַשַּׁי שֶׁלֹּא לַחְתֹּם; 
וְשֶׁלֹּא לַחְתֹּם, אֵינוּ רַשַּׁי לַחְתֹּם.

<small>יא</small>
מַזְכִּירִין יְצִיאַת מִצְרַיִם בַּלֵּילוֹת. 
אָמַר רְבִּי אֶלְעָזָר בֶּן עֲזַרְיָה: 
הֲרֵי אֲנִי כְבֶן שִׁבְעִים שָׁנָה, 
וְלֹא זָכִיתִי שֶׁתֵּאָמֵר יְצִיאַת מִצְרַיִם בַּלֵּילוֹת, 
עַד שֶׁדְּרָשָׁהּ בֶּן זוֹמָה, שֶׁנֶּאֱמַר: 
(דְּבָרִים טז, ג) "לְמַעַן תִּזְכֹּר אֶת יוֹם צֵאתְךָ מֵאֶרֶץ מִצְרַיִם, 
כֹּל יְמֵי חַיֶּיךָ", 
"יְמֵי חַיֶּיךָ" הַיָּמִים, 
"כֹּל יְמֵי חַיֶּיךָ" הַלֵּילוֹת. 
וַחֲכָמִים אוֹמְרִים: 
"יְמֵי חַיֶּיךָ" הָעוֹלָם הַזֶּה, 
"כֹּל יְמֵי חַיֶּיךָ" לְהָבִיא אֶת ימוֹת הַמָּשִׁיחַ.




Chapter 2

<small>א</small>
הָיָה קוֹרֵא בַתּוֹרָה, וְהִגִּיעַ זְמַן הַמִּקְרָא; 
אִם כִּוֵּן אֶת לִבּוֹ, יָצָא, 
וְאִם לָאו, לֹא יָצָא. 
וּבַפְּרָקִים, 
שׁוֹאֵל מִפְּנֵי הַכָּבוֹד וּמֵשִׁיב, 
וּבָאֶמְצַע, 
שׁוֹאֵל מִפְּנֵי הַיִּרְאָה וּמֵשִׁיב, 
דִּבְרֵי רְבִּי מֵאִיר. 
רְבִּי יְהוּדָה אוֹמֵר: 
בָּאֶמְצַע, 
שׁוֹאֵל מִפְּנֵי הַיִּרְאָה, וּמֵשִׁיב מִפְּנֵי הַכָּבוֹד, 
וּבַפְּרָקִים, 
שׁוֹאֵל מִפְּנֵי הַכָּבוֹד, וּמֵשִׁיב שָׁלוֹם לְכָל הָאָדָם.

אֵלּוּ הֵן בֵּין הַפְּרָקִים: 
בֵּין בְּרָכָה הָרִאשׁוֹנָה לַשְּׁנִיָּה, 
וּבֵין שְׁנִיָּה לִ"שְׁמַע", 
וּבֵין "שְׁמַע" לִ"וְהָיָה אִם שָׁמֹעַ", 
בֵּין "וְהָיָה אִם שָׁמֹעַ" לְ"וַיֹּאמֶר", 
בֵּין "וַיֹּאמֶר" לֶ"אֱמֶת וְיַצִּיב". 
רְבִּי יְהוּדָה אוֹמֵר: 
בֵּין "וַיֹּאמֶר" לֶ"אֱמֶת וְיַצִּיב", 
לֹא יַפְסִיק. 

<small>ג</small>
אָמַר רֶבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה: 
וְלָמָּה קָדְמָה "שְׁמַע" לִ"וְהָיָה אִם שָׁמֹעַ" ? 
אֶלָּא יְקַבֵּל עָלָיו מַלְכוּת שָׁמַיִם תְּחִלָּה, 
וְאַחַר כָּךְ יְקַבֵּל עָלָיו עֹל מִצְוֹת. 
"וְהָיָה אִם שָׁמֹעַ" לְ"וַיֹּאמֶר" ? 
שֶׁ"וְהָיָה אִם שָׁמֹעַ" נוֹהֵג בַּיּוֹם וּבַלַּיְלָה, 
"וַיֹּאמֶר" אֵינוּ נוֹהֵג אֶלָּא בַיּוֹם.

<small>ד</small>
הַקּוֹרֵא אֶת שְׁמַע וְלֹא הִשְׁמִיעַ לְאָזְנוֹ, 
יָצָא. 
רְבִּי יוֹסֵה אוֹמֵר: 
לֹא יָצָא. 
קָרָא וְלֹא דִקְדֵּק בְּאוֹתוֹתֶיהָ, 
רְבִּי יוֹסֵה אוֹמֵר: 
יָצָא. 
רְבִּי יְהוּדָה אוֹמֵר: 
לֹא יָצָא. 
הַקּוֹרֵא לְמַפְרֵעַ לֹא יָצָא. 
קָרָא וְטָעָה, יַחְזֹר לִמְקוֹם שֶׁטָּעָה. 
 
<small>ה</small>
הָאֻמָּנִים קוֹרִין בְּרֹאשׁ הָאִילָּן אוֹ בְרֹאשׁ הַנִּדְבָּךְ, 
מַה שֶּׁאֵינָן רַשָּׁיִים לַעֲשׁוֹת כֵּן בַּתְּפִלָּה.

<small>ו</small>
חָתָן פָּטוּר מִקִּרְיַת שְׁמַע בְּלַיְלָה הָרִאשׁוֹן, 
וְעַד מוֹצָאֵי שַׁבָּת, אִם לֹא עָשָׂה מַעֲשֶׂה. 
מַעֲשֶׂה בְרַבָּן גַּמְלִיאֵל שֶׁקָּרָא בְלַיְלָה שֶׁנָּשָׂא. 
אָמְרוּ לוֹ: 
לֹא לִמַּדְתָּנוּ שֶׁחָתָן פָּטוּר מִקִּרְיַת שְׁמַע בְּלַיְלָה הָרִאשׁוֹן? 
אָמַר לָהֶן: 
אֵינִי שׁוֹמֵעַ לָכֶם 
לְבַטֵּל מִמֶּנִּי מַלְכוּת שָׁמַיִם שָׁעָה אַחַת.

<small>ז</small>
רָחַץ בְּלַיְלָה הָרִאשׁוֹן שֶׁמֵּתָה אִשְׁתּוֹ. 
אָמְרוּ לוֹ: 
לֹא לִמַּדְתָּנוּ שֶׁאָבֵל אָסוּר לִרְחוֹץ? 
אָמַר לָהֶם: 
אֵינִי כִשְׁאָר כָּל הָאָדָם, אַסְטְנֵס אָנִי.

<small>ח</small>
וּכְשֶׁמֵּת טְבִי עַבְדּוֹ, 
קִבֵּל עָלָיו תַּנְחוּמִים. 
אָמְרוּ לוֹ תַלְמִידָיו: 
לֹא לִמַּדְתָּנוּ שֶׁאֵין מְקַבְּלִין תַּנְחוּמִין עַל הָעֲבָדִים? 
אָמַר לָהֶם: 
אֵין טְבִי עַבְדִּי כִשְׁאָר כָּל הָעֲבָדִים, 
כָּשֵׁר הָיָה.

<small>ט</small>
חָתָן, 
אִם רוֹצֶה לִקְרוֹת אֶת שְׁמַע בְּלַיְלָה הָרִאשׁוֹן, 
קוֹרֵא. 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
לֹא כָל הָרוֹצֶה לִטּוֹל אֶת הַשֵּׁם יִטֹּל.



 

Chapter 3

<small>א</small>
מִי שֶׁמֵּתוֹ מֻטָּל לְפָנָיו, 
פָּטוּר מִקִּרְיַת שְׁמַע וּמִן הַתְּפִלִּין. 
נוֹשְׂאֵי הַמִּטָּה וְחִלּוּפֵיהֶן וְחִלּוּפֵי חִלּוּפֵיהֶן, 
אֶת שֶׁלִּפְנֵי הַמִּטָּה, וְאֶת שֶׁלְּאַחַר הַמִּטָּה; 
אֶת שֶׁלַּמִּטָּה צֹרֶךְ בָּהֶן פְּטוּרִין, 
וְאֶת שֶׁאֵין לַמִּטָּה צֹרֶךְ בָּהֶן חַיָּבִין. 
אֵלּוּ וָאֵלּוּ פְּטוּרִין מִן הַתְּפִלָּה.

<small>ב</small>
קָבְרוּ אֶת הַמֵּת וְחָזָרוּ, 
אִם יְכוּלִין לְהַתְחִיל וְלִגְמוֹר 
עַד שֶׁלֹּא יַגִּיעוּ לַשּׁוּרָה, 
יַתְחִילוּ; 
וְאִם לָאו, 
לֹא יַתְחִילוּ. 
הָעוֹמְדִין בַּשּׁוּרָה, 
הַפְּנִימִין פְּטוּרִין, 
וְהַחִיצוֹנִין חַיָּבִין.

<small>ג</small>
נָשִׁים וַעֲבָדִים וּקְטַנִּים, 
פְּטוּרִין מִקִּרְיַת שְׁמַע וּמִן הַתְּפִלִּין, 
וְחַיָּבִין בַּתְּפִלָּה וּבַמְּזוּזָה וּבְבִרְכַּת הַמָּזוֹן.

<small>ד</small>
בַּעַל קֶרִי, 
מְהַרְהֵר בְּלִבּוֹ וְאֵינוּ מְבָרֵךְ, 
לֹא לְפָנֶיהָ וְלֹא לְאַחֲרֶיהָ; 
וְעַל הַמָּזוֹן, 
מְבָרֵךְ לְאַחֲרָיו, וְאֵינוּ מְבָרֵךְ לְפָנָיו. 
רֶבִּי יְהוּדָה אוֹמֵר: 
מְבָרֵךְ לִפְנֵיהֶם וּלְאַחֲרֵיהֶם.

<small>ה</small>
הָיָה עוֹמֵד בַּתְּפִלָּה, 
וְנִזְכַּר שֶׁהוּא בַּעַל קֶרִי, 
לֹא יַפְסִיק, אֶלָּא יְקַצֵּר. 
יָרַד לִטְבּוֹל, 
אִם יָכוֹל לַעֲלוֹת וּלְהִתְכַּסּוֹת, 
וְלִקְרוֹת עַד שֶׁלֹּא תָנֵץ הַחַמָּה, 
יַעֲלֶה וְיִתְכַּסֶּה וְיִקְרֵא; 
וְאִם לָאו, 
יִתְכַּסֶּה בַמַּיִם וְיִקְרֵא. 
אֲבָל לֹא יִתְכַּסֶּה 
לֹא בְמַיִם הָרָעִים וְלֹא בְמֵי הַמִּשְׁרָה, 
עַד שֶׁיָּטִיל לְתוֹכָן מַיִם. 
כַּמָּה יַרְחִיק מֵהֶן, מִן הַצּוֹאָה? 
אַרְבַּע אַמּוֹת.

<small>ו</small>
זָב שֶׁרָאָה קֶרִי, 
וְנִדָּה שֶׁפָּלְטָה שִׁכְבַת זֶרַע, 
וְהַמְשַׁמֶּשֶׁת שֶׁרָאַת נִדָּה, 
צְרִיכִין טְבִילָה; 
רֶבִּי יְהוּדָה פוֹטֵר.




Chapter 4

<small>א</small>
תְּפִלַּת הַשַּׁחַר, עַד חֲצוֹת; 
רֶבִּי יְהוּדָה אוֹמֵר: 
עַד אַרְבַּע שָׁעוֹת. 
תְּפִלַּת הַמִּנְחָה עַד הָעֶרֶב; 
רֶבִּי יְהוּדָה אוֹמֵר: 
עַד פְּלַג הַמִּנְחָה. 
תְּפִלַּת הָעֶרֶב, אֵין לָהּ קֶבַע. 
וְשֶׁלַּמּוּסָפִים כָּל הַיּוֹם; 
רְבִּי יְהוּדָה אוֹמֵר: 
עַד שֶׁבַע שָׁעוֹת.

<small>ב</small>
רֶבִּי נְחוֹנְיָא בֶן הַקָּנָה הָיָה מִתְפַּלֵּל 
בִּכְנֵיסָתוֹ לְבֵית הַמִּדְרָשׁ וּבִיצִיאָתוֹ תְפִלָּה קְצָרָה. 
אָמְרוּ לוֹ: 
מַה מָּקוֹם לִתְפִלָּה זוֹ? 
אָמַר לָהֶם: 
"בִּכְנֵיסָתִי אֲנִי מִתְפַּלֵּל 
שֶׁלֹּא תֶאֱרַע תַּקָּלָה עַל יָדִי, 
וּבִיצִיאָתִי אֲנִי נוֹתֵן הוֹדָיָה עַל חֶלְקִי."

<small>ג</small>
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
בְּכָל יוֹם מִתְפַּלֵּל אָדָם שְׁמוֹנֶה עֶשְׂרֵה. 
רֶבִּי יְהוֹשֻׁעַ אוֹמֵר: 
מֵעֵין שְׁמוֹנֶה עֶשְׂרֵה. 
רֶבִּי עֲקִיבָה אוֹמֵר: 
אִם שָׁגְרָה תְפִלָּתוֹ בְפִיו, 
מִתְפַּלֵּל שְׁמוֹנֶה עֶשְׂרֵה; 
וְאִם לָאו, 
מֵעֵין שְׁמוֹנֶה עֶשְׂרֵה.

<small>ד</small>
רֶבִּי אֱלִיעֶזֶר אוֹמֵר: 
הָעוֹשֶׂה תְפִלָּתוֹ קֶבַע, 
אֵין תְּפִלָּתוֹ תַחֲנוּנִים. 
רֶבִּי יְהוֹשֻׁעַ אוֹמֵר: 
הַמְּהַלֵּךְ בִּמְקוֹם סַכָּנָה, 
מִתְפַּלֵּל תְּפִלָּה קְצָרָה מֵעֵין שְׁמוֹנֶה עֶשְׂרֵה, 
וְאוֹמֵר: 
"הוֹשַׁע הַשֵּׁם אֶת עַמְּךָ אֵת יִשְׂרָאֵל; 
כָּל פָּרָשַׁת הָעִבּוּר צָרְכֵיהֶם לְפָנֶיךָ; 
בָּרוּךְ אַתָּה יי שׁוֹמֵעַ תְּפִלָּה".

<small>ה</small>
הָיָה רָכוּב עַל הַחֲמוֹר, 
יֵרֵד; 
אִם אֵינוּ יָכוֹל לֵירֵד, 
יַחְזִיר אֶת פָּנָיו; 
אִם אֵינוּ יָכוֹל לְהַחְזִיר אֶת פָּנָיו, 
יְכַוֵּן אֶת לִבּוֹ כְנֶגֶד בֵּית קֹדֶשׁ הַקֳּדָשִׁים.

<small>ו</small>
הָיָה יוֹשֵׁב בִּסְפִינָה, אוֹ בְקָרוֹן, אוֹ בְאַסְדָּא, 
יְכַוֵּן אֶת לִבּוֹ כְנֶגֶד בֵּית קֹדֶשׁ הַקֳּדָשִׁים.

רֶבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר: 
אֵין תְּפִלַּת הַמּוּסָפִים אֶלָּא בַחֲבַר עִיר. 
וַחֲכָמִים אוֹמְרִים: 
בַּחֲבַר עִיר וְשֶׁלֹּא בַחֲבַר עִיר. 
רֶבִּי יְהוּדָה אוֹמֵר מִשְּׁמוֹ: 
כָּל מָקוֹם שֶׁיֶּשׁ חֲבַר עִיר, 
הַיָּחִיד פָּטוּר מִתְּפִלַּת הַמּוּסָפִין.




Chapter 5

<small>א</small>
אֵין עוֹמְדִין לְהִתְפַּלֵּל אֶלָּא מִתּוֹךְ כֹּבֶד רֹאשׁ. 
חֲסִידִים הָרִאשׁוֹנִים 
הָיוּ שׁוֹהִים שָׁעָה אַחַת וּמִתְפַּלְּלִים, 
כְּדֵי שֶׁיְּכַוְּנוּ אֶת לִבָּם לַמָּקוֹם. 
וַאֲפִלּוּ הַמֶּלֶךְ שׁוֹאֵל בִּשְׁלוֹמוֹ, 
לֹא יְשִׁיבֶנּוּ; 
אֲפִלּוּ נָחָשׁ כָּרוּךְ עַל עֲקֵבוֹ, 
לֹא יַפְסִיק.

<small>ב</small>
מַזְכִּירִין גְּבוּרוֹת גְּשָׁמִים בִּתְּחַיַּת הַמֵּתִים, 
וְשׁוֹאֲלִים גְּשָׁמִים בְּבִרְכַּת הַשָּׁנִים, 
וְהַבְדָּלָה בְחוֹנֵן הַדַּעַת. 
רֶבִּי עֲקִיבָה אוֹמֵר: 
אוֹמְרָהּ בְּרָכָה רְבִיעִית לְעַצְמָהּ. 
רֶבִּי אֱלִיעֶזֶר אוֹמֵר: 
בַּהוֹדָיָה.

<small>ג</small>
הָאוֹמֵר: 
"עַל קַן צִפּוֹר יַגִּיעוּ רַחֲמֶיךָ", 
וְ"עַל טוֹב יִזָּכֵר שְׁמֶךָ", 
"מוֹדִים, מוֹדִים" 
מְשַׁתְּקִים אוֹתוֹ. 
הָעוֹבֵר לִפְנֵי הַתֵּבָה וְטָעָה, 
יַעֲבוֹר אַחֵר תַּחְתָּיו, 
לֹא יְהֵא סָרְבָן בְּאוֹתָהּ שָׁעָה. 
מְנַיִן הוּא מַתְחִיל? 
מִתְּחִלַּת הַבְּרָכָה שֶׁטָּעָה זֶה.

<small>ד</small>
הָעוֹבֵר לִפְנֵי הַתֵּבָה 
לֹא יַעֲנֶה אַחַר הַכֹּהֲנִים "אָמֵן", 
מִפְּנֵי הַטֵּרוּף. 
וְאִם אֵין שָׁם כֹּהֵן אֶלָּא הוּא, 
לֹא יִשָּׂא אֶת כַּפָּיו. 
וְאִם אַבְטָחָתוֹ הִיא 
שֶׁהוּא נוֹשֵׂא אֶת כַּפָּיו וְחוֹזֵר לִתְפִלָּתוֹ, 
רַשַּׁי.

<small>ה</small>
הַמִּתְפַּלֵּל וְטָעָה, 
סִימָן רַע לוֹ; 
וְאִם שְׁלִיחַ צִבּוּר הוּא, 
סִימָן רַע לְשׁוֹלְחָיו, 
שֶׁשְּׁלוּחוֹ שֶׁלָּאָדָם כְּמוֹתוֹ. 
אָמְרוּ עָלָיו, עַל רֶבִּי חֲנִינָא בֶן דּוֹסָה, 
שֶׁהָיָה מִתְפַּלֵּל עַל הַחוֹלִין, וְאוֹמֵר: 
"זֶה חַי", וְ"זֶה מֵת". 
אָמְרוּ לוֹ: 
מְנַיִן אַתָּה יוֹדֵעַ? 
אָמַר לָהֶם: 
אִם שָׁגְרָה תְפִלָּתִי בְּפִי, 
יוֹדֵעַ אֲנִי שֶׁהוּא מְקֻבָּל; 
וְאִם לָאו, 
יוֹדֵעַ אֲנִי שֶׁהוּא מְטֹרָף.




Chapter 6

<small>א</small>
כֵּיצַד מְבָרְכִין עַל הַפֵּרוֹת? 
עַל פֵּרוֹת הָאִילָן הוּא אוֹמֵר: 
"בּוֹרֵא פְרִי הָעֵץ", 
חוּץ מִן הַיַּיִן, 
שֶׁעַל הַיַּיִן הוּא אוֹמֵר: 
"בּוֹרֵא פְרִי הַגֶּפֶן". 
וְעַל פֵּרוֹת הָאָרֶץ הוּא אוֹמֵר: 
"בּוֹרֵא פְרִי הָאֲדָמָה", 
חוּץ מִן הַפַּת, 
שֶׁעַל הַפַּת הוּא אוֹמֵר: 
"הַמּוֹצִיא לֶחֶם מִן הָאָרֶץ". 
וְעַל הַיְרָקוֹת הוּא אוֹמֵר: 
"בּוֹרֵא פְרִי הָאֲדָמָה". 
רֶבִּי יְהוּדָה אוֹמֵר: 
"בּוֹרֵא מִינֵי דְשָׁאִים".

<small>ב</small>
בֵּרַךְ עַל פֵּרוֹת הָאִילָן "בּוֹרֵא פְרִי הָאֲדָמָה", 
יָצָא; 
וְעַל פֵּרוֹת הָאָרֶץ "בּוֹרֵא פְרִי הָעֵץ", 
לֹא יָצָא. 
וְעַל כֻּלָּם, אִם אָמַר "שֶׁהַכֹּל נִהְיָה בִדְבָרוֹ", 
יָצָא.

<small>ג</small>
עַל דָּבָר שֶׁאֵין גִּדּוּלָיו מִן הָאָרֶץ אוֹמֵר: 
"שֶׁהַכֹּל נִהְיָה בִדְבָרוֹ". 
עַל הַחֹמֶץ וְעַל הַגּוֹבַאי וְעַל הַנּוֹבְלוֹת הוּא אוֹמֵר: 
"שֶׁהַכֹּל נִהְיָה בִדְבָרוֹ". 
רֶבִּי יְהוּדָה אוֹמֵר: 
כָּל שֶׁהוּא מִין קְלָלָה, אֵין מְבָרְכִין עָלָיו.

<small>ד</small>
הָיוּ לְפָנָיו מִינִים הַרְבֵּה, 
רֶבִּי יְהוּדָה אוֹמֵר: 
אִם יֵשׁ בֵּינֵיהֶן מִמִּין שִׁבְעָה, 
עָלָיו הוּא מְבָרֵךְ. 
וַחֲכָמִים אוֹמְרִים: 
מְבָרֵךְ עַל אֵיזֶה מֵהֶן שֶׁיִּרְצֶה.

<small>ה</small>
בֵּרַךְ עַל הַיַּיִן שֶׁלִּפְנֵי הַמָּזוֹן, 
פָּטַר אֶת הַיַּיִן שֶׁלְּאַחַר הַמָּזוֹן. 
בֵּרַךְ עַל הַפַּרְפֶּרֶת שֶׁלִּפְנֵי הַמָּזוֹן, 
פָּטַר אֶת הַפַּרְפֶּרֶת שֶׁלְּאַחַר הַמָּזוֹן. 
בֵּרַךְ עַל הַפַּת, 
פָּטַר אֶת הַפַּרְפְּרָיוֹת; 
עַל הַפַּרְפְּרָיוֹת, 
לֹא פָטַר אֶת הַפַּת. 
בֵּית שַׁמַּי אוֹמְרִים: 
אַף לֹא מַעֲשֵׂה קְדֵרָה.

<small>ו</small>
הָיוּ יוֹשְׁבִין, 
כָּל אֶחָד וְאֶחָד מְבָרֵךְ לְעַצְמוֹ; 
הֵסַבּוּ, 
אֶחָד מְבָרֵךְ לְכֻלָּם. 
בָּא לָהֶם יַיִן בְּתוֹךְ הַמָּזוֹן, 
כָּל אֶחָד וְאֶחָד מְבָרֵךְ לְעַצְמוֹ; 
לְאַחַר הַמָּזוֹן, 
אֶחָד מְבָרֵךְ לְכֻלָּם, וְאוֹמֵר עַל הַמִּגְמָר, 
אַף עַל פִּי שֶׁאֵין מְבִיאִין אֶת הַמִּגְמָר 
אֶלָּא לְאַחַר הַסְּעוֹדָה.

<small>ז</small>
הֵבִיאוּ לוֹ מָלִיחַ כַּתְּחִלָּה וּפַת עִמּוֹ, 
מְבָרֵךְ עַל הַמָּלִיחַ וּפוֹטֵר אֶת הַפַּת, 
שֶׁהַפַּת טְפֵלָה לוֹ. 
זֶה הַכְּלָל: 
כָּל שֶׁהוּא עִקָּר וְעִמּוֹ טְפֵלָה, 
מְבָרֵךְ עַל הָעִקָּר וּפוֹטֵר אֶת הַטְּפֵלָה.

<small>ח</small>
אָכַל תְּאֵנִים וַעֲנָבִים וְרִמּוֹנִים, 
מְבָרֵךְ עֲלֵיהֶן שָׁלוֹשׁ בְּרָכוֹת. 
דִּבְרֵי רַבָּן גַּמְלִיאֵל. 
וַחֲכָמִים אוֹמְרִים: 
בְּרָכָה אַחַת. 
רֶבִּי עֲקִיבָה אוֹמֵר: 
אֲפִלּוּ שֶׁאָכַל שֶׁלֶק, וְהוּא מְזוֹנוֹ, 
מְבָרֵךְ אַחֲרָיו שָׁלֹשׁ בְּרָכוֹת. 
הַשּׁוֹתֶה מַיִם לְצִמְאוֹ, 
אוֹמֵר "שֶׁהַכֹּל נִהְיָה בִדְבָרוֹ". 
רֶבִּי טַרְפוֹן אוֹמֵר: 
"בּוֹרֵא נְפָשׁוֹת רַבּוֹת וְחֶסְרוֹנָן".




Chapter 7

<small>א</small>
שְׁלֹשָׁה שֶׁאָכְלוּ כְאַחַת חַיָּבִין לְזַמֵּן. 
אָכַל דְּמַאי, 
וּמַעֲשֵׂר רִאשׁוֹן שֶׁנִּטְּלָה תְרוּמָתוֹ, 
וּמַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁנִּפְדּוּ, 
הַשַּׁמָּשׁ שֶׁאָכַל כַּזַּיִת, 
וְהַכּוּתִי, 
מְזַמְּנִין עֲלֵיהֶן. 

<small>ב</small>
אֲבָל אָכַל טֶבֶל, 
וּמַעֲשֵׂר רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְרוּמָתוֹ, 
וּמַעֲשֵׂר שֵׁנִי וְהֶקְדֵּשׁ שֶׁלֹּא נִפְדּוּ, 
הַשַּׁמָּשׁ שֶׁאָכַל פָּחוּת מִכַּזַּיִת, 
וְהַנָּכְרִי, 
אֵין מְזַמְּנִין עֲלֵיהֶן.

נָשִׁים, וַעֲבָדִים, וּקְטַנִּים, 
אֵין מְזַמְּנִין עֲלֵיהֶן. 
עַד כַּמָּה הֵן מְזַמְּנִין? 
עַד כַּזַּיִת. 
רֶבִּי יְהוּדָה אוֹמֵר: 
עַד כַּבֵּיצָה.

<small>ד</small>
כֵּיצַד מְזַמְּנִין? 
בִּשְׁלֹשָׁה אוֹמֵר "נְבָרֵךְ"; 
בִּשְׁלֹשָׁה וָהוּא אוֹמֵר "בָּרְכוּ". 
בַּעֲשָׂרָה אוֹמֵר "נְבָרֵךְ לֵאלֹהֵינוּ"; 
בַּעֲשָׂרָה וָהוּא אוֹמֵר "בָּרְכוּ". 
אֶחָד עֲשָׂרָה וְאֶחָד עֶשֶׂר רִבּוֹא. 

<small>ה</small>
בְּמֵאָה אוֹמֵר "נְבָרֵךְ לַיי אֱלֹהֵינוּ"; 
בְּמֵאָה וָהוּא אוֹמֵר "בָּרְכוּ". 
בָּאֶלֶף אוֹמֵר "נְבָרֵךְ לַיי אֱלֹהֵינוּ אֱלֹהֵי יִשְׂרָאֵל"; 
בָּאֶלֶף וָהוּא אוֹמֵר "בָּרְכוּ". 
בָּרִבּוֹא אוֹמֵר "נְבָרֵךְ לַיי אֱלֹהֵינוּ אֱלֹהֵי יִשְׂרָאֵל, 
אֱלֹהֵי צְבָאוֹת"; 
בָּרִבּוֹא וָהוּא אוֹמֵר "בָּרְכוּ". 
כְּעִנְיָן שֶׁהוּא מְבָרֵךְ, כָּךְ עוֹנִין אַחֲרָיו: 
"בָּרוּךְ יי אֱלֹהֵינוּ אֱלֹהֵי יִשְׂרָאֵל, אֱלֹהֵי צְבָאוֹת, 
יוֹשֵׁב הַכְּרוּבִים, עַל הַמָּזוֹן שֶׁאָכַלְנוּ". 
רֶבִּי יוֹסֵה הַגָּלִילִי אוֹמֵר: 
לְפִי רֹב הַקָּהָל הֵם מְבָרְכִין, 
שֶׁנֶּאֱמַר: (תְּהִלִּים סח, כז) 
"בְּמַקְהֵלוֹת בָּרְכוּ אֱלהִים אֲדֹנָי מִמְּקוֹר יִשְׂרָאֵל". 

<small>ו</small>
אָמַר רֶבִּי עֲקִיבָה: 
מַה מָּצִינוּ בְּבֵית הַכְּנֶסֶת, 
אֶחָד מְרֻבִּים וְאֶחָד מְמֻעָטִים, 
אוֹמֵר "בָּרְכוּ אֶת יי". 
רֶבִּי יִשְׁמָעֵאל אוֹמֵר: 
"בָּרְכוּ אֶת יי הַמְּבֹרָךְ".

<small>ז</small>
שְׁלֹשָׁה שֶׁאָכְלוּ כְאַחַת, 
אֵינָן רַשָּׁיִים לֵחָלֵק; 
וְכֵן אַרְבָּעָה, וְכֵן חֲמִשָּׁה. 
שִׁשָּׁה נֶחְלָקִין, עַד עֲשָׂרָה. 
וַעֲשָׂרָה אֵינָן נֶחְלָקִין, עַד שֶׁיְּהוּ עֶשְׂרִים.

<small>ח</small>
שְׁתֵּי חֲבוֹרוֹת שֶׁהָיוּ אוֹכְלוֹת בְּבַיִת אֶחָד, 
בִּזְמַן שֶׁמִּקְצָתָן רוֹאִים אֵלּוּ אֶת אֵלּוּ, 
הֲרֵי אֵלּוּ מִצְטָרְפִין לַזִּמּוּן. 
אִם לָאו, 
אֵלּוּ מְזַמְּנִין לְעַצְמָן וְאֵלּוּ מְזַמְּנִין לְעַצְמָן. 
אֵין מְבָרְכִין עַל הַיַּיִן, 
עַד שֶׁיִּתֵּן לְתוֹכוֹ מַיִם, 
דִּבְרֵי רֶבִּי אֱלִיעֶזֶר. 
וַחֲכָמִים אוֹמְרִין: 
מְבָרְכִין.




Chapter 8

<small>א</small>
אֵלּוּ דְבָרִים שֶׁבֵּין בֵּית שַׁמַּי לְבֵית הֶלֵּל בִּסְעוֹדָה. 
בֵּית שַׁמַּי אוֹמְרִין: 
מְבָרֵךְ עַל הַיּוֹם, וְאַחַר כָּךְ מְבָרֵךְ עַל הַיַּיִן. 
וּבֵית הֶלֵּל אוֹמְרִין: 
מְבָרֵךְ עַל הַיַּיִן, וְאַחַר כָּךְ מְבָרֵךְ עַל הַיּוֹם.

<small>ב</small>
בֵּית שַׁמַּי אוֹמְרִין: 
נוֹטְלִים לַיָּדַיִם, וְאַחַר כָּךְ מוֹזְגִין אֶת הַכּוֹס. 
וּבֵית הֶלֵּל אוֹמְרִין: 
מוֹזְגִין אֶת הַכּוֹס, וְאַחַר כָּךְ נוֹטְלִים לַיָּדַיִם.

<small>ג</small>
בֵּית שַׁמַּי אוֹמְרִין: 
מְכַבְּדִין אֶת הַבַּיִת, וְאַחַר כָּךְ נוֹטְלִים לַיָּדַיִם. 
וּבֵית הֶלֵּל אוֹמְרִין: 
נוֹטְלִים לַיָּדַיִם, וְאַחַר כָּךְ מְכַבְּדִין אֶת הַבַּיִת.

<small>ד</small>
בֵּית שַׁמַּי אוֹמְרִין: 
מְקַנֵּחַ אֶת יָדָיו בַּמַּפָּא, וּמַנִּיחָהּ עַל הַשֻּׁלְחָן. 
וּבֵית הֶלֵּל אוֹמְרִין: 
עַל הַכֶּסֶת.

<small>ה</small>
בֵּית שַׁמַּי אוֹמְרִין: 
נֵר, וּמָזוֹן, וּבְשָׂמִים, וְהַבְדָּלָה. 
וּבֵית הֶלֵּל אוֹמְרִין: 
נֵר, וּבְשָׂמִים, וּמָזוֹן, וְהַבְדָּלָה. 
בֵּית שַׁמַּי אוֹמְרִין: 
"שֶׁבָּרָא מְאוֹר הָאֵשׁ". 
וּבֵית הֶלֵּל אוֹמְרִין: 
"בּוֹרֵא מְאוֹרֵי הָאֵשׁ".

<small>ו</small>
אֵין מְבָרְכִין 
לֹא עַל הַנֵּר וְלֹא עַל הַבְּשָׂמִים שֶׁלַּגּוֹיִם; 
וְלֹא עַל הַנֵּר וְלֹא עַל הַבְּשָׂמִים שֶׁלַּמֵּתִים; 
וְלֹא עַל הַנֵּר וְלֹא עַל הַבְּשָׂמִים שֶׁלִּפְנֵי עֲבוֹדָה זָרָה. 
אֵין מְבָרְכִין עַל הַנֵּר עַד שֶׁיֵּאוֹתוּ לְאוֹרוֹ.

<small>ז</small>
מִי שֶׁאָכַל וְשָׁכַח וְלֹא בֵרַךְ, 
בֵּית שַׁמַּי אוֹמְרִין: 
יַחְזֹר לִמְקוֹמוֹ וִיבָרֵךְ. 
וּבֵית הֶלֵּל אוֹמְרִין: 
יְבָרֵךְ בִּמְקוֹם שֶׁנִּזְכַּר. 
עַד אֶמָּתַי הוּא מְבָרֵךְ? 
עַד כְּדֵי שֶׁיִּתְאַכֵּל הַמָּזוֹן בְּמֵעָיו.

<small>ח</small>
בָּא לָהֶן יַיִן לְאַחַר הַמָּזוֹן, 
וְאֵין שָׁם אֶלָּא אוֹתוֹ הַכּוֹס, 
בֵּית שַׁמַּי אוֹמְרִין: 
מְבָרֵךְ עַל הַיַּיִן, וְאַחַר כָּךְ מְבָרֵךְ עַל הַמָּזוֹן. 
וּבֵית הֶלֵּל אוֹמְרִין: 
מְבָרֵךְ עַל הַמָּזוֹן, וְאַחַר כָּךְ מְבָרֵךְ עַל הַיַּיִן. 
עוֹנִין אָמֵן אַחַר יִשְׂרָאֵל הַמְבָרֵךְ, 
וְאֵין עוֹנִין אָמֵן אַחַר הַכּוּתִי הַמְבָרֵךְ, 
עַד שֶׁיִּשְׁמַע כָּל הַבְּרָכָה.




Chapter 9

<small>א</small>
הָרוֹאֶה מָקוֹם שֶׁנֶּעֱשׁוּ בוֹ נִסִּים לְיִשְׂרָאֵל, אוֹמֵר: 
"בָּרוּךְ שֶׁעָשָׂה נִסִּים לַאֲבוֹתֵינוּ בַּמָּקוֹם הַזֶּה". 
מָקוֹם שֶׁנֶּעְקְרָה מִמֶּנּוּ עֲבוֹדָה זָרָה, אוֹמֵר: 
"בָּרוּךְ שֶׁעָקַר עֲבוֹדָה זָרָה מֵאַרְצֵנוּ".

<small>ב</small>
עַל הַזִּיקִים, וְעַל הַזּוֹעוֹת, וְעַל הַבְּרָקִים, 
וְעַל הָרְעָמִים, וְעַל הָרוּחוֹת, הוּא אוֹמֵר: 
"בָּרוּךְ שֶׁכּחוֹ מָלֵא עוֹלָם". 
עַל הֶהָרִים, וְעַל הַגְּבָעוֹת, 
וְעַל הַיַּמִּים, וְעַל הַנְּהָרוֹת, 
וְעַל הַמִּדְבָּרוֹת, הוּא אוֹמֵר: 
"בָּרוּךְ עוֹשֶׂה בְרֵאשִׁית". 
רֶבִּי יְהוּדָה אוֹמֵר: 
הָרוֹאֶה אֶת הַיָּם הַגָּדוֹל אוֹמֵר: 
"בָּרוּךְ שֶׁעָשָׂה אֶת הַיָּם הַגָּדוֹל", 
בִּזְמַן שֶׁהוּא רוֹאֶהוּ לִפְרָקִים. 
עַל הַגְּשָׁמִים, 
וְעַל בְּשׁוֹרוֹת טוֹבוֹת הוּא אוֹמֵר: 
"בָּרוּךְ הַטּוֹב וְהַמֵּטִיב"; 
וְעַל שְׁמוּעוֹת הָרָעוֹת הוּא אוֹמֵר: 
"בָּרוּךְ דַּיַּן הָאֱמֶת".

<small>ג</small>
בָּנָה בַיִת חָדָשׁ, 
וְקָנָה כֵלִים חֲדָשִׁים, אוֹמֵר: 
"בָּרוּךְ שֶׁהִגִּיעָנוּ לִזְמַן הַזֶּה". 
מְבָרֵךְ עַל הָרָעָה מֵעֵין עַל הַטּוֹבָה, 
וְעַל הַטּוֹבָה מֵעֵין עַל הָרָעָה. 
הַצּוֹעֵק לְשֶׁעָבַר, 
הֲרֵי זוֹ תְפִלַּת שָׁוְא. 

<small>ד</small>
כֵּיצַד? 
הָיְתָה אִשְׁתּוֹ מְעֻבֶּרֶת, וְאָמַר: 
"יְהִי רָצוֹן שֶׁתֵּלֵד אִשְׁתִּי זָכָר", 
הֲרֵי זוֹ תְפִלַּת שָׁוְא. 
הָיָה בָא בַדֶּרֶךְ, 
וְשָׁמַע קוֹל צְוָחוֹת בָּעִיר, וְאָמַר: 
"יְהִי רָצוֹן שֶׁלֹּא יְהוּא אֵלּוּ בְתוֹךְ בֵּיתִי", 
הֲרֵי זוֹ תְפִלַּת שָׁוְא.

הַנִּכְנָס לִכְרַךְ מִתְפַּלֵּל שְׁתַּיִם, 
אַחַת בִּכְנֵיסָתוֹ וְאַחַת בִּיצִיאָתוֹ. 
בֶּן עַזַּי אוֹמֵר: 
אַרְבַּע, שְׁתַּיִם בִּכְנֵיסָתוֹ וּשְׁתַּיִם בִּיצִיאָתוֹ. 
וְנוֹתֵן הוֹדָיָה לְשֶׁעָבַר, 
וְצוֹעֵק לֶעָתִיד לָבֹא.

<small>ה</small>
חַיָּב לְבָרֵךְ עַל הָרָעָה, 
כַּשֵּׁם שֶׁהוּא מְבָרֵךְ עַל הַטּוֹבָה, 
שֶׁנֶּאֱמַר: (דברים ו,ה) 
"וְאָהַבְתָּ אֵת יי אֱלהֶיךָ, 
בְּכָל לְבָבְךָ וּבְכָל נַפְשְׁךָ וּבְכָל מְאֹדֶךָ". 
"בְּכָל לְבָבְךָ", בִּשְׁנֵי יְצָרֶיךָ, 
בְּיֵצֶר טוֹב וּבְיֵצֶר רָע; 
"בְּכָל נַפְשְׁךָ", אֲפִלּוּ הוּא נוֹטֵל אֶת נַפְשֶׁךָ; 
"בְּכָל מְאֹדֶךָ", בְּכָל מָמוֹנֶךָ. 
דָּבָר אַחֵר: 
"בְּכָל מְאֹדֶךָ", 
בְּכָל מִדָּה וּמִדָּה שֶׁהוּא מוֹדֵד לָךְ, 
בַּכֹּל הֱוֵי מוֹדֶה לוֹ בְּכָל מְאֹד מְאֹד. 

<small>ו</small>
לֹא יָקֵל אָדָם אֶת רֹאשׁוֹ כְנֶגֶד שַׁעַר הַמִּזְרָח, 
שֶׁהוּא מְכֻוָּן כְּנֶגֶד בֵּית קֹדֶשׁ הַקֳּדָשִׁים. 
לֹא יִכָּנֵס לְהַר הַבַּיִת בְּמַקְלוֹ וּבְמַנְעַלּוֹ, 
וּבַאֲפֻנְדָּתוֹ, וּבַאֲבַק שֶׁעַל רַגְלוֹ, 
וְלֹא יַעֲשֶׂנּוּ קַפַּנְדַּרְיָה, 
וּרְקִיקָה מִקַּל וָחֹמֶר. 

<small>ז</small>
כָּל חוֹתַם הַבְּרָכוֹת שֶׁהָיוּ בַמִּקְדָּשׁ הָיוּ: 
"מִן הָעוֹלָם". 
מִשֶּׁקִּלְקְלוּ הַמִּינִים וְאָמְרוּ: 
"אֵין עוֹלָם אֶלָּא אֶחָד", 
הִתְקִינוּ שֶׁיְּהוּא אוֹמְרִין: 
"מִן הָעוֹלָם וְעַד הָעוֹלָם". 
וְהִתְקִינוּ שֶׁיְּהֵא אָדָם שׁוֹאֵל אֶת שְׁלוֹם חֲבֵרוֹ בַשֵּׁם, 
שֶׁנֶּאֱמַר: (רות ב,ד) 
"וְהִנֵּה בֹעַז בָּא מִבֵּית לֶחֶם, 
וַיֹּאמֶר לַקּוֹצְרִים יי עִמָּכֶם, 
וַיֹּאמְרוּ לוֹ, יְבָרֶכְךָ יי". 
וְאוֹמֵר: (שופטים ו,יב) 
"יי עִמְּךָ גִּבּוֹר הֶחָיִל". 
וְאוֹמֵר: (משלי כג,כב) 
"אַל תָּבוּז כִּי זָקְנָה אִמֶּךָ". 
(תהלים קיט,קכו) "עֵת לַעֲשׁוֹת לַיי הֵפֵרוּ תוֹרָתֶךָ". 
רֶבִּי נָתָן אוֹמֵר: 
הֵפֵרוּ תוֹרָתֶךָ, עֵת לַעֲשׁוֹת לַיי.
