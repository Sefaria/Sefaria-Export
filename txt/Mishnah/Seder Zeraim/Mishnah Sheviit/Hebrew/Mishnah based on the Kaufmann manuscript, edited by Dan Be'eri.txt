משנה שביעית



Chapter 1

<small>א</small>
עַד אֶמָּתַי חוֹרְשִׁין שְׂדֵה הָאִילָן עֶרֶב שְׁבִיעִית? 
בֵּית שַׁמַּי אוֹמְרִים: 
כָּל זְמַן שֶׁהוּא יָפֶה לַפֶּרִי; 
בֵּית הֶלֵּל אוֹמְרִים: 
עַד הָעֲצֶרֶת. 
וּקְרוֹבִים דִּבְרֵי אֵלּוּ לִהְיוֹת כְּדִבְרֵי אֵלּוּ.

אֵי זוֹ הִיא שְׂדֵה הָאִילָן? 
כָּל שְׁלֹשָׁה אִילָנוֹת לְבֵית סְאָה, 
אִם רְאוּיִים לַעֲשׁוֹת כִּכַּר דְּבֵלָה 
שֶׁלְּשִׁשִּׁים מָנֶה בָאִיטַלְקִי, 
חוֹרְשִׁין כָּל בֵּית סְאָה בִשְׁבִילָן; 
פָּחוּת מִכֵּן, 
אֵין חוֹרְשִׁין לָהֶן 
אֶלָּא מְלֹא הָאוֹרֶה וְסַלּוֹ חוּצָה לוֹ.

<small>ב</small>
אֶחָד אִילַן סְרָק וְאֶחָד אִילַן מַאֲכָל, 
רוֹאִין אוֹתָן כְּאִלּוּ הֵן תְּאֵנִים: 
אִם רְאוּיִין לַעֲשׁוֹת כִּכַּר דְּבֵלָה 
שֶׁלְּשִׁשִּׁים מָנֶה בָאִיטַלְקִי, 
חוֹרְשִׁין כָּל בֵּית סְאָה בִשְׁבִילָן; 
פָּחוּת מִכֵּן, 
אֵין חוֹרְשִׁין לָהֶן אֶלָּא צָרְכָן.

<small>ג</small>
הָיָה אֶחָד עוֹשֶׂה כִכַּר דְּבֵלָה, 
אוֹ שְׁנַיִם עוֹשִׂין, 
אֵין חוֹרְשִׁין לָהֶן אֶלָּא צָרְכָן, 
עַד שֶׁיְּהוּא מִשְּׁלֹשָׁה וְעַד תִּשְׁעָה. 
הָיוּ עֲשָׂרָה, מֵעֲשָׂרָה וּלְמַעְלָן, 
בֵּין עוֹשִׂין וּבֵין שֶׁאֵינָן עוֹשִׂין, 
חוֹרְשִׁין כָּל בֵּית סְאָה בִשְׁבִילָן. 
שֶׁנֶּאֱמַר: (שמות לד, כא) 
"בֶּחָרִישׁ וּבַקָּצִיר תִּשְׁבּת". 

<small>ה</small>
אֵין צֹרֶךְ לוֹמַר חָרִישׁ וְקָצִיר שֶׁלִּשְׁבִיעִית, 
אֶלָּא חָרִישׁ שֶׁלְּעֶרֶב שְׁבִיעִית שֶׁהוּא נִכְנָס בַּשְּׁבִיעִית, 
וְקָצִיר שֶׁלִּשְׁבִיעִית שֶׁהוּא יוֹצֵא עִם מוֹצָאֵי שְׁבִיעִית. 
רְבִּי יִשְׁמָעֵאל אוֹמֵר: 
מָה חָרִישׁ רְשׁוּת, אַף קָצִיר רְשׁוּת, 
יָצָא קְצִיר הָעֹמֶר שֶׁהוּא חוֹבָה.

<small>ו</small>
שְׁלֹשָׁה אִילָנוֹת שֶׁלִּשְׁלֹשָׁה אֲנָשִׁים, 
הֲרֵי אֵלּוּ מִצְטָרְפִים, 
וְחוֹרְשִׁים כָּל בֵּית סְאָה בִשְׁבִילָן. 
כַּמָּה יְהֵא בֵינֵיהֶם? 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
כְּדֵי שֶׁיְּהֵא הַבָּקָר עוֹבֵר בְּכֵלָיו.

<small>ז</small>
עֶשֶׂר נְטִיעוֹת מְפֻזָּרוֹת בְּתוֹךְ בֵּית סְאָה, 
חוֹרְשִׁים כָּל בֵּית סְאָה עַד רֹאשׁ הַשָּׁנָה. 
הָיוּ עֲשׁוּיוֹת שׁוּרָה אוֹ מֻקָּפוֹת עֲטָרָה, 
אֵין חוֹרְשִׁין לָהֶן אֶלָּא צָרְכָן.

<small>ח</small>
הַנְּטִיעוֹת וְהַדִּלּוּעִים מִצְטָרְפִין לְתוֹךְ בֵּית סְאָה. 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
עֲשָׂרָה דִלּוּעִים לְבֵית סְאָה, 
חוֹרְשִׁין כָּל בֵּית סְאָה עַד רֹאשׁ הַשָּׁנָה.

<small>ט</small>
עַד אֶמָּתַי נִקְרְאוּ נְטִיעוֹת? 
רְבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹמֵר: 
עַד שֶׁיָּחוּלוּ; 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
עַד תֵּשַׁע שָׁנִים; 
רְבִּי עֲקִיבָה אוֹמֵר: 
נְטִיעָה כִשְׁמָהּ. 
אִילָן שֶׁנִּגְמַם וְהוֹצִיא חֲלִפִּים, 
מִטֶּפַח וּלְמַטָּן, כִּנְטִיעָה, 
מִטֶּפַח וּלְמַעְלָן, כָּאִילָן. 
דִּבְרֵי רְבִּי שִׁמְעוֹן.




Chapter 2

<small>א</small>
עַד אֶמָּתַי חוֹרְשִׁין בִּשְׂדֵה הַלָּבָן עֶרֶב שְׁבִיעִית? 
עַד שֶׁתִּכְלֶה הַלֵּחָה: 
כָּל זְמַן שֶׁבְּנֵי אָדָן חוֹרְשִׁין 
לִטַּע בַּמִּקְשָׁאוֹת וּבַמִּדְלָעוֹת. 
אָמַר רְבִּי שִׁמְעוֹן: 
נָתַתָּ תוֹרַת כָּל אֶחָד וְאֶחָד בְּיָדוֹ; 
אֶלָּא בִשְׂדֵה הַלָּבָן, עַד הַפֶּסַח, 
וּבִשְׂדֵה הָאִילָן עַד הָעֲצֶרֶת.

<small>ב</small>
מְזַבְּלִין וּמְעַדְּרִין בַּמִּקְשָׁאוֹת וּבַמִּדְלָעוֹת 
עַד רֹאשׁ הַשָּׁנָה. 
וְכֵן בֵּית הַשְּׁלָחִים. 
מְיַבְּלִים, מְפָרְקִים, וּמְאָבְקִים, 
וּמְעַשְּׁנִים עַד רֹאשׁ הַשָּׁנָה. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אַף נוֹטֵל הוּא אֶת הֶעָלֶה 
מִן הָאֶשְׁכּוֹל בַּשְּׁבִיעִית.

<small>ג</small>
מְסַקְּלִים עַד רֹאשׁ הַשָּׁנָה. 
מְקַרְסְמִין, מְזָרְדִין, 
וּמְפַסְּלִים עַד רֹאשׁ הַשָּׁנָה. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
כְּזֵרוּדָהּ וּכְפִסּוּלָהּ שֶׁלַּחֲמִישִׁית, 
כֵּן שֶׁלַּשִּׁשִּׁית. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כָּל זְמַן שֶׁאֲנִי רַשַּׁי בַּעֲבוֹדַת הָאִילָן, 
רַשַּׁי אֲנִי בְפִסּוּלוֹ.

<small>ד</small>
מְזַהֲמִין אֶת הַנְּטִיעוֹת, 
וְכוֹרְכִין אוֹתָן, וְקוֹטְמִין אוֹתָם, 
וְעוֹשִׂין לָהֶן בָּתִּים, 
וּמַשְׁקִין אוֹתָן עַד רֹאשׁ הַשָּׁנָה. 
רְבִּי לְעָזָר בִּרְבִּי צָדוֹק אוֹמֵר: 
אַף מַשְׁקֶה הוּא אֶת הַלּוּף בַּשְּׁבִיעִית, 
אֲבָל לֹא עַל הָעִקָּר.

<small>ה</small>
סָכִים אֶת הַפַּגִּים וּמְנַקְּבִין אוֹתָן 
עַד רֹאשׁ הַשָּׁנָה. 
פַּגֵּי עֶרֶב שְׁבִיעִית שֶׁנִּכְנְסוּ לַשְּׁבִיעִית, 
וְשֶׁלַּשְּׁבִיעִית שֶׁיָּצְאוּ לְמוֹצָאֵי שְׁבִיעִית, 
לֹא סָכִים וְלֹא מְנַקְּבִין אוֹתָן. 
רְבִּי יְהוּדָה אוֹמֵר: 
מָקוֹם שֶׁנָּהֲגוּ לָסוּךְ, אֵינָן סָכִים, 
מִפְּנֵי שֶׁהִיא עֲבוֹדָה; 
מָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לָסוּךְ, סָכִים. 
רְבִּי שִׁמְעוֹן מַתִּיר בָּאֵלּוּ, 
מִפְּנֵי שֶׁהוּא רַשַּׁי בַּעֲבוֹדַת הָאִילָן.

<small>ו</small>
אֵין נוֹטְעִין, אֵין מַבְרִיכִין, 
וְאֵין מַרְכִּיבִין עֶרֶב שְׁבִיעִית, 
פָּחוּת מִשְּׁלֹשִׁים יוֹם לִפְנֵי רֹאשׁ הַשָּׁנָה; 
אִם נָטַע אוֹ הִבְרִיךְ אוֹ הִרְכִּיב, יַעֲקֹר. 
רְבִּי יְהוּדָה אוֹמֵר: 
כָּל הַרְכָּבָה שֶׁאֵינָהּ קוֹלֶטֶת לִשְׁלֹשָׁה יָמִים, 
אֵינָה קוֹלֶטֶת. 
רְבִּי יוֹסֵה וּרְבִּי שִׁמְעוֹן אוֹמְרִים: 
לִשְׁתֵּי שַׁבָּתוֹת.

<small>ז</small>
הָאֹרֶז וְהַדֹּחַן וְהַפְּרָגִים וְהַשַּׁמְשְׁמִים 
שֶׁהִשְׁרִישׁוּ לִפְנֵי רֹאשׁ הַשָּׁנָה, 
מִתְעַשְּׂרִים לְשֶׁעָבַר, וּמֻתָּרִים בַּשְּׁבִיעִית; 
אִם לָאו, 
אֲסוּרִין בַּשְּׁבִיעִית וּמִתְעַשְּׂרִין לְשָׁנָה הַבָּאָה.

<small>ח</small>
רְבִּי שִׁמְעוֹן שֵׁזוּרִי אוֹמֵר: 
פּוּל מִצְרִי שֶׁזְּרָעוֹ לְזֶרַע מִתְּחִלָּה כַיּוֹצֵא בָהֶן. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אֲפוּנִים הַגַּמְלוֹנִים כַיּוֹצֵא בָהֶן. 
רְבִּי אֶלְעָזָר אוֹמֵר: 
אֲפוּנִין הַגַּמְלוֹנִין שֶׁתִּרְמְלוּ לִפְנֵי רֹאשׁ הַשָּׁנָה.

<small>ט</small>
הַבְּצָלִים הַסָּרִיסִים וּפוּל מִצְרִי, 
שֶׁמָּנַע מֵהֶן מַיִם שְׁלֹשִׁים יוֹם לִפְנֵי רֹאשׁ הַשָּׁנָה, 
מִתְעַשְּׂרִין לְשֶׁעָבַר וּמֻתָּרִים בַּשְּׁבִיעִית; 
וְאִם לָאו, 
אֲסוּרִין בַּשְּׁבִיעִית וּמִתְעַשְּׂרִין לְשָׁנָה הַבָּאָה. 

<small>י</small>
וְשֶׁלַּבַּעַל, שֶׁמָּנַע מֵהֶן שְׁתֵּי עוֹנוֹת. 
דִּבְרֵי רְבִּי מֵאִיר. 
וַחֲכָמִים אוֹמְרִים: 
שָׁלוֹשׁ.

<small>יא</small>
הַדִּלּוּעִים שֶׁקִּיְּמָן לְזֶרַע, 
אִם הִקְשׁוּ לִפְנֵי רֹאשׁ הַשָּׁנָה, 
וְנִפְסָלוּ מֵאֹכֶל אָדָן, 
מֻתָּר לְקַיְּמָן בַּשְּׁבִיעִית; 
וְאִם לָאו, 
אָסוּר לְקַיְּמָן בַּשְּׁבִיעִית. 
הַתְּמָרוֹת שֶׁלָּהֶן אֲסוּרוֹת בַּשְּׁבִיעִית. 
מְרַבְּצִים בְּעָפָר לָבָן. 
דִּבְרֵי רְבִּי שִׁמְעוֹן. 
רְבִּי אֱלִיעֶזֶר בֶּן יַעֲקב אוֹסֵר. 
מְמָרְסִים בָּאֹרֶז בַּשְּׁבִיעִית. 
דִּבְרֵי רְבִּי שִׁמְעוֹן. 
אֲבָל לֹא מְכַסְּחִים.




Chapter 3

<small>א</small>
מֵאֶמָּתַי מוֹצִיאִין זְבָלִים לְאַשְׁפַּתּוֹת? 
מִשֶּׁיִּפְסְקוּ עוֹבְדֵי עֲבוֹדָה. 
דִּבְרֵי רְבִּי מֵאִיר. 
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁיִּיבַשׁ הַמָּתוֹק. 
רְבִּי יוֹסֵה אוֹמֵר: 
מִשֶּׁיִּקָּשֵׁר.

עַד כַּמָּה מְזַבְּלִים? 
עַד שָׁלוֹשׁ שָׁלוֹשׁ אַשְׁפַּתּוֹת 
לְבֵית סְאָה שֶׁלְּעֶשֶׂר עֶשֶׂר מַשְׁפְּלוֹת, 
וְשֶׁלְּלֶתֶךְ לֶתֶךְ. 
מוֹסִיפִין עַל הַמַּשְׁפְּלוֹת, 
וְאֵין מוֹסִיפִין עַל הָאַשְׁפַּתּוֹת. 
רְבִּי יוּדָה אוֹמֵר: 
אַף עַל הָאַשְׁפַּתּוֹת.

<small>ב</small>
עוֹשֶׂה אָדָם אֶת שָׂדֵהוּ 
שָׁלוֹשׁ שָׁלוֹשׁ אַשְׁפַּתּוֹת לְבֵית סְאָה. 
יָתֵר מִכֵּן, (מַחֲצִיב). 
דִּבְרֵי רְבִּי שִׁמְעוֹן. 
וַחֲכָמִים אוֹסְרִין, 
עַד שֶׁיַּעֲמִיק שְׁלֹשָׁה, 
אוֹ עַד שֶׁיַּגְבִּיהַּ שְׁלֹשָׁה. 
עוֹשֶׂה אָדָם אֶת זִבְלוֹ אוֹצָר. 
רְבִּי מֵאִיר אוֹסֵר, 
עַד שֶׁיַּעֲמִיק שְׁלֹשָׁה, 
אוֹ עַד שֶׁיַּגְבִּיהַּ שְׁלֹשָׁה. 
הָיָה לוֹ דָבָר מְמֻעָט, 
מוֹסִיף עָלָיו וְהוֹלֵךְ. 
רְבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹסֵר, 
עַד שֶׁיַּעֲמִיק שְׁלֹשָׁה, 
אוֹ עַד שֶׁיַּגְבִּיהַּ שְׁלֹשָׁה, 
אוֹ עַד שֶׁיִּתֵּן עַל הַסֶּלַע.

<small>ג</small>
הַמְדַיֵּר אֶת שָׂדֵהוּ 
וְעוֹשֶׂה סַהַר לְבֵית סָאתַיִם, 
עוֹקֵר שָׁלוֹשׁ רוּחוֹת וּמַנִּיחַ אֶת הָאֶמְצָעִית, 
נִמְצָא מְדַיֵּר בֵּית אַרְבַּעַת סְאִים. 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
בֵּית שְׁמוֹנַת סְאִין. 
הָיְתָה כָל שָׂדֵהוּ בֵית אַרְבַּעַת סְאִים, 
מְשַׁיֵּר מִמֶּנּוּ מִקְצָת, מִפְּנֵי מַרְאִית הָעַיִן, 
וּמוֹצִיאִין מִן הַסַּהַר 
וְנוֹתֵן לְתוֹךְ שָׂדֵהוּ כְּדֶרֶךְ הַמְזַבְּלִים.

<small>ד</small>
לֹא יִפְתַּח אָדָם מַחְצֵב כַּתְּחִלָּה לְתוֹךְ שָׂדֵהוּ, 
עַד שֶׁיְּהוּ בוֹ שָׁלוֹשׁ מָרְבְּיוֹת, 
שֶׁהֵן שָׁלֹשׁ עַל שָׁלוֹשׁ עַל רוּם שָׁלוֹשׁ, 
שֵׁעוּרָם עֶשְׂרִים וָשֶׁבַע אֲבָנִים.

<small>ה</small>
גָּדֵר שֶׁיֶּשׁ בּוֹ עֶשֶׂר אֲבָנִים שֶׁלְּמַשְּׂוּי שְׁנַיִם שְׁנַיִם, 
הֲרֵי אֵלּוּ יִנָּטֵלוּ. 
שֵׁעוּר גָּדֵר עֲשָׂרָה טְפָחִין. 
פָּחוּת מִכֵּן, מַחְצֵב, 
וְגוֹמְמוֹ עַד פָּחוּת מֵהָאָרֶץ טֶפַח. 

<small>ו</small>
בַּמֶּה דְבָרִים אֲמוּרִים? 
מִתּוֹךְ שֶׁלּוֹ; 
אֲבָל מִתּוֹךְ שֶׁלַּחֲבֵרוֹ, 
מַה שֶּׁהוּא רוֹצֶה יִטֹּל. 
בַּמֶּה דְבָרִים אֲמוּרִים? 
בִּזְמַן שֶׁלֹּא הִתְחִיל בּוֹ מֵעֶרֶב שְׁבִיעִית, 
אֲבָל אִם הִתְחִיל בּוֹ מֵעֶרֶב שְׁבִיעִית, 
מַה שֶּׁהוּא רוֹצֶה יִטֹּל.

<small>ז</small>
אֲבָנִים שֶׁזִּעְזְעַתָּן הַמַּחֲרֵשָׁה, 
אוֹ שֶׁהָיוּ מְכֻסּוֹת וְנִתְגַּלּוּ, 
אִם יֵשׁ בָּהֶם שְׁתַּיִם שֶׁלְּמַשְּׂוּי שְׁנַיִם שְׁנַיִם, 
הֲרֵי אֵלּוּ יִנָּטֵלוּ. 
הַמְסַקֵּל אֶת שָׂדֵהוּ, 
נוֹטֵל אֶת הָעֶלְיוֹנוֹת, 
וּמַנִּיחַ אֶת הַנּוֹגְעוֹת בָּאָרֶץ. 
וְכֵן גַּרְגֵּשׁ שֶׁלִּצְרוֹרוֹת, 
נוֹטֵל אֶת הָעֶלְיוֹנוֹת 
וּמַנִּיחַ אֶת הַנּוֹגְעוֹת בָּאָרֶץ. 
אִם יֵשׁ תַּחְתֵּיהֶם סֶלַע אוֹ קַשׁ, 
הֲרֵי אֵלּוּ יִנָּטֵלוּ.

<small>ח</small>
אֵין בּוֹנִים מַדְרֵגוֹת עַל פִּי הַגֵּאָיוֹת 
עֶרֶב שְׁבִיעִית מִשֶּׁפָּסְקוּ גְשָׁמִים, 
מִפְּנֵי שֶׁהוּא מַתְקִינָן לַשְּׁבִיעִית; 
אֲבָל בּוֹנֶה הוּא בַשְּׁבִיעִית מִשֶּׁפָּסְקוּ גְשָׁמִים, 
מִפְּנֵי שֶׁהוּא מַתְקִינָן לְמוֹצָאֵי שְׁבִיעִית. 
וְלֹא יִסְמֹךְ בֶּעָפָר, 
אֲבָל עוֹשֶׂהוּ חַיִץ. 
כָּל אֶבֶן שֶׁהוּא יָכוֹל לִפְשֹׁט אֶת יָדוֹ וְלִטְּלָהּ, 
הֲרֵי זוֹ תִנָּטֵל.

<small>ט</small>
אַבְנֵי כָתֵף בָּאוֹת מִכָּל מָקוֹם. 
וְהַמְּקַבְּלָן מֵבִיאָן מִכָּל מָקוֹם. 
וְאֵלּוּ הֵן אַבְנֵי כָתֵף? 
כָּל שֶׁאֵינָה יְכוּלָה לְהִנָּטֵל בְּאַחַת יָד. 
דִּבְרֵי רְבִּי מֵאִיר. 
רְבִּי יוֹסֵה אוֹמֵר: 
אַבְנֵי כָתֵף כִּשְׁמָן, 
כָּל שֶׁהֵן נִטָּלוֹת שְׁתַּיִם וְשָׁלוֹשׁ עַל הַכָּתֵף.

<small>י</small>
הַבּוֹנֶה גָדֵר בֵּינוֹ לִרְשׁוּת הָרַבִּים, 
מֻתָּר לְהַעְמִיק עַד הַסֶּלַע. 
מַה יַּעֲשֶׂה בֶעָפָר? 
צוֹבְרוֹ בִרְשׁוּת הָרַבִּים וּמַתְקִינוֹ. 
דִּבְרֵי רְבִּי יְהוֹשֻׁעַ. 
רְבִּי עֲקִיבָה אוֹמֵר: 
כְּדֶרֶךְ שֶׁאֵין מְקַלְקְלִין בִּרְשׁוּת הָרַבִּים, 
כָּךְ לֹא יְתַקֵּן. 
מַה יַּעֲשֶׂה בֶעָפָר? 
צוֹבְרוֹ לְתוֹךְ שָׂדֵהוּ כְּדֶרֶךְ הַמְזַבְּלִין. 
וְכֵן הַחוֹפֵר בּוֹר וְשִׁיחַ וּמְעָרָה.




Chapter 4

<small>א</small>
בָּרִאשֹׁנָה הָיוּ אוֹמְרִין: 
מְלַקֵּט אָדָם עֵצִים וַעֲשָׂבִים מִתּוֹךְ שֶׁלּוֹ, 
כְּדֶרֶךְ שֶׁהוּא מְלַקֵּט מִתּוֹךְ שֶׁלַּחֲבֵרוֹ, 
אֶת הַגַּס הַגַּס. 
וּמִשֶּׁרַבּוּ עוֹבְרֵי עֲבֵרָה הִתְקִינוּ, 
שֶׁיְּהֵא זֶה מְלַקֵּט מִתּוֹךְ שֶׁלַּזֶּה שֶׁלֹּא בְטוֹבָה, 
וְאֵין צֹרֶךְ לוֹמַר שֶׁיְּקַצֵּץ לָהֶם מְזוֹנוֹת.

<small>ב</small>
שָׂדֶה שֶׁנִּתְקַוָּצָה, 
תִּזָּרַע מוֹצָאֵי שְׁבִיעִית; 
שֶׁנִּטַּיָּבָה אוֹ שֶׁנִּדַּיָּרָה, 
לֹא תִזָּרַע מוֹצָאֵי שְׁבִיעִית. 
שָׂדֶה שֶׁנִּטַּיָּבָה, בֵּית שַׁמַּי אוֹמְרִים: 
אֵין אוֹכְלִין פֵּרוֹתֶיהָ בַשְּׁבִיעִית; 
וּבֵית הֶלֵּל אוֹמְרִים: 
אוֹכְלִין. 
בֵּית שַׁמַּי אוֹמְרִים: 
אֵין אוֹכְלִין פֵּרוֹת שְׁבִיעִית בְּטוֹבָה; 
וּבֵית הֶלֵּל אוֹמְרִים: 
בְּטוֹבָה וְשֶׁלֹּא בְּטוֹבָה. 
רְבִּי יְהוּדָה אוֹמֵר: 
חִלּוּף הַדְּבָרִים, 
זֶה מִקָּלֵּי בֵית שַׁמַּי וּמֵחֻמְרֵי בֵית הֶלֵּל.

<small>ג</small>
חוֹכְרִים נִירִים מִן הַגּוֹיִם בַּשְּׁבִיעִית, 
אֲבָל לֹא מִיִּשְׂרָאֵל. 
וּמְחַזְּקִים יְדֵי גוֹיִם בַּשְּׁבִיעִית, 
אֲבָל לֹא יְדֵי יִשְׂרָאֵל. 
וְשׁוֹאֲלִין בִּשְׁלוֹמָם, 
מִפְּנֵי דַּרְכֵי שָׁלוֹם.

<small>ד</small>
הַמֵּדֵל בַּזֵּיתִים, 
בֵּית שַׁמַּי אוֹמְרִים: 
יָגֹם. 
וּבֵית הֶלֵּל אוֹמְרִים: 
יְשָׁרֵשׁ. 
מוֹדִים בְּמַחֲלִיק עַד שֶׁיָּגֹם. 
וְאֵי זֶה הוּא מֵדֵל? 
אֶחָד אוֹ שְׁנַיִם. 
הַמַּחֲלִיק? 
שְׁלֹשָׁה זֶה בְצַד זֶה. 
בַּמֶּה דְבָרִים אֲמוּרִים? 
מִתּוֹךְ שֶׁלּוֹ, 
אֲבָל מִתּוֹךְ שֶׁלַּחֲבֵרוֹ, 
אַף הַמַּחֲלִיק יְשָׁרֵשׁ.

<small>ה</small>
הַמַּבְקִיעַ בַּזַּיִת לֹא יְחַפֶּה בֶעָפָר, 
אֲבָל מְכַסֶּה הוּא בָאֲבָנִים אוֹ בַקַּשׁ. 
הַקּוֹצֵץ קוֹרוֹת שִׁקְמָה, לֹא יְחַפֶּה בֶעָפָר, 
אֲבָל מְכַסֶּה הוּא בָאֲבָנִים אוֹ בַקַּשׁ. 
אֵין קוֹצְצִין בְּתוּלַת שִׁקְמָה בַשְּׁבִיעִית, 
מִפְּנֵי שֶׁהִיא עֲבוֹדָה. 
רְבִּי יְהוּדָה אוֹמֵר: 
כְּדַרְכָּהּ אָסוּר; 
אֶלָּא מַגְבִּיהַּ עֲשָׂרָה טְפָחִים, 
אוֹ גוֹמֵם מֵעַם הָאָרֶץ.

<small>ו</small>
הַמְזַנֵּב בַּגְּפָנִים וְהַקּוֹצֵץ קָנִים, 
רְבִּי יוֹסֵה אוֹמֵר: 
יַרְחִיק טֶפַח. 
רְבִּי עֲקִיבָה אוֹמֵר: 
קוֹצֵץ כְּדַרְכּוֹ בַּקָּרְדּוֹם אוֹ בַמַּגָּל וּבַמְּגֵרָה, 
וּבְכָל מִין שֶׁיִּרְצֶה. 
אִילָן שֶׁנִּפְשַׁח, 
קוֹשְׁרִין אוֹתוֹ בַשְּׁבִיעִית. 
לֹא שֶׁיַּעֲלֶה, אֶלָּא שֶׁלֹּא יוֹסִיף.

<small>ז</small>
מֵאֶמָּתַי אוֹכְלִין פֵּרוֹת הָאִילָן בַּשְּׁבִיעִית? 
הַפַּגִּים, 
מִשֶּׁיַּזְרִיחוּ, אוֹכֵל בָּהֶן פִּתּוֹ בַשָּׂדֶה; 
בִּחֵלוּ, כּוֹנֵס לְתוֹךְ בֵּיתוֹ. 
וְכֵן כַּיּוֹצֵא בָהֶם בִּשְׁאָר שְׁנֵי שָׁבוּעַ, 
חַיָּבִים בַּמַּעַשְׂרוֹת.

<small>ח</small>
הַבֹּסֶר, 
מִשֶּׁהֵבִיא מַיִם, אוֹכֵל פִּתּוֹ בַשָּׂדֶה; 
הִבְאִישׁ, כּוֹנֵס לְתוֹךְ בֵּיתוֹ. 
וְכֵן כַּיּוֹצֵא בוֹ בִשְׁאָר שְׁנֵי שָׁבוּעַ, 
חַיָּב בַּמַּעַשְׂרוֹת.

<small>ט</small>
זֵיתִים שֶׁהִכְנִיסוּ רְבִיעִית לְסְאָה, 
פּוֹצֵעַ וְאוֹכֵל בַּשָּׂדֶה. 
הִכְנִיסוּ חֲצִי לֹג, כּוֹתֵשׁ וְסָךְ בַּשָּׂדֶה. 
הִכְנִיסוּ שְׁלִישׁ, 
כּוֹתֵשׁ בַּשָּׂדֶה וְכוֹנֵס לְתוֹךְ בֵּיתוֹ. 
וְכֵן כַּיּוֹצֵא בָהֶן בִּשְׁאָר שְׁנֵי שָׁבוּעַ, 
חַיָּבִים בַּמַּעַשְׂרוֹת. 
וּשְׁאָר כָּל פֵּרוֹת הָאִילָן, 
כְּעוֹנָתָן לַמַּעַשְׂרוֹת, כָּךְ עוֹנָתָן לַשְּׁבִיעִית.

<small>י</small>
מֵאֶמָּתַי אֵין קוֹצְצִין אֶת הָאִילָן בַּשְּׁבִיעִית? 
בֵּית שַׁמַּי אוֹמְרִים: 
כָּל הָאִילָן מִשֶּׁיּוֹצִיא. 
וּבֵית הֶלֵּל אוֹמְרִים: 
הֶחָרוּבִין מִשֶּׁיְּשַׁלֵּשׁוּ, 
וְהַגְּפָנִים מִשֶּׁיְּגָרֵעוּ, 
הַזֵּיתִים מִשֶּׁיָּנִיצוּ, 
וּשְׁאָר כָּל פֵּרוֹת הָאִילָן מִשֶּׁיּוֹצִיא. 
כָּל הָאִילָן, 
כֵּיוָן שֶׁבָּא לְעוֹנַת הַמַּעַשְׂרוֹת, 
מֻתָּר לְקוֹצְצוֹ. 
כַּמָּה יְהֵא בַזַּיִת וְלֹא יְקַצְּצֶנּוּ? 
רֹבַע. 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
הַכֹּל לְפִי זַיִת.




Chapter 5

<small>א</small>
בְּנוֹת שׁוּחַ, שְׁבִיעִית יֵשׁ לָהֶן שְׁנִיָּה, 
שֶׁהֵן עוֹשׁוֹת לְשָׁלוֹשׁ שָׁנִים. 
רְבִּי יוּדָה אוֹמֵר: 
הַפַּרְסָיוֹת, שְׁבִיעִית שֶׁלָּהֶן מוֹצָאֵי שְׁבִיעִית, 
שֶׁהֵן עוֹשׁוֹת לִשְׁתֵּי שָׁנִים. 
אָמְרוּ לוֹ: 
לֹא אָמְרוּ אֶלָּא בְנוֹת שׁוּחַ.

<small>ב</small>
הַטּוֹמֵן אֶת הַלּוּף בַּשְּׁבִיעִית, 
רְבִּי מֵאִיר אוֹמֵר: 
לֹא יִפְחֹת מִסָּאתַיִם, 
עַל גֹּבַהּ שְׁלֹשָׁה טְפָחִים, 
וְטֶפַח עָפָר עַל גַּבָּיו. 
וַחֲכָמִים אוֹמְרִים: 
לֹא יִפְחֲתוּ מֵאַרְבַּעַת קַבִּים, 
עַל גֹּבַהּ טֶפַח, 
וְטֶפַח עָפָר עַל גַּבָּיו. 
וְטוֹמְנוֹ מְקוֹם דְּרִיסַת הָאָדָם.

<small>ג</small>
לוּף שֶׁעָבְרָה עָלָיו שְׁבִיעִית, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אִם לָקְטוּ הָעֲנִיִּים אֶת עָלָיו, לָקְטוּ; 
וְאִם לָאו, 
יַעֲשֶׂה חֶשְׁבּוֹן לָעֲנִיִּים. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
אִם לָקְטוּ הָעֲנִיִּים אֶת עָלָיו, לָקְטוּ; 
וְאִם לָאו, 
אֵין לָעֲנִיִּים עִמּוֹ חֶשְׁבּוֹן.

<small>ד</small>
לוּף שֶׁלְּעֶרֶב שְׁבִיעִית שֶׁהוּא נִכְנָס לַשְּׁבִיעִית, 
וְכֵן בְּצָלִים הַקִּיצוֹנִים, 
וְכֵן פּוּאָה שֶׁלְּעִדִּית, 
בֵּית שַׁמַּי אוֹמְרִים: 
עוֹקְרִין אוֹתָהּ בְּמָרוּפוֹת שֶׁלָּעֵץ; 
וּבֵית הֶלֵּל אוֹמְרִים: 
בְּקַרְדֻּמּוֹת שֶׁלַּמַּתֶּכֶת. 
מוֹדִים בְּפוּאָה שֶׁלַּצְּלָעוֹת, 
שֶׁעוֹקְרִין אוֹתָהּ בְּקַרְדֻּמּוֹת שֶׁלַּמַּתֶּכֶת.

<small>ה</small>
מֵאֶמָּתַי מֻתָּר אָדָם לִקַּח לוּף מוֹצָאֵי שְׁבִיעִית? 
רְבִּי יְהוּדָה אוֹמֵר: 
מִיָּד. 
וַחֲכָמִים אוֹמְרִים: 
מִשֶּׁיִּרְבֶּה הֶחָדָשׁ.

<small>ו</small>
אֵלּוּ כֵלִים שֶׁאֵין הָאֻמָּן רַשַּׁי לְמָכְרָם בַּשְּׁבִיעִית: 
מַחֲרֵשָׁה וְכָל כֵּלֶיהָ, 
הָעֹל, הַמִּזְרֶה, וְהַדָּקָר. 
אֲבָל מוֹכֵר הוּא מַגַּל יָד, 
מַגַּל קָצִיר, עֲגָלָה וְכָל כֵּלֶיהָ. 
זֶה הַכְּלָל: 
כָּל שֶׁמְּלַאכְתּוֹ מְיֻחֶדֶת לַעֲבֵרָה, אָסוּר; 
לֶאֱסֹר וּלְהַתִּיר, מֻתָּר.

<small>ז</small>
הַיּוֹצֵר מוֹכֵר חָמֵשׁ כַּדֵּי שֶׁמֶן, 
וַחֲמֵשׁ עֶשְׂרֵה כַדֵּי יַיִן, 
שֶׁכֵּן דַּרְכּוֹ לְהָבִיא מִן הַהֶבְקֵר. 
אִם הֵבִיא יָתֵר מִכֵּן, מֻתָּר. 
וּמוֹכֵר לַגּוֹיִם בָּאָרֶץ, 
וּלְיִשְׂרָאֵל בְּחוּצָה לָאָרֶץ.

<small>ח</small>
בֵּית שַׁמַּי אוֹמְרִים: 
לֹא יִמְכֹּר לוֹ פָרָה חוֹרֶשֶׁת בַּשְּׁבִיעִית. 
בֵּית הֶלֵּל מַתִּירִין, 
מִפְּנֵי שֶׁהוּא יָכוֹל לְשָׁחְטָהּ. 
מוֹכֵר לוֹ פֵרוֹת אֲפִלּוּ בְשָׁעַת הַזֶּרַע. 
מַשְׁאִיל לוֹ סְאָתוֹ, 
אַף עַל פִּי שֶׁהוּא יוֹדֵעַ שֶׁיֶּשׁ לוֹ גֹרֶן. 
פּוֹרֵט לוֹ מָעוֹת, 
אַף עַל פִּי שֶׁהוּא יוֹדֵעַ שֶׁיֶּשׁ לוֹ פוֹעֲלִים. 
וְכֻלָּם בְּפֵרוּשׁ אֲסוּרִין.

<small>ט</small>
מַשְׁאֶלֶת אִשָּׁה 
לַחֲבֶרְתָּהּ הַחֲשׁוּדָה עַל הַשְּׁבִיעִית 
נָפָה וּכְבָרָה, וְרֵחַיִם וְתַנּוֹר; 
אֲבָל לֹא תָבֹר וְלֹא תִטְחַן עִמָּהּ. 
אֵשֶׁת חָבֵר מַשְׁאֶלֶת לְאֵשֶׁת עַם הָאָרֶץ 
נָפָה וּכְבָרָה; 
וּבוֹרֶרֶת וְטוֹחֶנֶת וּמְרַקֶּדֶת עִמָּהּ. 
אֲבָל מִשֶּׁתַּטִּיל אֶת הַמַּיִם, 
לֹא תִגַּע אֶצְלָהּ, 
שֶׁאֵין מְחַזְּקִין יְדֵי עוֹבְרֵי עֲבֵרָה. 
וְכֻלָּם לֹא אָמְרוּ אֶלָּא מִפְּנֵי דַרְכֵי שָׁלוֹם. 
מְחַזְּקִים יְדֵי גוֹיִם בַּשְּׁבִיעִית, 
אֲבָל לֹא יְדֵי יִשְׂרָאֵל. 
וְשׁוֹאֲלִים בִּשְׁלוֹמָם, 
מִפְּנֵי דַרְכֵי שָׁלוֹם.




Chapter 6

<small>א</small>
שָׁלוֹשׁ אֲרָצוֹת לַשְּׁבִיעִית: 
כָּל שֶׁהֶחֱזִיקוּ עוֹלֵי בָבֶל, 
מֵאֶרֶץ יִשְׂרָאֵל וְעַד גְּזִיב, 
לֹא נֶאֱכָל וְלֹא נֶעֱבָד; 
וְכָל שֶׁהֶחֱזִיקוּ עוֹלֵי מִצְרַיִם, 
מִגְּזִיב עַד הַנָּהָר וְעַד אֲמָנָה, 
נֶאֱכָל אֲבָל לֹא נֶעֱבָד; 
מֵהַנָּהָר וּמֵאֲמָנָה וְלִפְנִים, 
נֶאֱכָל וְנֶעֱבָד.

<small>ב</small>
עוֹשִׂים בַּתָּלוּשׁ בְּסוּרְיָה, 
אֲבָל לֹא בִמְחֻבָּר. 
דָּשִׁים וְזוֹרִים וְדוֹרְכִים וּמְעַמְּרִים, 
אֲבָל לֹא קוֹצְרִים וְלֹא בוֹצְרִים וְלֹא מוֹסְקִים. 
כְּלָל אָמַר רְבִּי עֲקִיבָה: 
כָּל שֶׁכַּיּוֹצֵא בוֹ מֻתָּר בָּאֶרֶץ, 
עוֹשִׂים אוֹתוֹ בְסוּרְיָה.

<small>ג</small>
בְּצָלִים שֶׁיָּרְדוּ עֲלֵיהֶן גְּשָׁמִים וְצִמֵּחוּ, 
אִם הָיוּ הֶעָלִים שֶׁלָּהֶן שְׁחוֹרִים, 
אֲסוּרִין; 
הוֹרִיקוּ, הֲרֵי אֵלּוּ מֻתָּרִין. 
רְבִּי חֲנַנְיָה בֶן אַנְטִיגְנָס אוֹמֵר: 
אִם יְכוּלִין לְהִתָּלֵשׁ בֶּעָלִין שֶׁלָּהֶן, 
אֲסוּרִין; 
כְּנֶגֶד כֵּן מוֹצָאֵי שְׁבִיעִית מֻתָּרִין.

<small>ד</small>
מֵאֶמָּתַי מֻתָּר אָדָם 
לִקַּח יָרָק מוֹצָאֵי שְׁבִיעִית? 
מִשֶּׁיַּעֲשֶׂה כַיּוֹצֵא בוֹ. 
עָשָׂה הַבַּכִּיר, 
הֻתַּר הָאָפִיל. 
רֶבִּי הִתִּיר לִקַּח יָרָק מוֹצָאֵי שְׁבִיעִית מִיָּד.

<small>ה</small>
אֵין מוֹצִיאִין שֶׁמֶן שְׂרֵפָה וּפֵרוֹת שְׁבִיעִית 
מֵהָאָרֶץ לְחוּצָה לָאָרֶץ. 
אָמַר רְבִּי שִׁמְעוֹן: 
שָׁמַעְתִּי בְּפֵרוּשׁ, 
שֶׁמּוֹצִיאִין לְסוּרְיָה, 
וְאֵין מוֹצִיאִין לְחוּצָה לָאָרֶץ.

<small>ו</small>
אֵין מְבִיאִין תְּרוּמָה מֵחוּצָה לָאָרֶץ לָאָרֶץ. 
אָמַר רְבִּי שִׁמְעוֹן: 
שָׁמַעְתִּי בְּפֵרוּשׁ, 
שֶׁמְּבִיאִין מִסּוּרְיָה, 
וְאֵין מְבִיאִין מֵחוּצָה לָאָרֶץ.




Chapter 7

<small>א</small>
כְּלָל גָּדוֹל אָמְרוּ בַשְּׁבִיעִית: 
כָּל שֶׁהוּא מֵאֹכֶל אָדָם, 
וּמֵאֹכֶל בְּהֵמָה מִמִּין הַצּוֹבְעִים, 
וְאֵינוּ מִתְקַיֵּם בָּאָרֶץ, 
יֶשׁ לוֹ שְׁבִיעִית וּלְדָמָיו שְׁבִיעִית, 
יֶשׁ לוֹ בֵעוּר וּלְדָמָיו בֵּעוּר. 
וְאֵי זֶה זֶה? 
זֶה עֲלֵי לוּף שׁוֹטֶה, 
וַעֲלֵה הַדַּנְדַּנָּא, 
הָעֻלְשִׁים, 
וְהַכְּרֵישִׁים, 
הָרְגִילָה, 
וְנֵץ הֶחָלָב; 
מֵאֹכֶל בְּהֵמָה, 
הַחוֹחִים וְהַדַּרְדָּרִים; 
וּמִמִּין הַצּוֹבְעִים, 
סִפְחֵי אֶסְטֵס וְקוֹצָא; 
יֵשׁ לָהֶם שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית, 
יֵשׁ לָהֶם בֵּעוּר וְלִדְמֵיהֶן בֵּעוּר.

<small>ב</small>
וְעוֹד כְּלָל אַחֵר אָמָרוּ: 
כָּל שֶׁהוּא מֵאֹכֶל אָדָם 
וּמֵאֹכֶל בְּהֵמָה וּמִמִּין הַצּוֹבְעִים, 
וּמִתְקַיֵּם בָּאָרֶץ, 
יֶשׁ לוֹ שְׁבִיעִית וּלְדָמָיו שְׁבִיעִית, 
אֵין לוֹ בֵעוּר וְלֹא לְדָמָיו בֵּעוּר. 
וְאֵי זֶה זֶה? 
זֶה עִקַּר הַלּוּף שׁוֹטֶה, 
וְעִקַּר הַדַּנְדַּנָּה, 
הָעַרְקְבָנִין, 
וְהַחַלְבֵּיצִין, 
וְהַבֻּכְרָיָה; 
מִמִּין הַצּוֹבְעִים, 
הַפּוּאָה וְהָרְכָפָה; 
יֵשׁ לָהֶן שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית, 
אֵין לָהֶן בֵּעוּר וְלֹא לִדְמֵיהֶן בֵּעוּר. 

<small>ג</small>
רְבִּי מֵאִיר אוֹמֵר: 
דְּמֵיהֶן מִתְבַּעֲרִין עַד רֹאשׁ הַשָּׁנָה. 
אָמְרוּ לוֹ: 
לָהֶן אֵין בֵּעוּר, 
קַל וָחֹמֶר לִדְמֵיהֶן.

<small>ד</small>
קְלִפֵּי רִמּוֹן וְהַנֵּץ שֶׁלּוֹ, 
קְלִפֵּי אֱגוֹזִים וְהַגַּלְעַנִּים, 
יֵשׁ לָהֶם שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית. 
הַצַּבָּע צוֹבֵעַ לְעַצְמוֹ; 
לֹא יִצְבַּע בְּשָׂכָר, 
שֶׁאֵין עוֹשִׂין סְחוֹרָה בְפֵרוֹת שְׁבִיעִית; 
וְלֹא בִבְכוֹרוֹת, 
וְלֹא בִתְרוּמוֹת, 
וְלֹא בִנְבֵלוֹת, 
וְלֹא בִטְרֵפוֹת, 
וְלֹא בִשְׁקָצִים, 
וְלֹא בִרְמָסִים. 
וְלֹא יִהֶא לוֹקֵחַ יַרְקוֹת שָׂדֶה וּמוֹכֵר בַּשּׁוּק, 
אֲבָל הוּא לוֹקֵט וּבְנוֹ מוֹכֵר עַל יָדוֹ. 
לָקַח לְעַצְמוֹ וְהוֹתִיר, 
מֻתָּר לְמָכְרוֹ.

<small>ה</small>
לָקַח בְּכוֹר לְמִשְׁתֵּה בְנוֹ אוֹ לָרֶגֶל, 
וְלֹא צָרַךְ לוֹ, מֻתָּר לְמָכְרוֹ. 
צָדֵי חַיָּה, עוֹפוֹת וְדָגִים, 
שֶׁנִּתְמַנּוּ לָהֶם מִינִים טְמֵאִים, 
מֻתָּרִין לְמָכְרָן. 
רְבִּי יְהוּדָה אוֹמֵר: 
אַף מִי שֶׁנִּתְמַנָּה לוֹ לְפִי דַרְכּוֹ, 
לוֹקֵחַ וּמוֹכֵר, 
וּבִלְבַד שֶׁלֹּא תְהֵא אֻמָּנוּתוֹ לְכֵן. 
וַחֲכָמִים אוֹסְרִין.

<small>ו</small>
לוּלַבֵּי זְרָדִים וְהֶחָרוּבִין, 
יֵשׁ לָהֶן שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית. 
לוּלַבֵּי הָאֵלָה וְהַבָּטְנָה וְהָאֲטָדִים, 
יֵשׁ לָהֶן שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית, 
אֵין לָהֶן בֵּעוּר וְלֹא לִדְמֵיהֶן בֵּעוּר; 
אֲבָל לֶעָלִין יֵשׁ בֵּעוּר, 
מִפְּנֵי שֶׁנּוֹשְׁרִין מֵאֲבִיהֶן.

<small>ז</small>
הַוֶּרֶד וְהַכֹּפֶר וְהַקֶּטֶב וְהַלֹּטֶם, 
יֵשׁ לָהֶן שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אֵין לַקֶּטֶב שְׁבִיעִית, 
מִפְּנֵי שֶׁאֵינוּ פְרִי.

<small>ח</small>
וֶרֶד חָדָשׁ שֶׁכְּבָשׁוֹ בְשֶׁמֶן יָשָׁן, 
יְלַקֵּט אֶת הַוֶּרֶד; 
יָשָׁן בְּחָדָשׁ, חַיָּב בַּבֵּעוּר. 
חָרוּבִים חֲדָשִׁים שֶׁכְּבָשָׁן בְּיַיִן יָשָׁן, 
וִישָׁנִים בְּחָדָשׁ, חַיָּבִין בַּבֵּעוּר. 
זֶה הַכְּלָל: 
כָּל שֶׁהוּא בְנוֹתֵן טַעַם, 
חַיָּבִים לְבַעֵר מִין בְּשֶׁאֵינוּ מִינוֹ; 
מִין בְּמִינוֹ, כָּל שֶׁהוּא. 
שְׁבִיעִית אוֹסֶרֶת כָּל שֶׁהִיא בְמִינָהּ, 
וְשֶׁלֹּא בְמִינָהּ, בְּנוֹתֵן טַעַם.




Chapter 8

<small>א</small>
כְּלָל גָּדוֹל אָמְרוּ בַשְּׁבִיעִית: 
כָּל הַמְיֻחָד לְאֹכֶל אָדָם, 
אֵין עוֹשִׂין מִמֶּנּוּ מְלָגְמָא לָאָדָם, 
וְאֵין צֹרֶךְ לוֹמַר לִבְהֵמָה; 
וְכָל שֶׁאֵינוּ מְיֻחָד לְאֹכֶל אָדָם, 
עוֹשִׂין מִמֶּנּוּ מְלָגְמָא לָאָדָם, 
אֲבָל לֹא לַבְּהֵמָה; 
וְכָל שֶׁאֵינוּ מְיֻחָד 
לֹא לְאֹכֶל אָדָם וְלֹא לְאֹכֶל בְּהֵמָה, 
חָשַׁב עָלָיו אֹכֶל אָדָם וְאֹכֶל בְּהֵמָה, 
נוֹתְנִין עָלָיו חֻמְרֵי אָדָם וְחֻמְרֵי בְהֵמָה; 
חָשַׁב עָלָיו לָעֵצִים, הֲרֵי הוּא כָּעֵצִים, 
כְּגוֹן הַסְּאָה הָאֵזוֹב וְהַקָּרְנִית.

<small>ב</small>
שְׁבִיעִית נִתְּנָה לַאֲכִילָה לִשְׁתִיָּה וְלִסִיכָה, 
לֹאכַל דָּבָר שֶׁדַּרְכּוֹ לֹאכַל, 
וְלָסוּךְ דָּבָר שֶׁדַּרְכּוֹ לָסוּךְ. 
לֹא יָסוּךְ יַיִן וָחֹמֶץ, 
אֲבָל סָךְ הוּא אֶת הַשֶּׁמֶן. 
וְכֵן בִּתְרוּמָה וּבְמַעֲשֵׂר שֵׁנִי. 
קַל מֵהֶן שְׁבִיעִית, 
שֶׁנִּתְּנָה לְהַדְלָקַת הַנֵּר.

<small>ג</small>
אֵין מוֹכְרִין פֵּרוֹת שְׁבִיעִית, 
לֹא בַמִּדָּה, וְלֹא בַמִּשְׁקָל, וְלֹא בַמִּנְיָן; 
וְלֹא תְאֵנִים בַּמִּנְיָן, 
וְלֹא יָרָק בַּמִּשְׁקָל. 
בֵּית שַׁמַּי אוֹמְרִים: 
אַף לֹא אֲגֻדּוֹת. 
וּבֵית הֶלֵּל אוֹמְרִים: 
אֶת שֶׁדַּרְכּוֹ לְהֵאָגֵד בַּבַּיִת, 
אוֹגְדִין אוֹתוֹ בַשּׁוּק, 
כְּגוֹן הַכְּרֵישִׁים וְנֵץ הֶחָלָב.

<small>ד</small>
הָאוֹמֵר לַפּוֹעֵל: 
"הֵילָךְ אִסָּר זֶה, וְלַקֵּט לִי יָרָק הַיּוֹם", 
שְׂכָרוֹ מֻתָּר; 
"לַקֵּט לִי בוֹ יָרָק הַיּוֹם", 
שְׂכָרוֹ אָסוּר. 
לָקַח מִן הַנַּחְתּוֹם מִכִּכָּר בְּפוֹנְדְּיוֹן: 
"כְּשֶׁאֶלְקֹט יַרְקוֹת שָׂדֶה אָבִיא לָךְ", 
מֻתָּר; 
לָקַח מִמֶּנּוּ סְתָם, 
לֹא יְשַׁלֵּם לוֹ מִדְּמֵי שְׁבִיעִית, 
שֶׁאֵין פּוֹרְעִין חוֹב מִדְּמֵי שְׁבִיעִית.

<small>ה</small>
אֵין נוֹתְנִין 
לֹא לַבַּיָּר, וְלֹא לַבַּלָּן, 
וְלֹא לַסַּפָּר, וְלֹא לַסַּפָּן; 
אֲבָל נוֹתֵן הוּא לַבַּיָּר לִשְׁתּוֹת. 
וּלְכֻלָּם הוּא נוֹתֵן מַתְּנַת חִנָּם.

<small>ו</small>
תְּאֵנִים שֶׁלַּשְּׁבִיעִית, 
אֵין קוֹצִין אוֹתָן בַּמֻּקְצֶה, 
אֲבָל קוֹצֶה הוּא בַחָרְבָּה. 
אֵין דּוֹרְכִין עֲנָבִים בַּגַּת, 
אֲבָל דּוֹרֵךְ הוּא בַעֲרֵבָה. 
וְאֵין עוֹשִׂין זֵיתִים בַּבַּד וּבַקָּטְכָא, 
אֲבָל כּוֹתֵשׁ וּמַכְנִיס לַבּוֹדֵידָה. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אַף טוֹחֵן הוּא בְּבֵית הַבַּד וּמַכְנִיס לַבּוֹדֵידָה.

<small>ז</small>
אֵין מְבַשְּׁלִין יָרָק שֶׁלַּשְּׁבִיעִית בְּשֶׁמֶן שֶׁלִּתְרוּמָה, 
שֶׁלֹּא יְבִיאֶנּוּ לִידֵי פְסוּל. 
רְבִּי שִׁמְעוֹן מַתִּיר. 
וְהָאַחֲרוֹן אַחֲרוֹן נִתְפָּשׁ בַּשְּׁבִיעִית, 
וְהַפְּרִי עַצְמוֹ אָסוּר.

<small>ח</small>
אֵין לוֹקְחִין עֲבָדִים וְקַרְקָעוֹת וּבְהֵמָה טְמֵאָה 
מִדְּמֵי שְׁבִיעִית; 
וְאִם לָקַח, יֹאכַל כְּנֶגְדּוֹ. 
אֵין מְבִיאִים קִנֵּי זָבִים, קִנֵּי זָבוֹת וְקִנֵּי יוֹלְדוֹת 
מִדְּמֵי שְׁבִיעִית; 
וְאִם הֵבִיא, יֹאכַל כְּנֶגְדּוֹ. 
אֵין סָכִים כֵּלִים בְּשֶׁמֶן שֶׁלַּשְּׁבִיעִית; 
וְאִם סָךְ, יֹאכַל כְּנֶגְדּוֹ.

<small>ט</small>
עוֹר שֶׁסָּכוֹ בְשֶׁמֶן שֶׁלַּשְּׁבִיעִית, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
יִדָּלֵק. 
וַחֲכָמִים אוֹמְרִים: 
יֹאכַל כְּנֶגְדּוֹ. 
אָמְרוּ לִפְנֵי רְבִּי עֲקִיבָה: 
אוֹמֵר הָיָה רְבִּי אֱלִיעֶזֶר: 
עוֹר שֶׁסָּכוֹ בְשֶׁמֶן שֶׁלַּשְּׁבִיעִית, 
יִדָּלֵק. 
אָמַר לָהֶם: 
שְׁתֹקוּ! 
לֹא אֹמַר לָכֶם מָה רַבִּי אֱלִיעֶזֶר אוֹמֵר בּוֹ.

<small>י</small>
וְעוֹד אָמְרוּ לְפָנָיו: 
אוֹמֵר הָיָה רְבִּי אֱלִיעֶזֶר: 
הָאוֹכֵל פַּת כּוּתִים כְּאוֹכֵל בְּשַׂר חֲזִיר. 
אָמַר לָהֶם: 
שְׁתֹקוּ! 
לֹא אֹמַר לָכֶם מָה רַבִּי אֱלִיעֶזֶר אוֹמֵר בּוֹ.

<small>יא</small>
מַרְחֵץ שֶׁהֻסָּקָה בַתֶּבֶן אוֹ בַקַּשׁ שֶׁלַּשְּׁבִיעִית, 
מֻתָּר לִרְחֹץ בָּהּ. 
וְאִם מִתְחַשֵּׁב הוּא, 
הֲרֵי זֶה לֹא יִרְחַץ.




Chapter 9

<small>א</small>
הַפִּיגָם וְהַיַּרְבּוּזִין הַשּׁוֹטִים, 
וְהַחֲלַגְלוֹגוֹת, כִּסְבָּר שֶׁבֶּהָרִים, 
וְכַרְפַּס שֶׁלַּנְּהָרוֹת, וְהַגַּרְגֵּר שֶׁלָּאַפָּר, 
פְּטוּרִין מִן הַמַּעַשְׂרוֹת, 
וְנִלְקָחִים מִכָּל אָדָם בַּשְּׁבִיעִית, 
שֶׁאֵין כַּיּוֹצֵא בָהֶן נִשְׁמָר. 
רְבִּי יְהוּדָה אוֹמֵר: 
סִפְחֵי חַרְדָּל מֻתָּרִין, 
שֶׁלֹּא נֶחְשְׁדוּ עֲלֵיהֶם עוֹבְרֵי עֲבֵרָה. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כָּל הַסְּפָחִין מֻתָּרִין, 
חוּץ מִסִּפְחֵי אֶכְרוּב, 
שֶׁאֵין כַּיּוֹצֵא בָהֶן בְּיַרְקוֹת שָׂדֶה. 
וַחֲכָמִים אוֹמְרִים: 
כָּל הַסְּפָחִין אֲסוּרִין.

<small>ב</small>
שָׁלוֹשׁ אֲרָצוֹת לַבֵּעוּר: 
יְהוּדָה, וְעֵבֶר הַיַּרְדֵּן, וְגָלִיל; 
וְשָׁלוֹשׁ שָׁלוֹשׁ אֲרָצוֹת לְכָל אַחַת וְאַחַת. 
גָּלִיל הָעֶלְיוֹן, וְגָלִיל הַתַּחְתּוֹן, וְהָעֵמֶק; 
מִכְּפַר חֲנַנְיָה וּלְמַעְלָן, 
כָּל שֶׁאֵינוּ מְגַדֵּל שִׁקְמִים, גָּלִיל הָעֶלְיוֹן; 
מִכְּפַר חֲנַנְיָה וּלְמַטָּן, 
כָּל שֶׁהוּא מְגַדֵּל שִׁקְמִים, גָּלִיל הַתַּחְתּוֹן; 
וּתְחוּם טְבֶרְיָה, הָעֵמֶק. 
וּבִיהוּדָה: 
הָהָר, וְהַשְּׁפֵלָה, וְהָעֵמֶק. 
וּשְׁפֵלַת לוּד כִּשְׁפֵלַת דָּרוֹם, 
וְהָהָר שֶׁלָּהּ כְּהַר הַמֶּלֶךְ. &lt;כההר&gt;
מִבֵּית חוֹרוֹן וְעַד הַיָּם, מְדִינָה אַחַת.

<small>ג</small>
וְלָמָּה אָמָרוּ שָׁלוֹשׁ אֲרָצוֹת? 
שֶׁיְּהוּ אוֹכְלִין מִכָּל אַחַת וְאַחַת, 
עַד שֶׁיִּכְלֶה הָאַחֲרוֹן שֶׁבָּהּ. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
לֹא אָמְרוּ שָׁלוֹשׁ אֲרָצוֹת אֶלָּא בִיהוּדָה, 
וּשְׁאָר כָּל הָאֲרָצוֹת כְּהַר הַמֶּלֶךְ. 
כָּל הָאֲרָצוֹת כְּאַחַת לַזֵּיתִים וְלַתְּמָרִים.

<small>ד</small>
אוֹכְלִים עַל הַמֻּבְקָר, 
אֲבָל לֹא מִן הַשָּׁמוּר. 
רְבִּי יוֹסֵה אוֹמֵר: 
אַף עַל הַשָּׁמוּר. 
אוֹכְלִים עַל הַטְּפָחִין וְעַל הַדִּפְרָא, 
אֲבָל לֹא עַל הַסִּתְוָנִיּוֹת. 
רְבִּי יְהוּדָה מַתִּיר כָּל זְמַן שֶׁבִּכֵּרוּ, 
עַד שֶׁלֹּא יִכְלֶה הַקַּיִץ.

<small>ה</small>
הַכּוֹבֵשׁ שְׁלֹשָׁה כְבָשִׁים בְּחָבִית אַחַת, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אוֹכְלִים עַל הָרִאשׁוֹן. 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
אַף עַל הָאַחֲרוֹן. 
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
כָּל שֶׁכָּלֶה מִינוֹ מִן הַשָּׂדֶה, 
יְבַעֵר מִינוֹ מִן הֶחָבִית; 
וַהֲלָכָה כִדְבָרָיו. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כָּל הַיָּרָק אֶחָד לַבֵּעוּר. 
אוֹכְלִין בִּרְגִילָה 
עַד שֶׁיִּכְלוּ הַסַּנָּרִיּוֹת מִבִּקְעַת בֵּית נְטוֹפָה.

<small>ו</small>
הַמְלַקֵּט עֲשָׂבִים לַחִים, 
עַד שֶׁיִּיבַשׁ הַמָּתִיק; 
וְהַמְגַבֵּב יָבֵשׁ, 
עַד שֶׁתֵּרֵד רְבֵיעָה שְׁנִיָּה. 
עֲלֵי קָנִים וַעֲלֵי גְפָנִים, 
עַד שֶׁיִּשְּׁרוּ מֵאֲבִיהֶם; 
הַמְגַבֵּב יָבֵשׁ, 
עַד שֶׁתֵּרֵד רְבֵיעָה שְׁנִיָּה. 
רְבִּי עֲקִיבָה אוֹמֵר: 
בְּכֻלָּן, עַד שֶׁתֵּרֵד רְבֵיעָה שְׁנִיָּה.

<small>ז</small>
כַּיּוֹצֵא בוֹ, 
הַמַּשְׂכִּיר בַּיִת לַחֲבֵרוֹ עַד הַגְּשָׁמִים, 
עַד שֶׁתֵּרֵד רְבֵיעָה שְׁנִיָּה. 
הַמֻּדָּר הֲנָיָה מֵחֲבֵרוֹ עַד הַגְּשָׁמִים, 
עַד שֶׁתֵּרֵד רְבֵיעָה שְׁנִיָּה. 
עַד אֶמָּתַי הָעֲנִיִּים נִכְנָסִים לַפַּרְדֵּסוֹת? 
עַד שֶׁתֵּרֵד רְבֵיעָה שְׁנִיָּה. 
מֵאֶמָּתַי נֶהְנִים וְשׁוֹרְפִים 
בַּתֶּבֶן וּבַקַּשׁ שֶׁלַּשְּׁבִיעִית? 
מִשֶּׁתֵּרֵד רְבֵיעָה שְׁנִיָּה.

<small>ח</small>
מִי שֶׁהָיוּ לוֹ פֵרוֹת שְׁבִיעִית, 
וְהִגִּיעָה שָׁעַת הַבֵּעוּר, 
מְחַלְּקִין מְזוֹן שָׁלוֹשׁ סְעוֹדוֹת לְכָל אֶחָד וְאֶחָד. 
וְהָעֲנִיִּים אוֹכְלִים אַחַר הַבֵּעוּר, 
אֲבָל לֹא עֲשִׁירִים. 
דִּבְרֵי רְבִּי יוּדָה. 
רְבִּי יוֹסֵה אוֹמֵר: 
אֶחָד עֲנִיִּים וְאֶחָד עֲשִׁירִים, 
אוֹכְלִין אַחַר הַבֵּעוּר.

<small>ט</small>
מִי שֶׁהָיוּ לוֹ פֵרוֹת שְׁבִיעִית, 
שֶׁנָּפְלוּ לוֹ לִירוּשָׁה אוֹ שֶׁנִּתְּנוּ לוֹ מַתָּנָה, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
יִנָּתְנוּ לְאוֹכְלֵיהֶם. 
וַחֲכָמִים אוֹמְרִים: 
אֵין הַחוֹטֵא נִשְׂכָּר, 
אֶלָּא יִמָּכְרוּ לְאוֹכְלֵיהֶם, 
וּדְמֵיהֶם יִתְחַלְּקוּ לְכָל אָדָם. 
הָאוֹכֵל מֵעִסַּת שְׁבִיעִית עַד שֶׁלֹּא הוּרָמָה חַלָּתָהּ, 
חַיָּב מִיתָה.




Chapter 10

<small>א</small>
שְׁבִיעִית מַשְׁמֶטֶת אֶת הַמַּלְוָה, 
בִשְׁטָר וְשֶׁלֹּא בִשְׁטָר. 
הַקָּפַת הֶחָנוּת אֵינָה מַשְׁמֶטֶת. 
וְאִם עֲשָׂאָהּ מַלְוָה, 
הֲרֵי זוֹ מַשְׁמֶטֶת. 
רְבִּי יְהוּדָה אוֹמֵר: 
הָרִאשׁוֹן מַשְׁמִיט. 
שְׂכַר שָׂכִיר אֵינוּ מַשְׁמִיט; 
וְאִם עֲשָׂאוֹ מַלְוָה, 
הֲרֵי זֶה מַשְׁמִיט. 
רְבִּי יוֹסֵה אוֹמֵר: 
כָּל מְלָאכָה שֶׁהִיא פוֹסֶקֶת לַשְּׁבִיעִית, 
מַשְׁמֶטֶת; 
וְשֶׁאֵינָהּ פּוֹסֶקֶת לַשְּׁבִיעִית, 
אֵינָה מַשְׁמֶטֶת.

<small>ב</small>
הַשּׁוֹחֵט אֶת הַפָּרָה, 
וְחִלְּקָהּ בְּרֹאשׁ הַשָּׁנָה, 
אִם הָיָה הַחֹדֶשׁ מְעֻבָּר, מַשְׁמִיט; 
וְאִם לָאו, אֵינוּ מַשְׁמִיט. 
הָאוֹנֵס, וְהַמְפַתֶּה, וְהַמּוֹצִיא שֵׁם רָע, 
וְכָל מַעֲשֵׂה בֵית דִּין, אֵינָן מַשְׁמִיטִין. 
הַמַּלְוֶה עַל הַמַּשְׁכּוֹן, 
וְהַמּוֹסֵר שְׁטָרוֹתָיו לְבֵית דִּין, 
אֵינָן מַשְׁמִיטִין.

<small>ג</small>
פְּרוֹזְבוֹל אֵינוּ מַשְׁמִיט. 
זֶה אֶחָד מִן הַדְּבָרִים שֶׁהִתְקִין הֶלֵּל הַזָּקֵן, 
שֶׁרָאָה שֶׁנִּמְנְעוּ הָעָם מִלַּלְווֹת זֶה אֶת זֶה, 
וְעוֹבְרִים עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה, שֶׁנֶּאֱמַר: (דברים טו,ט) 
"הִשָּׁמֶר לְךָ פֶּן יִהְיֶה דָבָר עִם לְבָבְךָ בְלִיַּעַל לֵאמֹר" וְגוֹמֵר, 
הִתְקִין הֶלֵּל פְּרוֹזְבוֹל.

<small>ד</small>
זֶה הוּא גוּפוֹ שֶׁלִּפְרוֹזְבוֹל: 
"מוֹסֵר אֲנִי לָכֶם, 
אִישׁ פְּלוֹנִי וּפְלוֹנִי וְהַדַּיָּנִים שֶׁבְּמָקוֹם פְּלוֹנִי, 
שֶׁכָּל חוֹב שֶׁיֶּשׁ לִי, 
שֶׁאֶגְבֶּנּוּ כָּל זְמַן שֶׁאֶרְצֶה", 
וְהַדַּיָּנִים חוֹתְמִין לְמַטָּן אוֹ הָעֵדִים.

<small>ה</small>
פְּרוֹזְבוֹל הַמֻּקְדָּם כָּשֵׁר, 
וְהַמְאֻחָר פָּסוּל. 
שְׁטָרֵי חוֹב הַמֻּקְדָּמִים פְּסוּלִין, 
וְהַמְאֻחָרִין כְּשֵׁרִין. 
אֶחָד לוֹוֶה מֵחֲמִשָּׁה, 
כּוֹתְבִין פְּרוֹזְבוֹל לְכָל אֶחָד וְאֶחָד. 
חֲמִשָּׁה לָוִים מִן הָאֶחָד, 
אֵינוּ כוֹתֵב אֶלָּא פְרוֹזְבוֹל אֶחָד לְכֻלָּם.

<small>ו</small>
אֵין כּוֹתְבִין פְּרוֹזְבוֹל אֶלָּא עַל הַקַּרְקַע. 
אִם אֵין לוֹ, 
מְזַכֵּהוּ לְתוֹךְ שָׂדֵהוּ כָל שֶׁהוּא. 
הָיְתָה לוֹ שָׂדֶה מְמֻשְׁכֶּנֶת בָּעִיר, 
כּוֹתְבִין עָלֶיהָ פְרוֹזְבוֹל. 
רְבִּי חָצְפִּית אוֹמֵר: 
כּוֹתְבִין לָאִישׁ עַל נִכְסֵי אִשְׁתּוֹ, 
וְלַיְתוֹמִים עַל נִכְסֵי אַפִיטְרוֹפִּים.

<small>ז</small>
כַּוֶּרֶת דְּבוֹרִים, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
הֲרֵי הִיא כַּקַּרְקַע, 
וְכוֹתְבִין עָלֶיהָ פְרוֹזְבוֹל, 
וְאֵינָהּ מְקַבֶּלֶת טֻמְאָה בִמְקוֹמָהּ, 
וְהָרוֹדֶה מִמֶּנָּה בַשַּׁבָּת, חַיָּב. 
וַחֲכָמִים אוֹמְרִים: 
אֵינָה כַּקַּרְקַע, 
וְאֵין כּוֹתְבִין עָלֶיהָ פְרוֹזְבוֹל, 
וּמְקַבֶּלֶת טֻמְאָה בִמְקוֹמָהּ, 
וְהָרוֹדֶה מִמֶּנָּה בַשַּׁבָּת, פָּטוּר.

<small>ח</small>
הַמַּחֲזִיר חוֹב בַּשְּׁבִיעִית, 
יֹאמַר לוֹ: 
"מַשְׁמִיט אָנִי". 
אָמַר לוֹ: 
"אַף עַל פִּי כֵן", 
יְקַבֵּל מִמֶּנּוּ, שֶׁנֶּאֱמַר: (דברים טו,ב) 
"וְזֶה דְּבַר הַשְּׁמִטָּה". 

<small>ט</small>
כַּיּוֹצֵא בוֹ, 
רוֹצֵחַ שֶׁגָּלָה לְעִיר הַמִּקְלָט 
וְרָצוּ אַנְשֵׁי הָעִיר לְכַבְּדוֹ, 
יֹאמַר לָהֶם: 
"רוֹצֵחַ אָנִי!" 
אָמְרוּ לוֹ: 
"אַף עַל פִּי כֵן", 
יְקַבֵּל מֵהֶן, שֶׁנֶּאֱמַר: (דברים יט, ד) 
"וְזֶה דְּבַר הָרוֹצֵחַ".

<small>י</small>
הַמַּחֲזִיר חוֹב בַּשְּׁבִיעִית, 
רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ. 
הַלּוֹוֶה מִן הַגֵּר שֶׁנִּתְגַּיְּרוּ בָנָיו עִמּוֹ, 
לֹא יַחְזִיר לְבָנָיו; 
וְאִם הֶחְזִיר, 
רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ. 
כָּל הַמִּטַּלְטְלִין נִקְנִין בִּמְשִׁיכָה; 
כָּל הַמְקַיֵּם אֶת דְּבָרוֹ, 
רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ. 

חסל
