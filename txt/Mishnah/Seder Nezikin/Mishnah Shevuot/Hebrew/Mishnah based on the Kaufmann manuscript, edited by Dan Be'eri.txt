משנה שבועות



Chapter 1

<small>א</small>
שְׁבוּעוֹת, שְׁתַּיִם שֶׁהֵן אַרְבַּע.  
יְדִיעוֹת הַטֻּמְאָה, שְׁתַּיִם שֶׁהֵן אַרְבַּע.  
יְצִיאוֹת הַשַּׁבָּת, שְׁתַּיִם שֶׁהֵן אַרְבַּע.  
מַרְאוֹת נְגָעִים, שְׁנַיִם שֶׁהֵן אַרְבָּעָה. 

<small>ב</small>
כָּל שֶׁיֶּשׁ בָּהּ יְדִיעָה בַתְּחִלָּה וִידִיעָה בַסּוֹף וְהֶעְלֵם בִּנְתַּיִם, 
הֲרֵי זֶה בְעוֹלֶה וְיוֹרֵד.  
יֶשׁ בָּהּ יְדִיעָה בַתְּחִלָּה וְאֵין בָּהּ יְדִיעָה בַסּוֹף, 
שָׂעִיר הַנֶּעֱשָׂה בִפְנִים יוֹם הַכִּפּוּרִים תּוֹלֶה, 
עַד שֶׁתִּוָּדַע לוֹ, וְיָבִיא בְּעוֹלֶה וְיוֹרֵד. 

<small>ג</small>
אֵין בָּהּ יְדִיעָה בַתְּחִלָּה, 
אֲבָל יֶשׁ בָּהּ יְדִיעָה בַסּוֹף, 
שָׂעִיר הַנֶּעֱשָׂה בַחוּץ יוֹם הַכִּפּוּרִים מְכַפֵּר, 
שֶׁנֶּאֱמַר: (במדבר כט,יא) 
"מִלְּבַד חַטַּאת הַכִּפֻּרִים". 
עַל מַה שֶּׁזֶּה מְכַפֵּר זֶה מְכַפֵּר: 
מַה הַפְּנִימִי, 
אֵינוּ מְכַפֵּר אֶלָּא עַל דָּבָר שֶׁיֶּשׁ בּוֹ יְדִיעָה, 
אַף הַחִיצוֹן, 
לֹא יְכַפֵּר אֶלָּא עַל דָּבָר שֶׁיֶּשׁ בּוֹ יְדִיעָה. &lt;בָּהּ&gt;

<small>ד</small>
וְעַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף, 
שְׂעִירֵי רְגָלִים וּשְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִים.  
דִּבְרֵי רְבִּי יְהוּדָה.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
שְׂעִירֵי רְגָלִים מְכַפְּרִים, 
אֲבָל לֹא שְׂעִירֵי רָאשֵׁי חֳדָשִׁים.  
וְעַל מַה שְּׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִים? 
עַל טָהוֹר שֶׁאָכַל טָמֵא.  &lt;טָּהוֹר&gt;
רְבִּי מֵאִיר אוֹמֵר: 
כָּל הַשְּׂעִירִים כַּפָּרָתָן שָׁוָה, 
עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו.  

<small>ה</small>
הָיָה רְבִּי שִׁמְעוֹן אוֹמֵר: 
שְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִים עַל טָהוֹר שֶׁאָכַל טָמֵא. 
וְשֶׁלָּרְגָלִים מְכַפְּרִים עַל שֶׁאֵין בָּהּ יְדִיעָה, 
לֹא בַתְּחִלָּה וְלֹא בַסּוֹף. 
וְשֶׁלְּיוֹם הַכִּפּוּרִים מְכַפְּרִין עַל שֶׁאֵין בָּהּ יְדִיעָה [בַתְּחִלָּה, 
אֲבָל יֶשׁ בָּהּ יְדִיעָה] בַסּוֹף.   

<small>ו</small>
אָמְרוּ לוֹ: 
מַה הֵן שֶׁיִּקָּרְבוּ זֶה בָזֶה? 
אָמַר לָהֶן: 
יִקָּרְבוּ.  
אָמְרוּ לוֹ: 
הוֹאִיל וְאֵין כַּפָּרָתָן שָׁוָה, 
הֵיאָךְ הֵן קְרֵבִין? 
אָמַר לָהֶן: 
כֻּלָּם בָּאִין לְכַפֵּר עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו. 

<small>ז</small>
רְבִּי שִׁמְעוֹן בֶּן יְהוּדָה אוֹמֵר מִשְּׁמוֹ: 
שְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִים עַל טָהוֹר שֶׁאָכַל טָמֵא. 
מוּסָף עֲלֵיהֶן שֶׁלָּרְגָלִין, 
שֶׁהֵן מְכַפְּרִין עַל טָהוֹר שֶׁאָכַל טָמֵא, 
וְעַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף. 
מוּסָף עֲלֵיהֶם שֶׁלְּיוֹם הַכִּפּוּרִים, 
שֶׁהֵן מְכַפְּרִין עַל טָהוֹר שֶׁאָכַל טָמֵא, 
וְעַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף, 
וְעַל שֶׁאֵין בָּהּ יְדִיעָה בַתְּחִלָּה, 
אֲבָל יֶשׁ בָּהּ יְדִיעָה בַסּוֹף.  

<small>ח</small>
אָמְרוּ לוֹ: 
מַה הֵן שֶׁיִּקָּרְבוּ זֶה בָזֶה? &lt;אוֹמֵר הָיָה&gt;
אָמַר לָהֶן: 
הִין.  
אָמְרוּ לוֹ: 
אִם כֵּן, 
יְהוּ שֶׁלְּיוֹם הַכִּפּוּרִים קְרֵבִים לְרָאשֵׁי חֳדָשִׁים, 
אֲבָל הֵיאָךְ שֶׁלְּרָאשֵׁי חֳדָשִׁים קְרֵבִים לְיוֹם הַכִּפּוּרִים, 
לְכַפֵּר כַּפָּרָה שֶׁאֵינָה שֶׁלָּהֶן? 
אָמַר לָהֶן: 
כֻּלָּם בָּאִין לְכַפֵּר עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו. &lt;כְּלוּם&gt;

<small>ט</small>
וְעַל זְדוֹן טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו, 
שָׂעִיר הַנֶּעֱשָׂה בִפְנִים וְיוֹם הַכִּפּוּרִים מְכַפֵּר.  
וְעַל שְׁאָר עֲבֵרוֹת שֶׁבַּתּוֹרָה, 
הַקַּלּוֹת וְהַחֲמוּרוֹת, 
הַזְּדוֹנוֹת וְהַשְּׁגָגוֹת, 
הוֹדַע וְלֹא הוֹדַע, 
עֲשֵׂה וְלֹא תַעֲשֶׂה, 
כָּרֵתוֹת וּמִיתוֹת בֵּית דִּין, 
שָׂעִיר הַמִּשְׁתַּלֵּחַ מְכַפֵּר. 

<small>י</small>
אֶחָד יִשְׂרְאֵלִים, וְאֶחָד כֹּהֲנִים, וְאֶחָד כֹּהֵן מָשִׁיחַ.  
אֶלָּא שֶׁדַּם הַפָּר מְכַפֵּר עַל הַכֹּהֲנִים 
וְעַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כַּשֵּׁם שֶׁדַּם הַשָּׂעִיר הַנֶּעֱשָׂה בִפְנִים מְכַפֵּר עַל יִשְׂרָאֵל, 
כָּךְ דַּם הַפָּר מְכַפֵּר עַל הַכֹּהֲנִים. 
כַּשֵּׁם שֶׁוִּדּוּיוֹ שֶׁלְּשָׂעִיר הַמִּשְׁתַּלֵּחַ מְכַפֵּר עַל יִשְׂרָאֵל, 
כָּךְ וִדּוּיוֹ שֶׁלַּפַּר מְכַפֵּר עַל הַכֹּהֲנִים. 





Chapter 2

<small>א</small>
יְדִיעוֹת הַטֻּמְאָה שְׁתַּיִם שֶׁהֵן אַרְבַּע: 
נִטַּמָּא וְיָדַע, 
וְנֶעְלְמָה מִמֶּנּוּ טֻמְאָה, וְזָכוּר לַקֹּדֶשׁ; 
נֶעְלַם מִמֶּנּוּ קֹדֶשׁ, וְזָכוּר לַטֻּמְאָה; 
נֶעְלְמוּ מִמֶּנּוּ זֶה וָזֶה, 
אָכַל אֶת הַקֹּדֶשׁ וְלֹא יָדַע, 
הֲרֵי זֶה בְעוֹלֶה וְיוֹרֵד.  

<small>ב</small>
נִטַּמָּא וְיָדַע, 
נֶעְלְמָה מִמֶּנּוּ טֻמְאָה, וְזָכוּר לַמִּקְדָּשׁ; 
נֶעְלַם מִמֶּנּוּ מִקְדָּשׁ, וְזָכוּר לַטֻּמְאָה; 
נֶעְלְמוּ מִמֶּנּוּ זֶה וָזֶה, 
נִכְנַס לַמִּקְדָּשׁ וְלֹא יָדַע, מִשֶּׁיָּצָא יָדַע, 
הֲרֵי זֶה בְעוֹלֶה וְיוֹרֵד. 

<small>ג</small>
אֶחָד הַנִּכְנָס לַעֲזָרָה וְאֶחָד הַנִּכְנָס לְתוֹסֶפֶת הָעֲזָרָה. 
שֶׁאֵין מוֹסִיפִין עַל הָעִיר וְעַל הָעֲזָרוֹת, 
אֶלָּא בְמֶלֶךְ, בַּנָּבִיא בָאוּרִים וּבְתֻמִּים, 
וּבְסֶנְהֶדְרִין שֶׁלְּשִׁבְעִים וְאֶחָד, 
וּבִשְׁתֵּי תוֹדוֹת וּבְשִׁיר.  
וּבֵית דִּין מְהַלְּכִין וּשְׁתֵּי תוֹדוֹת אַחֲרֵיהֶן, 
הַפְּנִימִית נֶאֱכֶלֶת וְהַחִיצוֹנָה נִשְׂרֶפֶת.  
וְכָל שֶׁלֹּא נֶעֱשֵׂת בְכָל אֵלּוּ, 
הַנִּכְנָס לַשָּׁם, אֵין חַיָּבִין עָלֶיהָ. 

<small>ד</small>
נִטַּמָּא בָעֲזָרָה, 
נֶעְלְמָה מִמֶּנּוּ טֻמְאָה, וְזָכוּר לַמִּקְדָּשׁ; 
נֶעְלַם מִמֶּנּוּ מִקְדָּשׁ, וְזָכוּר לַטֻּמְאָה; 
נֶעְלַם מִמֶּנּוּ זֶה וָזֶה, 
הִשְׁתַּחֲוָה, שֶׁשָּׁהָה בִכְדֵי הִשְׁתַּחֲוָיָה, &lt;הַשְׁתַּחֲוָיָה&gt;
בָּא לוֹ בָאֲרֻכָּה, חַיָּב, 
וּבַקְּצָרָה, פָּטוּר.  
זוֹ הִיא מִצְוַת עֲשֵׂה שֶׁבַּמִּקְדָּשׁ, 
שֶׁאֵין חַיָּבִין עָלֶיהָ. 

<small>ה</small>
אֵי זוֹ הִיא מִצְוַת עֲשֵׂה שֶׁבַּנִּדָּה, 
שֶׁחַיָּבִין עָלֶיהָ? 
הָיָה מְשַׁמֵּשׁ עִם הַטְּהוֹרָה, 
אָמְרָה לוֹ "נִטְמֵאתִי", 
אִם פֵּרַשׁ מִיָּד, 
חַיָּב, שֶׁיְּצִיאָתוֹ הֲנָיָה לוֹ כְבִיאָתוֹ. 

<small>ו</small>
רְבִּי אֱלִיעֶזֶר אוֹמֵר: (ויקרא ה,ב)  
הַ "שֶׁרֶץ... וְנֶעְלַם מִמֶּנּוּ", 
עַל הֶעְלֵם הַשֶּׁרֶץ הוּא חַיָּב, 
אֵינוּ חַיָּב עַל הֶעְלֵם הַמִּקְדָּשׁ.  
רְבִּי עֲקִיבָה אוֹמֵר: (ויקרא ה,ב)  
"וְנֶעְלַם מִמֶּנּוּ וְהוּא טָמֵא", 
עַל הֶעְלֵם הַטֻּמְאָה הוּא חַיָּב, 
אֵינוּ חַיָּב עַל הֶעְלֵם מִקְדָּשׁ.  
רְבִּי יִשְׁמָעֵאל אוֹמֵר: (ויקרא ה,ב-ג)
"וְנֶעְלַם מִמֶּנּוּ... וְנֶעְלַם מִמֶּנּוּ" שְׁנֵי פְעָמִים, 
לְחַיֵּב עַל הֶעְלֵם הַטֻּמְאָה וְעַל הֶעְלֵם הַמִּקְדָּשׁ. 





Chapter 3

<small>א</small>
שְׁבוּעוֹת, שְׁתַּיִם שֶׁהֵן אַרְבַּע: 
"שְׁבוּעָה שֶׁאֹכַל", וְ"שֶׁלֹּא אֹכַל", 
"שֶׁאָכַלְתִּי", וְ"שֶׁלֹּא אָכַלְתִּי".  
"שְׁבוּעָה שֶׁלֹּא אֹכַל", וְאָכַל כָּל שֶׁהוּא, 
חַיָּב.  
דִּבְרֵי רְבִּי עֲקִיבָה.  
אָמְרוּ לוֹ לִרְבִּי עֲקִיבָה: 
בְּאֵיכָן מָצִינוּ בְאוֹכֵל כָּל שֶׁהוּא, 
שֶׁהוּא חַיָּב, שֶׁזֶּה חַיָּב? 
אָמַר לָהֶן: 
בְּאֵיכָן מָצִינוּ בִמְדַבֵּר וּמֵבִיא קָרְבָּן, 
שֶׁזֶּה מְדַבֵּר וּמֵבִיא קָרְבָּן? 

<small>ב</small>
"שְׁבוּעָה שֶׁלֹּא אֹכַל", וְאָכַל וְשָׁתָה, 
אֵינוּ חַיָּב אֶלָּא אַחַת.  
"שְׁבוּעָה שֶׁלֹּא אֹכַל וְשֶׁלֹּא אֶשְׁתֶּה", וְאָכַל וְשָׁתָה, 
חַיָּב שְׁתַּיִם. 

<small>ג</small>
"שְׁבוּעָה שֶׁלֹּא אֹכַל", 
וְאָכַל פַּת חִטִּין וּפַת שְׂעוֹרִים וּפַת כֻּסְּמִים, 
אֵינוּ חַיָּב אֶלָּא אַחַת.  
"שְׁבוּעָה שֶׁלֹּא אֹכַל פַּת חִטִּין וּפַת שְׂעוֹרִים וּפַת כֻּסְּמִים", 
וְאָכַל, 
חַיָּב עַל כָּל אַחַת וְאַחַת. 

<small>ד</small>
"שְׁבוּעָה שֶׁלֹּא אֶשְׁתֶּה", 
וְשָׁתָה מַשְׁקִין הַרְבֵּה, 
אֵינוּ חַיָּב אֶלָּא אַחַת.  
"שְׁבוּעָה שֶׁלֹּא אֶשְׁתֶּה יַיִן וָשֶׁמֶן וּדְבַשׁ", 
וְשָׁתָה, 
חַיָּב עַל כָּל אַחַת וְאַחַת. 

<small>ה</small>
"שְׁבוּעָה שֶׁלֹּא אֹכַל", 
וְאָכַל אֳכָלִין שֶׁאֵינָן רְאוּיִן לַאֲכִילָה, 
וְשָׁתָה מַשְׁקִין שֶׁאֵינָן רְאוּיִן לִשְׁתִיָּה, 
פָּטוּר.  
"שְׁבוּעָה שֶׁלֹּא אֹכַל", 
וְאָכַל נְבֵלוֹת וּטְרֵפוֹת, וּשְׁקָצִים וּרְמָשִׂים, 
חַיָּב. 
רְבִּי שִׁמְעוֹן פּוֹטֵר.  
אָמַר "קֻנָּם אִשְׁתִּי נֶהְנֵית לִי אִם אָכַלְתִּי הַיּוֹם", 
וְהוּא שֶׁאָכַל נְבֵלוֹת וּטְרֵפוֹת, שְׁקָצִים וּרְמָשִׂים, 
הֲרֵי אִשְׁתּוֹ אֲסוּרָה. 

<small>ו</small>
אֶחָד דְּבָרִים שֶׁלְּעַצְמוֹ, 
וְאֶחָד דְּבָרִין שֶׁלַּאֲחֵרִים, 
וְאֶחָד דְּבָרִין שֶׁיֵּשׁ בָּהֶן מַמָּשׁ, 
וְאֶחָד דְּבָרִין שֶׁאֵין בָּהֶן מַמָּשׁ.  
כֵּיצַד? 
אָמַר "שְׁבוּעָה שֶׁאֶתֵּן לְאִישׁ פְּלוֹנִי", וְ"שֶׁלֹּא אֶתֵּן", 
"שֶׁנָּתַתִּי", וְ"שֶׁלֹּא נָתַתִּי", 
"שֶׁאִישַׁן", וְ"שֶׁלֹּא אִישַׁן", 
"שֶׁיָּשַׁנְתִּי, וְ"שֶׁלֹּא יָשַׁנְתִּי", 
"שֶׁאֶזְרֹק צְרוֹר לַיָּם", וְ"שֶׁלֹּא אֶזְרֹק", 
"שֶׁזָּרַקְתִּי", וְ"שֶׁלֹּא זָרַקְתִּי".  
רְבִּי יִשְׁמָעֵאל אוֹמֵר: 
אֵינוּ חַיָּב אֶלָּא לֶעָתִיד לָבֹא, שֶׁנֶּאֱמַר: (ויקרא ה,ד) 
"לְהָרַע אוֹ לְהֵיטִיב".  
אָמַר לוֹ רְבִּי עֲקִיבָה: 
אִם כֵּן, אֵין לִי אֶלָּא דְבָרִים שֶׁיֵּשׁ בָּהֶן הֲרָעָה וַהֲטָבָה. 
דְּבָרִים שֶׁאֵין בָּהֶן הֲרָעָה וַהֲטָבָה, מְנַיִן? 
אָמַר לוֹ: 
מֵרִבּוּי הַכָּתוּב.  
אָמַר לוֹ: 
אִם רִבָּה הַכָּתוּב לְכָךְ, 
רִבָּה הַכָּתוּב לְכָךְ. 

<small>ז</small>
נִשְׁבַּע לְבַטֵּל אֶת הַמִּצְוָה, וְלֹא בִטֵּל, 
פָּטוּר; 
לְקַיֵּם אֶת הַמִּצְוָה, וְלֹא קִיֵּם, 
פָּטוּר, שֶׁהָיָה בַדִּין שֶׁיְּהֵא חַיָּב, 
כְּדִבְרֵי רְבִּי יְהוּדָה בֶן בְּתִירָה:  
מָה אִם הָרְשׁוּת, שֶׁאֵינוּ מֻשְׁבָּע עָלֶיהָ מֵהַר סִינַי, 
הֲרֵי הוּא חַיָּב עָלֶיהָ, 
מִצְוָה, שֶׁהוּא מֻשְׁבָּע עָלֶיהָ מֵהַר סִינַי, 
אֵינוּ דִין שֶׁיְּהֵא חַיָּב עָלֶיהָ? 
אָמְרוּ לוֹ: 
לֹא! 
אִם אָמַרְתָּ בִשְׁבוּעַת הָרְשׁוּת, 
שֶׁכֵּן עָשָׂה בָהּ לָאו כְּהִין, 
תֹּאמַר בִּשְׁבוּעַת מִצְוָה, 
שֶׁלֹּא עָשָׂה בָהּ לָאו כְּהִין. 

<small>ח</small>
"שְׁבוּעָה שֶׁלֹּא אֹכַל כִּכָּר זוֹ", 
"שְׁבוּעָה שֶׁלֹּא אֹכְלֶנָּה", 
וַאֲכָלָהּ, 
אֵינוּ חַיָּב אֶלָּא אַחַת.  
זוֹ הִיא שְׁבוּעַת בִּטּוּי, 
שֶׁחַיָּבִין עַל זְדוֹנָהּ מַכּוֹת, 
וְעַל שִׁגְגָתָהּ קָרְבָּן עוֹלֶה וְיוֹרֵד.  

<small>ט</small>
שְׁבוּעַת שָׁוְא, 
חַיָּבִין עַל זְדוֹנָהּ מַכּוֹת, 
וְעַל שִׁגְגָתָהּ פָּטוּר. 

אֵיזוֹ הִיא שְׁבוּעַת שָׁוְא? 
נִשְׁבַּע לְשַׁנּוֹת אֶת הַיָּדוּעַ לָאָדָן: 
אָמַר עַל הָעַמּוּד שֶׁלָּאֶבֶן שֶׁהוּא שֶׁלַּזָּהָב, 
וְעַל הָאִישׁ שֶׁהוּא אִשָּׁה, 
וְעַל אִשָּׁה שֶׁהִיא אִישׁ. 
עַל דָּבָר שֶׁאֵי אֶפְשָׁר לוֹ: 
"אִם לֹא רָאִיתִי גָמָל שֶׁפּוֹרֵחַ בְּאָוֵיר", 
וְ"אִם לֹא רָאִיתִי נָחָשׁ כְּקוֹרַת בֵּית הַבַּד". 
אָמַר לָעֵדִים "בּוֹאוּ וְהַעִידוּנִי!" 
--"שְׁבוּעָה שֶׁלֹּא נְעִידֶךָ!" 
נִשְׁבַּע לְבַטֵּל אֶת הַמִּצְוָה: 
שֶׁלֹּא לַעֲשׁוֹת סֻכָּה, 
וְשֶׁלֹּא לִטּוֹל לוּלָב, 
וְשֶׁלֹּא לְהָנִיחַ תְּפִלִּין, 
זוֹ הִיא שְׁבוּעַת שָׁוְא, 
שֶׁחַיָּבִין עַל זְדוֹנָהּ מַכּוֹת, 
וְעַל שִׁגְגָתָהּ פָּטוּר. 

<small>י</small>
"שְׁבוּעָה שֶׁאֹכַל כִּכָּר זוֹ", 
"שְׁבוּעָה שֶׁלֹּא אֹכְלֶנָּה", 
הָרִאשׁוֹנָה שְׁבוּעַת בִּטּוּי, 
וְהַשְּׁנִיָּה שְׁבוּעַת שָׁוְא.  
אֲכָלָהּ, עָבַר עַל שְׁבוּעַת שָׁוְא. 
לֹא אֲכָלָהּ, עָבַר עַל שְׁבוּעַת בִּטּוּי. 

<small>יא</small>
שְׁבוּעַת בִּטּוּי נוֹהֶגֶת בָּאֲנָשִׁים, בַּנָּשִׁים, 
בָּרְחוֹקִים, בַּקְּרוֹבִים, 
בַּכְּשֵׁרִין, בַּפְּסוּלִים, 
בִּפְנֵי בֵית דִּין, וְשֶׁלֹּא בִפְנֵי בֵית דִּין, 
מִפִּי עַצְמוֹ. 
וְחַיָּבִין עַל זְדוֹנָהּ מַכּוֹת, 
וְעַל שִׁגְגָתָהּ קָרְבָּן עוֹלֶה וְיוֹרֵד. 

<small>יב</small>
שְׁבוּעַת שָׁוְא נוֹהֶגֶת בָּאֲנָשִׁין, בַּנָּשִׁים, 
בָּרְחוֹקִים, בַּקְּרוֹבִים, 
בַּכְּשֵׁרִים, בַּפְּסוּלִים, 
בִּפְנֵי בֵית דִּין, וְשֶׁלֹּא בִפְנֵי בֵית דִּין, 
מִפִּי עַצְמוֹ. 
וְחַיָּבִין עַל זְדוֹנָהּ מַכּוֹת, 
וְעַל שִׁגְגָתָהּ פָּטוּר.  
אַחַת זוֹ וְאַחַת זוֹ, 
הַמֻּשְׁבָּע מִפִּי אֲחֵרִים, חַיָּב.  
"אִם לֹא אָכַלְתִּי הַיּוֹם", 
וְ"לֹא הִנַּחְתִּי תְפִלִּין הַיּוֹם, מַשְׁבִּיעָךְ אָנִי", 
וְאָמַר "אָמֵן", חַיָּב. 





Chapter 4

<small>א</small>
שְׁבוּעַת הָעֵדוּת נוֹהֶגֶת בָּאֲנָשִׁים וְלֹא בַנָּשִׁים, 
בָּרְחוֹקִין וְלֹא בַקְּרוֹבִין, 
בַּכְּשֵׁרִין וְלֹא בַּפְּסוּלִין, 
וְאֵינָהּ נוֹהֶגֶת אֶלָּא בִּרְאוּיִים לְהָעִיד. 
בִּפְנֵי בֵית דִּין וְשֶׁלֹּא בִפְנֵי בֵית דִּין, 
מִפִּי עַצְמוֹ וּמִפִּי אֲחֵרִין. 
אֵינָן חַיָּבִין עַד שֶׁיִּכְפְּרוּ בוֹ בְּבֵית דִּין. &lt;שֶׁיְּכַפְּרוּ&gt; 
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
בֵּין מִפִּי עַצְמוֹ וּבֵין מִפִּי אֲחֵרִין, 
אֵינָן חַיָּבִין עַד שֶׁיִּכְפְּרוּ בוֹ בְּבֵית דִּין. &lt;שֶׁיְּכַפְּרוּ&gt;

<small>ב</small>
וְחַיָּבִין עַל זְדוֹן הַשְּׁבוּעָה, 
וְעַל שִׁגְגָתָהּ עִם זְדוֹן הָעֵדוּת, 
אֵינָן חַיָּבִין עַל שִׁגְגָתָהּ.  
וּמַה הֵן חַיָּבִין עַל זְדוֹנָהּ? 
קָרְבָּן עוֹלֶה וְיוֹרֵד. 

<small>ג</small>
שְׁבוּעַת הָעֵדוּת כֵּיצַד? 
אָמַר לָעֵדִין: 
"בּוֹאוּ וְהַעִידוּנִי!" 
--"שְׁבוּעָה שֶׁאֵין אָנוּ יוֹדְעִים לָךְ עֵדוּת!" 
אוֹ שֶׁאָמְרוּ לוֹ: 
"אֵין אָנוּ יוֹדְעִין לָךְ עֵדוּת", 
--"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם!" 
וְאָמְרוּ "אָמֵן", 
הֲרֵי אֵלּוּ חַיָּבִין.  
הִשְׁבִּיעַ עֲלֵיהֶן חֲמִשָּׁה פְעָמִים חוּץ לְבֵית דִּין, 
וּבָאוּ לְבֵית דִּין וְהוֹדוּ, 
פְּטוּרִים. 
כָּפְרוּ, 
חַיָּבִים עַל כָּל אַחַת וְאַחַת.  
הִשְׁבִּיעַ עֲלֵיהֶן חֲמִשָּׁה פְעָמִים, 
בֵּין בִּפְנֵי בֵית דִּין, בֵּין שֶׁלֹּא בִפְנֵי בֵית דִּין,
וְכָפְרוּ, 
אֵין חַיָּבִין אֶלָּא אַחַת.  
אָמַר רְבִּי שִׁמְעוֹן: 
מַה הַטַּעַם? 
מִפְּנֵי שֶׁאֵינָם יְכוּלִין לַחְזוֹר וּלְהוֹדוֹת. &lt;וּלְהוֹרוֹת&gt;

<small>ד</small>
כָּפְרוּ שְׁנֵיהֶן כְּאַחַת, 
שְׁנֵיהֶן חַיָּבִין; 
זֶה אַחַר זֶה, 
הָרִאשׁוֹן חַיָּב.  
כָּפַר אֶחָד וְהוֹדָה אֶחָד, 
הַכּוֹפֵר חַיָּב.  
הָיוּ שְׁתֵּי כִתֵּי עֵדִים, 
כָּפְרָה הָרִאשׁוֹנָה וְאַחַר כָּךְ כָּפְרָה הַשְּׁנִיָּה, 
שְׁתֵּיהֶן חַיָּבוֹת, 
מִפְּנֵי שֶׁהָעֵדוּת יְכוּלָה לְהִתְקַיֵּם בִּשְׁתֵּיהֶן. 

<small>ה</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם, 
אִם לֹא תָבֹאוּ וּתְעִידוּנִי, 
שֶׁיֶּשׁ לִי בְיַד פְּלוֹנִי פִקָּדוֹן וּתְשׁוּמֶת יָד וְגָזֵל וַאֲבֵדָה", 
--"שְׁבוּעָה שֶׁאֵין אָנוּ יוֹדְעִין לָךְ עֵדוּת", 
אֵינָן חַיָּבִין אֶלָּא אַחַת. 
--"שְׁבוּעָה שֶׁאֵין אָנוּ יוֹדְעִין, 
שֶׁיֶּשׁ לָךְ בְּיַד פְּלוֹנִי פִּקָּדוֹן וּתְשׁוּמֶת יָד, גָּזֵל וַאֲבֵדָה", 
חַיָּבִין עַל כָּל אַחַת וְאַחַת.  

<small>ו</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם, 
אִם לֹא תָבֹאוּ וּתְעִידוּנִי, 
שֶׁיֶּשׁ לִי בְיַד פְּלוֹנִי חִטִּין וּשְׂעוֹרִין וְכֻסְּמִין", 
--"שְׁבוּעָה שֶׁאֵין אָנוּ יוֹדְעִין לָךְ עֵדוּת", 
אֵינָן חַיָּבִין אֶלָּא אַחַת; 
--"שְׁבוּעָה שֶׁאֵין אָנוּ יוֹדְעִין לָךְ עֵדוּת, 
שֶׁיֶּשׁ לָךְ בְּיַד אִישׁ פְּלוֹנִי חִטִּים וּשְׂעוֹרִין וְכֻסְּמִין", 
חַיָּבִין עַל כָּל אַחַת וְאַחַת. 

<small>ז</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם אִם לֹא תָבֹאוּ וּתְעִידוּנִי, 
שֶׁיֶּשׁ לִי בְיַד פְּלוֹנִי נֶזֶק, וַחֲצִי נֶזֶק, 
תַּשְׁלוּמֵי כֶפֶל, 
וְתַשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה, 
וְשֶׁאָנַס אִישׁ פְּלוֹנִי וּפִּתָּה בִתִּי, 
וְשֶׁהִכַּנִי בְנִי, 
וְשֶׁחָבַל בִּי חֲבֵרִי, 
וְשֶׁהִדְלִיק אֶת גְּדִישִׁי בְיוֹם הַכִּפּוּרִים", 
הֲרֵי אֵלּוּ חַיָּבִין. 

<small>ח</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם אִם לֹא תָבֹאוּ וּתְעִידוּנִי, 
שֶׁאֲנִי כֹהֵן, 
שֶׁאֲנִי לֵוִי, 
שֶׁאֵינִי בֶן גְּרוּשָׁה, 
שֶׁאֵינִי בֶן חֲלוּצָה, 
שֶׁאִישׁ פְּלוֹנִי כֹהֵן, 
שֶׁאִישׁ פְּלוֹנִי לֵוִי, 
שֶׁאֵינוּ בֶן גְּרוּשָׁה, 
שֶׁאֵינוּ בֶן חֲלוּצָה, 
שֶׁאָנַס אִישׁ פְּלוֹנִי וּפִּתָּה בִתּוֹ, 
וְשֶׁחָבַל בּוֹ בְנוֹ, 
וְשֶׁהִדְלִיק גְּדִישִׁי בַשַּׁבָּת", 
הֲרֵי אֵלּוּ פְטוּרִין. 

<small>ט</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם אִם לֹא תָבֹאוּ וּתְעִידוּנִי, 
שֶׁאָמַר אִישׁ פְּלוֹנִי לִתֵּן לִי מָאתַיִם זוּז וְלֹא נָתַן", 
הֲרֵי אֵלּוּ פְטוּרִים, 
שֶׁאֵין חַיָּבִין [אֶלָּא] עַל תְּבִיעַת מָמוֹן כַּפִּקָּדוֹן. 

<small>י</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם, 
כְּשֶׁתֵּדְעוּ לִי עֵדוּת, שֶׁתָּבוֹאוּ וּתְעִידוּנִי", 
הֲרֵי אֵלּוּ פְטוּרִין, 
מִפְּנֵי שֶׁקָּדְמָה שְׁבוּעָה לָעֵדוּת. 

<small>יא</small>
עָמַד בְּבֵית הַכְּנֶסֶת, וְאָמַר: 
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם אִם יוֹדְעִין אַתֶּם לִי עֵדוּת, 
שֶׁתָּבוֹאוּ וּתְעִידוּנִי", 
הֲרֵי אֵלּוּ פְטוּרִין. 

<small>יב</small>
אָמַר לִשְׁנַיִם: 
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם, אִישׁ פְּלוֹנִי וּפְלוֹנִי, 
אִם יוֹדְעִין אַתֶּם לִי עֵדוּת, 
שֶׁתָּבוֹאוּ וּתְעִידוּנִי", 
וְהֵם שֶׁיּוֹדְעִין לוֹ עֵדוּת עֵד מִפִּי עֵד, &lt;לִי&gt;
אוֹ שֶׁהָיָה אֶחָד מֵהֶן קָרוֹב אוֹ פָסוּל, 
הֲרֵי אֵלּוּ פְטוּרִין. 

<small>יג</small>
שִׁלַּח בְּיַד עַבְדּוֹ, 
אוֹ שֶׁאָמַר לָהֶן הַנִּתְבָּע: 
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם, אִם יוֹדְעִין אַתֶּם לִי עֵדוּת, 
שֶׁתָּבוֹאוּ וּתְעִידוּהוּ!" 
הֲרֵי אֵלּוּ פְטוּרִים, 
עַד שֶׁיִּשְׁמְעוּ מִפִּי הַתּוֹבֵעַ. 

<small>יד</small>
"מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם", 
"מְצַוֶּה אֲנִי עֲלֵיכֶם", 
"אוֹסֶרְכֶם אֲנִי", 
הֲרֵי אֵלּוּ חַיָּבִין.  
"בַּשָּׁמַיִם וּבָאָרֶץ", 
הֲרֵי אֵלּוּ פְטוּרִין.  
"בְּאָלֶף דָלֶת", 
"בְּיוֹד הֵי", 
"בְּשַׁדַּי", 
"בִּצְבָאוֹת", 
"בְּחַנּוּן וּבְרַחוּם", 
"בְּאֶרֶךְ אַפַּיִם וְרַב חֶסֶד", 
וְכָל הַכִּנּוּיִים, 
הֲרֵי אֵלּוּ חַיָּבִין.  
הַמְקַלֵּל בְּכֻלָּם, חַיָּב. 
דִּבְרֵי רְבִּי מֵאִיר. 
וַחֲכָמִים פּוֹטְרִין.  
הַמְקַלֵּל אָבִיו וְאִמּוֹ בְכֻלָּם, חַיָּב. 
דִּבְרֵי רְבִּי מֵאִיר. 
וַחֲכָמִים פּוֹטְרִין.  
הַמְקַלֵּל עַצְמוֹ וַחֲבֵרוֹ בְכֻלָּם, 
עוֹבֵר בְּ'לֹא תַעֲשֶׂה'.  
"יַכְּכָה הָאֱלֹהִים!" 
זוֹ הִיא הָאָלָה הָאֲמוּרָה בַתּוֹרָה (ויקרא ה,א) 
"אַל יַכְּכָה", 
"יְבָרֶכְךָ וְיֵיטִיב לָךְ", 
רְבִּי מֵאִיר מְחַיֵּב, 
רְבִּי יְהוּדָה פוֹטֵר. 





Chapter 5

<small>א</small>
שְׁבוּעַת הַפִּקָּדוֹן נוֹהֶגֶת בָּאֲנָשִׁים וּבַנָּשִׁים, 
וּבָרְחוֹקִים וּבַקְּרוֹבִים, 
וּבַכְּשֵׁרִים וּבַפְּסוּלִים, 
בִּפְנֵי בֵית דִּין וְשֶׁלֹּא בִפְנֵי בֵית דִּין, 
מִפִּי עַצְמוֹ וּמִפִּי אַחֵר. 
אֵינוּ חַיָּב עַד שֶׁיִּכְפֹּר בּוֹ בְּבֵית דִּין.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
בֵּין מִפִּי עַצְמוֹ וּבֵין מִפִּי אַחֵר, 
כֵּיוָן שֶׁכָּפַר בּוֹ, חַיָּב.  
וְחַיָּב עַל זְדוֹן הַשְּׁבוּעָה, 
וְעַל שִׁגְגָתָהּ עִם זְדוֹן הַפִּקָּדוֹן, 
אֵינוּ חַיָּב עַל שִׁגְגָתָהּ.  
מַה הוּא חַיָּב עַל זְדוֹנוֹ? 
אָשָׁם בְּכֶסֶף שְׁקָלִים. 

<small>ב</small>
שְׁבוּעַת הַפִּקָּדוֹן כֵּיצַד? 
אָמַר לוֹ: 
"תֶּן לִי פִקָּדוֹן שֶׁיֶּשׁ לִי בְיָדָךְ", 
--"שְׁבוּעָה שֶׁאֵין לָךְ בְּיָדִי". 
אוֹ שֶׁאָמַר לוֹ: 
"אֵין לָךְ בְּיָדִי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
הֲרֵי זֶה חַיָּב.  
הִשְׁבִּיעַ עָלָיו חֲמִשָּׁה פְעָמִים, 
בִּפְנֵי בֵית דִּין, &lt;שֶׁלֹּא בִפְנֵי בֵית דִּין&gt; 
וְכָפַר, 
חַיָּב עַל כָּל אַחַת וְאַחַת.  
אָמַר רְבִּי שִׁמְעוֹן: 
מַה הַטַּעַם? 
מִפְּנֵי שֶׁהוּא יָכוֹל לַחֲזוֹר וּלְהוֹדוֹת. 

<small>ג</small>
הָיוּ חֲמִשָּׁה תוֹבְעִין אוֹתוֹ, וְאוֹמְרִין לוֹ: 
"תֶּן לָנוּ פִקָּדוֹן שֶׁיֶּשׁ לָנוּ בְיָדָךְ", 
--"שְׁבוּעָה שֶׁאֵין לָכֶם בְּיָדִי", 
אֵינוּ חַיָּב אֶלָּא אַחַת; 
--"שְׁבוּעָה שֶׁאֵין לָךְ בְּיָדִי, וְלֹא לָךְ וְלֹא לָךְ וְלֹא לָךְ", 
חַיָּב עַל כָּל אַחַת וְאַחַת. 
(רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
עַד שֶׁיֹּאמַר "שְׁבוּעָה" בָאַחֲרוֹנָה. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
עַד שֶׁיֹּאמַר "שְׁבוּעָה" לְכָל אֶחָד וְאֶחָד.)  

<small>ד</small>
"תֶּן לִי פִקָּדוֹן וּתְשׁוּמֶת יָד, וְגָזֵל וַאֲבֵדָה, 
שֶׁיֶּשׁ לִי בְיָדָךְ", 
--"שְׁבוּעָה שֶׁאֵין לָךְ בְּיָדִי", 
אֵינוּ חַיָּב אֶלָּא אַחַת. 
--"שְׁבוּעָה שֶׁאֵין לָךְ בְּיָדִי פִקָּדוֹן וּתְשׁוּמֶת יָד, 
וְגָזֵל וַאֲבֵדָה", 
חַיָּב עַל כָּל אַחַת וְאַחַת.  

<small>ה</small>
"תֶּן לִי חִטִּין וּשְׂעוֹרִין וְכֻסְּמִין שֶׁיֶּשׁ לִי בְיָדָךְ", 
--"שְׁבוּעָה שֶׁאֵין לָךְ בְּיָדִי", 
אֵינוּ חַיָּב אֶלָּא אַחַת. 
--"שְׁבוּעָה שֶׁאֵין לָךְ בְּיָדִי חִטִּין וּשְׂעוֹרִין וְכֻסְּמִין", 
חַיָּב עַל כָּל אַחַת וְאַחַת. 
רְבִּי מֵאִיר אוֹמֵר: 
אֲפִלּוּ אָמַר: 
"חִטָּה וּשְׂעוֹרָה וְכֻסֶּמֶת", 
חַיָּב עַל כָּל אַחַת וְאַחַת. &lt;לְכָל&gt;

<small>ו</small>
"אָנַסְתָּ וּפִתִּיתָ אֶת בִּתִּי", 
וְהוּא אוֹמֵר: 
"לֹא אָנַסְתִּי וְלֹא פִתִּיתִי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
חַיָּב.  
רְבִּי שִׁמְעוֹן פּוֹטֵר, 
שֶׁאֵינוּ מְשַׁלֵּם קְנָס עַל פִּי עַצְמוֹ.  
וְאָמְרוּ לוֹ: 
אַף עַל פִּי שֶׁאֵינוּ מְשַׁלֵּם קְנָס עַל פִּי עַצְמוֹ, 
מְשַׁלֵּם בֹּשֶׁת וּפְגָם עַל פִּי עַצְמוֹ. 

<small>ז</small>
"גָּנַבְתָּ אֶת שׁוֹרִי", 
וְהוּא אוֹמֵר: 
"לֹא גָנַבְתִּי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
חַיָּב.  
"גָּנַבְתִּי, אֲבָל לֹא טָבַחְתִּי וְלֹא מָכַרְתִּי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר.  

<small>ח</small>
"הֵמִית שׁוֹרָךְ אֶת שׁוֹרִי", 
וְהוּא אוֹמֵר: 
"לֹא הֵמִית", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
חַיָּב.  
"הֵמִית שׁוֹרָךְ אֶת עַבְדִּי", 
וְהוּא אוֹמֵר: 
"לֹא הֵמִית", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר.  
אָמַר לוֹ חֲבֵרוֹ: 
"חָבַלְתָּ בִי וְעָשִׂיתָ בִּי חַבּוּרָה", 
וְהוּא אוֹמֵר: 
"לֹא חָבַלְתִּי וְלֹא עָשִׂיתִי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
חַיָּב.  
אָמַר לוֹ עַבְדּוֹ: 
"הִפַּלְתָּ אֶת שִׁנִּי וְסִמִּיתָ אֶת עֵינִי", &lt;שִׁנַּי; עֵינַי&gt;
וְהוּא אוֹמֵר: 
"לֹא הִפַּלְתִּי וְלֹא סִמִּיתִי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר.  
זֶה הַכְּלָל: 
כָּל הַמְשַׁלֵּם עַל פִּי עַצְמוֹ חַיָּב, 
וְשֶׁאֵינוּ מְשַׁלֵּם עַל פִּי עַצְמוֹ פָּטוּר. 





Chapter 6

<small>א</small>
שְׁבוּעַת הַדַּיָּנִין, 
הַטַּעֲנָה שְׁתֵּי כֶסֶף, וְהַהוֹדָיָה בְשׁוֹוֶה פְרוּטָה. 
וְאִם אֵין הוֹדָיָה מִן הַטַּעֲנָה, 
פָּטוּר.  
כֵּיצַד? 
"שְׁתֵּי כֶסֶף לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי", 
פָּטוּר.  
"שְׁתֵּי כֶסֶף וּפְרוּטָה לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי", 
פָּטוּר.  
"שְׁתֵּי כֶסֶף וּפְרוּטָה לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי אֶלָּא פְרוּטָה", 
חַיָּב.  
"מָנֶה לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי", 
פָּטוּר.  
"מָנֶה לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי אֶלָּא חֲמִשִּׁים דִּינָר", 
חַיָּב.  
"מָנֶה לְאַבָּא בְיָדָךְ", 
--"אֵין לוֹ בְיָדִי אֶלָּא חֲמִשִּׁים דִּינָר", &lt;לָךְ&gt;
פָּטוּר, מִפְּנֵי שֶׁהוּא כְמֵשִׁיב אֲבֵדָה. 

<small>ב</small>
"מָנֶה לִי בְיָדָךְ", 
אָמַר לוֹ "הִין", 
וּלְמָחָר אָמַר לוֹ: 
"תְּנֵהוּ לִי!" 
--"נְתַתִּיו לָךְ", 
פָּטוּר. 
--"אֵין לָךְ בְּיָדִי", 
חַיָּב.  
"מָנֶה לִי בְיָדָךְ", 
אָמַר לוֹ "הִין", 
--"אַל תִּתְּנֵהוּ לִי אֶלָּא בִפְנֵי עֵדִים!" 
לְמָחָר אָמַר לוֹ: 
"תְּנֵהוּ לִי!" 
--"נְתַתִּיו לָךְ", 
חַיָּב, מִפְּנֵי שֶׁהוּא צָרִיךְ לִתְּנוֹ לוֹ בִפְנֵי עֵדִים. 

<small>ג</small>
"לִטְרָה זָהָב לִי בְיָדָךְ!" 
--"אֵין לָךְ בְּיָדִי אֶלָּא לִטְרָה כֶסֶף", 
פָּטוּר. 
"דִּינַר זָהָב לִי בְיָדָךְ!" 
--"אֵין לָךְ בְּיָדִי אֶלָּא דִינַר כֶּסֶף, 
וּטְרִיסוֹת, וּפָנְדְּיוֹן, וּפְרוּטָה", 
חַיָּב, שֶׁהַכֹּל מִין מַטְבֵּעַ אֶחָד.  

<small>ד</small>
"כּוֹר תְּבוּאָה יֶשׁ לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי אֶלָּא לֶתֶךְ קִטְנִית", 
פָּטוּר. 
"כּוֹר פֵּרוֹת יֶשׁ לִי בְיָדָךְ", 
--"אֵין לָךְ בְּיָדִי אֶלָּא לֶתֶךְ קִטְנִית", 
חַיָּב, שֶׁהַקִּטְנִית בִּכְלַל הַפֵּרוֹת.  
טְעָנוֹ חִטִּין, וְהוֹדָה לוֹ בִשְׂעוֹרִים, 
פָּטוּר. 
רַבָּן גַּמְלִיאֵל מְחַיֵּב,  
שֶׁהַטּוֹעֵן אֶת חֲבֵרוֹ בְכַדֵּי שֶׁמֶן, 
וְהוֹדָה לוֹ בְקִנְקְנִים, &lt;בקינקינים&gt;
אַדְמוֹן אוֹמֵר: 
הוֹאִיל וְהוֹדָה מִן הַטַּעֲנָה, יִשָּׁבַע. 
וַחֲכָמִים אוֹמְרִים: 
אֵין הוֹדָיָה מִן הַטַּעֲנָה.  
אָמַר רַבָּן גַּמְלִיאֵל: 
רוֹאֶה אֲנִי אֶת דִּבְרֵי אַדְמוֹן.  

<small>ה</small>
טְעָנוֹ כֵלִים וְקַרְקָעוֹת, 
הוֹדָה בַכֵּלִים וְכָפַר בַּקַּרְקָעוֹת, 
בַּקַּרְקָעוֹת וְכָפַר בַּכֵּלִים, 
[פָּטוּר].  
הוֹדָה בְּמִקְצָת הַכֵּלִים, 
חַיָּב, 
בְמִקְצָת הַקַּרְקָעוֹת, 
פָּטוּר, שֶׁהַנְּכָסִים שֶׁאֵין לָהֶן אַחְרָיוּת 
זוֹקְקִין אֶת הַנְּכָסִים שֶׁיֶּשׁ לָהֶן אַחְרָיוּת, 
יִשָּׁבַע עֲלֵיהֶן. 

<small>ו</small>
אֵין נִשְׁבָּעִין עַל טַעֲנַת חֵרֵשׁ שׁוֹטֶה וְקָטָן, 
וְאֵין מַשְׁבִּיעִין אֶת הַקָּטָן. 
אֲבָל נִשְׁבָּעִין לַקָּטָן וְלַהֶקְדֵּשׁ. 

<small>ז</small>
אֵלּוּ דְבָרִין שֶׁאֵין נִשְׁבָּעִים עֲלֵיהֶן: 
הָעֲבָדִים, וְהַשְּׁטָרוֹת, וְהַקַּרְקָעוֹת, וְהַהֶקְדֵּשׁוֹת.  
אֵין בָּהֶן לֹא תַשְׁלוּמֵי כֶפֶל, 
וְלֹא תַשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה.  
שׁוֹמֵר חִנָּם אֵינוּ נִשְׁבָּע, 
וְנוֹשֵׂא שָׂכָר אֵינוּ מְשַׁלֵּם.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
דְּבָרִים שֶׁהוּא חַיָּב בְּאַחְרָיוּתָן, 
נִשְׁבָּעִין עֲלֵיהֶן, 
וְשֶׁאֵינוּ חַיָּב בְּאַחֲרָיוּתָן, 
אֵין נִשְׁבָּעִין עֲלֵיהֶן. 

<small>ח</small>
רְבִּי מֵאִיר אוֹמֵר: 
יֵשׁ דְּבָרִין שֶׁהֵן כַּקַּרְקַע וְאֵינָן כַּקַּרְקַע. 
וְאֵין חֲכָמִים מוֹדִין לוֹ.  
כֵּיצַד? 
אָמַר לוֹ:
"עֶשֶׂר גְּפָנִים טְעוּנוֹת מָסַרְתִּי לָךְ", 
וְהַלָּה אוֹמֵר: 
"אֵינָן אֶלָּא חָמֵשׁ", 
רְבִּי מֵאִיר מְחַיֵּב, 
וַחֲכָמִים אוֹמְרִים: 
כָּל הַמְחֻבָּר לַקַּרְקַע הֲרֵי הוּא כַקַּרְקַע.  

<small>ט</small>
אֵין נִשְׁבָּעִין אֶלָּא עַל דָּבָר שֶׁהוּא בַמִּדָּה וּבַמִּשְׁקָל וּבַמִּנְיָן.  
כֵּיצַד? 
"בַּיִת מָלֵא מָסַרְתִּי לָךְ", 
וְ"כִּיס מָלֵא מָסַרְתִּי לָךְ", 
וְהַלָּה אוֹמֵר: 
"אֵינִי יוֹדֵעַ, אֶלָּא מַה שֶּׁהִנַּחְתָּ אַתָּה נוֹטֵל", 
פָּטוּר.  
זֶה אוֹמֵר "עַל הַזִּיז", 
וְזֶה אוֹמֵר "עַל הַחַלּוֹן", 
חַיָּב. 

<small>י</small>
הַמַּלְוֶה אֶת חֲבֵרוֹ עַל הַמַּשְׁכּוֹן, 
וְאָבַד הַמַּשְׁכּוֹן. 
אָמַר לוֹ: 
"סֶלַע הִלְוִיתִיךָ עָלָיו, וְשֶׁקֶל הָיָה שׁוֹוֶה", 
וְהַלָּה אוֹמֵר: 
"לֹא כִי, אֶלָּא סֶלַע הִלְוִיתַנִי עָלָיו, 
וְסֶלַע הָיָה שׁוֹוֶה", 
פָּטוּר.  

<small>יא</small>
"סֶלַע הִלְוִיתִיךָ עָלָיו, וְשֶׁקֶל הָיָה שׁוֹוֶה", 
וְהַלָּה אוֹמֵר: 
"לֹא כִי, אֶלָּא סֶלַע הִלְוִיתַנִי עָלָיו, 
וּשְׁלֹשָׁה דִינָרִים הָיָה שׁוֹוֶה", 
חַיָּב.  

<small>יב</small>
"סֶלַע הִלְוִיתַנִי עָלָיו, וּשְׁתַּיִם הָיָה שׁוֹוֶה", 
וְהַלָּה אוֹמֵר: 
"לֹא כִי, אֶלָּא סֶלַע הִלְוִיתִיךָ עָלָיו, 
וְסֶלַע הָיָה שׁוֹוֶה", 
פָּטוּר.  

<small>יג</small>
"סֶלַע הִלְוִיתַנִי עָלָיו, וּשְׁתַּיִם הָיָה שׁוֹוֶה", 
וְהַלָּה אוֹמֵר: 
"לֹא כִי, אֶלָּא סֶלַע הִלְוִיתִיךָ עָלָיו, 
וַחֲמִשָּׁה דִינָרִים הָיָה שׁוֹוֶה", 
חַיָּב.  
מִי הוּא הַנִּשְׁבָּע? 
מִי שֶׁהַפִּקָּדוֹן אֶצְלוֹ, 
שֶׁמֵּא יִשָּׁבַע זֶה, 
וְיוֹצִיא הַלָּה אֶת הַפִּקָּדוֹן. 





Chapter 7

<small>א</small>
כָּל הַנִּשְׁבָּעִין שֶׁבַּתּוֹרָה נִשְׁבָּעִין וְלֹא מְשַׁלְּמִין.  
אֵלּוּ נִשְׁבָּעִין וְנוֹטְלִין: 
הַשָּׂכִיר, וְהַנִּגְזָל, וְהַנֶּחְבָּל, 
וְשֶׁכְּנֶגְדּוֹ חָשׁוּד עַל הַשְּׁבוּעָה, 
וְהַחַנְוָנִי עַל פִּנְקְסוֹ.  
הַשָּׂכִיר כֵּיצַד? 
אָמַר לוֹ: 
"תֶּן לִי שְׂכָרִי שֶׁיֶּשׁ לִי בְיָדָךְ", 
הוּא אוֹמֵר "נָתַתִּי", 
וְהַלָּה אוֹמֵר "לֹא נָטַלְתִּי", 
הֲרֵי זֶה נִשְׁבָּע וְנוֹטֵל.  
רְבִּי יְהוּדָה אוֹמֵר: 
עַד שֶׁיְּהֵא שָׁם מִקְצָת הוֹדָיָה, 
וְיֹאמַר לוֹ: 
"תֶּן לִי שְׂכָרִי חֲמִשִּׁים דִּינָר שֶׁיֶּשׁ לִי בְיָדָךְ", 
וְהוּא אוֹמֵר: 
"הִתְקַבַּלְתָּ מֵהֶן דִּינַר זָהָב". 

<small>ב</small>
הַנִּגְזָל כֵּיצַד? 
הָיוּ מְעִידִין אוֹתוֹ שֶׁנִּכְנַס לְתוֹךְ בֵּיתוֹ, 
וּמִשְׁכְּנוֹ שֶׁלֹּא בִרְשׁוּת. 
אָמַר לוֹ: 
"תֵּן לִי כֵלַי שֶׁנָּטַלְתָּ", 
וְהוּא אוֹמֵר: 
"לֹא נָטַלְתִּי", 
הֲרֵי זֶה נִשְׁבָּע וְנוֹטֵל.  
רְבִּי יְהוּדָה אוֹמֵר: 
עַד שֶׁיְּהֵא שָׁם מִקְצָת הוֹדָיָה. 
וְיֹאמַר לוֹ: 
"שְׁנֵי כֵלַי נָטַלְתָּ", 
וְהוּא אוֹמֵר: 
"לֹא נָטַלְתִּי אֶלָּא אֶחָד". 

<small>ג</small>
הַנֶּחְבָּל כֵּיצַד? 
הָיוּ מְעִידִין אוֹתוֹ שֶׁנִּכְנַס לְתוֹךְ יָדוֹ שָׁלֵם, 
וְיָצָא חָבוּל. 
אָמַר לוֹ: 
"חָבַלְתָּ בִּי", 
וְהוּא אוֹמֵר: 
"לֹא חָבַלְתִּי", 
הֲרֵי זֶה נִשְׁבָּע וְנוֹטֵל.  
רְבִּי יְהוּדָה אוֹמֵר: 
עַד שֶׁיְּהֵא שָׁם מִקְצָת הוֹדָיָה. 
וְיֹאמַר לוֹ: 
"חָבַלְתָּ בִּי שְׁתַּיִם", 
וְהוּא אוֹמֵר: 
"לֹא חָבַלְתִּי אֶלָּא אַחַת". 

<small>ד</small>
וְשֶׁכְּנֶגְדּוֹ חָשׁוּד עַל הַשְּׁבוּעָה, כֵּיצַד? 
אֶחָד שְׁבוּעַת הָעֵדוּת וְאֶחָד שְׁבוּעַת הַפִּקָּדוֹן, &lt;אֶחָת&gt;
וַאֲפִלּוּ שְׁבוּעַת שָׁוְא.  
הָיָה אֶחָד מֵהֶם מְשַׂחֵק בַּקָּבְיָה, 
וּמַלְוֶה בָרִבִּית, וּמַפְרִיחַ יוֹנִים, וְסוֹחֲרֵי שְׁבִיעִית, 
שֶׁכְּנֶגְדּוֹ נִשְׁבָּע וְנוֹטֵל.  
הָיוּ שְׁנֵיהֶן חֲשׁוּדִים, 
חָזְרָה שְׁבוּעָה לִמְקוֹמָהּ.  
דִּבְרֵי רְבִּי יוֹסֵה.  
רְבִּי מֵאִיר אוֹמֵר: 
יַחֲלֹקוּ. 

<small>ה</small>
וְהַחַנְוָנִי עַל פִּנְקְסוֹ, כֵּיצַד? 
לֹא שֶׁיֹּאמַר לוֹ: 
"כָּתוּב בְּפִנְקְסִי, שֶׁאַתְּ חַיָּב לִי מָאתַיִם זוּז", 
אֶלָּא אוֹמֵר לוֹ: 
"תֵּן לִבְנִי סָאתַיִם חִטִּין, 
וּלְפוֹעֲלַי סֶלַע מָעוֹת". 
הוּא אוֹמֵר: 
"נָתַתִּי", 
וְהֵן אוֹמְרִים: 
"לֹא נָטַלְנוּ", 
הוּא נִשְׁבָּע וְנוֹטֵל, 
וְהֵן נִשְׁבָּעִין וְנוֹטְלִין.  
אָמַר בֶּן נַנָּס: 
כֵּיצַד? 
אֵלּוּ וָאֵלּוּ בָאִין לִידֵי שְׁבוּעַת שָׁוְא! 
אֶלָּא הוּא נוֹטֵל שֶׁלֹּא בִשְׁבוּעָה, 
וְהֵן נוֹטְלִין שֶׁלֹּא בִשְׁבוּעָה. 

<small>ו</small>
אָמַר לַחַנְוָנִי: 
"תֶּן לִי בְדִינָר פֵּרוֹת!" 
וְנָתַן לוֹ. 
אָמַר לוֹ: 
"תֶּן לִי אֶת הַדִּינָר!" 
אָמַר לוֹ: 
"נְתַתִּיו לָךְ, וּנְתַתּוֹ בְאַנְפָּלִי", 
יִשָּׁבַע בַּעַל הַבַּיִת.  
נָתַן לוֹ אֶת הַדִּינָר, אָמַר לוֹ: 
"תֶּן לִי אֶת הַפֵּרוֹת!" 
אָמַר לוֹ: 
"נְתַתִּים לָךְ, וְהוֹלַכְתָּם לְתוֹךְ בֵּיתָךְ", 
יִשָּׁבַע הַחַנְוָנִי.  
רַבִּי יְהוּדָה אוֹמֵר: &lt;יודה&gt;
כָּל שֶׁהַפֵּרוֹת בְּיָדוֹ, יָדוֹ לָעֶלְיוֹנָה.  

<small>ז</small>
אָמַר לַשֻּׁלְחָנִי: 
"תֶּן לִי בְדִינָר מָעוֹת!" 
וְנָתַן לוֹ. 
אָמַר לוֹ: 
"תֶּן לִי אֶת הַדִּינָר!" 
אָמַר לוֹ: 
"נְתַתִּיו לָךְ, וּנְתַתּוֹ בְאַנְפָּלִי", &lt;באנפולי&gt;
יִשָּׁבַע בַּעַל הַבַּיִת.  
נָתַן לוֹ אֶת הַדִּינָר, אָמַר לוֹ: 
"תֶּן לִי אֶת הַמָּעוֹת!" 
אָמַר לוֹ: 
"נְתַתִּים לָךְ, וְהִשְׁלַכְתָּם לְתוֹךְ כִּיסָךְ", 
יִשָּׁבַע הַשֻּׁלְחָנִי.  
רְבִּי יְהוּדָה אוֹמֵר: 
אֵין דֶּרֶךְ הַשֻּׁלְחָנִי לִהְיוֹת נוֹתֵן אִסָּר, 
עַד שֶׁהוּא נוֹטֵל דִּינָרוֹ. 

<small>ח</small>
כַּשֵּׁם שֶׁאָמְרוּ: 
הַפּוֹגֶמֶת כְּתֻבָּתָהּ, 
לֹא תִפָּרַע אֶלָּא בִשְׁבוּעָה, 
וְעֵד אֶחָד מְעִידָהּ שֶׁהִיא פְרוּעָה, 
לֹא תִפָּרַע אֶלָּא בִשְׁבוּעָה. 
מִנְּכָסִים מְשֻׁעְבָּדִין וּמִנִּכְסֵי יְתוֹמִים, 
וְהַנִּפְרַעַת שֶׁלֹּא בְּפָנָיו, 
לֹא תִפָּרַע אֶלָּא בִשְׁבוּעָה. 

<small>ט</small>
וְכֵן הַיְתוֹמִים, 
לֹא יִפָּרְעוּ אֶלָּא בִשְׁבוּעָה: 
"שְׁבוּעָה שֶׁלֹּא פִקְּדָנוּ אַבָּא, 
וְשֶׁלֹּא אָמַר לָנוּ אַבָּא, 
וְשֶׁלֹּא מָצִינוּ שְׁטָר בֵּין שְׁטָרוֹתָיו שֶׁלְּאַבָּא, 
שֶׁשְּׁטָר זֶה פָרוּעַ.  
הֵעִיד רְבִּי יוֹחָנָן בֶּן בְּרוֹקָה, 
שֶׁאֲפִלּוּ נוֹלַד בֵּן לְאַחַר מִיתַת הָאָב, 
הֲרֵי זֶה נִשְׁבָּע וְנוֹטֵל.  
אָמַר רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל: 
אִם יֵשׁ עֵדִין שֶׁאָמַר הָאָב בְּשָׁעַת מִיתָתוֹ: 
"שְׁטָר זֶה אֵינוּ פָרוּעַ", 
נוֹטֵל שֶׁלֹּא בִשְׁבוּעָה. 

<small>י</small>
אֵלּוּ נִשְׁבָּעִין שֶׁלֹּא בְטַעֲנָה: 
הַשּׁוּתָפִים, וְהָאֲרִיסִים, וְהָאַפִּטְרוֹפִּים, 
וְהָאִשָּׁה הַנּוֹשָׂא וְהַנּוֹתֶנֶת בְּתוֹךְ הַבַּיִת, 
וּבֶן הַבַּיִת.  
אָמַר לוֹ: 
"מָה אַתָּה טוֹעֲנֵנִי? 
רְצוֹנִי שֶׁתִּשָּׁבַע לִי", 
חַיָּב.  
חָלְקוּ הַשּׁוּתָפִים וְהָאֲרִיסִים, 
אֵינוּ יָכוֹל לְהַשְׁבִּיעָן. 
נִתְגַּלְגְּלָה לוֹ שְׁבוּעָה מִמָּקוֹם אַחֵר, 
מְגַלְגְּלִים עָלָיו אֶת הַכֹּל.  
וְהַשְּׁבִיעִית מַשְׁמֶטֶת אֶת הַשְּׁבוּעָה. 





Chapter 8

<small>א</small>
אַרְבָּעָה שׁוֹמְרִין הֵן: 
שׁוֹמֵר חִנָּם, וְהַשּׁוֹאֵל, נוֹשֵׂא שָׂכָר, וְהַשּׂוֹכֵר.  
שׁוֹמֵר חִנָּם נִשְׁבָּע עַל הַכֹּל, 
וְהַשּׁוֹאֵל מְשַׁלֵּם אֶת הַכֹּל, 
וְנוֹשֵׂא שָׂכָר וְהַשּׂוֹכֵר נִשְׁבָּעִין עַל הַשֶּׁבֶר, 
וְעַל הַשְּׁבוּיָה וְעַל הַמֵּתָה, 
וּמְשַׁלְּמִים אֶת הָאֲבֵדָה וְאֶת הַגְּנֵבָה. 

<small>ב</small>
אָמַר לְשׁוֹמֵר חִנָּם: 
"אֵיכָן שׁוֹרִי?" 
אָמַר לוֹ: 
"מֵת", 
וְהוּא שֶׁנִּשְׁבַּר אוֹ שֶׁנִּשְׁבָּה אוֹ נִגְנַב אוֹ אָבַד. 
"נִשְׁבַּר", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבָּה אוֹ נִגְנַב אוֹ אָבַד. 
"נִשְׁבָּה", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִגְנַב אוֹ אָבַד. 
"נִגְנַב", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה אוֹ אָבַד. 
"אָבַד", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה אוֹ נִגְנַב. 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר. 

<small>ג</small>
"אֵיכָן שׁוֹרִי?" 
אָמַר לוֹ: 
"אֵינִי יוֹדֵעַ מָה אַתָּה סָח", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה אוֹ נִגְנַב אוֹ אָבַד, 
--"מַשְׁבִּיעָךְ אָנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר. 

<small>ד</small>
"אֵיכָן שׁוֹרִי?" 
אָמַר לוֹ: 
"אָבַד". 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
הָעֵדִין מְעִידִין אוֹתוֹ שֶׁאֲכָלוֹ, 
מְשַׁלֵּם אֶת הַקֶּרֶן.  
הוֹדָה מֵעַצְמוֹ, 
מְשַׁלֵּם קֶרֶן וְחֹמֶשׁ וְאָשָׁם.  

<small>ה</small>
"אֵיכָן שׁוֹרִי?" 
וְאָמַר לוֹ: 
"נִגְנַב". 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
וְהָעֵדִים מְעִידִים אוֹתוֹ שֶׁגְּנָבוֹ, 
מְשַׁלֵּם תַּשְׁלוּמֵי כֶפֶל.  
הוֹדָה מֵעַצְמוֹ, 
מְשַׁלֵּם קֶרֶן וְחֹמֶשׁ וְאָשָׁם. 

<small>ו</small>
אָמַר לְאֶחָד בַּשּׁוּק: 
"אֵיכָן שׁוֹרִי שֶׁגָּנַבְתָּ?" 
וְהוּא אוֹמֵר: 
"לֹא גָנַבְתִּי", 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
וְהָעֵדִים מְעִידִין אוֹתוֹ שֶׁגְּנָבוֹ, 
מְשַׁלֵּם תַּשְׁלוּמֵי כֶפֶל. 
טָבַח וּמָכַר, 
מְשַׁלֵּם תַּשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה. &lt;ארבעה חמשה&gt; 
רָאָה עֵדִים מְמַשְׁמְשִׁים וּבָאִים, וְאָמַר: 
"גָּנַבְתִּי, אֲבָל לֹא טָבַחְתִּי וְלֹא מָכַרְתִּי", 
אֵינוּ מְשַׁלֵּם אֶלָּא קֶרֶן. 

<small>ז</small>
אָמַר לַשּׁוֹאֵל: 
"אֵיכָן שׁוֹרִי?" 
אָמַר לוֹ: 
"מֵת", 
וְהוּא שֶׁנִּשְׁבַּר אוֹ נִשְׁבָּה אוֹ נִגְנַב אוֹ אָבַד. 
"נִשְׁבַּר", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבָּה אוֹ נִגְנַב אוֹ אָבַד. 
"נִשְׁבָּה", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִגְנַב אוֹ אָבַד. 
"נִגְנַב", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה אוֹ אָבַד. 
"אָבַד", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה אוֹ נִגְנַב. 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר. 

<small>ח</small>
"אֵיכָן שׁוֹרִי?" 
אָמַר לוֹ: 
"אֵינִי יוֹדֵעַ מָה אַתָּה סָח", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה אוֹ נִגְנַב אוֹ אָבַד. 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
חַיָּב.  

<small>ט</small>
אָמַר לְנוֹשֵׂא שָׂכָר וְלַשּׂוֹכֵר: 
"אֵיכָן שׁוֹרִי?" 
אָמַר לוֹ: 
"מֵת", 
וְהוּא שֶׁנִּשְׁבַּר אוֹ שֶׁנִּשְׁבָּה. 
"נִשְׁבַּר", 
וְהוּא שֶׁמֵּת אוֹ שֶׁנִּשְׁבָּה. 
"נִשְׁבָּה", 
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר. 
"נִגְנַב", 
וְהוּא שֶׁאָבַד. 
"אָבַד", 
וְהוּא שֶׁנִּגְנַב, 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר.  

<small>י</small>
"מֵת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה", 
וְהוּא שֶׁנִּגְנַב אוֹ שֶׁאָבַד. 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
חַיָּב.  
"אָבַד אוֹ נִגְנַב", &lt;שֶׁנִּגְנַב&gt;
וְהוּא שֶׁמֵּת אוֹ נִשְׁבַּר אוֹ נִשְׁבָּה, 
--"מַשְׁבִּיעָךְ אֲנִי", 
וְאָמַר "אָמֵן", 
פָּטוּר.  
(זֶה הַכְּלָל: 
כָּל הַמְשַׁנֶּה מֵחוֹבָה לְחוֹבָה, 
וּמִפְּטוּר לִפְטוּר, 
וּמִפְּטוּר לְחוֹבָה, פָּטוּר; 
מֵחוֹבָה לִפְטוּר, חַיָּב.)  
זֶה הַכְּלָל: 
כָּל הַנִּשְׁבָּע לְהָקֵל עַל עַצְמוֹ, 
חַיָּב; 
לְהַחְמִיר עַל עַצְמוֹ, 
פָּטוּר.
