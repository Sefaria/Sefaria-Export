משנה כלים



Chapter 1

<small>א</small>
אֲבוֹת הַטְּמָאוֹת: 
הַשֶּׁרֶץ, 
וְשִׁכְבַת זֶרַע, 
וּטְמֵא מֵת, 
וּמְצֹרָע בִּימֵי סְפָרוֹ, 
וּמֵי חַטָּאת שֶׁאֵין בָּהֶן כְּדֵי הַזָּיָה, 
הֲרֵי אֵלּוּ מְטַמִּין אָדָם וְכֵלִים בְּמַגָּע, 
וּכְלֵי חֶרֶשׂ בְּאָוֵיר, 
וְאֵינָן מְטַמִּין בְּמַשָּׂא. 

<small>ב</small>
לְמַעְלָה מֵהֶן: 
נְבֵלָה, 
וּמֵי חַטָּאת שֶׁיֵּשׁ בָּהֶן כְּדֵי הַזָּיָה, 
שֶׁהֵן מְטַמִּין אֶת הָאָדָם בְּמַשָּׂא, 
לְטַמֵּא בְגָדִים וַחֲסוּכֵי בְגָדִים בְּמַגָּע. 

<small>ג</small>
לְמַעְלָה מֵהֶן: 
בּוֹעֵל נִדָּה, 
שֶׁהוּא מְטַמֵּא מִשְׁכָּב תַּחְתּוֹן כָּעֶלְיוֹן.  
לְמַעְלָה מֵהֶן: 
זוֹבוֹ שֶׁלַּזָּב, וְרוֹקוֹ, וְשִׁכְבַת זַרְעוֹ, וּמֵימֵי רַגְלָיו, 
וְדַם הַנִּדָּה, 
שֶׁהֵן מְטַמִּין בְּמַגָּע וּבְמַשָּׂא.  
לְמַעְלָה מֵהֶן: 
מִרְכָּב, 
שֶׁהוּא מְטַמֵּא תַחַת אֶבֶן מְסָמָא.  
לְמַעְלָה מִן הַמִּרְכָּב: 
מִשְׁכָּב, 
שֶׁשָּׁוֶה מַגָּעוֹ לְמַשָּׂאוֹ.  
לְמַעְלָה מִן הַמִּשְׁכָּב: 
הַזָּב, 
שֶׁהַזָּב עוֹשֶׂה מִשְׁכָּב, 
וְאֵין מִשְׁכָּב עוֹשֶׂה מִשְׁכָּב. 

<small>ד</small>
לְמַעְלָה מִן הַזָּב: 
זָבָה, 
שֶׁהִיא מְטַמָּא אֶת בּוֹעֲלָהּ.  
לְמַעְלָה מִן הַזָּבָה: 
מְצֹרָע, 
שֶׁהוּא מְטַמֵּא בַבִּיאָה.  
לְמַעְלָה מִן הַמְּצֹרָע: 
עֶצֶם כִּשְׂעוֹרָה, 
שֶׁהוּא מְטַמֵּא טֻמְאַת שִׁבְעָה.  
חָמוּר מִכֻּלָּם: 
הַמֵּת, 
שֶׁהוּא מְטַמֵּא בְאָהֵל, 
מַה שֶּׁאֵין כֻּלָּם מְטַמִּין. &lt;מִיטַּמִּין&gt;

<small>ה</small>
עֶשֶׂר טְמָאוֹת פּוֹרְשׁוֹת מִן הָאָדָם: 
מְחֻסַּר כִּפּוּרִים, אָסוּר בַּקֹּדֶשׁ, 
וּמֻתָּר בִּתְרוּמָה וּבַמַּעֲשֵׂר.  
חָזַר לִהְיוֹת טְבוּל יוֹם, 
אָסוּר בַּקֹּדֶשׁ וּבַתְּרוּמָה, 
וּמֻתָּר בַּמַּעֲשֵׂר.  
חָזַר לִהְיוֹת בַּעַל קֶרִי, 
אָסוּר בִּשְׁלָשְׁתָּן.  

<small>ו</small>
חָזַר לִהְיוֹת בּוֹעֵל נִדָּה, 
מְטַמֵּא מִשְׁכָּב תַּחְתּוֹן כָּעֶלְיוֹן.  
חָזַר לִהְיוֹת זָב שֶׁרָאָה שְׁתֵּי רְאִיּוֹת, 
מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב, 
וְצָרִיךְ בִּיאַת מַיִם חַיִּים, 
וּפָטוּר מִן הַקָּרְבָּן; 
רָאָה שָׁלוֹשׁ, חַיָּב בַּקָּרְבָּן.  

<small>ז</small>
חָזַר לִהְיוֹת מְצֹרָע מֻסְגָּר, 
מְטַמֵּא בַבִּיאָה, 
וּפָטוּר מִן הַפְּרִיעָה, 
וּמִן הַפְּרִימָה, 
וּמִן הַתִּגְלַחַת, 
וּמִן הַצִּפֳּרִין. 
וְאִם הָיָה מֻחְלָט, 
חַיָּב בְּכֻלָּן.  

<small>ח</small>
פָּרַשׁ מִמֶּנּוּ אֵבֶר שֶׁאֵין עָלָיו בָּשָׂר כָּרָאוּי, 
מְטַמֵּא בְמַגָּע וּבְמַשָּׂא, 
וְאֵינוּ מְטַמֵּא בְאָהֵל. 
אִם יֵשׁ עָלָיו בָּשָׂר כָּרָאוּי, 
מְטַמֵּא בְמַגָּע וּבְמַשָּׂא וּבְאָהֵל.  

<small>ט</small>
שֵׁעוּר הַבָּשָׂר, 
כְּדֵי לַעֲלוֹת אֲרוּכָה.  
רְבִּי יְהוּדָה אוֹמֵר: 
אִם יֵשׁ בְּמָקוֹם אֶחָד כְּדֵי לְהַקִּיפוֹ בְחוּט עֵרֶב, 
יֵשׁ בּוֹ לַעֲלוֹת אֲרוּכָה. 

<small>י</small>
עֶשֶׂר קְדֻשּׁוֹת הֵן: 
אֶרֶץ יִשְׂרָאֵל מְקֻדֶּשֶׁת מִכָּל הָאֲרָצוֹת.  
מַה הִיא קְדֻשָּׁתָהּ? 
שֶׁמְּבִיאִין מִמֶּנָּה הָעֹמֶר, וְהַבִּכּוּרִים, וּשְׁתֵּי הַלֶּחֶם, 
מַה שֶּׁאֵין מְבִיאִין כֵּן מִכָּל הָאֲרָצוֹת. 

עֲיָרוֹת מֻקָּפוֹת חוֹמָה מְקֻדָּשׁוֹת מִמֶּנָּה, 
שֶׁמְּשַׁלְּחִין מִתּוֹכָן מְצֹרָעִין, 
וּמְסַבְּבִין לְתוֹכָן מֵת עַד שֶׁיִּרְצוּ; 
יָצָא, אֵין מַחְזִירִין אוֹתוֹ. 

<small>יא</small>
לִפְנִים מִן הַחוֹמָה מְקֻדָּשׁ מֵהֶן, 
שֶׁאוֹכְלִים שָׁם קָדָשִׁים קַלִּים וּמַעֲשֵׂר שֵׁנִי.  
הַר הַבַּיִת מְקֻדָּשׁ מִמֶּנּוּ, 
שֶׁאֵין זָבִים וְזָבוֹת, נִדּוֹת וְיוֹלְדוֹת נִכְנָסִים לַשָּׁם.  
הַחֵיל מְקֻדָּשׁ מִמֶּנּוּ, 
שֶׁאֵין גּוֹיִם וּטְמֵא מֵת נִכְנָסִין לַשָּׁם.  

<small>יב</small>
עֲזָרַת הַנָּשִׁים מְקֻדֶּשֶׁת מִמֶּנּוּ, 
שֶׁאֵין טְבוּל יוֹם נִכְנָס לַשָּׁם, 
וְאֵין חַיָּבִין עָלֶיהָ חַטָּאת.  
עֲזָרַת יִשְׂרָאֵל מְקֻדֶּשֶׁת מִמֶּנָּה, 
שֶׁאֵין מְחֻסַּר כִּפּוּרִין נִכְנָס לַשָּׁם, 
וְחַיָּבִין עָלֶיהָ חַטָּאת.  
עֲזָרַת הַכֹּהֲנִים מְקֻדֶּשֶׁת מִמֶּנָּה, 
שֶׁאֵין יִשְׂרָאֵל נִכְנָסִין לַשָּׁם אֶלָּא בְשָׁעַת צָרְכֵיהֶן: 
לִסְמִיכָה, לִשְׁחִיטָה, לִתְנוּפָה. 

<small>יג</small>
בֵּין הָאוּלָם וְלַמִּזְבֵּחַ מְקֻדָּשׁ מִמֶּנָּה, 
שֶׁאֵין בַּעֲלֵי מוּמִין וּפְרוּעֵי רֹאשׁ נִכְנָסִין לַשָּׁם.  
הַהֵיכָל מְקֻדָּשׁ מִמֶּנּוּ, 
שֶׁאֵין נִכְנָס לַשָּׁם אֶלָּא רְחוּץ יָדַיִם וְרַגְלַיִם.  

<small>יד</small>
בֵּית קֹדֶשׁ הַקֳּדָשִׁים מְקֻדָּשׁ מֵהֶן, 
שֶׁאֵין נִכְנָס לַשָּׁם 
אֶלָּא כֹהֵן גָּדוֹל בְּיוֹם הַכִּפּוּרִים בְּשָׁעַת הָעֲבוֹדָה.  
אָמַר רְבִּי יוֹסֵה: 
חֲמִשָּׁה דְבָרִים בֵּין הָאוּלָם וְלַמִּזְבֵּחַ שָׁוֶה לַהֵיכָל: 
שֶׁאֵין בַּעֲלֵי מוּמִין וּפְרוּעֵי רֹאשׁ וּשְׁתוּיֵי יַיִן, 
וְשֶׁלֹּא רְחוּץ יָדַיִם וְרַגְלַיִם נִכְנָסִין לַשָּׁם. 
וּפוֹרְשִׁין בֵּין הָאוּלָם וְלַמִּזְבֵּחַ בְּשָׁעַת הַקְטָרָה. 





Chapter 2

<small>א</small>
כְּלֵי עֵץ, כְּלֵי עוֹר, כְּלֵי עֶצֶם, כְּלֵי זְכוֹכִית, 
פְּשׁוּטֵיהֶן טְהוֹרִין, 
וּמְקַבְּלֵיהֶן טְמֵאִים.  
נִשְׁבָּרוּ, טָהָרוּ.  
חָזַר וְעָשָׂה מֵהֶן כֵּלִים, 
מְקַבְּלִין טֻמְאָה מִכָּן וְלַבָּא.  
כְּלֵי חֶרֶשׂ וּכְלֵי נֶתֶר טֻמְאָתָן שָׁוָה, 
מִטַּמִּין וּמְטַמִּין בְּאָוֵיר, 
וּמִטַּמִּין מֵאַחֲרֵיהֶן, 
וְאֵינָן מִטַּמִּין מִגַּבֵּיהֶן, 
וּשְׁבִירָתָן הִיא טַהֲרָתָן. 

[<small>ב</small>]
הַדַּקִּין שֶׁבִּכְלֵי חֶרֶשׂ, 
וְקַרְקְרוֹתֵיהֶן, וּדְפָנוֹתֵיהֶן, 
יוֹשְׁבִין עַל שׁוּלֵיהֶן,
יוֹשְׁבִין שֶׁלֹּא מְסֻמָּכִין: 
שֵׁעוּרָן, מִכְּדֵי סִיכַת קָטָן, וְעַד לֹג.  
מִלֹּג וְעַד סְאָה, בָּרְבִיעִית.  
מִסְּאָה וְעַד סָאתַיִם, כַּחֲצִי לֹג.  
מִסָּאתַיִם וְעַד שָׁלוֹשׁ, וְעַד חָמֵשׁ סְאִים, בַּלֹּג.  
דִּבְרֵי רְבִּי יִשְׁמָעֵאל.  
רְבִּי עֲקִיבָה אוֹמֵר: 
אֲנִי אֵינִי נוֹתֵן בָּהֶן מִדָּה, 
אֶלָּא הַדַּקִּין שֶׁבִּכְלֵי חֶרֶשׂ, 
וְקַרְקְרוֹתֵיהֶן, וּדְפָנוֹתֵיהֶן, 
שֶׁלֹּא מְסֻמָּכִין, 
שֵׁעוּרָן מִכְּדֵי סִיכַת קָטָן, 
וְעַד קְדֵרוֹת הַדַּקּוֹת.  
מִקְּדֵרוֹת הַדַּקּוֹת וְעַד חֲבִיּוֹת לוּדִיּוֹת, 
בָּרְבִיעִית.  
מִלּוּדִיּוֹת וְעַד לַחְמִיּוֹת, בַּחֲצִי לֹג.  
מִלַּחְמִיּוֹת וְעַד חֲצָבִים גְּדוֹלִים, בַּלֹּג.
רַבָּן יוֹחָנָן בֶּן זַכַּי אוֹמֵר: 
חֲצָבִים גְּדוֹלִים, 
שֵׁעוּרָן כִּשְׁנֵי לָגִּין; 
הַפַּכִּין הַגְּלִילִים וְהַחֲבִיּוֹנוֹת, 
שֵׁעוּר קַרְקְרוֹתֵיהֶן כָּל שֶׁהֵן, 
וְאֵין לָהֶן דְּפָנוֹת. 

<small>ג</small>
הַטְּהוֹרִים שֶׁבִּכְלֵי חֶרֶשׂ: 
וְטַבְלָה שֶׁאֵין לָהּ לִזְבֵּז, 
וּמַחְתָּה פְרוּצָה, 
וְאַבּוּב שֶׁלַּקַּלָּאִים, 
וְסִלּוֹנוֹת, אַף עַל פִּי כְפוּפִין, 
אַף עַל פִּי מְקַבְּלִין; 
וְכַבְכָּב שֶׁעֲשָׂאוֹ לְסַל הַפַּת, 
וְטָפֵי שֶׁהִתְקִינוֹ לָעֲנָבִים, 
וְחָבִית שֶׁלַּשַּׁיָּטִין, 
וְחָבִית דְּפוּנָה בְשׁוּלֵי הַמַּחַץ, 
וְהַמִּטָּה, 
וְהַכִּסֵּא, 
וְהַסַּפְסֵל, 
וְהַשֻּׁלְחָן, 
וְהַסְּפִינָה, 
וְהַמְּנוֹרָה שֶׁלַּחֶרֶשׂ, 
הֲרֵי אֵלּוּ טְהוֹרִים.  
זֶה הַכְּלָל: 
כָּל שֶׁאֵין לוֹ תוֹךְ בִּכְלֵי חֶרֶשׂ, 
אֵין לוֹ אֲחוֹרַיִם. 

<small>ד</small>
פַּנֵס שֶׁיֶּשׁ בּוֹ בֵית קִבּוּל שֶׁמֶן, טָמֵא, 
וְשֶׁאֵין בּוֹ, טָהוֹר.  
מְגוּפַת הַיּוֹצְרִים שֶׁהוּא פוֹתֵחַ בָּהּ, טְהוֹרָה, 
וְשֶׁהוּא גּוֹמֵר בָּהּ, טְמֵאָה.  
מַשְׁפֵּךְ שֶׁלְּבַעֲלֵי בָתִּים, טָהוֹר, 
וְשֶׁלָּרוֹכְלִין, טָמֵא, מִפְּנֵי שֶׁהוּא שֶׁלַמִּדָּה.  
דִּבְרֵי רְבִּי יְהוּדָה בֶן בְּתִירָה.  
רְבִּי עֲקִיבָה אוֹמֵר: 
מִפְּנֵי שֶׁהוּא מַטֵּהוּ עַל צִדּוֹ, 
וּמֵרֵיחַ בּוֹ לַלּוֹקֵחַ. 

<small>ה</small>
כְּסוּיֵי כַדֵּי יַיִן וְכַדֵּי שֶׁמֶן, 
וּכְסוּיֵי חֲבִית נְיָרוֹת, טְהוֹרִין.  
אִם הִתְקִינָן לְתַשְׁמִישׁ, טְמֵאִין.  
כְּסוּי הַלַּפָּס: 
בִּזְמַן שֶׁהוּא נָקוּב, וְיֵשׁ לוֹ חַדּוּד, טָהוֹר; 
אִם אֵינוּ נָקוּב וְאֵין לוֹ חַדּוּד, טָמֵא, 
מִפְּנֵי שֶׁהִיא מְסַנֶּנֶת לְתוֹכוֹ אֶת הַיָּרָק.  
רְבִּי אֱלִיעֶזֶר בִּרְבִּי צָדוֹק אוֹמֵר: 
מִפְּנֵי שֶׁהִיא הוֹפֶכֶת עָלָיו אֶת הָרוֹנְקֵי. 

<small>ו</small>
גִּצְטְרָה שֶׁנִּמְצֵאת בַּכִּבְשָׁן, 
עַד שֶׁלֹּא נִגְמְרָה מְלַאכְתָּהּ, טְהוֹרָה; 
מִשֶּׁנִּגְמְרָה מְלַאכְתָּהּ, טְמֵאָה.  
טִטְרוֹס, 
רְבִּי אֶלְעָזָר בִּרְבִּי צָדוֹק מְטַהֵר. 
רְבִּי יוֹסֵה מְטַמֵּא, 
מִפְּנֵי שֶׁהוּא מוֹצִיא פְּרוּטוֹת. 

<small>ז</small>
הַטְּמֵאִין שֶׁבִּכְלֵי חֶרֶשׂ: 
וְטַבְלָה שֶׁיֶּשׁ לָהֶן לִזְבֵּז, 
וּמַחְתָּה שְׁלֵמָה, 
וְטַבְלָה שֶׁהִיא מְלֵאָה קְעָרוֹת.  
נִטַּמָּאת אַחַת מֵהֶן בַּשֶּׁרֶץ, לֹא נִטַּמּוּ כֻלָּם.  
אִם יֶשׁ לָהּ לִזְבֵּז עוֹדֵף, 
נִטַּמָּאת אַחַת מֵהֶן, נִטַּמּוּ כֻלָּם.  

<small>ח</small>
וְכֵן בֵּית תַּבְלִין שֶׁלַּחֶרֶשׂ, 
וְקָלָמָרִין הַמַּתְאֶמֶת.  
בֵּית תַּבְלִין שֶׁלָּעֵץ שֶׁנִּטַּמָּא אֶחָד בְּמַשְׁקֶה, &lt;אַחַת&gt;
לֹא נִטַּמָּא חֲבֵרוֹ.  
רְבִּי יוֹחָנָן בֶּן נוּרִי אוֹמֵר: 
חוֹלְקִין אֶת עָבְיוֹ, 
הַמְשַׁמֵּשׁ לַטָּמֵא, טָמֵא, 
וְהַמְשַׁמֵּשׁ לַטָּהוֹר, טָהוֹר.  
אִם יֵשׁ עָלָיו לִזְבֵּז עוֹדֵף, 
נִטַּמָּא אֶחָד מֵהֶן, נִטַּמָּא חֲבֵרוֹ. 

<small>ט</small>
הַלַּפִּיד, טָמֵא.  
וּבֵית שֶׁקָעוֹ שֶׁלַּנֵּר, 
מְטַמֵּא בְאָוֵיר.  
הַמַּסְרֵק שֶׁלַּצַּרְצוּר, 
רְבִּי אֱלִיעֶזֶר מְטַהֵר, 
וַחֲכָמִים מְטַמִּין. 





Chapter 3

<small>א</small>
שֵׁעוּר כְּלִי חֶרֶשׂ לִטַּהֵר: 
הֶעָשׂוּי לָאֳכָלִים, 
שֵׁעוּרוֹ בַזֵּיתִים.  
הֶעָשׂוּי לַמַּשְׁקִין, 
שֵׁעוּרוֹ בַמַּשְׁקִין.  
הֶעָשׂוּי לְכָךְ וּלְכָךְ, 
מְטִילִין אוֹתוֹ לְחָמְרוֹ בַזֵּיתִין. 

<small>ב</small>
חָבִית, 
שֵׁעוּרָהּ בִּגְרוֹגְרוֹת. &lt;כִּגְרוֹגֶרֶת&gt; 
דִּבְרֵי רְבִּי שִׁמְעוֹן.  
רְבִּי יְהוּדָה אוֹמֵר: 
בָּאֱגוֹזִים.  
רְבִּי מֵאִיר אוֹמֵר: 
בַּזֵּיתִין.  
הַלַּפָּס וְהַקְּדֵרָה, 
שֵׁעוּרָן בַּזֵּיתִים.  
הַפַּךְ וְהַטָּפִי, 
שֵׁעוּרָן בַּשֶּׁמֶן.  
הַצַּרְצוּר, 
שֵׁעוּרוֹ בַמַּיִם.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
שְׁלָשְׁתָּן בַּזֵּרְעוֹנִים.  

<small>ג</small>
נֵר, שֵׁעוּרוֹ בַשֶּׁמֶן.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
בִּפְרוּטָה קְטַנָּה.  
נֵר שֶׁנִּטַּל פִּיו, 
טָהוֹר, 
וְשֶׁלָּאֲדָמָה שֶׁהֻסַּק פִּיו בַּפְּתִילָה, 
טָהוֹר. 

<small>ד</small>
חָבִית שֶׁנִּקָּבָה וַעֲשָׂאָהּ בַּזֶּפֶת, וְנִשְׁבָּרָה, 
אִם יֵשׁ בִּמְקוֹם הַזֶּפֶת מַחְזִיק רְבִיעִית, 
טְמֵאָה, 
מִפְּנֵי שֶׁלֹּא בָטַל שֵׁם כֶּלִי מֵעָלֶיהָ.  
חֶרֶשׂ שֶׁנִּקַּב וַעֲשָׂאוֹ בַזֶּפֶת, 
אַף עַל פִּי מַחֲזִיק רְבִיעִית, 
טָהוֹר, 
מִפְּנֵי שֶׁבָּטַל שֵׁם כֶּלִי מֵעָלָיו. 

<small>ה</small>
חָבִית שֶׁנִּתְרוֹעָעָה וּטְפָלָהּ בַּגְּלָלִים, 
אַף עַל פִּי שֶׁהוּא נוֹטֵל אֶת הַגְּלָלִים וַחֲרָשִׂים נוֹפְלִים, 
טְמֵאָה, 
מִפְּנֵי שֶׁלֹּא בָטַל שֵׁם כֶּלִי מֵעָלֶיהָ.  
נִשְׁבָּרָה וְדִבֵּק מִמֶּנָּה חֲרָשִׂין, 
אוֹ שֶׁהֵבִיא חֲרָשִׂין מִמָּקוֹם אַחֵר וּטְפָלָן בַּגְּלָלִים, 
אַף עַל פִּי שֶׁהוּא נוֹטֵל אֶת הַגְּלָלִים וַחֲרָשִׂים עוֹמְדִים, 
טְהוֹרָה, 
מִפְּנֵי שֶׁבָּטַל שֵׁם כֶּלִי מֵעָלֶיהָ.  
הָיָה בָהּ חֶרֶשׂ מַחֲזִיק רְבִיעִית, 
כֻּלָּהּ מְטַמָּא בְמַגָּע, 
וּכְנֶגְדּוֹ מְטַמֵּא בְאָוֵיר. 

<small>ו</small>
הַטּוֹפֵל כְּלִי חֶרֶשׂ הַבָּרִיא, 
רְבִּי מֵאִיר וּרְבִּי שִׁמְעוֹן מְטַמִּין. 
וַחֲכָמִים אוֹמְרִים: 
הַטּוֹפֵל אֶת הַבָּרִיא, 
טָהוֹר, 
וְאֶת הָרָעוּעַ, &lt;הָרֹאוע&gt;
טָמֵא.  
וְכֵן בְּחִדּוּק קְרוּיָה. 

<small>ז</small>
יַבָּלִית שֶׁטּוֹפְלִים בָּהּ אֶת הַפִּדְסִים, 
הַנּוֹגֵעַ בָּהּ, טָמֵא.  
מְגוּפַת הֶחָבִית אֵינָה חִבּוּר.  
הַנּוֹגֵעַ בִּטְפֵלוֹ שֶׁלַּתַּנּוּר, טָמֵא. 

<small>ח</small>
מֵחַם שֶׁטְּפָלוֹ בַחֵמָר וּבַחַרְשִׂית, 
הַנּוֹגֵעַ בַּחֵמָר, טָמֵא, 
וּבַחַרְשִׂית, טָהוֹר.  
קֻמְקוּם שֶׁנִּקַּב וַעֲשָׂאוֹ בַזֶּפֶת, 
רְבִּי יוֹסֵה מְטַהֵר, 
שֶׁאֵינוּ יָכוֹל לְקַבֵּל אֶת הַחַמִּים כַּצּוֹנִין.  
וְכֵן הָיָה אוֹמֵר בִּכְלֵי זֶפֶת, 
כְּלֵי נְחֹשֶׁת שֶׁזְּפָתָם, טְהוֹרִין.  
וְאִם לַיַּיִן, טְמֵאִין. 

<small>ט</small>
חָבִית שֶׁנִּקָּבָה וַעֲשָׂאָהּ בַזֶּפֶת יָתֵר מִצָּרְכָהּ, 
הַנּוֹגֵעַ בְּצָרְכָהּ, טָמֵא, 
יָתֵר מִצָּרְכָהּ, טָהוֹר.  
זֶפֶת שֶׁנָּטְפָה עַל הֶחָבִית, 
הַנּוֹגֵעַ בָּהּ, טָהוֹר.  
מַשְׁפֵּךְ שֶׁלָּעֵץ וְשֶׁלַּחֶרֶשׂ שֶׁפְּקָקָן בַּזֶּפֶת, 
רְבִּי אֶלְעָזָר בֶּן עֲזַרְיָה מְטַמֵּא.  
רְבִּי עֲקִיבָה מְטַמֵּא בְשֶׁלָּעֵץ, 
וּמְטַהֵר בְּשֶׁלַּחֶרֶשׂ.  
רְבִּי יוֹסֵה מְטַהֵר בִּשְׁנֵיהֶן. 





Chapter 4

<small>א</small>
הַחֶרֶשׁ שֶׁאֵינוּ יָכוֹל לַעֲמֹד מִפְּנֵי אָזְנוֹ, 
אוֹ שֶׁהָיָה בוֹ חַדּוּד וְהַחַדּוּד מַכְרִיעוֹ,  
נִטְּלָה הָאֹזֶן, נִשְׁבַּר הַחַדּוּד, טָהוֹר.  
רְבִּי יוֹסֵה מְטַמֵּא.  
חָבִית שֶׁנִּפְחָתָה, 
וְהִיא מְקַבֶּלֶת עַל דְּפָנוֹתֶיהָ, 
אוֹ שֶׁנֶּחְלְקָה כְמִין שְׁתֵּי עֲרֵבוֹת, 
רְבִּי יְהוּדָה מְטַהֵר, 
וַחֲכָמִים מְטַמִּין. 

<small>ב</small>
חָבִית שֶׁנִּתְרוֹעָעָה, 
וְאֵינָה יְכוּלָה לְהִטַּלְטֵל בַּחֲצִי קַב גְּרוֹגְרוֹת, 
טְהוֹרָה.  
גַּצְטְרָה שֶׁנִּתְרוֹעָעָה וְאֵינָה מְקַבֶּלֶת מַשְׁקִין, 
אַף עַל פִּי מְקַבֶּלֶת אֳכָלִים, 
טְהוֹרָה, שֶׁאֵין שִׁירִים לַשִּׁירִים. 

<small>ג</small>
וְאֵיזוֹ הִיא גַצְרָה? 
כָּל שֶׁנִּטְּלוּ אָזְנֶיהָ.  
הָיוּ בָהּ חַדּוּדִים יוֹצְאִין, 
כָּל הַמְקַבֵּל עִמָּהּ בַּזֵּיתִין, 
מְטַמֵּא בְמַגָּע, &lt;מִטַּמֵּא&gt;
וּכְנֶגְדּוֹ מְטַמֵּא בְאָוֵיר; 
וְכָל שֶׁאֵינוּ מְקַבֵּל עִמָּהּ בַּזֵּיתִין, 
מְטַמֵּא בְמַגָּע, 
וְאֵין כְּנֶגְדּוֹ מְטַמֵּא בְאָוֵיר.  
הָיְתָה מֻטָּה עַל צִדָּהּ כְּמִין קָתֶדְרָה, 
כָּל הַמְקַבֵּל עִמָּהּ בַּזֵּיתִים, 
מְטַמֵּא בְמַגָּע, 
וּכְנֶגְדּוֹ מְטַמֵּא בְאָוֵיר; 
וְכָל שֶׁאֵינוּ מְקַבֵּל עִמָּהּ בַּזֵּיתִים, 
מְטַמֵּא בְמַגָּע, 
וְאֵין כְּנֶגְדּוֹ מְטַמֵּא בְאָוֵיר.  
שׁוּלֵי הַקָּרְפָּיוֹת, שׁוּלֵי הַקּוֹסִין הַצִּידוֹנִיִּין, 
אַף עַל פִּי שֶׁאֵינָם יְכוּלִין לֵישֵׁב שֶׁלֹּא מְסֻמָּכִין, 
טְמֵאִין, שֶׁלְּכֵן נֶעֱשׂוּ מִתְּחִלָּתָן. 

<small>ד</small>
כְּלִי חֶרֶשׂ שֶׁיֶּשׁ לוֹ שָׁלוֹשׁ סְפָיוֹת, 
הַפְּנִימִית עוֹדֶפֶת, הַכֹּל טָהוֹר; 
הַחִיצוֹנָה עוֹדֶפֶת, הַכֹּל טָמֵא; 
הָאֶמְצָעִית עוֹדֶפֶת, 
מִמֶּנָּה וְלִפְנִים טָמֵא, מִמֶּנָּה וְלַחוּץ טָהוֹר.  
הָיוּ שָׁווֹת, 
רְבִּי יְהוּדָה אוֹמֵר: 
חוֹלְקִין אֶת הָאֶמְצָעִית. 
וַחֲכָמִים אוֹמְרִים: 
הַכֹּל טָהוֹר.  
כְּלֵי חֶרֶס, מֵאֶמָּתַי מְקַבְּלִין טֻמְאָה? 
מִשֶּׁיִּצְטָרְפוּ בַכִּבְשָׁן, 
וְהִיא גְּמַר מְלַאכְתָּן. 





Chapter 5

<small>א</small>
תַּנּוּר, 
תְּחִלָּתוֹ אַרְבָּעָה, וְשִׁירָיו אַרְבָּעָה.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
בַּמֵּי דְבָרִים אֲמוּרִים? 
בַּגָּדוֹל, אֲבָל בַּקָּטָן, 
תְּחִלָּתוֹ כָל שֶׁהוּא, וְשִׁירָיו וְרֻבּוֹ, 
מִשֶּׁתִּגָּמֵר מְלַאכְתּוֹ.  
אֵי זוֹ הִיא גְמַר מְלַאכְתּוֹ? 
מִשֶּׁיַּסִּיקֶנּוּ כְּדֵי לְאוֹפוֹת בּוֹ סִפְגָּנִים.  
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁיַּסִּיק אֶת הֶחָדָשׁ כְּדֵי לְאוֹפוֹת בַּיָּשָׁן סִפְגָּנִים. 

<small>ב</small>
כִּירָה, 
תְּחִלָּתָהּ שָׁלוֹשׁ, וְשִׁירֶיהָ שָׁלוֹשׁ, 
מִשֶּׁתִּגָּמֵר מְלַאכְתָּהּ.  
אֵי זוֹ הִיא גְמַר מְלַאכְתָּהּ? 
מִשֶּׁיַּסִּיקֶנָּה 
כְּדֵי לְבַשֵּׁל עָלֶיהָ בֵיצָה קַלָּה שֶׁבַּבֵּיצִים, 
טְרוּפָה וּנְתוּנָה בַלַּפָּס.  
כֻּפָּח: 
וַעֲשָׂאוֹ לַאֲפִיָּה, שֵׁעוּרוֹ כַתַּנּוּר; &lt;וַעֲשָׂאָהּ&gt;
עֲשָׂאוֹ לְבִשּׁוּל, שֵׁעוּרוֹ כַכִּירָה.  &lt;שֵׁעוּר&gt;
הָאֶבֶן הַיּוֹצָא מִן הַתַּנּוּר טֶפַח, &lt;הַיּוֹצֵא&gt;
וּמִן הַכִּירָה שָׁלוֹשׁ אֶצְבָּעוֹת, חִבּוּר.  
הַיּוֹצָא מִן הַכֻּפָּח, &lt;הַיּוֹצֵא&gt;
וַעֲשָׂאוֹ לַאֲפִיָּה, שֵׁעוּרוֹ כַתַּנּוּר; 
עֲשָׂאוֹ לְבִשּׁוּל, שֵׁעוּרוֹ כַכִּירָה.  
אָמַר רְבִּי יְהוּדָה: 
לֹא אָמְרוּ 'טֶפַח' אֶלָּא מִן הַתַּנּוּר לַכֹּתֶל.  
הָיוּ שְׁנֵי תַנּוּרִים סְמוּכִים זֶה לָזֶה, 
נוֹתֵן לָזֶה טֶפַח וְלָזֶה טֶפַח, 
וְהַשְּׁאָר טָהוֹר. 

<small>ג</small>
עֲטֶרֶת כִּירָה, טְהוֹרָה.  
טִירַת הַתַּנּוּר, 
בִּזְמַן שֶׁהִיא גְבוֹהָה עֲשָׂרָה טְפָחִים, 
מְטַמֵּא בְמַגָּע וּבְאָוֵיר; 
פְּחוּתָה מִכֵּן, טְהוֹרָה.  
אִם חִבְּרָהּ לוֹ, 
אֲפִלּוּ עַל שָׁלוֹשׁ אֲבָנִים, טְמֵאָה.  
בֵּית הַפַּךְ, 
וּבֵית הַתַּבְלִים, 
וּבֵית הַנֵּר שֶׁבַּכִּירָה, 
מְטַמִּין בְּמַגָּע, 
וְאֵינָן מְטַמִּין בְּאָוֵיר.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יִשְׁמָעֵאל מְטַהֵר. 

<small>ד</small>
תַּנּוּר שֶׁהֻסַּק מֵאֲחוֹרָיו, 
אוֹ שֶׁהֻסַּק שֶׁלֹּא לְדַעְתּוֹ, 
אוֹ שֶׁהֻסַּק מִבֵּית הָאֻמָּן, 
טָמֵא.  
מַעֲשֶׂה שֶׁנָּפְלָה דְלֵקָה 
בְתַנּוּרֵי כְפַר סִגְנָה, &lt;סיגנייה&gt;
וּבָא מַעֲשֶׂה לְיַבְנֶה, 
וְטִמָּן רַבָּן גַּמְלִיאֵל. 

<small>ה</small>
מוּסַף הַתַּנּוּר שֶׁלְּבַעֲלֵי בָתִּים, 
טָהוֹר, 
וְשֶׁלַּנַּחְתּוֹמִים, 
טָמֵא, מִפְּנֵי שֶׁהוּא סוֹמֵךְ עָלָיו אֶת הַשְּׁפוּד.  
רְבִּי יוֹחָנָן הַסַּנְדַּלָּר אוֹמֵר: 
מִפְּנֵי שֶׁהוּא אוֹפֶה בוֹ כְשֶׁהוּא נִדְחָק.  
כַּיּוֹצֵא בוֹ, 
מוּסַף הַיּוֹרָה שֶׁלְּשׁוֹלְקֵי זֵיתִים, טָמֵא.  
וְשֶׁלַּצַּבָּעִים, טָהוֹר. 

<small>ו</small>
תַּנּוּר שֶׁנָּתַן בּוֹ עָפָר עַד חֶצְיוֹ, 
מִן הֶעָפָר וּלְמַטָּן, מְטַמֵּא בְמַגָּע, 
מִן הֶעָפָר וּלְמַעְלָן, מְטַמֵּא בְאָוֵיר.  
נְתָנוֹ עַל פִּי הַבּוֹר, אוֹ עַל פִּי הַחַדּוּת, 
וְנָתַן שָׁם אֶבֶן, 
רְבִּי יְהוּדָה אוֹמֵר: 
אִם מַסִּיק מִלְּמַטָּן, 
וְהוּא נִסּוֹק מִלְמַעְלָן, 
טָמֵא.  
וַחֲכָמִים אוֹמְרִים: 
הוֹאִיל וְהוּא נִסּוֹק מִכָּל מָקוֹם, 
טָמֵא. 

<small>ז</small>
תַּנּוּר שֶׁנִּטַּמָּא, 
כֵּיצַד מְטַהֲרִין אוֹתוֹ? 
חוֹלְקוֹ לִשְׁלוֹשָׁה, 
וְגוֹרֵר אֶת הַטְּפֵלָה עַד שֶׁיְּהֵא בָאָרֶץ.  
רְבִּי מֵאִיר אוֹמֵר: 
אֵינוּ צָרִיךְ לָגוֹר אֶת הַטְּפֵלָה, 
וְלֹא עַד שֶׁיְּהֵא בָאָרֶץ, 
אֶלָּא מְמַעֲטוֹ מִבִּפְנִים אַרְבָּעָה טְפָחִים.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
צָרִיךְ לְהַסִּיעוֹ.  

<small>ח</small>
חֲלָקוֹ לִשְׁנַיִם, 
אֶחָד גָּדוֹל וְאֶחָד קָטֹן, 
הַגָּדוֹל טָמֵא, וְהַקָּטֹן טָהוֹר.  
חֲלָקוֹ לִשְׁלֹשָׁה, 
אֶחָד גָּדוֹל כִּשְׁנַיִם, 
הַגָּדוֹל טָמֵא, 
וּשְׁנַיִם קְטַנִּים טְהוֹרִים. 

<small>ט</small>
חִתְּכוֹ חֲלָיוֹת לְרָחְבּוֹ, 
פָּחוּת מֵאַרְבָּעָה טְפָחִים, 
טָהוֹר.  
מֵרְחוֹ בַטִּיט, 
מְקַבֵּל טֻמְאָה 
מִשֶּׁיַּסִּיקֶנּוּ כְּדֵי לְאוֹפוֹת בּוֹ סִפְגָּנִים.  
הִרְחִיק מִמֶּנּוּ אֶת הַטְּפֵלָה, 
וְנָתַן חוֹל אוֹ צְרוֹר בִּנְתַּיִם, 
בַּזֶּה אָמָרוּ: 
הַנִּדָּה וְהַטְּהוֹרָה אוֹפוֹת בּוֹ, 
וְהוּא טָהוֹר. 

<small>י</small>
תַּנּוּר שֶׁבָּא מְחֻתָּךְ מִבֵּית הָאֻמָּן, 
עָשָׂה בוֹ לִמּוּדִין, נוֹתְנָן עָלָיו, 
וְהוּא טָהוֹר.  
נִטַּמָּא, 
סִלֵּק אֶת הַלִּמּוּדִין, טָהוֹר.  
הֶחֱזִירָן לוֹ, טָהוֹר.  
מֵרְחוֹ בַטִּיט, מְקַבֵּל טֻמְאָה. 
אֵינוּ צָרִיךְ לְהַסִּיקֶנּוּ, 
שֶׁכְּבָר הֻסַּק. 

<small>יא</small>
חִתְּכוֹ חֲלָיוֹת וְנָתַן חוֹל בֵּין חָלְיָא לְחָלְיָא, 
רְבִּי אֱלִיעֶזֶר מְטַהֵר, 
וַחֲכָמִים מְטַמִּין.  
זֶה תַנּוּרוֹ שֶׁלַּחֲכִינַי.  
יוֹרוֹת הָעַרְבִיִּים, 
שֶׁהוּא חוֹפֵר בָּאָרֶץ וְטָח בַּטִּיט, 
אִם יָכוֹל הַטִּיחַ לַעֲמֹד בִּפְנֵי עַצְמוֹ, 
טָמֵא; 
וְאִם לָאו, טָהוֹר.  
וְזֶה תַנּוּרוֹ שֶׁלְּבֶן דִּינַאי. 

תַּנּוּר שֶׁלָּאֶבֶן וְשֶׁלַּמַּתֶּכֶת, טָהוֹר; 
וְטָמֵא מִשֵּׁם כְּלֵי מַתְּכוֹת.  
נִקַּב, 
נִפְגַּם, 
נִסְדַּק; 
עָשָׂה לוֹ טְפֵלָה, 
אוֹ מוּסָף שֶׁל טִיט, טָמֵא.  
כַּמָּה יְהֵא בַנֶּקֶב? 
כְּדֵי שֶׁיֵּצֵא בוֹ הָאוּר.  
וְכֵן בְּכִירָה, 

<small>יג</small>
כִּירָה שֶׁלָּאֶבֶן וְשֶׁלַּמַּתֶּכֶת, טְהוֹרָה; 
וּטְמֵאָה מִשֵּׁם כְּלִי מַתְּכוֹת.  
נִקָּבָה, נִפְגָּמָה, נִסְדָּקָה, 
עָשָׂה לָהּ פַּטְפּוּטִים, טְמֵאָה.  
מְרָחָהּ בַּטִּיט מִבִּפְנִים וּמִבַּחוּץ, טְהוֹרָה.  
רְבִּי יְהוּדָה אוֹמֵר: 
מִבִּפְנִים, טְמֵאָה, 
וּמִבַּחוּץ, טְהוֹרָה. 





Chapter 6

<small>א</small>
הָעוֹשֶׂה שְׁלֹשָׁה פַטְפּוּטִים בָּאָרֶץ, 
וְחִבְּרָן בַּטִּיט, 
לִהְיוֹת שׁוֹפֵת עֲלֵיהֶן אֶת הַקְּדֵרָה, 
טְמֵאָה.  
קָבַע שְׁלֹשָׁה מַסְמְרִים בָּאָרֶץ, 
לִהְיוֹת שׁוֹפֵת עֲלֵיהֶן אֶת הַקְּדֵרָה, 
אַף עַל פִּי שֶׁעָשָׂה בָרִאשׁוֹן מָקוֹם 
שֶׁתְּהֵא הַקְּדֵרָה יוֹשֶׁבֶת, טְהוֹרָה.  

<small>ב</small>
הָעוֹשֶׂה שְׁתֵּי אֲבָנִים כִּירָה, 
וְחִבְּרָן בַּטִּיט, טְמֵאָה.  
רְבִּי יְהוּדָה מְטַהֵר, 
עַד שֶׁיַּעֲשֶׂה שְׁלִישִׁי, 
אוֹ עַד שֶׁיִּסְמוֹךְ לַכֹּתֶל.  
אַחַת בַּטִּיט, 
וְאַחַת שֶׁלֹּא בַטִּיט, טְהוֹרָה. 

<small>ג</small>
הָאֶבֶן שֶׁהָיָה שׁוֹפֵת עָלֶיהָ וְעַל הַתַּנּוּר, 
עָלֶיהָ וְעַל הַכִּירָה, 
עָלֶיהָ וְעַל הַכֻּפָּח, 
טְמֵאָה.  
עָלֶיהָ וְעַל הָאֶבֶן, 
עָלֶיהָ וְעַל הַסֶּלַע, 
עָלֶיהָ וְעַל הַכֹּתֶל, 
טְהוֹרָה.  
זוֹ הָיְתָה כִּירַת נְזִירִים שֶׁבִּירוּשָׁלַיִם, 
שֶׁכְּנֶגֶד הַסֶּלַע.  
כִּירַת הַטַּבָּחִים, 
שֶׁהוּא נוֹתֵן אֶבֶן בְּצַד אֶבֶן, 
נִטַּמָּאת אַחַת מֵהֶן, 
לֹא נִטַּמּוּ כֻלָּם. 

<small>ד</small>
שָׁלוֹשׁ אֲבָנִים שֶׁעֲשָׂאָן שְׁתֵּי כִירַיִם, 
נִטַּמָּאת אַחַת מִן הַחִיצוֹנוֹת; 
הָאֶמְצָעִית, 
הַמְשַׁמֵּשׁ לַטָּמֵא, טָמֵא; 
הַמְשַׁמֵּשׁ לַטָּהוֹר, טָהוֹר.  
נִטְלָה טְהוֹרָה, הָחְלָטָה הָאֶמְצָעִית לַטֻּמְאָה.  
נִטְלָה טְמֵאָה, הָחְלָטָה הָאֶמְצָעִית לְטַהֲרָה.  
נִטַּמּוּ שְׁתַּיִם הַחִיצוֹנוֹת, 
אִם הָיְתָה הָאֶמְצָעִית גְּדוֹלָה, 
נוֹתֵן לָזוֹ כְדֵי שְׁפִיתָה מִכָּן, 
וְלָזוֹ כְדֵי שְׁפִיתָה מִכָּן, 
וְהַשְּׁאָר טָהוֹר; 
אִם הָיְתָה קְטַנָּה, הַכֹּל טָמֵא.  
נִטְּלָה הָאֶמְצָעִית, 
אִם יָכוֹל לִשְׁפּוֹת עָלֶיהָ יוֹרָה גְדוֹלָה, 
טְמֵאָה.  
הֶחֱזִירָהּ, טְהוֹרָה.  
מֵרְחָהּ בַּטִּיט, 
מְקַבֶּלֶת טֻמְאָה 
מִשֶּׁיַּסִּיקֶנָּה כְדֵי לְבַשֵּׁל עָלֶיהָ אֶת הַבֵּיצָה. 

<small>ה</small>
שְׁתֵּי אֲבָנִים שֶׁעֲשָׂאָן כִּירָה וְנִטַּמּוּ, 
סָמַךְ לָזוֹ אֶבֶן אַחַת מִכָּן, 
וְלָזוֹ אֶבֶן אַחַת מִכָּן, 
חֶצְיָהּ שֶׁלָּזוֹ טָמֵא וְחֶצְיָהּ טָהוֹר, 
וְחֶצְיָהּ שֶׁלָּזוֹ טָמֵא וְחֶצְיָהּ טָהוֹר.  
נִטְּלוּ טְהוֹרוֹת, חָזְרוּ לְטֻמְאָתָן. 





Chapter 7

<small>א</small>
הַקַּלָּתוֹת שֶׁלְּבַעֲלֵי בָתִּים 
שֶׁנִּפְחֲתָה פָחוּת מִשְׁלֹשָׁה טְפָחִים, טְמֵאָה, 
שֶׁהוּא מַסִּיק מִלְּמַטָּן וּקְדֵרָה בְשֵׁלָה מִלְמַעְלָן; 
יָתֵר מִכֵּן, טְהוֹרָה.  
נָתַן אֶבֶן אוֹ צְרוֹר, טְהוֹרָה.  
מֵרְחָהּ בַּטִּיט, 
מְקַבֶּלֶת טֻמְאָה מִכָּן וְלַבָּא.  
זוֹ הָיְתָה תְשׁוּבַת רְבִּי יְהוּדָה, 
בַּתַּנּוּר שֶׁנְּתָנוֹ עַל פִּי הַבּוֹר אוֹ עַל פִּי הַחַדּוּת. 

<small>ב</small>
דָּכוֹן שֶׁיֶּשׁ בּוֹ בֵית קִבּוּל קְדֵרוֹת, 
טָהוֹר מִשֵּׁם כִּירָה, 
וְטָמֵא מִשֵּׁם כְּלִי קִבּוּל.  
הַצְּדָדִין שֶׁלּוֹ, 
הַנּוֹגֵעַ בָּהֶן אֵינוּ טָמֵא מִשֵּׁם כִּירָה.  
הָרַחַב שֶׁלּוֹ, 
רְבִּי מֵאִיר מְטַהֵר, 
רְבִּי יְהוּדָה מְטַמֵּא.  
וְכֵן הַכּוֹפֶה אֶת הַסַּל וְעוֹשֶׂה עַל גַּבָּיו כִּירָה. 

<small>ג</small>
כִּירָה שֶׁנֶּחְלְקָה לְאָרְכָּהּ, טְהוֹרָה; 
לְרָחְבָּהּ, טְמֵאָה.  
כֻּפָּח שֶׁנֶּחֱלַק, 
בֵּין לְאָרְכּוֹ בֵין לְרָחְבּוֹ, טָהוֹר.  
חֲצַר הַכִּירָה, 
בִּזְמַן שֶׁהִיא גְבוֹהָה שָׁלוֹשׁ אֶצְבָּעוֹת, 
מְטַמָּא בְמַגָּע וּבְאָוֵיר.  
פְּחוּתָה מִכֵּן, 
מְטַמָּא בְמַגָּע וְאֵינָהּ מְטַמָּא בְאָוֵיר.  
כֵּיצַד מְשַׁעֲרִין אוֹתָהּ? 
רְבִּי יִשְׁמָעֵאל אוֹמֵר: 
נוֹתֵן אֶת הַשְּׁפוּד מִלְּמַטָּן לְמַעְלָן, 
וּכְנֶגְדּוֹ מְטַמֵּא בְאָוֵיר.  
רְבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר: 
נִטַּמָּאת הַכִּירָה, נִטַּמָּאת הֶחָצֵר; 
נִטַּמָּאת הֶחָצֵר, לֹא נִטַּמָּאת הַכִּירָה. 

<small>ד</small>
הָיְתָה מֻפְרֶשֶׁת מִן הַכִּירָה, 
בִּזְמַן שֶׁהִיא גְבוֹהָה שָׁלוֹשׁ אֶצְבָּעוֹת, 
מְטַמָּא בְמַגָּע וּבְאָוֵיר; 
פְּחוּתָה מִכֵּן, אוֹ שֶׁהָיְתָה חֲלָקָה, 
טְהוֹרָה.  
פַּטְפּוּטֵי כִירָה, 
שְׁלֹשָׁה שֶׁלְּשָׁלוֹשׁ אֶצְבָּעוֹת, 
מְטַמִּין בְּמַגָּע וּבְאָוֵיר. &lt;מיטמין&gt; 
פָּחוּת מִכֵּן, 
כָּל שֶׁכֵּן הֵן טְמֵאִין, 
אֲפִלּוּ הֵן אַרְבָּעָה. 

<small>ה</small>
נִטַּל אֶחָד מֵהֶן, 
מְטַמִּין בְּמַגָּע וְאֵינָן מְטַמִּין בְּאָוֵיר.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי שִׁמְעוֹן מְטַהֵר.  

<small>ו</small>
עָשָׂה שְׁנַיִם זֶה כְנֶגֶד זֶה, 
מְטַמִּין בְּמַגָּע וּבְאָוֵיר.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי שִׁמְעוֹן מְטַהֵר.  

<small>ז</small>
הָיוּ גְבוֹהִים שָׁלשׁ אֶצְבָּעוֹת, 
מִשָּׁלשׁ וּלְמַטָּן, 
מְטַמִּין בְּמַגָּע וּבְאָוֵיר; 
מִשָּׁלשׁ וּלְמַעְלָן, 
מְטַמִּין בְּמַגָּע, וְאֵינָן מְטַמִּין בְּאָוֵיר.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי שִׁמְעוֹן מְטַהֵר.  

<small>ח</small>
הָיוּ מְשׁוּכִין מִן הַשָּׂפָה, 
בְּתוֹךְ שָׁלוֹשׁ אֶצְבָּעוֹת, 
מְטַמִּין בְּמַגָּע וּבְאָוֵיר; 
חוּץ מִשָּׁלשׁ אֶצְבָּעוֹת, &lt;שָׁלֹשׁ&gt;
מְטַמִּין בְּמַגָּע, וְאֵין מְטַמִּין בְּאָוֵיר.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי שִׁמְעוֹן מְטַהֵר. 

<small>ט</small>
כֵּיצַד מְשַׁעֲרִין אוֹתָן? 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
נוֹתֵן כַּנָּה בֵינֵיהֶם, 
מִן הַכַּנָּה וְלַחוּץ, 
טָהוֹר; 
מִן הַכַּנָּה וְלִפְנִים, וּמְקוֹם הַכַּנָּה, 
טָמֵא. 





Chapter 8

<small>א</small>
תַּנּוּר שֶׁחֲצָצוֹ בִנְסָרִים אוֹ בִירִיעוֹת, 
וְנִמְצָא הַשֶּׁרֶץ בְּמָקוֹם אֶחָד, 
הַכֹּל טָמֵא.  
כַּוֶּרֶת פְּחוּתָה וּפְקוּקָה בַקַּשׁ, 
וּמְשֻׁלְשֶׁלֶת לַאֲוֵיר הַתַּנּוּר, 
הַשֶּׁרֶץ בְּתוֹכָהּ, 
הַתַּנּוּר טָמֵא; 
הַשֶּׁרֶץ בַּתַּנּוּר, 
אֳכָלִין שֶׁבְּתוֹכָהּ טְמֵאִין.  
רְבִּי אֱלִיעֶזֶר מְטַהֵר.  
אָמַר רְבִּי אֱלִיעֶזֶר: 
אִם הִצִּילָה בְמֵת הֶחָמוּר, &lt;הֶחָמַר&gt;
לֹא תַצִּיל בִּכְלִי חֶרֶשׂ הַקַּל? 
אָמְרוּ לוֹ: 
אִם הִצִּילָה בְמֵת הֶחָמוּר, &lt;הֶחָמָר&gt;
[שֶׁכֵּן חוֹלְקִים אהָלִים, 
תַּצִּיל בִּכְלִי חֶרֶס הַקַּל,] 
שֶׁאֵין חוֹלְקִין כְּלִי חֶרֶשׂ? 

<small>ב</small>
הָיְתָה שְׁלֵמָה, 
וְכֵן הַקֻּפָּה, וְכֵן הַחֵמֶת, 
הַשֶּׁרֶץ בְּתוֹכָהּ, 
הַתַּנּוּר טָהוֹר.  
הַשֶּׁרֶץ בַּתַּנּוּר, 
אֳכָלִין שֶׁבְּתוֹכָהּ טְהוֹרִין.  
נִקָּבוּ, 
הֶעָשׂוּי לָאֳכָלִין, שֵׁעוּרוֹ בַזֵּיתִין.  
הֶעָשׂוּי לַמַּשְׁקִין, שֵׁעוּרוֹ בַמַּשְׁקִין.  
הֶעָשׂוּי לְכָךְ וּלְכָךְ, 
מְטִילִין אוֹתוֹ לְחָמְרוֹ, 
בְּכוֹנֵס מַשְׁקֶה. 

<small>ג</small>
סְרֵידָה שֶׁהִיא נְתוּנָה עַל פִּי הַתַּנּוּר, 
שׁוֹקַעַת בְּתוֹכוֹ וְאֵין לָהּ גַּפַּיִם, 
הַשֶּׁרֶץ בְּתוֹכָהּ, הַתַּנּוּר טָמֵא. 
הַשֶּׁרֶץ בַּתַּנּוּר, אֳכָלִים שֶׁבְּתוֹכָהּ טְמֵאִים, 
שֶׁאֵין מַצִּילִין מִיַּד כְּלִי חֶרֶשׂ אֶלָּא כֵלִים.  
חָבִית שֶׁהִיא מְלֵאָה מַשְׁקִין טְהוֹרִין, 
וּנְתוּנָה לְמַטָּן מִנְּחֻשְׁתּוֹ שֶׁלַּתַּנּוּר, 
הַשֶּׁרֶץ בַּתַּנּוּר, 
חָבִית וְהַמַּשְׁקִין טְהוֹרִין.  
הָיְתָה כְפוּיָה וּפִיהָ לַאֲוֵיר הַתַּנּוּר, 
הַשֶּׁרֶץ בַּתַּנּוּר, 
מַשְׁקֶה טוֹפֵחַ שֶׁבְּשׁוּלֵי הֶחָבִית טָהוֹר. 

<small>ד</small>
קְדֵרָה שֶׁהִיא נְתוּנָה בַתַּנּוּר, 
וְהַשֶּׁרֶץ בַּתַּנּוּר, הַקְּדֵרָה טְהוֹרָה, 
שֶׁאֵין כְּלִי חֶרֶשׂ מְטַמֵּא כֵּלִים.  
הָיָה בָהּ מַשְׁקֶה טוֹפֵחַ, נִטַּמָּא, 
וְטִמְּאָהּ.  
הֲרֵי זֶה אוֹמֵר: 
מְטַמֶּיךָ לֹא טִמּוּנִי, 
וְאַתָּה טִמֵּאתַנִי. 

<small>ה</small>
תַּרְנְגוֹל שֶׁבָּלַע אֶת הַשֶּׁרֶץ, 
וְנָפַל לַאֲוֵיר הַתַּנּוּר, טָהוֹר.  
אִם מֵת, טָמֵא.  
הַשֶּׁרֶץ שֶׁנִּמְצָא בַתַּנּוּר, 
הַפַּת שֶׁבְּתוֹכוֹ שְׁנִיָּה, 
שֶׁהַתַּנּוּר תְּחִלָּה. 

<small>ו</small>
בֵּית שְׂאוֹר מֻקָּף צָמִיד פָּתִיל וְנָתוּן לְתוֹךְ הַתַּנּוּר, 
הַשְּׂאוֹר וְהַשֶּׁרֶץ בְּתוֹכוֹ וְהַקֶּרֶץ בִּנְתַּיִם, 
הַתַּנּוּר טָמֵא, וְהַשְּׂאוֹר טָהוֹר.  
אִם הָיָה כַזַּיִת מִן הַמֵּת, 
הַתַּנּוּר וְהַבַּיִת טָמֵא, 
וְהַשְּׂאוֹר טָהוֹר.  
אִם יֶשׁ שָׁם פּוֹתֵחַ טֶפַח, 
הַכֹּל טָמֵא. 

<small>ז</small>
הַשֶּׁרֶץ שֶׁנִּמְצָא בְעַיִן שֶׁלַּתַּנּוּר, 
בְּעַיִן שֶׁלַּכִּירָה, בְּעַיִן שֶׁלַּכֻּפָּח, 
מִן הַשָּׂפָה הַפְּנִימִית וְלַחוּץ, טָהוֹר.  
וְאִם הָיָה בְאָוֵיר, 
אֲפִלּוּ כַזַּיִת מִן הַמֵּת, טָהוֹר.  
וְאִם יֶשׁ שָׁם פּוֹתֵחַ טֶפַח, 
הַכֹּל טָמֵא. 

<small>ח</small>
נִמְצָא לִמְקוֹם הַנָּחַת הָעֵצִים, 
רְבִּי יְהוּדָה אוֹמֵר: 
מִן הַשָּׂפָה הַחִיצוֹנָה וְלִפְנִים, 
טָמֵא.  
וַחֲכָמִים אוֹמְרִים: 
מִשָּׂפָה הַפְּנִימִית וְלַחוּץ, 
טָהוֹר.  
רְבִּי יוֹסֵה אוֹמֵר: 
מִכְּנֶגֶד שְׁפִיתַת הַקְּדֵרָה וְלִפְנִים, 
טָמֵא; 
מִכְּנֶגֶד שְׁפִיתַת הַקְּדֵרָה וְלַחוּץ, 
טָהוֹר.  

<small>ט</small>
נִמְצָא מְקוֹם יְשִׁיבַת הַבַּלָּן, 
מְקוֹם יְשִׁיבַת הַצַּבָּע, 
מְקוֹם יְשִׁיבַת שֶׁלְּשׁוֹלְקֵי זֵיתִים, 
טָהוֹר; 
אֵין טָמֵא אֶלָּא מִן הַסְּתוּמָה וְלִפְנִים. 

<small>י</small>
בּוֹר שֶׁיֶּשׁ בּוֹ בֵית שְׁפִיתָה, 
טָמֵא.  
וְשֶׁלְּעוֹשֵׂי זְכוֹכִית, 
אִם יֶשׁ בּוֹ בֵית שְׁפִיתָה, 
טָמֵא.  
כִּבְשָׁן שֶׁלַּסַּיָּדִין, וְשֶׁלַּזַּגָּגִין, וְשֶׁלַּיּוֹצְרִין, 
טְהוֹרָה.  
פּוּרְנָה, אִם יֶשׁ לָהּ לִזְבֵּז, 
טְמֵאָה; 
רְבִּי יְהוּדָה אוֹמֵר: 
אִם יֶשׁ לָהּ אַסְטְגָיוֹת; 
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
אִם יֶשׁ לָהּ סְפִיּוֹת. 

<small>יא</small>
מַגַּע טְמֵא מֵת, 
שֶׁהָיוּ אֳכָלִין וּמַשְׁקִין לְתוֹךְ פִּיו, 
הִכְנִיס רֹאשׁוֹ לַאֲוֵיר הַתַּנּוּר טָהוֹר, 
טִמָּהוּ.  
וְטָהוֹר שֶׁהָיוּ אֳכָלִין וּמַשְׁקִין לְתוֹךְ פִּיו, 
הִכְנִיס רֹאשׁוֹ לַאֲוֵיר הַתַּנּוּר נִטְמָא, 
נִטַּמּוּ.  
הָיָה אוֹכֵל דְּבֵלָה בְיָדַיִן מְסֹאָבוֹת, 
הִכְנִיס יָדוֹ לְתוֹךְ פִּיו לִטּוֹל אֶת הַצְּרוֹר, 
רְבִּי מֵאִיר מְטַמֵּא. 
רְבִּי יְהוּדָה מְטַהֵר.  
רְבִּי יוֹסֵה אוֹמֵר: 
אִם הָפַךְ, טָמֵא; 
אִם לֹא הָפַךְ, טָהוֹר.  
אִם הָיְתָה פוֹנְדְּיוֹן לְתוֹךְ פִּיו, 
רְבִּי יוֹסֵה אוֹמֵר: 
אִם לִצְמָאוֹ, טָמֵא. 

<small>יב</small>
הָאִשָּׁה שֶׁנָּטַף חָלָב מִדַּדֶּיהָ, 
וְנָפַל לַאֲוֵיר הַתַּנּוּר, 
טָמֵא, 
שֶׁהַמַּשְׁקֶה מְטַמֵּא לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן.  
הָיְתָה גוֹרְפַתּוּ, וְהִכָּהּ הַקּוֹץ, 
וְיָצָא מִמֶּנָּה דָם, 
אוֹ שֶׁנִּכְוַת וְנָתְנָה אֶצְבָּעָהּ לְתוֹךְ פִּיהָ, 
נִטַּמָּא. 





Chapter 9

<small>א</small>
מַחַט אוֹ טַבַּעַת שֶׁנִּמְצְאוּ בִנְחֻשְׁתּוֹ שֶׁלַּתַּנּוּר, 
נִרְאִין אֲבָל לֹא יוֹצְאִין, 
אִם אוֹפֶה אֶת הַבָּצֵק וְהוּא נוֹגֵעַ בָּהֶם, 
טָמֵא.  
בְּאֵיזֶה בָצֵק אָמָרוּ? 
בְּבָצֵק הַבֵּינוֹנִי.  
נִמְצְאוּ בִטְפֵלַת הַתַּנּוּר מֻקָּף צָמִיד פָּתִיל, 
בַּטָּמֵא, טְמֵאִין, וּבַטָּהוֹר, טְהוֹרִין.  
נִמְצְאוּ בִמְגוּפַת הֶחָבִית, 
מִצִּדֶּיהָ, טְמֵאִין; 
מִכְּנֶגֶד פִּיהָ, טְהוֹרִין.  
נִרְאִין בְּתוֹכָהּ, אֲבָל לֹא לַאֲוִירָהּ, 
טְהוֹרִין.  
שׁוֹקְעִין בְּתוֹכָהּ, וְתַחְתֵּיהֶן כִּקְלִפַּת הַשּׁוּם, 
טְהוֹרִין. 

<small>ב</small>
חָבִית שֶׁהִיא מְלֵאָה מַשְׁקִין טְהוֹרִין, 
וּמֵנֶקֶת בְּתוֹכָהּ, &lt;ומניקית&gt;
מֻקֶּפֶת צָמִיד פָּתִיל וּנְתוּנָה בְאָהֵל הַמֵּת, 
בֵּית שַׁמַּי אוֹמְרִים: 
הֶחָבִית וְהַמַּשְׁקִין טְהוֹרִין, 
וּמֵנֶקֶת טְמֵאָה. &lt;ומניקית&gt; 
וּבֵית הֶלֵּל אוֹמְרִים: 
אַף וּמֵנֶקֶת טְהוֹרָה. &lt;מניקית&gt; 
חָזְרוּ בֵית הֶלֵּל לְהוֹרוֹת כְּדִבְרֵי בֵית שַׁמַּי. 

<small>ג</small>
שֶׁרֶץ שֶׁנִּמְצָא לְמַטָּה מִנְּחֻשְׁתּוֹ שֶׁלַּתַּנּוּר, 
טָהוֹר, שֶׁאֲנִי אוֹמֵר: &lt;שני&gt; 
חַי נָפַל וְעַכְשָׁיו מֵת.  
מַחַט אוֹ טַבַּעַת, 
שֶׁנִּמְצְאוּ לְמַטָּה מִנְּחֻשְׁתּוֹ שֶׁלַּתַּנּוּר, 
טָהוֹר, שֶׁאֲנִי אוֹמֵר: &lt;שני&gt; 
שָׁם הָיוּ עַד שֶׁלֹּא בָא הַתַּנּוּר.  
נִמְצְאוּ בְאֵפֶר מַקְלָה, 
טָמֵא, שֶׁאֵין לוֹ בְמָה יִתְלֶה. 

<small>ד</small>
סְפוֹג שֶׁבָּלַע מַשְׁקִין טְמֵאִין וְנָגוּב מִבַּחוּץ, 
וְנָפַל לַאֲוֵיר הַתַּנּוּר, 
טָמֵא, שֶׁסּוֹף מַשְׁקֶה לָצֵאת.  
וְכֵן חֲתִכָּה שֶׁלַּלֶּפֶת וְשֶׁלַּגְּמִי.  
רְבִּי שִׁמְעוֹן מְטַהֵר בִּשְׁנֵי אֵלּוּ. 

<small>ה</small>
חֲרָשִׂין שֶׁנִּשְׁתַּמַּשׁ בָּהֶן מַשְׁקִין טְמֵאִין 
וְנָפְלוּ לַאֲוֵיר הַתַּנּוּר, וְהֻסַּק הַתַּנּוּר, 
טָמֵא, שֶׁסּוֹף מַשְׁקֶה לָצֵאת.  
וְכֵן בְּגֶפֶת חֲדָשָׁה, 
אֲבָל בִּישָׁנָה, טָהוֹר.  
אִם יָדוּעַ שֶׁיּוֹצִיא מִמֶּנָּה מַשְׁקִין, &lt;שֶׁיּוֹצִא&gt;
אֲפִלּוּ לְאַחַר שָׁלוֹשׁ שָׁנִים, 
נִטְמָא. 

<small>ו</small>
הַגֶּפֶת וְהַזֻּגִּין שֶׁנֶּעֱשׁוּ בְטַהֲרָה, 
וְהִלְּכוּ עֲלֵיהֶן טְמֵאִין, 
וְאַחַר כָּךְ יָצְאוּ מֵהֶן מַשְׁקִין, 
טְהוֹרִין, שֶׁמִּתְּחִלָּתָן נֶעֱשׂוּ בְטַהֲרָה.  
כּוּשׁ שֶׁבָּלַע אֶת הַצִּינֵירָה, 
מַלְמָד שֶׁבָּלַע אֶת הַדָּרְבָן, 
לְבֵנָה שֶׁבָּלְעָה אֶת הַטַּבַּעַת, 
הָיוּ טְהוֹרִין, 
נִכְנְסוּ לְאֹהֶל הַמֵּת, 
נִטַּמּוּ.  
הִסִּיטָן הַזָּב, נִטַּמּוּ. &lt;הַסִּיטוֹן ְהַזָּב&gt;  
נָפְלוּ לַאֲוֵיר הַתַּנּוּר טָהוֹר, טִמּוּהוּ.  
וְנָגַע בָּהֶן כִּכָּר שֶׁלִּתְרוּמָה, טָהוֹר. 

<small>ז</small>
סְרֵידָה שֶׁהִיא נְתוּנָה עַל פִּי הַתַּנּוּר מֻקָּף צָמִיד פָּתִיל, 
נִסְדַּק בֵּין הַתַּנּוּר לִסְרֵידָה, 
שֵׁעוּרוֹ מְלֹא פִי מַרְדֵּא שֶׁלֹּא נִכְנָס.  
רְבִּי יְהוּדָה אוֹמֵר: 
נִכְנָס.  
נִסְדְּקָה סְרֵידָה, 
שֵׁעוּרוֹ מְלֹא פִי מַרְדֵּעַ נִכְנָס.  
רְבִּי יְהוּדָה אוֹמֵר: 
שֶׁלֹּא נִכְנָס.  
הָיָה עָגֹל, 
אֵין רוֹאִין אוֹתוֹ אָרוּךְ, 
אֶלָּא שֵׁעוּר כִּמְלֹא פִי מַרְדֵּאַ נִכְנָס. 

<small>ח</small>
תַּנּוּר שֶׁנִּקַּב מֵעֵינוֹ, 
שֵׁעוּרוֹ מְלֹא כוּשׁ, 
נִכְנָס וְיוֹצֵא, 
דּוֹלֵק.  
רְבִּי יְהוּדָה אוֹמֵר: 
שֶׁלֹּא דוֹלֵק.  
נִקַּב מִצִּדּוֹ, 
שֵׁעוּרוֹ כִמְלֹא כוּשׁ, 
נִכְנָס וְיוֹצֵא, 
שֶׁלֹּא דוֹלֵק.  
רְבִּי יְהוּדָה אוֹמֵר: 
דּוֹלֵק.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
מִן הָאֶמְצַע, נִכְנָס; 
מִן הַצַּד, אֵינוּ נִכְנָס.  

ט &lt;ד&gt;
וְכֵן הָיָה אוֹמֵר בִּמְגוּפַת הֶחָבִית שֶׁנִּקָּבָה, 
שֵׁעוּרָהּ מְלֹא מִצָּה שְׁנִיָּה שֶׁלַּשִּׁיפוֹן, 
מִן הָאֶמְצַע, נִכְנָס; 
מִן הַצַּד, אֵינוּ נִכְנָס.  
וְכֵן הָיָה אוֹמֵר בַּחֲצָבִים גְּדוֹלִים שֶׁנִּקָּבוּ, 
שֵׁעוּרָן מְלֹא מִצָּה שֶׁלַּקָּנֶה, 
מִן הָאֶמְצַע, נִכְנָס; 
מִן הַצַּד, אֵינוּ נִכְנָס.  

<small>י</small>
בַּמֵּי דְבָרִים אֲמוּרִים? 
בִּזְמַן שֶׁנֶּעֱשׁוּ לַיַּיִן, 
אֲבָל אִם נֶעֱשׂוּ לִשְׁאָר מַשְׁקִין, 
אֲפִלּוּ כָל שֶׁהֵן, טְמֵאִים.  
בַּמֵּי דְבָרִים אֲמוּרִים? 
בִּזְמַן שֶׁלֹּא נֶעֱשׂוּ בִידֵי אָדָם, 
[אֲבָל אִם נֶעֱשׂוּ בִידֵי אָדָם,] 
אֲפִלּוּ כָל שֶׁהֵן, טְמֵאִין.  
נִקָּבוּ, 
הֶעָשׂוּי לָאֳכָלִים, שֵׁעוּרוֹ בַּזֵּיתִין.  
הֶעָשׂוּי לַמַּשְׁקִין, שֵׁעוּרוֹ בְמַשְׁקִין.  
הֶעָשׂוּי לְכָךְ וּלְכָךְ, 
מְטִילִין אוֹתוֹ לְחָמְרוֹ בְצָמִיד פָּתִיל 
וּבְכוֹנֵס מַשְׁקֶה. 





Chapter 10

<small>א</small>
אֵלּוּ כֵלִים מַצִּילִים בְּצָמִיד פָּתִיל: 
כְּלֵי גְלָלִים, 
כְּלֵי אֲבָנִים, 
כְּלֵי אֲדָמָה, 
כְּלֵי חֶרֶשׂ, 
כְּלֵי נֶתֶר, 
וְעַצְמוֹת חַיָּה שֶׁבַּיָּם וְעוֹרָהּ, 
וּכְלֵי הָעֵץ הַטְּהוֹרִין.  
מַצִּילִין בֵּין מִפִּיהֶם, בֵּין מִצְּדָדֵיהֶן, 
בֵּין יוֹשְׁבִין עַל שׁוּלֵיהֶם, 
בֵּין מֻטִּין עַל צִדְדֵיהֶם.  
הָיוּ כְפוּיִים עַל פִּיהֶם, 
מַצִּילִים כָּל שֶׁתַּחְתֵּיהֶם עַד הַתְּהוֹם.  
רְבִּי אֱלִיעֶזֶר מְטַמֵּא.  
עַל הַכֹּל מַצִּילִים, 
חוּץ מִכְּלִי חֶרֶשׂ, 
שֶׁאֵינוּ מַצִּיל 
אֶלָּא עַל הָאֳכָלִים וְעַל הַמַּשְׁקִין, 
וְעַל כְּלִי חֶרֶשׂ. 

<small>ב</small>
בַּמָּה מַקִּיפִים? 
בַּסִּיד, 
וּבַגִּפְּסַס, 
בַּזֶּפֶת, 
וּבְשַׁעֲוָה, 
בַּטִּיט, 
וּבַצּוֹאָה, 
בַּחֵמָר, 
וּבַחַרְסִית, 
וּבְכָל דָּבָר הַמִּתְמָרֵחַ.  
אֵין מַקִּיפִין לֹא בַבַּעַץ, 
וְלֹא בָעוֹפֶרֶת, 
מִפְּנֵי שֶׁהוּא פָתִיל וְאֵינוּ צָמִיד.  
אֵין מַקִּיפִין לֹא בִדְבֵלָה שְׁמֵנָה, 
וְלֹא בְּבָצֵק שֶׁנִּלּוֹשׁ בְּמֵי פֵרוֹת, 
שֶׁלֹּא יְבִיאֶנּוּ לִידֵי פְסוּל.  
וְאִם הִקִּיף, הִצִּיל. 

<small>ג</small>
מְגוּפַת הֶחָבִית הַמְחֻלְחֶלֶת וְאֵינָה נִשְׁמֶטֶת, 
רְבִּי יְהוּדָה אוֹמֵר: 
מַצֶּלֶת. 
וַחֲכָמִים אוֹמְרִים: 
אֵינָה מַצֶּלֶת.

הַשֶּׁרֶץ בֶּחָבִית, אֳכָלִין שֶׁבְּתוֹכָהּ טְמֵאִין. 

<small>ד</small>
הַכַּדּוּר וְהַפַּקַּעַת שֶׁלַּגְּמִי,

הִצִּיל.

רְבִּי יְהוּדָה אוֹמֵר: 
אֵינָן מַצִּילִין; 
וַחֲכָמִים אוֹמְרִים: 
מַצִּילִין.

נֶסֶר שֶׁהוּא נָתוּן עַל פִּי הַתַּנּוּר,

חָדָשׁ בְּתוֹךְ יָשָׁן,

הָיוּ בְכוֹנֵס מַשְׁקֶה,

Chapter 11

<small>א</small>
כְּלֵי מַתְּכוֹת,

נִשְׁבָּרוּ, טָהָרוּ.  
חָזַר וְעָשָׂה מֵהֶן כֵּלִים, 
חָזְרוּ לְטֻמְאָתָן הַיְשָׁנָה.  
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
לֹא לְכָל טֻמְאָה, 
אֶלָּא לְטֻמְאַת נֶפֶשׁ.

כָּל כְּלִי מַתֶּכֶת שֶׁיֶּשׁ לוֹ שֵׁם בִּפְנֵי עַצְמוֹ,

וְהַנֶּגֶר,

וּמֵהָגְנֵי כֵלִים,

טְהוֹרִין.  
רְבִּי יוֹחָנָן בֶּן נוּרִי אוֹמֵר:

מִשִּׁבְרֵי כֵלִים, &lt;מִשְׁבְּרֵי&gt;
מִן הַגְּרוֹטֵי, 
וּמִן הַמַּסְמְרוֹת שֶׁיָּדוּעַ שֶׁנֶּעֱשׁוּ מִכֵּלִים, 
טְמֵאִים.  
מִן הַמַּסְמְרוֹת, 
בֵּית שַׁמַּי מְטַמִּין, 
וּבֵית הֶלֵּל מְטַהֲרִין.

וְכֵן מִן הַחַלָּמָה וּמִן הַגְּלָלִין.  
קְלוֹסְטְרָה, טְמֵאָה, 
וּמְצֻפָּה, טְהוֹרָה.  
הַפִּין, 
וְהַפּוּרְנָה, טְמֵאִין.  
קְלוֹסְטְרָה, 
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
שׁוֹמְטָהּ מִפֶּתַח זֶה וְתוֹלֶה בַחֲבֵרוֹ בַשַּׁבָּת.

עַקְרָב שֶׁלִּפְרָמְבְּיָה, טְמֵאָה,

וּמְצֻפָּה, טְהוֹרָה.

וּבְשָׁעַת חִבּוּרָן, הַכֹּל טָמֵא.

וּלְחָיַיִם, טְהוֹרִין.

<small>י</small>

וְכַעֲדָשָׁה מִלְמַעְלָן, וְנִפְרַק, 
כִּקְדֵרָה, 
טָמֵא מִשֵּׁם כְּלִי קִבּוּל; 
וְכַעֲדָשָׁה, 
טָמֵא בִפְנֵי עַצְמוֹ.  
חוּט צִנֵּירָה, טְהוֹרָה.  
הֶעָשׂוּי כְּמִין אֶשְׁכּוֹל, וְנִפְרַק, 
טָהוֹר. 





Chapter 12

<small>א</small>

וְשֶׁלָּאֲסִירִים, טְהוֹרָה.

הָעֲשׁוּיָה לִכְפִיתָה, טְהוֹרָה.  
שַׁלְשֶׁלֶת שֶׁלַּסִּיטוֹנוֹת, טְמֵאָה; 
שֶׁלְּבַעֲלֵי בָתִּים, טְהוֹרָה.  
אָמַר רְבִּי יוֹסֵה: 
אֶמָּתַי? 
בִּזְמַן שֶׁהוּא מַפְתֵּחַ אֶחָד, 
אֲבָל אִם הָיוּ שְׁנַיִם, 
אוֹ שֶׁקָּשַׁר חַלָּזוֹן בְּרֹאשָׁהּ, טְמֵאָה. 

<small>ב</small>
קְנֵה מֹאזְנַיִם שֶׁלַּסָּרוֹקוֹת, טָמֵא, 
מִפְּנֵי אָנְקְיָיוֹת; 
וְשֶׁלְּבַעֲלֵי בָתִּים, 
אִם יֶשׁ בּוֹ אָנְקְיָיוֹת, טָמֵא.  
אָנְקְלֵי שֶׁלַּכַּתָּפִין, טְהוֹרָה, 
שֶׁלָּרוֹכְלִים, טְמֵאָה.  
רְבִּי יְהוּדָה אוֹמֵר: 
שֶׁלָּרוֹכְלִים, שֶׁלְּפָנָיו טְמֵאָה, 
וְשֶׁלְּאַחֲרָיו טְהוֹרָה.  

<small>ג</small>
אָנְקְלֵי שֶׁלַּדַּרְגֵּשׁ, טְמֵאָה; 
וְשֶׁלְּנִקְלִיטִים, טְהוֹרָה; 
שֶׁלַּשִּׁדָּה, טְמֵאָה; 
וְשֶׁלָּאָקוֹן, טְהוֹרָה; 
שֶׁלַּשֻּׁלְחָן, טְמֵאָה; 
וְשֶׁלִּמְנוֹרַת הָעֵץ, טְהוֹרָה.  
זֶה הַכְּלָל: 
הַמְחֻבָּר לַטָּמֵא, טָמֵא, 
הַמְחֻבָּר לַטָּהוֹר, טָהוֹר.  
וְכֻלָּם אַחַת וְאַחַת בִּפְנֵי עַצְמָן, טְהוֹרוֹת. 

ד &lt;ט&gt;
כְּסוּי טְנִי שֶׁלַּמַּתֶּכֶת, שֶׁלְּבַעֲלֵי בָתִּים, 
רַבָּן גַּמְלִיאֵל מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין.  
וְשֶׁלָּרוֹפְאִין, טָמֵא.  
הַדֶּלֶת שֶׁבַּמִּגְדָּל שֶׁלְּבַעֲלֵי בָתִּים, טְהוֹרָה; 
וְשֶׁלָּרוֹכְלִין, טְמֵאָה.  
הַיַּתֶּכִּים, טְמֵאִין.  
וְהַפֵּרַכִּין, טְהוֹרִין.  
עַקְרַב בֵּית הַבַּד, טְמֵאָה.  
אָנְקְלֵי שֶׁבִּכְתָלִים, טְהוֹרָה. 

<small>ה</small>
מַסְמֵר הַגָּרָע, טָמֵא. 
שֶׁלְּאֶבֶן שָׁעוֹת, טָהוֹר.  
רְבִּי צָדוֹק מְטַמֵּא.  
מַסְמֵר הַגִּרְדִּי, טָמֵא.  
וְאָרוֹן שֶׁלַּגָּרוֹסוֹת, 
רְבִּי צָדוֹק מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין.  
הָיְתָה עֲגָלָה שֶׁלַּמַּתֶּכֶת, טְמֵאָה. 

<small>ו</small>
מַסְמֵר שֶׁהִתְקִינוֹ לִהְיוֹת פּוֹתֵחַ וְנוֹעֵל בּוֹ, טָמֵא. 
הֶעָשׂוּי לִשְׁמִירָה, טָהוֹר.  
מַסְמֵר שֶׁהִתְקִינוֹ לִהְיוֹת פּוֹתֵחַ בּוֹ אֶת הֶחָבִית, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין, 
עַד שֶׁיִּצְרְפֶנּוּ.  
מַסְמֵר שֶׁלַּשֻּׁלְחָנִי, טָהוֹר. 
רְבִּי צָדוֹק מְטַמֵּא.  

<small>ז</small>
שְׁלֹשָׁה דְבָרִים 
רְבִּי צָדוֹק מְטַמֵּא וַחֲכָמִים מְטַהֲרִין: 
מַסְמֵר שֶׁלַּשֻּׁלְחָנִי, 
וְאָרוֹן שֶׁלַּגָּרוֹסוֹת, 
וּמַסְמֵר שֶׁלְּאֶבֶן שָׁעוֹת, 
רְבִּי צָדוֹק מְטַמֵּא, וַחֲכָמִים מְטַהֲרִין. 

<small>ח</small>
אַרְבָּעָה דְבָרִים רַבָּן גַּמְלִיאֵל מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין: 
כְּסוּי טְנִי שֶׁלַּמַּתֶּכֶת שֶׁלְּבַעֲלֵי בָתִּים, 
וּתְלוּי הַמַּגְרְדוֹת, 
וְגָלְמֵי כְלֵי מַתְּכוֹת, 
וְטַבְלָה שֶׁנֶּחְלְקָה לִשְׁנַיִם.  
וּמוֹדִים חֲכָמִים לְרַבָּן גַּמְלִיאֵל 
בְּטַבְלָה שֶׁנֶּחְלְקָה לִשְׁנַיִם, 
אֶחָד גָּדוֹל וְאֶחָד קָטָן, 
הַגָּדוֹל טָמֵא, וְהַקָּטָן טָהוֹר. 

<small>ט</small>
דִּינָר שֶׁנִּפְסַל, 
וְהִתְקִינוֹ לִתְלוֹתוֹ בְצַוַּאר קְטַנָּה, 
טָמֵא.  
סֶלַע שֶׁנִּפְסְלָה, 
וְהִתְקִינָהּ לִהְיוֹת שׁוֹקֵל בָּהּ, 
טְמֵאָה.  
עַד כַּמָּה תִפָּסֵל וִיהֵא רַשַּׁי לְקַיְּמָהּ? 
עַד שְׁנֵי דִינָרִים.  
פָּחוּת מִכֵּן, יָקֹץ. 

<small>י</small>
הָאוֹלָר וְהַקּוֹלְמוֹס, &lt;הָאֵוֹלָר&gt;
וְהַמְּטֻלְטֶלֶת, וְהַמִּשְׁקָלוֹת, 
וְהַכּידִים וְהַכַּן וְהַכַּנָּה, 
טְמֵאִין.  
וְכָל גָּלְמֵי כְלֵי עֵץ, 
טְמֵאִים, 
חוּץ מִשֶּׁלַּפִרְשָׁע.  
רְבִּי יְהוּדָה אוֹמֵר: 
גְּרוֹפִית שֶׁלַּזַּיִת, 
טְהוֹרָה, עַד שֶׁתִּשָּׁלֵק. 





Chapter 13

<small>א</small>
הַסַּיִף, וְהַסְּכִין, וְהַפִּגְיוֹן, וְהָרֹמַח, 
מַגַּל יָד, מַגַּל קָצִיר, וְהַשְּׁחִיר, 
זוֹג שֶׁלַּסַּפָּרִין, שֶׁנֶּחְלְקוּ, 
הֲרֵי אֵלּוּ טְמֵאִין.  
רְבִּי אוֹמֵר: 
סָמוּךְ לַיָּד, טָמֵא, 
סָמוּךְ לָרֹאשׁ, טָהוֹר.  
מַסְפֶּרֶת שֶׁנֶּחְלְקָה לִשְׁנַיִם, 
רְבִּי יְהוּדָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין. 

<small>ב</small>
קֳלִגְרָפוֹן שֶׁנִּטְּלָה כַפָּהּ, 
טְמֵאָה, מִפְּנֵי שִׁנָּהּ; 
נִטְּלָה שִׁנָּהּ, 
טְמֵאָה, מִפְּנֵי כַפָּהּ.  
מַכְחָל שֶׁנִּטְּלָה הַכַּף, 
טָמֵא, מִפְּנֵי הַזָּכָר; 
נִטַּל הַזָּכָר, 
טָמֵא, מִפְּנֵי הַכַּף.  
מִכְתָּב שֶׁנִּטַּל הַכּוֹתֵב, 
טָמֵא, מִפְּנֵי הַמּוֹחֵק.  
נִטַּל הַמּוֹחֵק, 
טָמֵא, מִפְּנֵי הַכּוֹתֵב.  
זוֹמָאנִצְטְרָה שֶׁנִּטְּלָה כַפָּהּ, 
טְמֵאָה, מִפְּנֵי הַמַּזְלֵג.  
נִטְּלָה הַמַּזְלֵג, 
טְמֵאָה, מִפְּנֵי כַפָּהּ.  
וְכֵן הַשֵּׁן שֶׁלַּמַּעְדֵּר.  
שֵׁעוּר כֻּלָּן, כְּדֵי לַעֲשׁוֹת מְלַאכְתָּן. 

<small>ג</small>
חַרְחוּר שֶׁנִּפְגַּם, 
טָמֵא, עַד שֶׁיִּנָּטֵל רֻבּוֹ.  
נִשְׁבַּר מַקּוּפוֹ, 
טָהוֹר.  
קָרְדֹּם שֶׁנִּטַּל עָשְׁפּוֹ, 
טָמֵא, מִפְּנֵי בֵית בִּקּוּעוֹ; 
נִטַּל בֵּית בִּקּוּעוֹ, 
טָמֵא, מִפְּנֵי עָשְׁפּוֹ.  
נִשְׁבַּר מַקּוּפוֹ, 
טָהוֹר. 

<small>ד</small>
מַגְרֵפָה שֶׁנִּטְּלָה כַפָּהּ, טְמֵאָה, 
מִפְּנֵי שֶׁהִיא כְקָרְנָס.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים מְטַהֲרִין.  
מְגֵרָה שֶׁנִּטְּלוּ שִׁנֶּיהָ אַחַת מִבִּנְתַּיִם, טְהוֹרָה.  
נִשְׁתַּיַּר בָּהּ מְלֹא הַסִּיט בְּמָקוֹם אֶחָד, טְמֵאָה.  
הַמַּעְצָד, וְהָאַזְמֵל, וְהַמַּפְסֶלֶת, 
וְהַמַּקְדֵּחַ שֶׁנִּפְגָּמוּ, טְמֵאִין.  
נִטַּל חַסוּמָּן, טְהוֹרִין.  
וְכֻלָּם שֶׁנֶּחְלְקוּ לִשְׁנַיִם, טְמֵאִין, 
חוּץ מִן הַמַּקְדֵּחַ.  
וְהָרוֹקְנָה בִפְנֵי עַצְמָהּ, טְהוֹרָה. 

<small>ה</small>
מַחַט שֶׁנִּטַּל חֲרִירָהּ אוֹ עָקְצָהּ, 
טְהוֹרָה.  
אִם הִתְקִינָהּ לְמִיתָה, 
טְמֵאָה.  
שֶׁלְּסַקָּיִין שֶׁנִּטַּל חֲרִירָהּ, 
טְמֵאָה, מִפְּנֵי שֶׁהוּא כוֹתֵב בָּהּ. 
נִטַּל עָקְצָהּ, 
טְהוֹרָה.  
שֶׁלַּמִתּוֹן, 
בֵּין כָּךְ וּבֵין כָּךְ טְמֵאָה.  

<small>ו</small>
מַחַט שֶׁהֶעֱלַת חֲלוּדָה, 
אִם מְעַכֶּבֶת אֶת הַתְּפִירָה, 
טְהוֹרָה; 
וְאִם לָאו, טְמֵאָה.  
צִנֵּירָה שֶׁפְּשָׁטָהּ, 
טְהוֹרָה.  
כְּפָפָהּ, 
חָזְרָה לְטֻמְאָתָהּ. 

<small>ז</small>
עֵץ הַמְשַׁמֵּשׁ אֶת הַמַּתֶּכֶת, 
טָמֵא. 
וְהַמַּתֶּכֶת וְהַמְשַׁמֶּשֶׁת אֶת הָעֵץ, 
טָהוֹר.  
כֵּיצַד? 
פּוֹתַחַת שֶׁלָּעֵץ, 
וְהַפִּין שֶׁלָּהּ שֶׁלַּמַּתֶּכֶת, 
אֲפִלּוּ אַחַת, 
טְמֵאָה.  
פּוֹתַחַת שֶׁלַּמַּתֶּכֶת, 
וְהַפִּין שֶׁלָּהּ שֶׁלָּעֵץ, 
טְהוֹרָה.  
טַבַּעַת שֶׁלַּמַּתֶּכֶת, 
וְחוֹתָם שֶׁלָּהּ שֶׁלָּאַלְמוֹג, 
טְמֵאָה.  
טַבַּעַת שֶׁלָּאַלְמוֹג, 
וְחוֹתָם שֶׁלָּהּ שֶׁלַּמַּתֶּכֶת, 
טְהוֹרָה.  
הַשֵּׁן שֶׁבַּטַּס שֶׁבַּפּוֹתַחַת, וְשֶׁבַּמַּפְתֵּחַ, 
טְמֵאָה בִפְנֵי עַצְמָהּ. 

<small>ח</small>
הַכְּדֻמִּים הָאַשְׁקְלוֹנִים שֶׁנִּשְׁבָּרוּ, 
וְהָאָנְקְלֵי שֶׁלָּהֶם קַיֶּמֶת, 
הֲרֵי אֵלּוּ טְמֵאִין.  
הַמַּעְבֵּר, וְהַמִּזְרֶה, וְהַמַּגּוֹב, 
וְכֵן מַשְׂרֵק שֶׁלָּרֹאשׁ, 
שֶׁנִּטְּלָה אַחַת מִשִּׁנֵּיהֶן, 
וַעֲשָׂאָן שֶׁלַּמַּתֶּכֶת, 
הֲרֵי אֵלּוּ טְמֵאִין.  
וְעַל כֻּלָּם אָמַר רְבִּי יְהוֹשֻׁעַ: 
דָּבָר חָדָשׁ חִדְּשׁוּ סוֹפְרִין, 
וְאֵין לִי מָה אָשִׁיב. 

<small>ט</small>
מַשְׂרֵק שֶׁלַּפִּשְׁתָּן שֶׁנִּטְּלוּ שִׁנָּיו, 
נִשְׁתַּיְּרוּ בוֹ שְׁתַּיִם, טָמֵא, 
וְאַחַת, טָהוֹר. 
וְכֻלָּם, אַחַת וְאַחַת בִּפְנֵי עַצְמָן, 
טְמֵאוֹת.  

<small>י</small>
וְשֶׁלַּצֶּמֶר שֶׁנִּטְּלוּ שִׁנָּיו וְאַחַת מִבִּנְתַּיִם, 
טָהוֹר. 
נִשְׁתַּיַּר בּוֹ שָׁלוֹשׁ בְּמָקוֹם אֶחָד, 
טָמֵא.  
הָיְתָה הַחִיצוֹנָה אַחַת מֵהֶן, 
טָהוֹר.  
נִטְּלוּ מִמֶּנּוּ שְׁתַּיִם וַעֲשָׂאָן לַמַּלְקֶטֶת, 
טְמֵאוֹת; 
אַחַת, וְהִתְקִינָהּ לַנֵּר, אוֹ לַמִּיתָה,

Chapter 14

<small>א</small>
כְּלֵי מַתְּכוֹת, 
כַּמָּה הוּא שֵׁעוּרָן? 
הַדְּלִי, כְּדֵי לְמַלֹּאות בּוֹ.  
קֻמְקוּמוֹס, כְּדֵי לָחוּם בּוֹ.  
מֵחַם, כְּדֵי לְקַבֵּל סְלָעִים.  
הַלְּבֵס, כְּדֵי לְקַבֵּל קִיתוֹנוֹת.  
קִיתוֹנוֹת, כְּדֵי לְקַבֵּל פְּרוּטוֹת.  
מִדּוֹת הַיַּיִן, בְּיַיִן, 
וּמִדּוֹת הַשֶּׁמֶן, בְּשֶׁמֶן; 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
כֻּלָּם בִּפְרוּטוֹת.  
רְבִּי עֲקִיבָה אוֹמֵר: 
הַמְחֻסָּר חֲטָפָה, טָמֵא, 
וְהַמְחֻסָּר לְטִישָׁה, טָהוֹר. 

<small>ב</small>
מַקֵּל שֶׁעָשָׂה בְרֹאשׁוֹ מַסְמֵר כְּמִין חֲזַיְנָה, 
טָמֵא; 
סִמְּרוֹ, טָמֵא. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
עַד שֶׁיַּעֲשֶׂה בוֹ שְׁלֹשָׁה סְדָרִים. 
וְכֻלָּם, שֶׁעֲשָׂאָן לַנּוֹי, 
טְהוֹרִין.  
עָשָׂה בְרֹאשׁוֹ מְנִיקִית, וְכֵן בַּדֶּלֶת, 
טְהוֹרָה; 
הָיְתָה כֶלִי וְחִבְּרָהּ לוֹ, 
טְמֵאָה.  
מֵאֶמָּתַי הִיא טַהֲרַתּוּ? 
בֵּית שַׁמַּי אוֹמְרִים: 
מִשֶּׁיְּחַבֵּל. 
וּבֵית הֶלֵּל אוֹמְרִים: 
מִשֶּׁיְּחַבֵּר. 

<small>ג</small>
הַקַּנְטֵר שֶׁלַּבַּנַּאי, 
וְהַדָּקוֹר שֶׁלֶּחָרָשׁ, &lt;חֶרֶשׂ&gt;
הֲרֵי אֵלּוּ טְמֵאִין.  
יִתְדוֹת אֲהָלִין, 
וְיִתְדוֹת הַמָּשׁוֹחוֹת, 
טְמֵאוֹת.  
שַׁלְשֶׁלֶת שֶׁלַּמָּשׁוֹחוֹת, 
טְמֵאָה. 
הָעֲשׁוּיָה לָעֵצִים, 
טְהוֹרָה.  
שַׁלְשֶׁלֶת דְּלִי גָדוֹל, 
אַרְבָּעָה טְפָחִים; 
וְשֶׁלַּקָּטוֹן, עֲשָׂרָה.  
חֲמוֹר שֶׁלַּנַּפָּחִים,  
טְמֵאָה. 
מְגֵרָה שֶׁעָשָׂה שִׁנֶּיהָ בְתוֹךְ הַחֹר, 
טְמֵאָה; 
עֲשָׂאָן מִלְּמַטָּן לְמַעְלָן, 
טְהוֹרָה.  
כָּל הַכְּסוּיִין טְהוֹרִין, 
חוּץ מִשֶּׁלַּמֵּחַם. &lt;מִשַּׁלְּמֵּיחֶם&gt;

<small>ד</small>
הַטְּמֵאִין שֶׁבָּעֲגָלָה: 
הָעֹל שֶׁלַּמַּתֶּכֶת, 
וְהַקַּטְרֵב, 
וְהַכְּנָפַיִם הַמְקַבְּלוֹת אֶת הָרְצוּעוֹת, &lt;הַמְקַבֶּלֶות&gt;
וּבַרְזֶל שֶׁתַּחַת צַוְּארֵי בְהֵמָה, 
הַסּוֹמֵךְ, וְהַמְחַגֵּר, 
וְהַתַּמְחוּיוֹת, וְהָעַנְבוּל, וְהַצִּנֵּירָה, 
וּמַסְמֵר מְחַבֵּר אֶת כֻּלָּם. 

<small>ה</small>
הַטְּהוֹרִים שֶׁבָּעֲגָלָה, 
וְהָעֹל הַמְצֻפֶּה, 
וּכְנָפַיִם הָעֲשׁוּיוֹת לַנּוֹי, 
וּשְׁפוֹפְרוֹת הַמַּשְׁמִיעוֹת אֶת הַקּוֹל, 
וְהָאֲבָר שֶׁבְּצַד צַוְּארֵי בְהֵמָה, 
הַסּוֹבֵב שֶׁלַּגַּלְגַּל, 
וְהַטָּסִין, וְהַצִּפּוּיִין, 
וּשְׁאָר כָּל הַמַּסְמְרוֹת, 
טְהוֹרִין.  
סַנְדַּלֵּי בְהֵמָה שֶׁלַּמַּתֶּכֶת, 
טְמֵאִין, 
שֶׁלַּשַּׁעַם, טְהוֹרִין.  
הַסֵּיף, מֵאֶמָּתַי מְקַבֵּל טֻמְאָה? 
מִשֶּׁיְּשׁוּפֶנּוּ.  
וְהַסְּכִין? 
מִשֶּׁיַּשְׁחִיזֶנָּה. 

<small>ו</small>
כְּסוּי טְנִי שֶׁלַּמַּתֶּכֶת שֶׁעָשָׂה בוֹ מַרְאָה, 
רְבִּי יְהוּדָה מְטַהֵר, 
וַחֲכָמִים מְטַמִּין.  
מַרְאָה שֶׁנִּשְׁבָּרָה, 
אִם אֵינָה מַרְאָה אֶת רֹב הַפָּנִים, 
טְהוֹרָה. 

<small>ז</small>
כְּלֵי מַתְּכוֹת מִטַּמִּין וּמִטַּהֲרִין שְׁבָרִין.  
דִּבְרֵי רְבִּי אֱלִיעֶזֶר.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
אֵינָן מִטַּהֲרִין אֶלָּא שְׁלֵמִים.  
כֵּיצַד? 
הִזָּה עֲלֵיהֶן, 
וְנִשְׁבָּרוּ בוֹ בַיּוֹם, 
הִתִּיכָן, חָזַר וְהִזָּה עֲלֵיהֶן שְׁנִיָּה בוֹ בַיּוֹם, 
הֲרֵי אֵלּוּ טְהוֹרִין.  
דִּבְרֵי רְבִּי אֱלִיעֶזֶר.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
אֵין הַזָּיָה פָחוּת מִשְּׁלִישִׁי וּמִשְּׁבִיעִי. 

<small>ח</small>
מַפְתֵּחַ שֶׁלָּאַרְכֻּבָּה שֶׁנִּשְׁבְּרָה מִתּוֹךְ אַרְכֻּבָּתוֹ, 
טָהוֹר.  
רְבִּי יְהוּדָה מְטַמֵּא, 
מִפְּנֵי שֶׁהוּא פוֹתֵחַ בּוֹ מִבִּפְנִים.  
וְשֶׁלַּגַּם שֶׁנִּשְׁבַּר מִתּוֹךְ גֻּמּוֹ, 
טָהוֹר.  
הָיוּ בוֹ חָפִין וּנְקָבִין, 
טָמֵא.  
נִטְּלוּ חָפִין, 
טָמֵא, מִפְּנֵי נְקָבִין.  
נִסְתַּתְּמוּ נְקָבִין, טָמֵא, 
מִפְּנֵי חָפִין; 
נִטְּלוּ חָפִין וְנִסְתַּתְּמוּ נְקָבִין, 
אוֹ שֶׁפְּרָצָן זֶה לְתוֹךְ זֶה, 
טָהוֹר.  
מַסְנֶנֶת שֶׁלַּחַרְדָּל 
שֶׁנִּפְרְצוּ בָהּ שְׁלֹשָׁה נְקָבִים מִלְּמַטָּן זֶה לְתוֹךְ זֶה, 
טְהוֹרָה.  
וְאַרְפַּכְס שֶׁלַּמַּתֶּכֶת, 
טְמֵאָה. 





Chapter 15

<small>א</small>
כְּלֵי עֵץ, 
כְּלֵי עוֹר, 
כְּלֵי עֶצֶם, 
כְּלֵי זְכוֹכִית, 
פְּשׁוּטֵיהֶן טְהוֹרִין, 
וּמְקַבְּלֵיהֶן טְמֵאִין.  
נִשְׁבָּרוּ, טָהָרוּ.  
חָזַר וְעָשָׂה מֵהֶן כֵּלִים, 
מְקַבְּלִין טֻמְאָה מִכָּן וְלַבָּא.  
הַשִּׁדָּה, 
וְהַתֵּבָה, 
וְהַמִּגְדָּל, 
כַּוֶּרֶת הַקַּשׁ, 
וְכַוֶּרֶת הַקָּנִים, 
וּבוֹר סְפִינָה אֲלֶכְּסַנְדְּרִית, 
שֶׁיֵּשׁ לָהֶם שׁוּלַיִם, 
וְהֵן מַחֲזִיקִים אַרְבָּעִים לַח, 
שֶׁהֵם כּוֹרַיִם בַּיָּבֵשׁ, 
הֲרֵי אֵלּוּ טְהוֹרִין.  
וּשְׁאָר כָּל הַכֵּלִים, 
בֵּין מְקַבְּלִין בֵּין אֵינָן מְקַבְּלִין, 
טְמֵאִים.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יְהוּדָה אוֹמֵר: 
דַּרְדּוּר עֲגָלָה, 
וְקָסֻטּוֹת הַמְּלָכִים, &lt;וקוסטות&gt;
וַעֲרֵבַת הָעֻבְּדָן, 
וּבוֹר סְפִינָה קְטַנָּה, 
וְהָאָרוֹן, 
אַף עַל פִּי מְקַבְּלִין, 
טְמֵאִין, 
שֶׁאֵינָן עֲשׁוּיִים לְהִטַּלְטֵל אֶלָּא בְמַה שֶּׁבְּתוֹכָן.  
וּשְׁאָר כָּל הַכֵּלִים, 
הַמְקַבְּלִין, טְהוֹרִין, 
וְשֶׁאֵינָן מְקַבְּלִין, טְמֵאִין.  
אֵין בֵּין דִּבְרֵי רְבִּי מֵאִיר לְדִבְרֵי רְבִּי יְהוּדָה 
אֶלָּא עֲרֵבַת בַּעַל הַבַּיִת. 

<small>ב</small>
אֲרֻבּוֹת שֶׁלַּנַּחְתּוֹמִים, טְמֵאוֹת, 
וְשֶׁלְּבַעֲלֵי בָתִּים, טְהוֹרוֹת. 
סְרָקָן אוֹ כִרְכְּמָן, טְמֵאוֹת. &lt;סירקן&gt; 
דַּף שֶׁלַּנַּחְתּוֹמִים שֶׁקְּבָעוֹ בַכֹּתֶל, 
רְבִּי אֱלִיעֶזֶר מְטַהֵר, 
וַחֲכָמִים מְטַמִּין.  
סְרוּד שֶׁלַּנַּחְתּוֹמִים, טָמֵא, 
וְשֶׁלְּבַעֲלֵי בָתִּים, טָהוֹר. 
גִּפְּפוֹ מֵאַרְבַּע רוּחוֹתָיו, טָמֵא. 
נִפְרַץ מֵרוּחַ אַחַת, טָהוֹר. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אִם הִתְקִינוֹ לִהְיוֹת פּוֹרֵץ עָלָיו, טָמֵא.  
וְכֵן הַמַּעֲרוֹךְ, טָמֵא. 

<small>ג</small>
יָם נָפָה שֶׁלְּסַלָּתִים, טָמֵא, 
וְשֶׁלְּבַעֲלֵי בָתִּים, טָהוֹר.  
רְבִּי יְהוּדָה אוֹמֵר: 
אַף שֶׁלַּגַּדֶּלֶת טָמֵא מוֹשָׁב, 
מִפְּנֵי שֶׁהַבָּנוֹת יוֹשְׁבוֹת לְתוֹכוֹ וְגוֹדְלוֹת. 

<small>ד</small>
כָּל הַתְּלוּיִן טְמֵאִין, 
חוּץ מִתְּלוּי נָפָה וּכְבָרָה שֶׁלְּבַעַל הַבַּיִת.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
כֻּלָּם טְהוֹרִין, 
חוּץ מִתְּלוּי נָפָה שֶׁלְּסַלָּתִין, 
וּתְלוּי כְּבָרַת גֳּרָנוֹת, 
וּתְלוּי מַגַּל יָד, 
וּתְלוּי מַקֵּל הַבָּלְשִׁין, 
מִפְּנֵי שֶׁהֵן מְסַיְּעִין בְּשָׁעַת מְלָאכָה.  
זֶה הַכְּלָל: 
הֶעָשׂוּי לְסִיּוּעַ בְּשָׁעַת מְלָאכָה, טָמֵא; 
הֶעָשׂוּי לְתָלוּי, טָהוֹר. 

<small>ה</small>
רַחַת הַגָּרוֹסוֹת, טְמֵאָה, 
וְשֶׁלְּאוֹצָרוֹת, טְהוֹרָה.  
זֶה הַכְּלָל: 
הֶעָשׂוּי לְקַבָּלָה, טָמֵא, 
וּלְכִנּוּס, טָהוֹר. 

<small>ו</small>
נִבְלֵי הַשָּׁרָה, טְמֵאִין, 
וְנִבְלֵי בְנֵי לֵוִי, טְהוֹרִין.  
כָּל הַמַּשְׁקִין, טְמֵאִין, 
וּמַשְׁקֵה בֵית מַטְבְּחִיָּה, טְהוֹרִין.  
כָּל הַסְּפָרִים מְטַמִּין אֶת הַיָּדַיִם, 
חוּץ מִסֵּפֶר הָעֲזָרָה.  
הַמַּרְכּוּב טָהוֹר.  
הַבַּטְנוֹן וְהַנִּקְטְמוֹן וְהָאֵרוֹס, 
הֲרֵי אֵלּוּ טְמֵאִין.  
רְבִּי יְהוּדָה אוֹמֵר: 
הָאֵרוֹס טָמֵא מוֹשָׁב, 
מִפְּנֵי שֶׁהָאַלָּאִית יוֹשֶׁבֶת עָלָיו.  
מְצוֹדַת הַחֻלְדָּה, טְמֵאָה, 
וְשֶׁלָּעַכְבָּרִין, טְהוֹרָה. 





Chapter 16

<small>א</small>
כָּל כְּלִי עֵץ שֶׁנֶּחֱלַק לִשְׁנַיִם, טָהוֹר, 
חוּץ מִשֻּׁלְחָן הַסָּפוּל, 
וְתַמְחוּי הַמַּזְנוֹן, 
וָאַפִּיפּוֹדִין שֶׁלְּבַעַל הַבַּיִת.  
רְבִּי יְהוּדָה אוֹמֵר: 
אַף הַמְּגֵס וְקוֹד הַבַּבְלִי כַּיּוֹצֵא בָהֶן.  
כְּלֵי עֵץ, מֵאֶמָּתַי מְקַבְּלִין טֻמְאָה? 
הַמִּטָּה וְהָעֲרִיסָה, מִשֶּׁיְּשׁוּפֵם בְּעוֹר הַדָּג.  
גָּמַר שֶׁלֹּא לָשׁוּף, טְמֵאָה.  
רְבִּי מֵאִיר אוֹמֵר: 
הַמִּטָּה, מִשֶּׁיְּסָרֵג בָּהּ שְׁלֹשָׁה בָתִּים. 

<small>ב</small>
הַסַּלִּים שֶׁלָּעֵץ, מִשֶּׁיַּחְסֹם וִיקַנֵּב; 
וְשֶׁלַּתְּמָרָה, 
אַף עַל פִּי שֶׁלֹּא קִנֵּב מִבִּפְנִים, 
טָמֵא, שֶׁכֵּן מְקַיְּמִים.  
כַּלְכַּלָּה, 
מִשֶּׁיַּחְסֹם וִיקַנֵּב וְיִגְמֹר אֶת הַתְּלוּיָה.  
בֵּית הַלָּגִינִין, וּבֵית הַכּוֹסוֹת, 
אַף עַל פִּי שֶׁלֹּא קִנֵּב מִבִּפְנִים, 
טָמֵא, שֶׁכֵּן מְקַיְּמִים. 

<small>ג</small>
הַקְּנוּנִין הַקְּטַנִּים, וְהַקְּלָתוֹת, 
מִשֶּׁיַּחְסֹם וִיקַנֵּב.  
הַקְּנוּנִין הַגְּדוֹלִין, וְהַסּוֹגִין הַגְּדוֹלִים, 
מִשֶּׁיַּעֲשֶׂה שְׁנֵי דוּרִין לָרַחַב שֶׁלָּהֶן.  
יָם, נָפָה, וּכְבָרָה, וְכַף שֶׁלְּמֹאזְנַיִם, 
מִשֶּׁיַּעֲשֶׂה דוּר אֶחָד לָרַחַב שֶׁלָּהֶן.  
הַקֻּפָּה, 
מִשֶּׁיַּעֲשֶׂה שְׁתֵּי צְפִירוֹת לָרַחַב שֶׁלָּהּ.  
וְהָעֲרָק, 
מִשֶּׁיַּעֲשֶׂה בוֹ צְפִירָה אַחַת. 

<small>ד</small>
כְּלֵי עוֹר, מֵאֶמָּתַי מְקַבְּלִין טֻמְאָה? 
הַתָּרְמֵל, מִשֶּׁיַּחְסֹם וִיקַנֵּב, 
וְיַעֲשֶׂה קֵיחוֹתָיו. 
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁיַּעֲשֶׂה אָזְנָיו. 
סְקֻרְטְיָה, 
מִשֶּׁיַּחְסֹם וִיקַנֵּב, 
וְיַעֲשֶׂה אֶת צִיצִיתָהּ. 
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁיַּעֲשֶׂה אֶת טַבְּעוֹתֶיהָ.  
קָטְבֻלְיָה, מִשֶּׁיַּחְסֹם וִיקַנֵּב. 
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁיַּעֲשֶׂה אֶת קֵיחוֹתֶיהָ.  

<small>ה</small>
הַכַּר וְהַכֶּסֶת שֶׁלָּעוֹר, 
מִשֶּׁיַּחְסֹם וִיקַנֵּב. 
רְבִּי יְהוּדָה אוֹמֵר: 
מִשֶּׁיִּתְפְּרֵם 
וִישַׁיֵּר בָּהֶן פָּחוּת מֵחֲמִשָּׁה טְפָחִים. 

<small>ו</small>
פָּטַלְיָה טְמֵאָה, &lt;מחוק ומתוקן. כנראה היה פטילייא&gt; 
וַחֲסִינָה טְהוֹרָה.  
סִגְנָיוֹת שֶׁלֶּעָלִים, טְהוֹרוֹת, 
וְשֶׁלִּנְצָרִין, טְמֵאוֹת.  
חוֹתָל שֶׁהוּא נוֹתֵן לְתוֹכוֹ וְנוֹטֵל מִתּוֹכוֹ, 
טָמֵא. 
אֵינוּ יָכוֹל עַד שֶׁיִּקְרָעֶנּוּ, עַד שֶׁיַּתִּירֶנּוּ, 
טָהוֹר. 

<small>ז</small>
קִסְיָה שֶׁלְּזוֹרֵי גֳרָנוֹת, 
שֶׁלְּהוֹלְכֵי דְרָכִים, 
שֶׁלְּעוֹשֵׂי פִשְׁתָּן, טְמֵאָה. 
אֲבָל שֶׁלַּצַּבָּעִים וְשֶׁלַּנַּפָּחִים, טְהוֹרָה.  
רְבִּי יוֹסֵה אוֹמֵר: 
אַף שֶׁלַּגָּרוֹסוֹת כַּיּוֹצֵא בָהֶן.  
זֶה הַכְּלָל: 
הֶעָשׂוּי לְקַבָּלָה, טָמֵא; 
מִפְּנֵי הַזֵּעָה, טָהוֹר. 

<small>ח</small>
הַמַּקְלוֹט שֶׁלַּבָּקָר, 
וְהַחִסּוּם שֶׁלּוֹ, 
וְהַמַּדָּף שֶׁלַּדְּבוֹרִים, 
וְהַמְּנָפָה, 
הֲרֵי אֵלּוּ טְהוֹרִין.  
כְּסוּי קַפְצָא, טָמֵא.  
כְּסוּי קַמְטְרָא, טָהוֹר.  
כְּסוּי תֵּבָה, כְּסוּי טְנִי, 
וְהַמַּכְבֵּשׁ שֶׁלֶּחָרָשׁ, 
וְהַכִּסֵּא שֶׁתַּחַת הַתֵּבָה, 
וְהַקַּמְּרוֹן שֶׁלָּהּ, 
וַאֲנַגְלִים שֶׁלַּסֵּפֶר, 
וּבֵית הַנַּגָּר, 
וּבֵית הַמַּנְעוּל, 
וּבֵית הַמְּזוּזָה, 
תֵּיק נְבָלִים, 
תֵּיק כִּנּוֹרוֹת, 
וְאָמוּם שֶׁלְּגוֹדְלֵי מִצְנְפוֹת, 
וְהַמַּרְכּוּג שֶׁלַּזַּמָּר, 
וּרְבִיעִין שֶׁלָּאַלָּאִית, 
וְגוֹגֶנֶת הֶעָנִי, 
וּסְמוּכוֹת הַמִּטָּה, 
וּטְפָס שֶׁלַּתְּפִלָּה, 
וַאֲמוּם שֶׁלְּעוֹשֵׂי סָבוֹת, 
הֲרֵי אֵלּוּ טְהוֹרִין.  
זֶה הַכְּלָל אָמַר רְבִּי יוֹסֵה: 
כָּל מְשַׁמְּשֵׁי מְשַׁמְּשָׁיו שֶׁלָּאָדָם, 
בְּשָׁעַת מְלָאכָה וְשֶׁלֹּא בְשָׁעַת מְלָאכָה, 
טָמֵא.  
וְכָל שֶׁאֵינוּ אֶלָּא בְשָׁעַת מְלָאכָה, 
טָהוֹר. 

<small>ט</small>
תֵּיק הַסַּיִף וְהַסְּכִין וְהַפִּגְיוֹן, 
תֵּיק מִסְפֶּרֶת וּמִסְפְּרַיִם וְהַתַּעַר, 
תֵּיק מַכְחָל, 
וּבֵית הַכֹּחַל, 
תֵּיק מִכְתָּב, 
וּתְרֻנְתֵּק, 
תֵּיק טַבְלָה וּסְקוֹרְטְיָה, 
בֵּית הַחִצִּים, 
וּבֵית הַפָּגוֹשׁוֹת, 
הֲרֵי אֵלּוּ טְמֵאִין.  
תֵּיק סִימְפוֹנְיָא, 
בִּזְמַן שֶׁהוּא נוֹתְנָהּ מִלְמַעְלָה, טָמֵא; 
מִצִּדּוֹ, טָהוֹר.  
תֵּיק חֲלִילִים, 
רְבִּי יְהוּדָה מְטַהֵר, 
מִפְּנֵי שֶׁהוּא נוֹתְנוֹ מִצִּדּוֹ. &lt;נוֹתְנָהּ&gt; 
חֲפוּי הָאַלָּה, 
וְהַקֶּשֶׁת, 
וְהָרֹמַח, 
הֲרֵי אֵלּוּ טְהוֹרִין.  
זֶה הַכְּלָל: 
הֶעָשׂוּי לְתֵיק, טָמֵא, 
לַחֲפוּי, טָהוֹר. 





Chapter 17

<small>א</small>
כָּל כְּלֵי בַעֲלֵי בָתִּים, 
שֵׁעוּרָן בָּרִמּוֹנִים; 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
בְּמַה שֶּׁהֵן.  
קֻפּוֹת הַגַּנָּנִים, 
שֵׁעוּרָן בַּאֲגֻדּוֹת שֶׁלַּיָּרָק, 
שֶׁלְּבַעֲלֵי בָתִּים, בַּתֶּבֶן, 
שֶׁלַּבַּלָּנִין, בַּגְּבָבָה.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
כֻּלָּם בָּרִמּוֹנִים. 

<small>ב</small>
הַחֵמֶת, 
שֵׁעוּרָהּ בְּפַקָּעִיּוֹת שֶׁלַּשֶּׁתִי.  
אִם אֵינָהּ מְקַבֶּלֶת שֶׁלַּשֶּׁתִי, 
אַף עַל פִּי מְקַבֶּלֶת שֶׁלָּעֵרֶב, 
טְמֵאָה.  
בֵּית קְעָרוֹת שֶׁאֵינוּ מְקַבֵּל קְעָרוֹת, 
אַף עַל פִּי מְקַבֵּל תַּמְחוּיִים, 
טָמֵא.  
בֵּית הָרֶאִי שֶׁאֵינוּ מְקַבֵּל מַשְׁקִים, &lt;=הָרֵעִי&gt;
אַף עַל פִּי שֶׁמְּקַבֵּל אֶת הָרֶאִי, 
טָמֵא.  
רַבָּן גַּמְלִיאֵל מְטַהֵר, 
שֶׁהֵן מְקַיְּמִין אוֹתוֹ. 

<small>ג</small>
הַסַּלִּים שֶׁלַּפַּת, 
שֵׁעוּרָן בְּכִכָּרוֹת שֶׁלַּפַּת.  
פִּיפְּיָרוֹת 
שֶׁעָשָׂה לָהּ קָנִים מִלְּמַטָּן לְמַעְלָן לְחִזּוּק, 
טְהוֹרָה.  
עָשָׂה לָהּ גַּפַּיִם כָּל שֶׁהֵן, 
טְמֵאָה.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אִם אֵינָה יְכוּלָה לְהִנָּטֵל בְּגַפַּיִם, 
טְהוֹרָה. 

<small>ד</small>
הָרִמּוֹנִים שֶׁאָמָרוּ, 
שְׁלֹשָׁה אֲחוּזִים זֶה בָזֶה.  
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
בְּנָפָה וּבִכְבָרָה, כְּדֵי שֶׁיִטֹּל וִיהַלֵּךְ.  
וּבַקֻּפָּה, כְּדֵי שֶׁיַּפְשִׁיל לְאַחֲרָיו.  
וּשְׁאָר כָּל הַכֵּלִים שֶׁאֵינָן יְכוּלִים לְקַבֵּל רִמּוֹנִים, 
כְּגוֹן רֹבַע, וַחֲצִי הָרֹבַע, 
הַקְּנוֹנִין הַקְּטַנִּים, 
שֵׁעוּרָן בְּרֻבָּן.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
בַּזֵּיתִין.  
נִפְרָצוּ, שֵׁעוּרָן בַּזֵּיתִים.  
נִגְמָמוּ, שֵׁעוּרָן בְּמַה שֶּׁהֵן. 

<small>ה</small>
הָרִמּוֹן שֶׁאָמָרוּ, 
לֹא גָדוֹל וְלֹא קָטֹן, אֶלָּא בֵינוֹנִי.  
וּלְמָה הָזְכָּרוּ רִמּוֹנֵי בָדָן? 
שֶׁיְּהוּ מְקַדְּשִׁין כָּל שֶׁהֵן.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יוֹחָנָן בֶּן נוּרִי אוֹמֵר: 
לְשַׁעֵר בָּהֶם אֶת הַכֵּלִים.  
רְבִּי עֲקִיבָה אוֹמֵר: 
לְכָךְ וּלְכָךְ הָזְכָּרוּ: 
לְשַׁעֵר בָּהֶן אֶת הַכֵּלִים, 
וְשֶׁיְּהוּ מְקַדְּשִׁין כָּל שֶׁהֵן.  
אָמַר רְבִּי יוֹסֵה: 
לֹא שֶׁיְּהוּ רִמּוֹנֵי בָדָן וַחֲצִירֵי גֶבַע, 
אֶלָּא שֶׁיְּהוּ מִתְעַשְּׂרִין וַדַּי בְּכָל מָקוֹם. 

ו &lt;ז&gt;
כַּבֵּיצָה שֶׁאָמָרוּ, 
לֹא גְדוֹלָה וְלֹא קְטַנָּה, אֶלָּא בֵינוֹנִית.  
רְבִּי יְהוּדָה אוֹמֵר: 
מֵבִיא גְדוֹלָה שֶׁבַּגְּדוֹלוֹת וּקְטַנָּה שֶׁבַּקְּטַנּוֹת, 
וְנוֹתֵן לְתוֹךְ הַמַּיִם, 
וְחוֹלֵק אֶת הַמַּיִם.  
אָמַר רְבִּי יוֹסֵה: 
וְכִי מִי מוֹדִיעֵנִי אֵי זוֹ גְדוֹלָה וְאֵי זוֹ קְטַנָּה? 
אֶלָּא הַכֹּל לְפִי דַעְתּוֹ שֶׁלָּרוֹאֶה. 

ז &lt;ח&gt;
כַּגְּרוֹגֶרֶת שֶׁאָמָרוּ, 
לֹא גְדוֹלָה וְלֹא קְטַנָּה, אֶלָּא בֵינוֹנִית.  
רְבִּי יְהוּדָה אוֹמֵר: 
הַגְּדוֹלָה שֶׁבְּאֶרֶץ יִשְׂרָאֵל, 
הִיא הַבֵּינוֹנִית שֶׁבַּמְּדִינוֹת. 

ח &lt;ט&gt;
כַּזַּיִת שֶׁאָמָרוּ, 
לֹא גָדוֹל וְלֹא קָטֹן, אֶלָּא בֵינוֹנִי, 
זֶה אֲגוֹרִי.  
כִּשְׂעוֹרָה שֶׁאָמָרוּ, 
לֹא גְדוֹלָה וְלֹא קְטַנָּה, אֶלָּא בֵינוֹנִית, 
זוֹ מִדְבָּרִית.  
כָּעֲדָשָׁה שֶׁאָמָרוּ, 
לֹא גְדוֹלָה וְלֹא קְטַנָּה, אֶלָּא בֵינוֹנִית, 
זוֹ מִצְרִית.  
כָּל הַמִּטַּלְטְלִין 
מְבִיאִין אֶת הַטֻּמְאָה בְעֹבִי הַמַּרְדֵּעַ, 
לֹא גָדוֹל וְלֹא קָטֹן, אֶלָּא בֵינוֹנִי.  
וְאֵי זֶה הוּא הַבֵּינוֹנִי? 
כָּל שֶׁהֶקֵּפוֹ טֶפַח. 

<small>ט</small>
הָאַמָּה שֶׁאָמָרוּ, 
בָּאַמָּה הַבֵּינוֹנִית.  
שְׁתֵּי אַמּוֹת הָיוּ בְשׁוּשַׁן הַבִּירָה: 
אַחַת עַל קֶרֶן מִזְרָחִית צְפוֹנִית, 
וְאַחַת עַל קֶרֶן מִזְרָחִית דְּרוֹמִית; 
שֶׁעַל קֶרֶן מִזְרָחִית צְפוֹנִית, &lt;דְּרוֹמִית&gt;
הָיְתָה יְתֵרָה עַל שֶׁלְּמֹשֶׁה חֲצִי אֶצְבַּע; 
וְשֶׁעַל קֶרֶן מִזְרָחִית דְּרוֹמִית, 
הָיְתָה יְתֵרָה עָלֶיהָ חֲצִי אֶצְבַּע. 
נִמְצֵאת יְתֵרָה עַל שֶׁלְּמֹשֶׁה אֶצְבַּע.  

<small>י</small>
וְלָמָּה אָמָרוּ 'אַחַת גְּדוֹלָה וְאַחַת קְטַנָּה'? 
אֶלָּא שֶׁהָאֻמָּנִים נוֹטְלִים בַּקְּטַנָּה, 
וּמַחְזִירִין בַּגְּדוֹלָה, 
כְּדֵי שֶׁלֹּא יָבוֹאוּ לְבֵית הַמְּעִילָה. 

רְבִּי מֵאִיר אוֹמֵר: 
כָּל הָאַמּוֹת הָיוּ בַּבֵּינוֹנִית, 
חוּץ מִמִּזְבַּח הַזָּהָב, 
וְקֶרֶן וְהַסּוֹבֵב וְהַיְסוֹד.  
רְבִּי יְהוּדָה אוֹמֵר: 
אַמַּת הַבִּנְיָן, שִׁשָּׁה טְפָחִים, 
וְשֶׁלַּכֵּלִים, חֲמִשָּׁה. 

<small>יא</small>
וְיֵשׁ שֶׁאָמָרוּ בְּמִדָּה דַקָּה: 
מִדּוֹת הַלַּח וְהַיָּבֵשׁ, 
שֵׁעוּרָן בָּאִיטַלְקִי, 
זוֹ מִדְבָּרִית.  
וְיֵשׁ שֶׁאָמָרוּ: 
הַכֹּל לְפִי מַה שֶּׁהוּא אָדָם: 
קוֹמֵץ אֶת הַמִּנְחָה, &lt;הקומץ&gt;
וְחוֹפְנֵי הַקְּטֹרֶת, 
וְהַשּׁוֹתֶה כִמְלֹא לָגְמָיו בְּיוֹם הַכִּפּוּרִים, 
כִּמְזוֹן שְׁתֵּי סְעוֹדוֹת לָעֵרוּב.  
מְזוֹנוֹ לַחֹל אֲבָל לֹא לַשַּׁבָּת, 
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יְהוּדָה אוֹמֵר: 
לַשַּׁבָּת אֲבָל לֹא לַחֹל.  
אֵלּוּ וָאֵלּוּ מִתְכַּוְּנִין לְהָקֵל.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
מִשְּׁתֵי יָדוֹת לַכִּכָּר, 
מִשָּׁלשׁ לַקַּב.  
רְבִּי יוֹחָנָן בֶּן בְּרוֹקָה אוֹמֵר: 
מִכִּכָּר בְּפוֹנְדְּיוֹן וּמֵאַרְבַּע בַּסֶּלַע. 

<small>יב</small>
וְיֵשׁ שֶׁאָמָרוּ מִדָּה גַסָּה: 
מְלֹא תַרְוָד רָקָב, &lt;תורווד&gt;
מְלֹא תַרְוָד גָּדוֹל שֶׁלָּרוֹפְאִים; 
גְּרִיס נְגָעִים, כְּגָרִיס הַקִּלְקִי; 
הָאוֹכֵל בְּיוֹם הַכִּפּוּרִים כְּכוֹתֶבֶת הַגַּסָּה, 
כָּמוֹהָ וּכְגַלְעַנָּתָהּ. 
נוֹדוֹת יַיִן וְשֶׁמֶן, 
שֵׁעוּרָן בְּפִיקָה גְדוֹלָה שֶׁלָּהֶן. 
וּמָאוֹר שֶׁלֹּא נֶעֱשָׂה בִידֵי אָדָם, 
שֵׁעוּרוֹ מְלֹא אֶגְרוֹף גָּדוֹל, 
זֶה אֶגְרוֹפוֹ שֶׁלְּבֶן בַּטִּיחַ.  
אָמַר רְבִּי יוֹסֵה: 
יֶשְׁנוֹ כְרֹאשׁ גָּדוֹל שֶׁלָּאָדָם. 

<small>יג</small>
וְשֶׁנֶּעֱשָׂה בִידֵי אָדָם, 
שֵׁעוּרוֹ מְלֹא מַקְדֵּחַ גָּדוֹל שֶׁלַּלִּשְׁכָּה, 
שֶׁהוּא כְפוֹנְדְּיוֹן הָאִיטַלְקִי, 
וּכְסֶלַע נְרוֹנִית, 
כִּמְלֹא נֶקֶב שֶׁבָּעֹל. 

<small>יד</small>
כָּל שֶׁבַּיָּם טָהוֹר, 
חוּץ מִכֶּלֶב הַיָּם, 
מִפְּנֵי שֶׁהוּא בוֹרֵחַ לַיִּיבָשׁ.  
דִּבְרֵי רְבִּי עֲקִיבָה.  
הָעוֹשֶׂה כֵלִים מִן הַגָּדֵל בַּיָּם 
וְחִבֵּר לָהֶם מִן הַגָּדֵל בָּאָרֶץ, 
אֲפִלּוּ חוּט, אֲפִלּוּ מְשִׁיחָה, 
דָּבָר שֶׁהוּא מְקַבֵּל טֻמְאָה, 
טָמֵא. 

<small>יה</small>
וְיֵשׁ בְּמַה שֶּׁנִּבְרָא בַיּוֹם הָרִאשׁוֹן טֻמְאָה; 
בַּשֵּׁנִי, אֵין בּוֹ טֻמְאָה, 
בַּשְּׁלִישִׁי, יֶשׁ בּוֹ טֻמְאָה, 
בָּרְבִיעִי וּבַחֲמִישִׁי, אֵין בּוֹ טֻמְאָה, 
חוּץ מִכְּנַף הָעוּז וּבֵיצַת הַנַּעֲמִית הַמְצֻפָּה.  
אָמַר רְבִּי יוֹחָנָן בֶּן נוּרִי: 
מַה שָּׁנַת כְּנַף הָעוּז מִכָּל כְּנָפַיִם? 
כָּל שֶׁנִּבְרָא בַיּוֹם הַשִּׁשִּׁי, טָמֵא. 

<small>יו</small>
הָעוֹשֶׂה כְלִי קִבּוּל מִכָּל מָקוֹם, 
טָמֵא.  
הָעוֹשֶׂה מִשְׁכָּב וּמוֹשָׁב מִכָּל מָקוֹם, 
טָמֵא.  
הָעוֹשֶׂה כִיס מִן הָעוֹר וּמִן הַמַּצָּה וּמִן הַנְּיָר, 
טָמֵא.  
הָרִמּוֹן, הָאַלּוֹן וְהָאֱגוֹז, 
שֶׁחֲקָקוּם הַתִּינוֹקוֹת לָמוֹד בָּהֶן אֶת הֶעָפָר, 
אוֹ שֶׁהִתְקִינוּם לְכַף מֹאזְנַיִם, 
טָמֵא, שֶׁיֵּשׁ לָהֶם מַעֲשֶׂה, 
וְאֵין לָהֶם מַחֲשָׁבָה. 

<small>יז</small>
קְנֵה מֹאזְנַיִם וְהַמָּחוֹק 
שֶׁיֵּשׁ בָּהֶן בֵּית קִבּוּל מַתֶּכֶת, 
וְהָאֲסָל שֶׁיֶּשׁ בָּהּ בֵּית קִבּוּל מָעוֹת, 
וְקָנֶה שֶׁלֶּעָנִי שֶׁיֶּשׁ לוֹ בֵית קִבּוּל מַיִם, 
מַקֵּל שֶׁיֶּשׁ בָּהּ בֵּית קִבּוּל מְזוּזָה וּמַרְגָּלִיּוֹת, 
הֲרֵי אֵלּוּ טְמֵאִין.  
וְעַל כֻּלָּם אָמַר רַבָּן יוֹחָנָן בֶּן זַכַּי: 
אִי לִי אִם אֹמַר, 
אִי לִי אִם לֹא אֹמַר. 

<small>יח</small>
תַּחְתִּית הַצּוֹרְפִין, טְמֵאָה, 
וְשֶׁלַּנַּפָּחִין, טְהוֹרָה.  
מַשְׁחֶזֶת שֶׁיֶּשׁ בָּהּ בֵּית קִבּוּל שֶׁמֶן, 
טְמֵאָה, 
וְשֶׁאֵין בָּהּ, טְהוֹרָה.  
פִּנְקֵס 
שֶׁיֶּשׁ בָּהּ בֵּית קִבּוּל שַׁעֲוָה, טְמֵאָה, 
וְשֶׁאֵין בָּהּ, טְהוֹרָה.  
מַחֲצֶלֶת הַקַּשׁ וּשְׁפוֹפֶרֶת הַקַּשׁ, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
רְבִּי יוֹחָנָן בֶּן נוּרִי מְטַהֵר.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אַף שֶׁלַּפַּקּוֹעוֹת כַּיּוֹצֵא בָהֶן.  
מַחְצֶלֶת הַקָָּנִים וְשֶׁלַּחֵלָף, טְהוֹרָה.  
שְׁפוֹפֶרֶת שֶׁלַּקָּנֶה שֶׁחֲתָכָהּ לְקַבָּלָה, 
טְהוֹרָה, עַד שֶׁיּוֹצִיא אֶת כָּל הַכָּכַי. 





Chapter 18

<small>א</small>
הַשִּׁדָּה, 
בֵּית שַׁמַּי אוֹמְרִים: 
נִמְדֶּדֶת מִבִּפְנִים, 
וּבֵית הֶלֵּל אוֹמְרִים: 
נִמְדֶּדֶת מִבַּחוּץ. 
וּמוֹדִים אֵלּוּ וָאֵלּוּ, 
שֶׁאֵין עֹבִי הָרַגְלַיִם וְעֹבִי לִזְבֵּזִים נִמְדָּד.  
רְבִּי יוֹסֵה אוֹמֵר: 
מוֹדִים שֶׁעֹבִי הָרַגְלַיִם וְעֹבִי לִזְבֵּזִים נִמְדָּד, 
[וּבֵינֵיהֶן אֵינוּ נִמְדָּד.]
רְבִּי שִׁמְעוֹן הַשֵּׁזוּרִי אוֹמֵר: 
אִם הָיוּ הָרַגְלַיִם גְּבוֹהוֹת טֶפַח, 
אֵין בֵּינֵיהֶן נִמְדָּד, 
וְאִם לָאו, בֵּינֵיהֶם נִמְדָּד. 

<small>ב</small>
הַמֶּכְנֵי שֶׁלָּהּ, 
בִּזְמַן שֶׁהִיא נִשְׁמֶטֶת, 
אֵינָהּ חִבּוּר לָהּ, 
וְאֵינָה נִמְדֶּדֶת עִמָּהּ, 
וְאֵינָה מַצֶּלֶת עִמָּהּ בְּאֹהֶל הַמֵּת, 
וְאֵין גּוֹרְרִין אוֹתָהּ בַּשַּׁבָּת 
בִּזְמַן שֶׁיֵּשׁ בְּתוֹכָהּ מָעוֹת. 
וְאִם אֵינָה נִשְׁמֶטֶת, 
חִבּוּר לָהּ, 
וְנִמְדֶּדֶת עִמָּהּ, 
וּמַצֶּלֶת עִמָּהּ בְּאֹהֶל הַמֵּת, 
וְגוֹרְרִים אוֹתָהּ בַּשַּׁבָּת, 
אַף עַל פִּי שֶׁיֵּשׁ בְּתוֹכָהּ מָעוֹת.  
הַקַּמְּרוֹן שֶׁלָּהּ, 
בִּזְמַן שֶׁהוּא קָבוּעַ, 
חִבּוּר לָהּ וְנִמְדָּד עִמָּהּ. 
אִם אֵינוּ קָבוּעַ, 
אֵינוּ חִבּוּר לָהּ, וְאֵינוּ נִמְדָּד עִמָּהּ.  
כֵּיצַד מוֹדְדִין אוֹתוֹ? 
רֹאשׁ תּוֹר.  
רְבִּי יְהוּדָה אוֹמֵר: 
אִם אֵינָה יְכוּלָה לַעֲמֹד בִּפְנֵי עַצְמָהּ, 
טְהוֹרָה. 

<small>ג</small>
הַשִּׁדָּה וְהַתֵּבָה וְהַמִּגְדָּל 
שֶׁנִּטְּלָה אַחַת מֵרַגְלֵיהֶן, 
אַף עַל פִּי מְקַבְּלִין, טְהוֹרִין, 
שֶׁאֵינָן מְקַבְּלִין כְּדַרְכָּן; 
רְבִּי יוֹסֵה מְטַמֵּא.  
נִקְלְטֵי הַמִּטָּה, וַחֲמוֹר, וַחֲפוּי, 
טְהוֹרִין.  
אֵין טָמֵא אֶלָּא מִטָּה וּמַלְבֵּן.  
וּמַלְבְּנֵי בְנֵי לֵוִי, טְהוֹרִין. 

<small>ד</small>
מַלְבֵּן שֶׁנְּתָנוֹ עַל לְשׁוֹנוֹת, 
רְבִּי מֵאִיר וּרְבִּי יְהוּדָה מְטַמִּין, 
רְבִּי יוֹסֵה וּרְבִּי שִׁמְעוֹן מְטַהֲרִין.  
אָמַר רְבִּי יוֹסֵה: 
מַה שָּׁנָה זֶה מִמַּלְבְּנֵי בְנֵי לֵוִי, 
שֶׁמַּלְבְּנֵי בְנֵי לֵוִי טְהוֹרִין? 

<small>ה</small>
מִטָּה שֶׁהָיְתָה טְמֵאָה מִדְרָס, 
נִטְּלָה קְצָרָה וּשְׁתֵּי כְרָעַיִם, 
טְמֵאָה, 
אֲרֻכָּה וּשְׁתֵּי כְרָעַיִם, 
טְהוֹרָה.  
רְבִּי נְחֶמְיָה מְטַמֵּא.  
גָּרַר שְׁתֵּי לְשׁוֹנוֹת לוֹכְסָן, 
גָּרַר שְׁתֵּי כְרָעַיִם טֶפַח עַל טֶפַח לוֹכְסָן, 
אוֹ שֶׁמִּעֲטָהּ פָּחוּת מִטֶּפַח, 
טְהוֹרָה. 

<small>ו</small>
מִטָּה שֶׁהָיְתָה טְמֵאָה מִדְרָס, 
נִשְׁבְּרָה אֲרֻכָּה וְתִקְּנָהּ, 
טְמֵאָה מִדְרָס; 
נִשְׁבְּרָה שְׁנִיָּה וְתִקְּנָהּ, 
טְהוֹרָה מִן הַמִּדְרָס, 
אֲבָל טְמֵאָה מַגַּע מִדְרָס.  
לֹא הִסְפִּיק לְתַקֵּן אֶת הָרִאשׁוֹנָה 
עַד שֶׁנִּשְׁבְּרָה שְׁנִיָּה, 
טְהוֹרָה. 

<small>ז</small>
כֶּרַע שֶׁהָיְתָה טְמֵאָה מִדְרָס, 
וְחִבְּרָהּ לַמִּטָּה, 
כֻּלָּהּ טְמֵאָה מִדְרָס.  
פָּרָשָׁה, הִיא טְמֵאָה מִדְרָס, &lt;שֶׁהִיא&gt;
[וְהַמִּטָּה טְמֵאָה מַגַּע מִדְרָס.]  
הָיְתָה טְמֵאָה טֻמְאַת שִׁבְעָה, 
וְחִבְּרָהּ לַמִּטָּה, 
כֻּלָּהּ טְמֵאָה טֻמְאַת שִׁבְעָה.  
פָּרָשָׁה, הִיא טְמֵאָה טֻמְאַת שִׁבְעָה, 
וְהַמִּטָּה טְמֵאָה טֻמְאַת עֶרֶב.  
[הָיְתָה טְמֵאָה טֻמְאַת עֶרֶב,] 
וְחִבְּרָהּ לַמִּטָּה, 
כֻּלָּהּ טְמֵאָה טֻמְאַת עֶרֶב.  
פָּרָשָׁה, הִיא טְמֵאָה טֻמְאַת עֶרֶב, 
וְהַמִּטָּה טְהוֹרָה.  
וְכֵן הַשֵּׁן שֶׁלַּמַּעְדֵּר. 

<small>ח</small>
תְּפִלָּה אַרְבָּעָה כֵלִים, 
הִתִּיר קְצִיצָה רִאשׁוֹנָה וְתִקְּנָהּ, 
טְמֵאָה טְמֵא מֵת, 
וְכֵן שְׁנִיָּה, 
וְכֵן שְׁלִישִׁית.  
הִתִּיר אֶת הָרְבִיעִית, 
טְהוֹרָה מִטְּמֵא מֵת, 
אֲבָל טְמֵאָה מַגַּע טְמֵא מֵת.  
חָזַר לָרִאשׁוֹנָה, 
וְהִתִּירָהּ וְתִקְּנָהּ, 
טְמֵאָה בְמַגָּע, 
וְכֵן הַשְּׁנִיָּה.  
הִתִּיר אֶת הַשְּׁלִישִׁית, 
טְהוֹרָה, שֶׁהָרְבִיעִית בְּמַגָּע, 
וְאֵין מַגָּע עוֹשֶׂה מַגָּע. 

<small>ט</small>
מִטָּה שֶׁנִּגְנַב חֶצְיָהּ, 
אוֹ אָבַד חֶצְיָהּ, 
וַחֲלָקוּהָ אַחִים אוֹ שׁוּתָפִים, 
טְהוֹרָה.  
הֶחֱזִירוּהָ, 
מְקַבֶּלֶת טֻמְאָה מִכָּן וְלַבָּא.  
מִטָּה מִטַּמַּת חֲבִלָּה, 
וּמִטַּהֶרֶת חֲבִלָּה.  
דִּבְרֵי רְבִּי אֱלִיעֶזֶר.  
וַחֲכָמִים אוֹמְרִים: 
מִטַּמַּת אֵבָרִים וּמִטַּהֶרֶת אֵבָרִים. 





Chapter 19

<small>א</small>
הַמְפָרֵק אֶת הַמִּטָּה לְהַטְבִּילָהּ, 
וְהַנּוֹגֵעַ בַּחֲבָלִים, 
טָהוֹר.  
הַחֶבֶל, מֵאֶמָּתַי הוּא חִבּוּר לַמִּטָּה? 
מִשֶּׁיְּסָרֵג בָּהּ שְׁלֹשָׁה בָתִּים.  
הַנּוֹגֵעַ מִן הַקֶּשֶׁר וְלִפְנִים, 
טָמֵא; 
מִן הַקֶּשֶׁר וְלַחוּץ, 
טָהוֹר.  
נוּמֵי הַקֶּשֶׁר, 
הַנּוֹגֵעַ בְּצָרְכּוֹ, טָמֵא.  
וְכַמָּה הוּא צָרְכוֹ? 
רְבִּי יְהוּדָה אוֹמֵר: 
שָׁלוֹשׁ אֶצְבָּעוֹת. 

<small>ב</small>
הַחֶבֶל הַיּוֹצֵא מִן הַמִּטָּה, 
עַד חֲמִשָּׁה טְפָחִים, טָהוֹר; 
מֵחֲמִשָּׁה וְעַד עֲשָׂרָה, טָמֵא.  
מֵעֲשָׂרָה וְלַחוּץ, טָהוֹר, 
שֶׁבּוֹ קוֹשְׁרִין אֶת הַפְּסָחִין, 
וּמְשַׁלְשְׁלִין אֶת הַמִּטּוֹת. 

<small>ג</small>
מֵזְרָן הַיּוֹצֵא מִן הַמִּטָּה, 
כָּל שֶׁהוּא.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יוֹסֵה אוֹמֵר: 
עֲשָׂרָה טְפָחִים.  
שִׁירֵי מֵזְרָן, 
שִׁבְעָה טְפָחִים, 
כְּדֵי לַעֲשׁוֹתוֹ חֲבָק לַחֲמוֹר. 

<small>ד</small>
נִשָּׂא הַזָּב עַל הַמִּטָּה וְעַל הַמֵּזְרָן, 
מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יוֹסֵה אוֹמֵר: 
נִשָּׂא הַזָּב עַל הַמִּטָּה וְהַמֵּזְרָן, 
עֲשָׂרָה טְפָחִים, 
מְטַמֵּא שְׁנַיִם, וּפוֹסֵל אֶחָד. 
מֵעֲשָׂרָה וְלַחוּץ, 
מְטַמֵּא אֶחָד, וּפוֹסֵל אֶחָד.  
נִשָּׂא עַל הַמֵּזְרָן, 
מֵעֲשָׂרָה וְלִפְנִים, טָמֵא; 
מֵעֲשָׂרָה וְלַחוּץ, טָהוֹר. 

<small>ה</small>
מִטָּה שֶׁהָיְתָה טְמֵאָה מִדְרָס, 
וְכָרַךְ לָהּ מֵזְרָן, 
כֻּלָּהּ טְמֵאָה מִדְרָס.  
פָּרָשָׁה, 
הִיא טְמֵאָה מִדְרָס, 
וְהַמֵּזְרָן מַגַּע מִדְרָס.  
הָיְתָה טְמֵאָה טֻמְאַת שִׁבְעָה, 
וְכָרַךְ לָהּ מֵזְרָן, 
כֻּלָּהּ טְמֵאָה טֻמְאַת שִׁבְעָה. &lt;מִדְרָס&gt; 
פָּרָשָׁה, 
הִיא טְמֵאָה טֻמְאַת שִׁבְעָה, 
וְהַמֵּזְרָן טָמֵא טֻמְאַת עֶרֶב.  
הָיְתָה טְמֵאָה טֻמְאַת עֶרֶב, 
וְכָרַךְ לָהּ מֵזְרָן, 
כֻּלָּהּ טְמֵאָה טֻמְאַת עֶרֶב.  
פָּרָשָׁה, 
הִיא טְמֵאָה טֻמְאַת עֶרֶב, 
וְהַמֵּזְרָן טָהוֹר. 

<small>ו</small>
מִטָּה שֶׁכָּרַךְ לָהּ מֵזְרָן, 
וְנָגַע בָּהֶן הַמֵּת, 
טְמֵאִים טֻמְאַת שִׁבְעָה.  
פָּרָשׁוּ, 
טְמֵאִים טֻמְאַת שִׁבְעָה.  
נָגַע בָּהֶן הַשֶּׁרֶץ, 
טְמֵאִין טֻמְאַת עֶרֶב. &lt;שִׁבְעָה ערב&gt; 
פָּרָשׁוּ, 
טְמֵאִים טֻמְאַת עֶרֶב.  

<small>ז</small>
מִטָּה שֶׁנִּטְּלוּ שְׁתֵּי אֲרֻכּוֹת שֶׁלָּהּ, 
וְעָשָׂה לָהּ חֲדָשׁוֹת וְלֹא שִׁנָּה אֶת הַנְּקָבִים; 
נִשְׁתַּבְּרוּ חֲדָשׁוֹת, טְמֵאָה, 
וִישָׁנוֹת, טְהוֹרָה, 
שֶׁהַכֹּל הוֹלֵךְ אַחַר הַיְשָׁנוֹת. 

<small>ח</small>
תֵּבָה שֶׁפִּתְחָהּ מִלְּמַעְלָן, 
טְמֵאָה טְמֵא מֵת.  
נִפְחָתָה מִלְּמַעְלָן, 
טְמֵאָה טְמֵא מֵת.  
נִפְחָתָה מִלְּמַטָּן, 
טְהוֹרָה.  
מְגוֹרוֹת שֶׁבָּהּ, 
טְמֵאוֹת, וְאֵינָן חִבּוּר לָהּ. 

<small>ט</small>
תָּרְמֵל שֶׁנִּפְחַת, 
הַכִּיס שֶׁבְּתוֹכוֹ טָמֵא, 
וְאֵינוּ חִבּוּר לוֹ.  
הַחֵמֶת שֶׁהַבֵּיצִים שֶׁלָּהּ מְקַבְּלוֹת עִמָּהּ, 
וְנִפְחָתוּ, 
טְהוֹרוֹת, שֶׁאֵינָן מְקַבְּלוֹת כְּדַרְכָּן. 

<small>י</small>
תֵּבָה שֶׁנִּפְתְחָה מִצִּדָּהּ, 
טְמֵאָה מִדְרָס וּטְמֵא מֵת.  
אָמַר רְבִּי יוֹסֵה: 
אֵמָתַי? 
בִּזְמַן שֶׁאֵינָה גְבוֹהָה עֲשָׂרָה טְפָחִים, 
אוֹ שֶׁאֵין לָהּ לִיזְבֵן טֶפַח.  
נִתְפַּחֲתָה מִלְּמַעְלָן, 
טְמֵאָה טְמֵא מֵת.  
נִפְחָתָה מִלְּמַטָּן, 
רְבִּי מֵאִיר מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין, 
מִפְּנֵי שֶׁבָּטַל הָעִקָּר, 
וּבָטְלָה הַטְּפֵלָה. 

<small>יא</small>
מַשְׁפֶּלֶת שֶׁנִּפְחָתָה מִלְּקַבֵּל רִמּוֹנִים, 
רְבִּי מֵאִיר מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִים, 
מִפְּנֵי שֶׁבָּטַל הָעִקָּר, 
וּבָטְלָה הַטְּפֵלָה. 





Chapter 20

<small>א</small>
הַכָּרִים וְהַכְּסָתוֹת, 
וְהַשַּׂקִים וְהַמַּרְצַפִּין שֶׁנִּפְחָתוּ, 
הֲרֵי אֵלּוּ טְמֵאִין מִדְרָס.  
הַקְּלוֹסְטֵר אַרְבַּעַת קַבִּים, 
הַתָּרְמֵל חֲמֵשֶׁת קַבִּים, 
הַכְּרֵתִית סְאָה, 
הַחֵמֶת שֶׁלְּשִׁבְעַת קַבִּים. 
רְבִּי יְהוּדָה אוֹמֵר: 
אַף הָרַבְצֵל וְהַמְּזָרָה כָל שֶׁהֵן, 
הֲרֵי אֵלּוּ טְמֵאִין מִדְרָס.  
וְכֻלָּן שֶׁנִּפְחָתוּ, טְהוֹרִין, 
מִפְּנֵי שֶׁבָּטַל הָעִקָּר, 
וּבָטְלָה הַטְּפֵלָה. 

<small>ב</small>
חֲמַת חֲלִילִים, 
טְהוֹרָה מִן הַמִּדְרָס.  
עֲרֵבַת פִּסּוֹנוֹת, 
בֵּית שַׁמַּי אוֹמְרִים: 
מִדְרָס. 
וּבֵית הֶלֵּל אוֹמְרִים: 
טְמֵא מֵת.  
עֲרֵבָה, 
מִשְּׁנֵי לֹג עַד תִּשְׁעַת קַבִּים שֶׁנִּסְדְּקָה, 
טְמֵאָה מִדְרָס.  
הִנִּיחָהּ בַּגְּשָׁמִים וְנִתְפְּחָה, 
טְמֵאָה טְמֵא מֵת.  
בַּקָּדִים וְנִסְדְּקָה, 
מְקַבֶּלֶת מִדְרָס.  
זֶה חֹמֶר בְּשִׁירֵי כְלֵי עֵץ מִבִּתְחִלָּתָן.  
וְחֹמֶר בִּכְלֵי נְצָרִין מִבִּתְחִלָּתָן, 
תְּחִלָּתָן אֵינָן מְקַבְּלִין טֻמְאָה עַד שֶׁיִּתְחַסְּמוּ.  
נִתְחַסְּמוּ, 
אַף עַל פִּי שֶׁנָּשְׁרוּ שִׂפְתוֹתֵיהֶן כָּל שֶׁהֵן, 
טְמֵאִין. 

<small>ג</small>
מַקֵּל שֶׁעֲשָׂאוֹ יָד לְקָרְדּוֹם, 
חִבּוּר לַטֻּמְאָה בְשָׁעַת מְלָאכָה.  
הַדְּיַסְטֵר, 
חִבּוּר לַטֻּמְאָה בְשָׁעַת מְלָאכָה.  
קְבָעוֹ בִכְלוֹנָס, 
טָמֵא, וְאֵינוּ חִבּוּר לוֹ.  
עָשָׂה בוֹ דְּיַסְטֵר, 
אֵין טָמֵא אֶלָּא צָרְכוֹ.  
כִּסֵּא שֶׁקְּבָעוֹ בִכְלוֹנָס, 
טָמֵא, וְאֵינוּ חִבּוּר לוֹ.  
עָשָׂה בוֹ כִסֵּא, 
אֵין טָמֵא אֶלָּא מְקוֹמוֹ.  
קְבָעוֹ בְקוֹרַת בֵּית הַבַּד, 
טָמֵא, וְאֵינוּ חִבּוּר לָהּ.  
עָשָׂה בְרֹאשָׁהּ כִּסֵּא, 
טְהוֹרָה, מִפְּנֵי שֶׁאוֹמְרִין לוֹ: 
"עֲמֹד, וְנַעֲשֶׂה אֶת מְלָאכְתֵּנוּ!" 

<small>ד</small>
עֲרֵבָה גְדוֹלָה שֶׁנִּפְחָתָה מִלְּקַבֵּל רִמּוֹנִים, 
הִתְקִינָהּ לִישִׁיבָה, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין, 
עַד שֶׁיְּקַצַּע.  
עֲשָׂאָהּ אֵבוּס לַבְּהֵמָה, 
אַף עַל פִּי שֶׁקְּבָעָהּ בַּכּתֶל, 
טְמֵאָה. 

<small>ה</small>
כֹּפֶת שֶׁקְּבָעוֹ בַנִּדְבָּךְ, 
קְבָעוֹ וְלֹא בָנָה עָלָיו, 
בָּנָה עָלָיו וְלֹא קְבָעוֹ, 
טָמֵא.  
קְבָעוֹ וּבָנָה עָלָיו, 
טָהוֹר.  
מַפֵּץ שֶׁנְּתָנוֹ עַל גַּבֵּי הַקּוֹרוֹת, 
קְבָעוֹ וְלֹא נָתַן עָלָיו אֶת הַמַּעֲזֵיבָה, 
נָתַן עָלָיו אֶת הַמַּעֲזֵיבָה וְלֹא קְבָעוֹ, 
טָמֵא.  
קְבָעוֹ וְנָתַן עָלָיו אֶת הַמַּעֲזֵיבָה, 
טָהוֹר.  
קְעָרָה שֶׁקְּבָעָהּ בַּשִּׁדָּה, בַּתֵּבָה וּבַמִּגְדָּל, 
כְּדֶרֶךְ קַבָּלָתָהּ, טְמֵאָה; 
שֶׁלֹּא כְדֶרֶךְ קַבָּלָתָהּ, טְהוֹרָה. 

<small>ו</small>
סָדִין שֶׁהוּא טָמֵא מִדְרָס, 
וַעֲשָׂאוֹ וֵילָן, 
טָהַר מִן הַמִּדְרָס, 
אֲבָל טָמֵא טְמֵא מֵת.  
מֵאֵמָתַי הִיא טַהֲרָתוֹ? 
בֵּית שַׁמַּי אוֹמְרִים: 
מִשֶּׁיִּתָּבֵר. &lt;מִשֶּׁיּתבֵּר&gt; 
וּבֵית הֶלֵּל אוֹמְרִים: 
מִשֶּׁיִּקָּשֵׁר.  
רְבִּי עֲקִיבָה אוֹמֵר: 
מִשֶּׁיִּקְבַּע. 

<small>ז</small>
מַחְצֶלֶת שֶׁעָשָׂה לָהּ קָנִים לְאָרְכָּהּ, 
טְהוֹרָה.  
וַחֲכָמִים אוֹמְרִים: 
עַד שֶׁיַּעֲשֶׂם כְּמִין כִי.  
עֲשָׂאָן לְרָחְבָּהּ, 
אִם אֵין בֵּין קָנֶה לַחֲבֵרוֹ אַרְבָּעָה טְפָחִים, 
טְהוֹרָה.  
נֶחְלְקָה לְרָחְבָּהּ, 
רְבִּי יְהוּדָה מְטַהֵר.  
וְכֵן הַמַּתִּיר רָאשֵׁי הַמַּעֲדַנִּים, 
טְהוֹרָה.  
נֶחְלְקָה לְאָרְכָּהּ, 
נִשְׁתַּיְּרוּ בָהּ שְׁלֹשָׁה מַעֲדַנִּים שֶׁהֵן שִׁשָּׁה טְפָחִים, 
טְמֵאָה.  
מַחְצֶלֶת, מֵאֵמָתַי מְקַבֶּלֶת טֻמְאָה? 
מִשֶּׁתִּתְקַנֵּב, וְהִיא גְּמָר מְלַאכְתָּהּ. 





Chapter 21

<small>א</small>
הַנּוֹגֵעַ בַּכָּבֵד הָעֶלְיוֹן, 
בַּכָּבֵד הַתַּחְתּוֹן, 
בַּנִּירַיִם, 
וּבַקִּירוֹס, 
וּבַחוּט שֶׁהֶעֱבִירוֹ עַל גַּבֵּי אַרְגָּמָן, 
וּבָעִירָה שֶׁאֵינוּ עָתִיד לְהַחְזִירָהּ, 
טָהוֹר.  
בְּנֶפֶשׁ הַמַּסֶּכֶת, 
וּבִשְׁתִי הָעוֹמֵד, 
וּבַכָּפוּל שֶׁהֶעֱבִירוֹ עַל גַּבֵּי אַרְגָּמָן, 
וּבָעִירָה שֶׁהוּא עָתִיד לְהַחְזִירָהּ, 
טָמֵא.  
הַנּוֹגֵעַ בַּצֶּמֶר שֶׁעַל הָעִמָּה, בְּאַשְׁוְיָה, 
טָהוֹר.  
הַנּוֹגֵעַ בַּפִּקָּה, 
עַד שֶׁלֹּא פֵרֵעָה, טָמֵא; 
מִשֶּׁפֵּרֵעָה, טָהוֹר. 

<small>ב</small>
הַנּוֹגֵעַ בָּעֹל, 
וּבַקַּטְרֵב, 
בָּעַיִן, 
וּבָעֲבוֹת, 
אֲפִלּוּ בְשָׁעַת מְלָאכָה, 
טָהוֹר.  
בַּחֶרֶב, 
וּבַבֹּרֶךְ, 
וּבַיָּצוֹל, 
טָמֵא.  
בָּעַיִן שֶׁלַּמַּתֶּכֶת, 
וּבִלְחָיַיִם, 
וּבָעֲרָיִים, &lt;אולי עדיים&gt;
טָמֵא.  
רְבִּי יְהוּדָה מְטַהֵר בִּלְחָיַיִם, 
שֶׁאֵינָן עֲשׂוּיִים אֶלָּא לְרַבּוֹת אֶת הֶעָפָר. 

<small>ג</small>
הַנּוֹגֵעַ בְּיַד מְגֵרָה, 
מִכָּן וּמִכָּן, 
טָמֵא.  
בַּחוּט וּבַמְּשִׁיחָה, 
בָּאַמָּה וּבַסְּנִיפִין, 
בְּמַכְבֵּשׁ שֶׁלֶּחָרָשׁ וּבַקַּשְׁטָנִית, 
טָהוֹר.  
רְבִּי יְהוּדָה אוֹמֵר: 
אַף הַנּוֹגֵעַ בַּמַּלְבֵּן שֶׁלְּמַסָּר הַגָּדוֹל, 
טָהוֹר.  
הַנּוֹגֵעַ בַּיֶּתֶר וּבַקֶּשֶׁת, 
אַף עַל פִּי מְתוּחָה, 
טָהוֹר.  
מְצוֹדוֹת הָאִשּׁוּת, 
טְהוֹרוֹת.  
רְבִּי יְהוּדָה אוֹמֵר: 
כָּל זְמַן שֶׁהִיא מְתוּחָה, חִבּוּר. 





Chapter 22

<small>א</small>
הַשֻּׁלְחָן וְהַדַּלְפְקִי שֶׁנִּפְחָתוּ, 
אוֹ שֶׁחִפָּן בַּשַּׁיִשׁ 
וְשִׁיֵּר בָּהֶן מְקוֹם הַנָּחַת הַכּוֹסוֹת, 
טְמֵאִין.  
רְבִּי יְהוּדָה אוֹמֵר: 
מְקוֹם הַנָּחַת חֲתִכּוֹת. 

<small>ב</small>
שֻׁלְחָן שֶׁנִּטְּלָה אַחַת מֵרַגְלָיו, טָהוֹר; 
נִטְּלָה שְׁנִיָּה, טָהוֹר; 
נִטְּלָה הַשְּׁלִישִׁית, 
טָמֵא עַד שֶׁיַּחְשֹׁב עָלָיו.  
רְבִּי יוֹסֵה אוֹמֵר: 
אֵין צָרִיךְ מַחֲשָׁבָה.  
וְכֵן הַדָּלְפְקִי. 

<small>ג</small>
סַפְסֵל שֶׁנִּטַּל אֶחָד מֵרָאשָׁיו, טָהוֹר; 
נִטַּל הַשֵּׁנִי, טָהוֹר; 
אִם יֶשׁ בּוֹ גֹבַהּ טֶפַח, טָמֵא.  
שְׁרַפְרֵף שֶׁנִּטַּל אֶחָד מֵרָאשָׁיו, 
טָמֵא.  
וְכֵן כִּסֵּא שֶׁלִּפְנֵי קָתֶדְרָה. 

<small>ד</small>
כִּסֵּא שֶׁלַּכַּלָּה שֶׁנִּטְּלוּ חֲפוּיָיו, 
בֵּית שַׁמַּי מְטַמִּין, 
וּבֵית הֶלֵּל מְטַהֲרִין.  
שַׁמַּי אוֹמֵר: 
אַף מַלְבֵּן שֶׁלַּכִּסֵּא, טָמֵא.  
כִּסֵּא שֶׁקְּבָעוֹ בָעֲרֵבָה, 
בֵּית שַׁמַּי מְטַמִּין, 
וּבֵית הֶלֵּל מְטַהֲרִין.  
שַׁמַּי אוֹמֵר: 
אַף הֶעָשׂוּי בָּהּ. 

<small>ה</small>
כִּסֵּא שֶׁלֹּא הָיוּ חֲפוּיָיו יוֹצְאִין, וְנִטָּלוּ, 
טָמֵא, שֶׁכֵּן דַּרְכּוֹ 
לִהְיוֹת מַטֵּהוּ עַל צִדּוֹ וְיוֹשֵׁב עָלָיו. 

<small>ו</small>
כִּסֵּא שֶׁנִּטַּל חֲפוּיוֹ הָאֶמְצָעִי, 
וְהַחִיצוֹנִין קַיָּמִין, 
טָמֵא.  
נִטְּלוּ הַחִיצוֹנִין וְהָאֶמְצָעִי קַיָּם, 
טָמֵא.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אִם הָיָה רָחָב טֶפַח. 

<small>ז</small>
כִּסֵּא שֶׁנִּטְּלוּ שְׁנַיִם מֵחֲפוּיָיו זֶה בְצַד זֶה, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין.  
אָמַר רְבִּי יְהוּדָה: 
אַף כִּסֵּא שֶׁלַּכַּלָּה שֶׁנִּטְּלוּ חֲפוּיָיו, 
נִשְׁתַּיַּר בּוֹ בֵית קַבָּלָה, 
טָהוֹר, 
מִפְּנֵי שֶׁבָּטַל הָעִקָּר, 
וּבָטְלָה הַטְּפֵלָה. 

<small>ח</small>
שִׁדָּה שֶׁנִּטַּל הָעֶלְיוֹן, 
טְמֵאָה, מִפְּנֵי הַתַּחְתּוֹן. 
נִטַּל הַתַּחְתּוֹן, 
טְמֵאָה, מִפְּנֵי הָעֶלְיוֹן.  
נִטַּל הָעֶלְיוֹן וְהַתַּחְתּוֹן, 
רְבִּי יְהוּדָה מְטַמֵּא, מִפְּנֵי הַדַּפִּים, 
וַחֲכָמִים מְטַהֲרִין.  
יְשִׁיבַת הַסַּתָּת, טְמֵאָה מִדְרָס. 

כֹּפֶת שֶׁסְּרָקוֹ וְכִרְכְּמוֹ, וַעֲשָׂאוֹ פָנִים, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין, 
עַד שֶׁיָּחֹק בּוֹ.  
הַסַּל וְהַכַּלְכַּלָּה שֶׁמִּלָן תֶּבֶן אוֹ מֻכִּין, 
הִתְקִינָן לִישִׁיבָה, טְהוֹרִין.  
סֵרְגָן בַּגְּמִי אוֹ בִמְשִׁיחָה, טְמֵאִין. 

<small>ט</small>
הַסַּלָּה טְמֵאָה מִדְרָס וּטְמֵא מֵת.  
פָּרָשָׁה, הִיא טְמֵאָה מִדְרָס, 
וְהַבַּרְזֶל טָמֵא טְמֵא מֵת.  
טְרַסְקֵל שֶׁחֲפוּיָיו שֶׁלָּעוֹר, 
טָמֵא מִדְרָס וְטָמֵא טְמֵא מֵת.  
פָּרָשָׁה, 
הָעוֹר טָמֵא מִדְרָס, 
וְהַטְּרַסְקֵל טָהוֹר מִכְּלוּם. &lt;מִכֻּלָם&gt; 
סַפְסֵלִים שֶׁבַּמַּרְחֵץ שֶׁשְּׁתֵּי רַגְלָיו שֶׁלָּעֵץ, 
טָמֵא.  
אַחַת שֶׁלָּעֵץ וְאַחַת שֶׁלָּאֶבֶן, 
טָהוֹר.  
נְסָרִין שֶׁבַּמַּרְחֵץ שֶׁשִּׁגְּמָן, 
רְבִּי מֵאִיר מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין, 
שֶׁאֵינָם עֲשׁוּיִן אֶלָּא שֶׁיְּהוּ הַמַּיִם מְהַלְּכִין תַּחְתֵּיהֶן.  
קַנְקֵילָן 
שֶׁיֶּשׁ בָּהּ בֵּית קַבָּלַת כְּסוּת, 
טְמֵאָה; 
וְהָעֲשׁוּיָה כְּכַוֶּרֶת, 
טְהוֹרָה. 





Chapter 23

<small>א</small>
הַכַּדּוּר וְהָאֱמוֹן וְהַקָּמִיעַ וּתְפִלִּין שֶׁנִּקְרָעוּ, 
הַנּוֹגֵעַ בָּהֶן טָמֵא, 
וּמַה שֶּׁבְּתוֹכָן טָהוֹר.  
הָאֱכוֹף שֶׁנִּקְרַע, 
הַנּוֹגֵעַ בְּמַה שֶּׁבְּתוֹכוֹ טָמֵא, 
מִפְּנֵי שֶׁהַתָּפִיר מְחַבְּרוֹ. 

<small>ב</small>
אֵלּוּ טְמֵאִין מִשֵּׁם מֶרְכָּב: 
זֶרֶז הָאֶשְׁקְלוֹנִי, 
וּמְרֵיכָה הַמָּדִית, 
וְעָבִיט שֶׁלַּגָּמָל, 
וּטְפֵטָן שֶׁלַּסּוּס.  
רְבִּי יוֹסֵה אוֹמֵר: 
אַף טַפֵּיטָן שֶׁלַּסּוּס טָמֵא מוֹשָׁב, 
מִפְּנֵי שֶׁעוֹמְדִין עָלָיו בְּקוֹמְפוֹן.  
אֲבָל אֱכוֹף שֶׁלַּנָּקָה, טָמֵא. 

<small>ג</small>
מַה בֵּין מֶרְכָּב לְמוֹשָׁב? 
מֶרְכָּב, חָלַק מַגָּעוֹ מִמַּשָּׂאוֹ, 
וּמוֹשָׁב, לֹא חָלַק.  
תָּפִית שֶׁלַּחֲמוֹר, 
שֶׁהוּא יוֹשֵׁב עָלֶיהָ, טְהוֹרָה.  
שִׁנָּה בָהּ אֶת הַנְּקָבִים, 
אוֹ שֶׁפְּרָצָן זֶה בְצַד זֶה, טְמֵאָה. 

<small>ד</small>
הַמִּטָּה, וְהַכַּר, וְהַכֶּסֶת שֶׁלַּמֵּת, 
הֲרֵי אֵלּוּ טְמֵאִין מִדְרָס.  
כִּסֵּא שֶׁלַּכַּלָּה, 
וּמַשְׁבֵּר שֶׁלֶּחָיָה, 
וְכִסֵּא שֶׁלַּכּוֹבֵס 
שֶׁהוּא כוֹרֵם עָלָיו אֶת הַכֵּלִים, 
אָמַר רְבִּי יוֹסֵה: 
אֵין בּוֹ מִשֵּׁם מוֹשָׁב. 

<small>ה</small>
הַחֵרֶם טָמֵא מִפְּנֵי זוֹטוֹ.  
הָרְשָׁתוֹת וְהַמַּכְמְרוֹת, 
וְהַמַּדָּף וְהַפְּלָצוּר, 
מְצוֹדוֹת שְׂכָרִין, טְמֵאִין.  
הָאַקּוֹן וְהָרָטוֹב וְהַכְּלוּב, טְהוֹרִין. 





Chapter 24

<small>א</small>
שְׁלֹשָׁה תְרִיסִים הֵם: 
תְּרִיס הַכָּפוּף, טָמֵא מִדְרָס; 
שֶׁמְּשַׂחֲקִין בַּקּוֹמְפּוֹן, טָמֵא טְמֵא מֵת; 
וְדֵיצַת הָעַרְבִיִּים, טְהוֹרָה מִכְּלוּם. 

<small>ב</small>
שָׁלֹשׁ עֲגָלוֹת הֵן: 
הָעֲשׁוּיָה כְּקָתֶדְרָה, טְמֵאָה מִדְרָס; 
כַּמִּטָּה, טְמֵאָה טְמֵא מֵת; 
וְשֶׁלָּאֲבָנִים, טְהוֹרָה מִכְּלוּם. 

<small>ג</small>
שָׁלֹשׁ עֲרֵבוֹת הֵן: 
עֲרֵבָה מִשְּׁנֵי לֹג וְעַד תִּשְׁעַת קַבִּין שֶׁנִּסְדָּקָה, 
טְמֵאָה מִדְרָס; 
שְׁלֵמָה, טְמֵאָה טְמֵא מֵת; 
וְהַבָּאָה בַמִּדָּה, טְהוֹרָה מִכְּלוּם. 

<small>ד</small>
שָׁלֹשׁ תֵּבוֹת הֵן: 
תֵּבָה שֶׁפִּתְחָהּ מִצִּדָּהּ, טְמֵאָה מִדְרָס; 
מִלְּמַעְלָן, טְמֵאָה טְמֵא מֵת; 
וְהַבָּאָה בַמִּדָּה, טְהוֹרָה מִכְּלוּם. 

<small>ה</small>
שְׁלֹשָׁה תַרְכּוּסִין הֵן: 
שֶׁלַּסַּפָּרִין, טָמֵא מִדְרָס; 
שֶׁאוֹכְלִין עָלָיו, טָמֵא טְמֵא מֵת; 
וְשֶׁלַּזֵּיתִין, טָהוֹר מִכְּלוּם. 

<small>ו</small>
שָׁלֹשׁ בָּסִיסָיוֹת הֵן: 
שֶׁלִּפְנֵי הַמִּטָּה, 
וְשֶׁלִּפְנֵי הַסּוֹפְרִים, 
טְמֵאָה מִדְרָס; 
וְשֶׁלַּדָּלְפְקֵי, 
טְמֵאָה טְמֵא מֵת; 
וְשֶׁלַּמִּגְדָּל, 
טְהוֹרָה מִכְּלוּם. 

<small>ז</small>
שָׁלוֹשׁ פִּנַקְסִיוֹת הֵן: 
הָאַפִּיפּוֹדִין, טְמֵאָה מִדְרָס; 
וְשֶׁיֶּשׁ בָּהּ בֵּית קִבּוּל שַׁעֲוָה, 
טְמֵאָה טְמֵא מֵת; 
וַחֲלָקָה, טְהוֹרָה מִכְּלוּם. 

<small>ח</small>
שָׁלֹשׁ מִטּוֹת הֵן: 
הָעֲשׁוּיָה לִשְׁכִיבָה, טְמֵאָה מִדְרָס; 
וְשֶׁלַּזַּגָּגִין, טְמֵאָה טְמֵא מֵת; 
וְשֶׁלַּסָּרָגִין, טְהוֹרָה מִכְּלוּם. 

ט &lt;ח&gt;
שָׁלוֹשׁ מַשְׁפְּלוֹת הֵן: 
שֶׁלַּזֶּבֶל, טְמֵאָה מִדְרָס; 
שֶׁלַּתֶּבֶן, טְמֵאָה טְמֵא מֵת; 
פּוֹחֲלָץ שֶׁלַּגְּמַלִּים, טָהוֹר מִכְּלוּם. 

י &lt;ט&gt;
שְׁלֹשָׁה מַפְּצִין הֵן: 
הֶעָשׂוּי לִישִׁיבָה, טָמֵא מִדְרָס; 
שֶׁלַּצַּבָּעִין, טָמֵא טְמֵא מֵת; 
וְשֶׁלֶַּגִּתּוֹת, טָהוֹר מִכְּלוּם. 

יא &lt;י&gt;
שָׁלוֹשׁ חֲמָתוֹת וּשְׁלֹשָׁה תָרְמָלִים הֵן: 
מְקַבְּלִין כַּשֵּׁעוּר, טְמֵאִין מִדְרָס; 
וְשֶׁאֵין מְקַבְּלִין כַּשֵּׁעוּר, טְמֵאִין טְמֵא מֵת; 
וְשֶׁלְּעוֹר הַדָּג, טָהוֹר מִכְּלוּם. 

<small>יב</small> 
שְׁלֹשָׁה עוֹרוֹת הֵן: 
הֶעָשׂוּי לַשָּׁטִיחַ, טָמֵא מִדְרָס; 
שֶׁלְּתַכְרִיךְ כֵּלִים, טָמֵא טְמֵא מֵת; 
וְשֶׁלָּרְצוּעוֹת וְשֶׁלַּסַּנְדַּלִּים, טָהוֹר מִכְּלוּם. 

<small>יג</small>
שְׁלֹשָׁה סְדִינִים הֵן: 
הֶעָשׂוּי לִשְׁכִיבָה, טָמֵא מִדְרָס; 
לְוֵילָן, טָמֵא טְמֵא מֵת; 
וְשֶׁלַּצּוּרוֹת, טָהוֹר מִכְּלוּם. 

<small>יו</small>
שָׁלוֹשׁ מִטְפָּחוֹת הֵן: 
שֶׁלַּיָּדַיִם, טָמֵא מִדְרָס; 
וְשֶׁלַּסְּפָרִין, טָמֵא טְמֵא מֵת; 
וְשֶׁלְּתַכְרִיךְ וְנִבְלֵי בְנֵי לֵוִי, טָהוֹר מִכְּלוּם. 

<small>יד</small> 
שְׁלֹשָׁה פְרַקְלִינוֹן הֵן: 
שֶׁלְּצָדֵי חַיָּה וָעוֹף, טָמֵא מִדְרָס; &lt;שֶׁלְּצֵדֵי&gt;
שֶׁלַּחֲגָבִים, טָמֵא טְמֵא מֵת; 
וְשֶׁלַּקּוֹצִין, טָהוֹר מִכְּלוּם. 

<small>יה</small>
שָׁלוֹשׁ סְבָכוֹת הֵן: 
שֶׁלַּיַּלְדָּה, טְמֵאָה טֻמְאַת מִדְרָס; 
שֶׁלַּזְּקֵנָה, טְמֵאָה טְמֵא מֵת; 
וְשֶׁלְּיוֹצֵאת הַחוּץ, טְהוֹרָה מִכְּלוּם. 


<small>יז</small>
שָׁלוֹשׁ קֻפּוֹת הֵן: 
מְהוּהָה שֶׁטְּלָיָהּ עַל הַבִּרְיָה, &lt;מהודה! לא מנוקד&gt;
הוֹלְכִים אַחַר הַבִּרְיָה; 
קְטַנָּה עַל הַגְּדוֹלָה, 
הוֹלְכִים אַחַר הַגְּדוֹלָה; 
הָיוּ שָׁווֹת, הוֹלְכִים אַחַר הַפְּנִימִית.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כַּף שֶׁלְּמֹאזְנַיִם שֶׁטְּלָיָהּ בְּשׁוּלֵי הַמֵּחַם, 
מִבִּפְנִים, טָמֵא, 
וּמִבַּחוּץ, טָהוֹר.  
טְלָיָהּ עַל צִדּוֹ, 
בֵּין מִבִּפְנִים, בֵּין מִבַּחוּץ, טָהוֹר. 





Chapter 25

<small>א</small>
כָּל הַכֵּלִים יֵשׁ לָהֶם אֲחוֹרַיִם וָתוֹךְ, 
כְּגוֹן: 
הַכָּרִים, וְהַכְּסָתוֹת, 
וְהַשַּׂקִין, וְהַמַּרְצַפִּים.  
דִּבְרֵי רְבִּי יְהוּדָה.  
רְבִּי מֵאִיר אוֹמֵר: 
כָּל שֶׁיֶּשׁ לוֹ תוֹבָרוֹת, 
[יֵשׁ לוֹ אֲחוֹרַיִם וָתוֹךְ; 
וְכָל שֶׁאֵין לוֹ תוֹבְרוֹת,] 
אֵין לוֹ אֲחוֹרַיִם וָתוֹךְ.  

<small>ב</small>
הַשֻּׁלְחָן וְהַדָּלְפְקֵי, 
יֵשׁ לוֹ אֲחוֹרַיִם וָתוֹךְ.  
דִּבְרֵי רְבִּי יְהוּדָה.  
רְבִּי מֵאִיר אוֹמֵר: 
אֵין לָהֶן.  
וְכֵן טַבְלָה שֶׁאֵין לָהּ לִזְבֵּז. 

<small>ג</small>
הַמַּרְדֵּעַ, 
יֵשׁ לוֹ אֲחוֹרַיִם וָתוֹךְ: 
מִשִּׁבְעָה לְחַרְחוּר, 
וּמֵאַרְבָּעָה לַדָּרְבָן.  
דִּבְרֵי רְבִּי יְהוּדָה.  
רְבִּי מֵאִיר אוֹמֵר: 
אֵין לָהֶן, 
לֹא הָזְכָּרוּ שִׁבְעָה וְאַרְבָּעָה אֶלָּא לַשִּׁירַיִם. 

<small>ד</small>
מִדּוֹת יַיִן וָשֶׁמֶן, 
וְזוֹמָנֶצְטְרָה, 
וּמְסַנֶּנֶת שֶׁלַּחַרְדָּל, 
וּמְשַׁמֶּרֶת שֶׁלַּיַּיִן, 
יֵשׁ לָהֶן אֲחוֹרַיִם וָתוֹךְ.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי יְהוּדָה אוֹמֵר: 
אֵין לָהֶן.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
יֵשׁ לָהֶן, 
נִטַּמּוּ מֵאֲחוֹרֵיהֶן, 
מַה שֶּׁבְּתוֹכָן טָהוֹר, 
וְצָרִיךְ לְהַטְבִּיל. 

<small>ה</small>
הָרֹבַע, וַחֲצִי רֹבַע; 
נִטַּמָּא הָרֹבַע, וְלֹא נִטַּמָּא חֲצִי הָרֹבַע, 
נִטַּמָּא חֲצִי הָרֹבַע, וְלֹא נִטַּמָּא הָרֹבַע,  
אָמְרוּ לִפְנֵי רְבִּי עֲקִיבָה: 
הוֹאִיל וַחֲצִי הָרֹבַע אֲחוֹרַיִם לָרֹבַע, 
כֶּלִי שֶׁנִּטַּמָּא תוֹכוֹ, 
לֹא נִטַּמּוּ אֲחוֹרָיו? 
אָמַר לָהֶן: 
שַׁלְחַת קוֹדְמִין הִיא? 
אוֹ שֶׁמֵּא הָרֹבַע אֲחוֹרַיִם לַחֲצִי הָרֹבַע, 
כְּלִי שֶׁנִּטַּמּוּ אֲחוֹרָיו, 
לֹא נִטַּמָּא תוֹכוֹ! 

<small>ו</small>
נִטַּמָּא הָרֹבַע, 
הָרֹבַע וַאֲחוֹרָיו טְמֵאִין, 
חֲצִי הָרֹבַע וַאֲחוֹרָיו טְהוֹרִין.  
נִטַּמָּא חֲצִי הָרֹבַע, 
חֲצִי הָרֹבַע וַאֲחוֹרָיו טְמֵאִין, 
וְהָרֹבַע וַאֲחוֹרָיו טְהוֹרִין.  
נִטַּמּוּ אֲחוֹרֵי הָרֹבַע, 
אֲחוֹרֵי חֲצִי הָרֹבַע טְהוֹרִין.  
דִּבְרֵי רְבִּי מֵאִיר.  
וַחֲכָמִים אוֹמְרִים: 
אֵין חוֹלְקִין אֶת הַגַּבַּיִם.  
כְּשֶׁהוּא מַטְבִּיל, מַטְבִּיל אֶת כֻּלּוֹ. 

<small>ז</small>
כַּנֵּי כֵלִים, 
וְהָגְנֵיהֶם, 
וְאָזְנֵיהֶם, 
וִידוֹת הַכֵּלִים הַמְקַבְּלִים, 
שֶׁנָּפְלוּ עֲלֵיהֶן מַשְׁקִין, 
מְנַגְּבָן וְהֵן טְהוֹרִין.  
וּשְׁאָר כָּל הַכֵּלִים שֶׁאֵין לָהֶם אֲחוֹרַיִם וָתוֹךְ, 
וְשֶׁנָּפְלוּ מַשְׁקִין עַל מִקְצָתוֹ, 
כֻּלּוֹ טָמֵא.  
כֶּלִי שֶׁנִּטַּמּוּ אֲחוֹרָיו בְּמַשְׁקִין, 
אֲחוֹרָיו טְמֵאִין; 
תּוֹכוֹ, וְהָגְנוֹ, וְאָזְנָיו, וְיָדָיו, 
טְהוֹרִין.  
נִטַּמָּא מִתּוֹכוֹ, כֻּלּוֹ טָמֵא. 

<small>ח</small>
כָּל הַכֵּלִים יֵשׁ לָהֶם אֲחוֹרַיִם וָתוֹךְ, 
וְיֵשׁ לָהֶן בֵּית צְבִיעָה.  
רְבִּי טַרְפוֹן אוֹמֵר: 
לַעֲרֵבָה גְדוֹלָה שֶׁלָּעֵץ.  
רְבִּי עֲקִיבָה אוֹמֵר: 
לַכּוֹסוֹת.  
רְבִּי מֵאִיר אוֹמֵר: 
לַיָּדַיִם הַטְּמֵאוֹת וְהַטְּהוֹרוֹת.  
אָמַר רְבִּי יוֹסֵה: 
לֹא אָמְרוּ אֶלָּא לַיָּדַיִם הַטְּהוֹרוֹת בִּלְבַד. 

<small>ט</small>
כֵּיצַד? 
הָיוּ יָדָיו טְהוֹרוֹת, 
וַאֲחוֹרֵי הַכּוֹס טְמֵאִין, 
אֲחָזוֹ בְבֵית צְבִיעָתוֹ, 
אֵינוּ חוֹשֵׁשׁ שֶׁמֵּא נִטַּמּוּ יָדָיו בַּאֲחוֹרֵי הַכּוֹס.  
הָיָה שׁוֹתֶה בְּכוֹס שֶׁאֲחוֹרָיו טְמֵאִין, 
אֵינוּ חוֹשֵׁשׁ שֶׁמֵּא נִטַּמָּא מַשְׁקֶה שֶׁבְּפִיו 
בַּאֲחוֹרֵי הַכּוֹס, וְחָזַר וְטִמֵּא אֶת הַכּוֹס.  
קֻמְקוּם שֶׁהוּא מַרְתֵּחַ, 
אֵינוּ חוֹשֵׁשׁ שֶׁמֵּא יָצְאוּ מַשְׁקִים מִתּוֹכוֹ, 
וְנָגְעוּ בַאֲחוֹרָיו, וְחָזְרוּ לְתוֹכוֹ. 

<small>י</small>
כְּלֵי הַקֹּדֶשׁ, 
אֵין לָהֶן אֲחוֹרַיִם וָתוֹךְ, 
וְאֵין לָהֶן בֵּית צְבִיעָה.  
וְאֵין מַטְבִּילִים כֵּלִים בְּתוֹךְ כֵּלִים בַּקֹּדֶשׁ.  
כָּל הַכֵּלִים יוֹרְדִים לִידֵי טֻמְאָתָן בְּמַחֲשָׁבָה, 
וְאֵינָן עוֹלִין מִידֵי טֻמְאָתָן אֶלָּא בְשִׁנּוּי מַעֲשֶׂה. 
הַמַּעֲשֶׂה מְבַטֵּל מִיַּד הַמַּעֲשֶׂה וּמִיַּד הַמַּחֲשָׁבָה, 
הַמַּחֲשָׁבָה אֵינָה מְבַטֶּלֶת 
לֹא מִיַּד הַמַּעֲשֶׂה וְלֹא מִיַּד הַמַּחֲשָׁבָה. 


 



Chapter 26

<small>א</small>
סַנְדָּל עִמְקִי, וְכִיס שֶׁלִּשְׁנָצוֹת, 
רְבִּי יְהוּדָה אוֹמֵר: 
אַף קְפֵיפָה מִצְרִית; 
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
אַף סַנְדָּל לַדִּיקִי כַיּוֹצֵא בָהֶן, 
הֲרֵי אֵלּוּ מִטַּמִּין וּמִטַּהֲרִין שֶׁלֹּא בָאֻמָּן. &lt;מְטַמִּין וּמְטַהֲרִין&gt; 
אָמַר רְבִּי יוֹסֵה: 
וַהֲלֹא כָל הַכֵּלִים מִטַּמִּין וּמִטַּהֲרִין שֶׁלֹּא בָאֻמָּן? 
אֲבָל אֵלּוּ, אַף עַל פִּי מֻתָּרִין, 
טְמֵאִין, שֶׁהַהֶדְיוֹט יָכוֹל לְהַחְזִירָן.  
לֹא אָמְרוּ אֶלָּא בִקְפֵיפָה מִצְרִית, 
שֶׁאַף הָאֻמָּן אֵינוּ יָכוֹל לְהַחְזִירָהּ. 

<small>ב</small>
כִּיס שֶׁלִּשְׁנָצוֹת שֶׁנִּטָּלוּ שְׁנָצָיו, טָמֵא;  
נִפְשַׁט, טָהוֹר.  
טָלָה עָלָיו מַטְלֵת מִלְּמַטָּן, טָמֵא.  
כִּיס בְּתוֹךְ כִּיס שֶׁנִּטַּמָּא אֶחָד מֵהֶן בְּמַשְׁקֶה, 
לֹא נִטַּמָּא חֲבֵרוֹ.  
צְרוֹר הַמַּרְגָלִית, טָמֵא.  
צְרוֹר הַמָּעוֹת, 
רְבִּי אֱלִיעֶזֶר מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין. 

<small>ג</small>
כַּף לוֹקְטֵי קוֹצִים, טְהוֹרָה.  
הַזּוֹן וְהַבַּרְכְּיָר, טְמֵאִין,   
וְהַשַּׁרְוָלִין, טְמֵאִין.  
וְהַפְּרַקְנֵימִין, טְהוֹרִין.  
וְכָל בֵּית אֶצְבָּעוֹת, טְהוֹרוֹת, 
חוּץ מִשֶּׁלַּקַּיָּצִים, 
מִפְּנֵי שֶׁהִיא מְקַבֶּלֶת הָאוֹג. 
נִקְרָעָה, 
אִם אֵינָהּ מְקַבֶּלֶת אֶת רֹב הָאוֹג, 
טְהוֹרָה. 

<small>ד</small>
סַנְדָּל שֶׁנִּפְסְקָה אַחַת מֵאָזְנָיו, וְתִקְּנָהּ, 
טְמֵאָה מִדְרָס; &lt;טמא&gt;
נִפְסְקָה שְׁנִיָּה, וְתִקְּנָהּ, 
טָהֲרָה מִן הַמִּדְרָס, &lt;מחוק: טהור&gt;
אֲבָל טְמֵאָה מַגָּע. &lt;טמא&gt; 
לֹא הִסְפִּיק לְתַקֵּן אֶת הָרִאשׁוֹנָה 
עַד שֶׁנִּפְסְקָה שְׁנִיָּה, 
נִפְסַק עֲקֵבוֹ, 
נִטַּל חָטְמוֹ, אוֹ שֶׁנֶּחֱלַק לִשְׁנַיִם, 
טָהוֹר.  
סֶלְיָם שֶׁנִּפְסַק מִכָּל מָקוֹם, &lt;סוליים&gt;
טָהוֹר.  
מַנְעָל שֶׁנִּפְחַת, וְאֵינוּ מְקַבֵּל אֶת רֹב הָרֶגֶל, 
טָהוֹר.  
מַנְעָל שֶׁעַל הָאָמוּם, 
רְבִּי אֱלִיעֶזֶר מְטַהֵר, 
וַחֲכָמִים מְטַמִּין.  

<small>ה</small>
כָּל חֲמָתוֹת צְרוּרוֹת, טְהוֹרוֹת, 
חוּץ מִשֶּׁלָּעַרְבִיִּים.  
רְבִּי מֵאִיר אוֹמֵר: 
צְרוֹר שָׁעָה, טְהוֹרוֹת.  
צְרוֹר עוֹלָם, טְמֵאוֹת.  
רְבִּי יוֹסֵה אוֹמֵר: 
כָּל חֲמָתוֹת צְרוּרוֹת, טְהוֹרוֹת. 

<small>ו</small>
אֵלּוּ עוֹרוֹת טְמֵאִים מִדְרָס: 
עוֹר שֶׁחָשַׁב עָלָיו לַשָּׁטִיחַ, 
עוֹר סְקֻרְטְיָה, 
עוֹר קַטְבֻלְיָה, 
עוֹר הַחַמָּר, &lt;הגהה מעל: עוֹר הַחֲמוֹר&gt;
עוֹר הַכַּתָּן, 
עוֹר הַכַּתָּף, &lt;הַקֶּתֶף&gt;
עוֹר הָרוֹפֵא, 
עוֹר הָעֲרֵיסָה, 
עוֹר הַלֵּב שֶׁלְּקָטָן, 
עוֹר הַכַּר, 
וְעוֹר הַכֶּסֶת, 
מִדְרָס.  
עוֹר הַסָּרוֹק, 
וְעוֹר הַסּוֹרֵק, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
מִדְרָס. 
וַחֲכָמִים אוֹמְרִים: 
טְמֵא מֵת. 

<small>ז</small>
עַב כְּסוּת, וְתַכְרִיךְ כְּסוּת, 
מִדְרָס.  
עַב אַרְגָּמָן, וְתַכְרִיךְ אַרְגָּמָן, 
בֵּית שַׁמַּי אוֹמְרִים: 
מִדְרָס. 
וּבֵית הֶלֵּל אוֹמְרִים: 
טְמֵא מֵת.  
עוֹר שֶׁעֲשָׂאוֹ חֲפוּי לַכֵּלִים, טָהוֹר, 
וְלַמִּשְׁקָלוֹת, טָמֵא.  
רְבִּי יוֹסֵה מְטַהֵר מִשֵּׁם אָבִיו. 

<small>ח</small>
כָּל מָקוֹם שֶׁאֵין חֶסְרוֹן מְלָאכָה, 
הַמַּחֲשָׁבָה מְטַמְּאַתָּן. 
וְכָל מָקוֹם שֶׁיֵּשׁ חֶסְרוֹן מְלָאכָה, 
אֵין הַמַּחֲשָׁבָה מְטַמְּאַתָּן, 
אֶלָּא הָעַצְבָּה. &lt;העוצבה&gt;

<small>ט</small>
עוֹרוֹת בַּעַל הַבַּיִת, 
הַמַּחֲשָׁבָה מְטַמְּאַתָּן. 
וְשֶׁלָּעָבְּדָן, 
אֵין הַמַּחֲשָׁבָה מְטַמְּאַתָּן, 
וְשֶׁלַּגַּנָּב, 
הַמַּחֲשָׁבָה מְטַמְּאַתָּן. 
וְשֶׁלַּגָּזְלָן, 
אֵין הַמַּחֲשָׁבָה מְטַמְּאַתָּן. 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
חִלּוּף הַדְּבָרִים: 
שֶׁל גַּזְלָן, 
הַמַּחֲשָׁבָה מְטַמְּאַתָּן. 
וְשֶׁל גַּנָּב, 
אֵין הַמַּחֲשָׁבָה מְטַמְּאַתָּן, 
מִפְּנֵי שֶׁלֹּא נִתְיָאֲשׁוּ הַבְּעָלִים. 

<small>י</small>
עוֹר שֶׁהוּא טָמֵא מִדְרָס, 
וְחָשַׁב עָלָיו לִרְצוּעוֹת הַסַּנְדַּלִּים, 
כֵּיוָן שֶׁנָּתַן בּוֹ אֶת הָאַזְמֵל, 
טָהַר.  
דִּבְרֵי רְבִּי יְהוּדָה.  
וַחֲכָמִים אוֹמְרִים: 
עַד שֶׁיַּמְעִיטֶנּוּ פָחוּת מֵחֲמִשָּׁה טְפָחִים.  
רְבִּי אֶלְעָזָר בִּרְבִּי צָדוֹק אוֹמֵר: 
אַף הָעוֹשֶׂה מִטְפַּחַת מִן הָעוֹר, 
טְמֵאָה, 
מִן הַכְּסָתוֹת, טְהוֹרָה. 





Chapter 27

<small>א</small>
הַבֶּגֶד מִטַּמֵּא מִשֵּׁם חֲמִשָּׁה שֵׁמוֹת, &lt;מְטַמֵּא&gt;
הַסַּק מִשֵּׁם אַרְבָּעָה, 
הָעוֹר מִשֵּׁם שְׁלֹשָׁה, 
הָעֵץ מִשֵּׁם שְׁנַיִם, 
כְּלִי חֶרֶשׂ מִשֵּׁם אֶחָד.  
כְּלִי חֶרֶשׂ מִטַּמֵּא מִשֵּׁם כְּלִי קִבּוּל. &lt;מְטַמֵּא&gt; 
וְכָל שֶׁאֵין לוֹ תוֹךְ בִּכְלִי חֶרֶשׂ, 
אֵין לוֹ אֲחוֹרַיִם.  

<small>ב</small>
מוֹסִיף עָלָיו הָעֵץ, &lt;מוסף! גרסת הסופר מוסף, והנקדן ניקד מוסף&gt;
שֶׁהוּא מִטַּמֵּא מִשֵּׁם מוֹשָׁב. &lt;מְטַמֵּא&gt;  
וְכֵן טַבְלָה שֶׁאֵין לָהּ לִזְבֵּז, 
בִּכְלִי עֵץ, טְמֵאָה, 
וּבִכְלִי חֶרֶשׂ, טְהוֹרָה.  

<small>ג</small>
מוֹסִיף עָלָיו הָעוֹר, &lt;מוסף&gt;
שֶׁהוּא מִטַּמֵּא מִשֵּׁם אֲהָלִים. &lt;מְטַמֵּא&gt;   
מוֹסִיף עָלָיו הַשַּׂק, &lt;מוסף&gt;
שֶׁהוּא מִטַּמֵּא מִשֵּׁם אָרִיג. &lt;מְטַמֵּא&gt; 
מוֹסִיף עָלָיו הַבֶּגֶד, &lt;מוסף&gt;
שֶׁהוּא מִטַּמֵּא מִשֵּׁם שָׁלוֹשׁ עַל שָׁלוֹשׁ. 

<small>ד</small>
הַבֶּגֶד מִטַּמֵּא מִשֵּׁם שְׁלֹשָׁה עַל שְׁלֹשָׁה &lt;מְטַמֵּא&gt;
לְמִדְרָס, 
וּמִשּׁוּם שָׁלוֹשׁ עַל שָׁלוֹשׁ לִטְמֵא מֵת.  
הַשַּׂק, אַרְבָּעָה עַל אַרְבָּעָה, 
הָעוֹר, חֲמִשָּׁה עַל חֲמִשָּׁה, 
מַפָּץ, שִׁשָּׁה עַל שִׁשָּׁה, 
שָׁוִין לְמִדְרָס וְלִטְמֵא מֵת.  
רְבִּי מֵאִיר אוֹמֵר: 
הַשַּׂק, שִׁירָיו אַרְבָּעָה, 
וּתְחִלָּתוֹ מִשֶּׁיִּגָּמֵר. 

<small>ה</small>
הָעוֹשֶׂה שְׁנַיִם מִן הַבֶּגֶד וְאֶחָד מִן הַשַּׂק, 
שְׁלֹשָׁה מִן הַשַּׂק וְאֶחָד מִן הָעוֹר, 
אַרְבָּעָה מִן הָעוֹר וְאֶחָד מִן הַמַּפָּץ, 
טָהוֹר.  
חֲמִשָּׁה מִן הַמַּפָּץ וְאֶחָד מִן הָעוֹר, 
אַרְבָּעָה מִן הָעוֹר וְאֶחָד מִן הַשַּׂק, 
שְׁלֹשָׁה מִן הַשַּׂק וְאֶחָד מִן הַבֶּגֶד, 
טָמֵא.  
זֶה הַכְּלָל: 
כָּל שֶׁחִבֵּר לוֹ מִן הֶחָמוּר מִמֶּנּוּ, טָמֵא, 
וּמִן הַקַּל מִמֶּנּוּ, טָהוֹר. 

<small>ו</small>
הַמְקַצֵּעַ מִכֻּלָּם טֶפַח עַל טֶפַח, 
טָמֵא.  
מִשּׁוּלֵי הַקֻּפָּה טֶפַח עַל טֶפַח, 
טָמֵא.  
מִצִּדְדֵי הַקֻּפָּה, 
רְבִּי שִׁמְעוֹן מְטַהֵר, 
וַחֲכָמִים אוֹמְרִים: 
הַמְקַצֵּעַ טֶפַח עַל טֶפַח בְּכָל מָקוֹם, 
טָמֵא. 

<small>ז</small>
בְּלוּיֵי נָפָה וּכְבָרָה שֶׁהִתְקִינָן לִישִׁיבָה, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין, עַד שֶׁיְּקַצַּע.  
כִּסֵּא שֶׁלַּקָּטָן שֶׁיֶּשׁ לוֹ רַגְלַיִם, 
אַף עַל פִּי שֶׁאֵין בּוֹ גֹבַהּ טֶפַח, 
טָמֵא.  
חָלוּק שֶׁלַּקָּטָן, 
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
כָּל שֶׁהוּא. 
וַחֲכָמִים אוֹמְרִים: 
עַד שֶׁיְּהֵא בוֹ כַשֵּׁעוּר, 
וְנִמְדָּד כָּפוּל. 

<small>ח</small>
אֵלּוּ נִמְדָּדִין כְּפוּלִים: 
אֲנַפְלְיָה, &lt;מחוק: אמפלייה&gt; 
וּפִמְלַנְיָה, 
וּמִכְנְסַיִם, 
וְכוֹבַע, 
וְכִיס שֶׁלְּפֻנְדָּה.  
וּמַטְלֵית שֶׁטְּלָיָהּ עַל הַשָּׂפָה, 
אִם פְּשׁוּטָה, נִמְדֶּדֶת פְּשׁוּטָה, 
אִם כְּפוּלָה, נִמְדֶּדֶת כְּפוּלָה. 

<small>ט</small>
בֶּגֶד שֶׁאָרַג בּוֹ שְׁלֹשָׁה עַל שְׁלֹשָׁה, 
נִטַּמָּא מִדְרָס, 
וְהִשְׁלִים עָלָיו אֶת כָּל הַבֶּגֶד, 
נָטַל חוּט אֶחָד מִתְּחִלָּתוֹ, 
טָהַר מִן הַמִּדְרָס, 
אֲבָל טָמֵא מַגַּע מִדְרָס.  
נָטַל חוּט אֶחָד מִתְּחִלָּתוֹ, 
וְאַחַר כָּךְ הִשְׁלִים עָלָיו אֶת כָּל הַבֶּגֶד, 
טָמֵא מַגַּע מִדְרָס. 

<small>י</small>
וְכֵן בֶּגֶד שֶׁאָרַג בּוֹ שָׁלוֹשׁ עַל שָׁלוֹשׁ, 
נִטַּמָּא טְמֵא מֵת, 
וְהִשְׁלִים עָלָיו אֶת כָּל הַבֶּגֶד, 
נָטַל חוּט אֶחָד מִתְּחִלָּתוֹ, 
טָהַר מִטְּמֵא מֵת, 
אֲבָל טָמֵא מַגַּע טְמֵא מֵת.  
נָטַל חוּט אֶחָד מִתְּחִלָּתוֹ, 
וְאַחַר כָּךְ הִשְׁלִים עָלָיו אֶת כָּל הַבֶּגֶד, 
טָהוֹר.  

<small>יא</small>
מִפְּנֵי שֶׁאָמָרוּ: 
שָׁלוֹשׁ עַל שָׁלוֹשׁ שֶׁנִּתְמָעֲטָה, טְהוֹרָה.  
אֲבָל שְׁלֹשָׁה עַל שְׁלֹשָׁה שֶׁנִּתְמָעַט, 
אַף עַל פִּי שֶׁטָּהַר מִן הַמִּדְרָס, 
טָמֵא בְכָל הַטְּמָאוֹת. 

<small>יב</small>
סָדִין שֶׁהוּא טָמֵא מִדְרָס, 
וַעֲשָׂאוֹ וֵילָן,  
טָהַר מִן הַמִּדְרָס, 
אֲבָל טָמֵא מַגַּע מִדְרָס.  
אָמַר רְבִּי יוֹסֵה: 
וְכִי בְּאֵי זֶה מִדְרָס נָגַע זֶה? 
אֶלָּא אִם כֵּן נָגַע בּוֹ זָב, 
טָמֵא מַגָּע זָב. 

<small>יג</small>
שְׁלֹשָׁה עַל שְׁלֹשָׁה שֶׁנֶּחֱלַק, 
טָהַר מִן הַמִּדְרָס, 
אֲבָל טָמֵא מַגַּע מִדְרָס.  
אָמַר רְבִּי יוֹסֵה: 
וְכִי בְּאֵי זֶה מִדְרָס נָגַע זֶה? 
אֶלָּא אִם כֵּן נָגַע בּוֹ הַזָּב, 
וְטָמֵא מַגָּע זָב. 

<small>יד</small>
שְׁלֹשָׁה עַל שְׁלֹשָׁה בָאַשְׁפּוֹת, 
בָּרִיא וְצוֹרֵר מֶלַח, 
וּבַבַּיִת, 
אוֹ בָרִיא, אוֹ צוֹרֵר מֶלַח.  
כַּמָּה מֶלַח יְהֵא צוֹרֵר? 
רֹבַע.  
רְבִּי יְהוּדָה אוֹמֵר: 
דַּקָּה. 
וַחֲכָמִים אוֹמְרִים: 
גַּסָּה.  
אֵלּוּ וָאֵלּוּ מִתְכַּוְּנִין לְהָקֵל.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
שָׁוִים שְׁלֹשָׁה עַל שְׁלֹשָׁה בָאַשְׁפּוֹת 
לְשָׁלֹשׁ עַל שָׁלוֹשׁ שֶׁבַּבַּיִת. 

<small>יה</small>
שְׁלֹשָׁה עַל שְׁלֹשָׁה שֶׁנִּקְרַע, 
אִם נוֹתְנוֹ עַל הַכִּסֵּא, 
וּבְשָׂרוֹ נוֹגֵעַ בַּכִּסֵּא, טָהוֹר; 
וְאִם לָאו, טָמֵא.  
שָׁלוֹשׁ עַל שָׁלוֹשׁ שֶׁנִּמְהָה מִמֶּנָּה חוּט אֶחָד, 
אוֹ שֶׁנִּמְצָא בָהּ קֶשֶׁר, 
אוֹ שְׁנֵי חוּטִין מַתְאִימִין, טְהוֹרָה.  
שָׁלוֹשׁ עַל שָׁלוֹשׁ שֶׁהִשְׁלִיכָהּ בָּאַשְׁפּוֹת, 
טְהוֹרָה.  
אִם הֶחֱזִירָהּ, טְמֵאָה.  
לְעוֹלָם הַשְׁלָכָתָהּ מְטַהֲרַתָּה, 
וַחֲזָרָתָהּ מְטַמְּאַתָּהּ, 
חוּץ מִשֶּׁלָּאַרְגָּמָן וְשֶׁלִּזְהוֹרִית טוֹבָה.  
רְבִּי אֱלִיעֶזֶר אוֹמֵר: 
אַף מַטְלֵת חֲדָשָׁה כַּיּוֹצֵא בָהֶן.  
רְבִּי שִׁמְעוֹן אוֹמֵר: 
כֻּלָּם טְמֵאִין, 
לֹא הָזְכָּרוּ אֶלָּא מִפְּנֵי הָשֵׁב אֲבֵדָה. 





Chapter 28

<small>א</small>
שָׁלוֹשׁ עַל שָׁלוֹשׁ שֶׁנָּתְנָה בַכַּדּוּר, 
אוֹ שֶׁעֲשָׂאָהּ כַּדּוּר בִּפְנֵי עַצְמָהּ, 
טְהוֹרָה.  
אֲבָל שְׁלֹשָׁה עַל שְׁלֹשָׁה שֶׁנִּתָּנוּ בַכַּדּוּר, 
טָמֵא.  
עֲשָׂאוֹ כַדּוּר בִּפְנֵי עַצְמוֹ, 
טָהוֹר, מִפְּנֵי שֶׁהַתָּפִיר מְמַעֲטוֹ. 

<small>ב</small>
פָּחוּת מִשְׁלֹשָׁה עַל שְׁלֹשָׁה 
שֶׁהִתְקִינוֹ לָפֹק בּוֹ אֶת הַמַּרְחֵץ, 
לְנַעֵר בּוֹ אֶת הַקְּדֵרָה, 
לְקַנַּח בּוֹ אֶת הָרֵחַיִם, 
בֵּין מִן מוּכָן וּבֵין שֶׁאֵינוּ מִן מוּכָן, 
טָמֵא.  
דִּבְרֵי רְבִּי אֱלִיעֶזֶר.  
רְבִּי יְהוֹשֻׁעַ אוֹמֵר: 
בֵּין מִן מוּכָן וּבֵין שֶׁאֵינוּ מִן מוּכָן, 
טָהוֹר.  
רְבִּי עֲקִיבָה אוֹמֵר: 
בְּמוּכָן, טָמֵא, 
וּבְשֶׁאֵינוּ מוּכָן, טָהוֹר. 

<small>ג</small>
הָעוֹשֶׂה אַסְפְּלָנִית, 
בֵּין בַּבֶּגֶד וּבֵין בָּעוֹר, 
טְהוֹרָה.  
מְלֻגְמָה, 
בַּבֶּגֶד, טְהוֹרָה, 
וּבָעוֹר, טְמֵאָה.  
רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר: 
אַף בַּבֶּגֶד, טְמֵאָה, 
מִפְּנֵי שֶׁהִיא נִנְעֶרֶת. 

<small>ד</small>
מִטְפְּחוֹת סְפָרִים, 
בֵּין מְצֻיָּרוֹת וּבֵין שֶׁאֵינָן מְצֻיָּרוֹת, 
טְמֵאוֹת. 
דִּבְרֵי בֵית שַׁמַּי.  
בֵּית הֶלֵּל אוֹמְרִים: 
מְצֻיָּרוֹת, טְהוֹרוֹת, 
וְשֶׁאֵינָן מְצֻיָּרוֹת, טְמֵאוֹת.  
רַבָּן גַּמְלִיאֵל אוֹמֵר: 
אֵלּוּ וָאֵלּוּ טְהוֹרוֹת. 

<small>ה</small>
כַּפָּח שֶׁהוּא טָמֵא מִדְרָס, 
וּנְתָנוֹ עַל הַסֵּפֶר, 
טָהוֹר מִן הַמִּדְרָס, 
אֲבָל טָמֵא טְמֵא מֵת.  
חֵמֶת שֶׁעֲשָׂאָהּ שָׁטִיחַ, 
וְשָׁטִיחַ שֶׁעֲשָׂאוֹ חֵמֶת, 
טָהוֹר.  
חֵמֶת שֶׁעֲשָׂאָהּ תָּרְמֵל, 
וְתָרְמֵל שֶׁעֲשָׂאוֹ חֵמֶת, 
כַּר שֶׁעֲשָׂאוֹ סָדִין, 
וְסָדִין שֶׁעֲשָׂאוֹ כַר, 
כֶּסֶת שֶׁעֲשָׂאָהּ מִטְפַּחַת, 
וּמִטְפַּחַת שֶׁעֲשָׂאָהּ כֶּסֶת, 
טָמֵא.  
זֶה הַכְּלָל: 
כָּל שֶׁשִּׁנָּהוּ לִשְׁמוֹ, טָמֵא; 
לְשֵׁם אַחֵר, טָהוֹר. 

<small>ו</small>
מַטְלֵית שֶׁטְּלָיָהּ עַל הַקֻּפָּה, 
מְטַמָּא אֶחָד, וּפוֹסֶלֶת אֶחָד.  
הִפְרִישָׁהּ מִן הַקֻּפָּה, 
הַקֻּפָּה מְטַמָּא אֶחָד וּפוֹסֶלֶת אֶחָד, 
וְהַמַּטְלֵית טְהוֹרָה.  
טְלָיָהּ עַל הַבֶּגֶד, 
מְטַמָּא שְׁנַיִם וּפוֹסֶלֶת אֶחָד.  
הִפְרִישָׁהּ מִן הַבֶּגֶד, 
הַבֶּגֶד מְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד, 
וְהַמַּטְלֵית מְטַמָּא שְׁנַיִם וּפוֹסֶלֶת אֶחָד.  
וְכֵן הַטּוֹלֶה עַל הַשַּׂק, אוֹ עַל הָעוֹר.  
דִּבְרֵי רְבִּי מֵאִיר.  
רְבִּי שִׁמְעוֹן מְטַהֵר.  
רְבִּי יוֹסֵה אוֹמֵר: 
עַל הָעוֹר, טָהוֹר, 
וְעַל הַשַּׂק, טָמֵא, 
מִפְּנֵי שֶׁהוּא אָרִיג. 

<small>ז</small>
שָׁלוֹשׁ עַל שָׁלוֹשׁ שֶׁאָמָרוּ, 
חוּץ מִן הַמְּלָל.  
דִּבְרֵי רְבִּי שִׁמְעוֹן.  
וַחֲכָמִים אוֹמְרִים: 
שָׁלוֹשׁ עַל שָׁלוֹשׁ מְכֻוָּנוֹת.  
טְלָיָהּ עַל הַבֶּגֶד מֵרוּחַ אַחַת, אֵינוּ חִבּוּר; 
מִשְּׁתֵי רוּחוֹת זוֹ כְנֶגֶד זוֹ, חִבּוּר.  
עֲשָׂאָהּ כְמִין גַּם, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
וַחֲכָמִים מְטַהֲרִין.  
אָמַר רְבִּי יְהוּדָה: 
בַּמֵּי דְבָרִים אֲמוּרִים? 
בַּטַּלִּית, אֲבָל בֶּחָלוּק, 
מִלְמַעְלָן, חִבּוּר, וּמִלְּמַטָּן, אֵינָהּ חִבּוּר. 

<small>ח</small>
בִּגְדֵי עֲנִיִּים, 
אַף עַל פִּי שֶׁאֵין בָּהֶן שָׁלוֹשׁ עַל שָׁלוֹשׁ, 
הֲרֵי אֵלּוּ טְמֵאִין מִדְרָס.  
טַלֵּית שֶׁהִתְחִיל בָּהּ לְקָרְעָהּ, 
כֵּיוָן שֶׁנִּקְרַע רֻבָּהּ, 
אֵינָה חִבּוּר.  
הֶעָבִים הָרַכִּים, 
אֵין בָּהֶן מִשֵּׁם שָׁלוֹשׁ עַל שָׁלוֹשׁ. 

<small>ט</small>
כֶּסֶת הַסַּבָּלִין, &lt;הַסַּבַּלִּין&gt;
טְמֵאָה מִדְרָס.  
מְשַׁמֶּרֶת שֶׁלַּיַּיִן, 
אֵין בָּהּ מִשֵּׁם מוֹשָׁב.  
שְׂבָכָה שֶׁלַּזְּקֵנָה, 
טְמֵאָה מִדְרָס מִשֵּׁם מוֹשָׁב.  
חָלוּק שֶׁלְּיוֹצְאוֹת הַחוּץ, 
הֶעָשׂוּי כִּשְׂבָכָה, 
טָהוֹר.  
הָעוֹשֶׂה בֶגֶד מִן הַחֵרֶם, טָהוֹר; 
וּמִן הַזּוֹטוֹ, טָמֵא.  
רְבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר: 
אַף הָעוֹשֶׂה בֶגֶד מִן הַחֵרֶם וּכְפָלוֹ, 
טָמֵא. 

<small>י</small>
שְׂבָכָה שֶׁהִתְחִיל בָּהּ מִפִּיהָ, 
טְהוֹרָה, עַד שֶׁיִּגְמֹר אֶת קָרְקָרְתָּהּ.  
הִתְחִיל בָּהּ מִקָּרְקָרְתָּהּ, 
טְהוֹרָה, עַד שֶׁיִּגְמֹר אֶת פִּיהָ.  
שְׁבִיס שֶׁלָּהּ, 
טָמֵא בִפְנֵי עַצְמוֹ.  
חוּטִין שֶׁלָּהּ, 
טְמֵאִין מִשֵּׁם חִבּוּר.  
שְׂבָכָה שֶׁנִּקְרְעָה, 
אִם אֵינָה מְקַבֶּלֶת אֶת רֹב הַשֵּׂעָר, 
טְהוֹרָה. 





Chapter 29

<small>א</small>
נוּמֵי הַסָּדִין וְהַסּוּדָרִין וְהַטַּרְטִין וְהַפַּלְיוֹן שֶׁלָּרֹאשׁ, 
שֵׁשׁ אֶצְבָּעוֹת; 
שֶׁלִּפְקַרְסִין, עֶשֶׂר.  &lt;אפיקרסין, סדין&gt;
נוּמֵי הַסָּגִי וְהָרָדִיד וְהֶחָלוּק וְהַטַּלֵּית, 
שָׁלוֹשׁ אֶצְבָּעוֹת.  
נוּמֵי הַכַּפָּח שֶׁלַּזְּקֵנָה, 
וְהַגֻּמְדִין שֶׁלָּעַרְבִיִּים, 
וְהַקֻּלְקִיס, וְהַפָּנְדָּה, 
וְהַמַּעְפֹּרֶת, וְהַפַּרְגּוֹד, 
נוּמֵיהֶן כָּל שֶׁהֵן. 

<small>ב</small>
שָׁלוֹשׁ כְּסָתוֹת שֶׁלַּצֶּמֶר, 
שֵׁשׁ שֶׁלַּפִּשְׁתָּן, 
שְׁלֹשָׁה סְדִינִים, 
שְׁתֵּים עֶשְׂרֵה מִטְפָּחוֹת, 
שְׁנֵי סֻוְרִיקִין, 
חָלוּק אֶחָד, 
טַלֵּית אַחַת, 
וּקְלֻבְקָרִין אַחַת, 
חִבּוּר לַטֻּמְאָה וְלַהַזָּיָה; 
יָתֵר מִכֵּן, 
חִבּוּר לַטֻּמְאָה, וְאֵינוּ חִבּוּר לַהַזָּיָה.  
רְבִּי יוֹסֵה אוֹמֵר: 
אַף לֹא לַטֻּמְאָה. 

<small>ג</small>
חוּט מִשְׁקֶלֶת, שְׁנֵים עָשָׂר; 
שֶׁלֶּחָרָשִׁים, שְׁמוֹנָה עָשָׂר; 
וְשֶׁלַּבַּנָּיִין, חֲמִשִּׁים אַמָּה.  
יָתֵר מִכֵּן, 
אִם רָצָה לְקַיֵּם, טָהוֹר.  
וְשֶׁלַּסַּיָּדִים וְשֶׁלַּצַּיָּרִים, 
כָּל שֶׁהֵן. 

<small>ד</small>
חוּט מֹאזְנַיִם שֶׁלַּחַנְוָנִי, 
וְחוּט מֹאזְנַיִם שֶׁלִּזְהָבִים, 
וְשֶׁלְּשׁוֹקְלֵי אַרְגָּמָן טוֹב, 
שָׁלוֹשׁ אֶצְבָּעוֹת.  
יַד הַקָּרְדּוֹם מֵאַחֲרָיו, 
שָׁלוֹשׁ אֶצְבָּעוֹת.  
רְבִּי יוֹסֵה אוֹמֵר: 
טֶפַח, טָהוֹר. 

<small>ה</small>
חוּט מֹאזְנַיִם שֶׁלַּשֻּׁלְחָנִי, 
שֶׁלְּבַעֲלֵי בָתִּים, 
טֶפַח.  
יַד קָרְדּוֹם מִלְּפָנָיו, 
טֶפַח.  
שִׁירֵי יַד פַּרְגָּל, 
טֶפַח.  
יַד מַקֶּבֶת שֶׁלִּמְפַתְּחֵי אֲבָנִים, &lt;שֶׁלְּמַפְתְּחֵי&gt;
טֶפַח. 

<small>ו</small>
חוּט מֹאזְנַיִם שֶׁלַּצַּמָּרִים, 
וְשֶׁלְּשׁוֹקְלֵי זְכוֹכִית, 
טִפְחַיִם.  
יַד הַמַּקּוֹר, 
טִפְחַיִם.  
יַד הַמַּעְצָד שֶׁלַּלִּגְיוֹנוֹת, 
טִפְחַיִם.  
יַד הַקָּרְנָס שֶׁלִּזְהָבִים, 
טִפְחַיִם; 
וְשֶׁלֶּחָרָשִׁין, 
שְׁלֹשָׁה טְפָחִים. 

<small>ז</small>
שִׁירֵי הַדָּרְבָן מִלְּמַעְלָן, אַרְבָּעָה.  
יַד הַבָּדִיד, יַד הַקָּרְדּוֹם שֶׁלַּנָּכִישׁ, חֲמִשָּׁה.  
יַד בֶּן הַפַּטִּישׁ, חֲמִשָּׁה; 
וְהַפַּטִּישׁ, שִׁשָּׁה.  
יַד קָרְדּוֹם שֶׁלַּבִּקּוּעַ וְשֶׁלֶּעָדִיר, שִׁשָּׁה.  
יַד מַקֶּבֶת, שִׁשָּׁה. 

<small>ח</small>
שִׁירֵי חַרְחוּר מִלְּמַטָּן, שִׁבְעָה.  
יַד מַגְרֵפָה שֶׁלְּבַעֲלֵי בָתִּים, 
בֵּית שַׁמַּי אוֹמְרִים: 
שִׁבְעָה, 
וּבֵית הֶלֵּל אוֹמְרִים: 
שְׁמוֹנָה.  
וְשֶׁלַּסַּיָּדִין, 
בֵּית שַׁמַּי אוֹמְרִים: 
תִּשְׁעָה; 
וּבֵית הֶלֵּל אוֹמְרִים: 
עֲשָׂרָה.  
יָתֵר מִכֵּן, אִם רָצָה לְקַיֵּם, טָמֵא.  
יַד מְשַׁמְּשֵׁי הָאוּר, כָּל שֶׁהֵן. 





Chapter 30

<small>א</small>
כְּלֵי זְכוֹכִית, 
פְּשׁוּטֵיהֶן טְהוֹרִין, וּמְקַבְּלֵיהֶן טְמֵאִים.  
נִשְׁבָּרוּ, טָהָרוּ.  
חָזַר וְעָשָׂה מֵהֶן כֵּלִים, 
מְקַבְּלִין טֻמְאָה מִכָּאן וְלַבָּא.  
הַטַּבְלָה וְהָאַסְקֻטְלָא שֶׁלִּזְכוֹכִית, טְהוֹרִין.  
אִם יֶשׁ לָהֶן לִזְבֵּז, טְמֵאִין.  
וְשׁוּלֵי קְעָרָה וְשׁוּלֵי אַסְקֻטְלָא שֶׁלִּזְכוֹכִית 
שֶׁהִתְקִינָן לְתַשְׁמִישׁ, טְהוֹרִין.  
קִרְטְסָם, וְשָׁפָן בְּשׁוֹפִין, טְמֵאִין. 

<small>ב</small>
סְפַקְלַרְיָה, טְהוֹרָה.  
וְתַמְחוּי שֶׁעֲשָׂאוֹ סְפַקְלַרְיָה, טָמֵא.  
וְאִם מִתְּחִלָּה עֲשָׂאוֹ לְשֵׁם סְפֶקְלַרְיָה, טָהוֹר.  
תַּרְוָד שֶׁהוּא נָתוּן עַל הַשֻּׁלְחָן, 
אִם מְקַבֵּל כָּל שֶׁהוּא, טָמֵא, 
וְאִם לָאו, 
רְבִּי עֲקִיבָה מְטַמֵּא, 
רְבִּי יוֹחָנָן בֶּן נוּרִי מְטַהֵר. 

<small>ג</small>
כּוֹס שֶׁנִּפְגַּם רֻבּוֹ, שָׁלוֹשׁ בְּרֻבּוֹ, 
טָהוֹר; 
רְבִּי שִׁמְעוֹן אוֹמֵר: 
אִם מְפַזֵּר הוּא אֶת רֹב הַמַּיִם, 
טָהוֹר.  
נִקַּב, וַעֲשָׂאוֹ בֵּין בַּבַּעַץ בֵּין בַּזֶּפֶת, 
טָהוֹר.  
רְבִּי יוֹסֵה אוֹמֵר: 
בַּבַּעַץ, טָמֵא, 
וּבַזֶּפֶת, טָהוֹר. 

<small>ד</small>
צְלוֹחִית קְטַנָּה שֶׁנִּטַּל פִּיהָ, טְהוֹרָה; 
שֶׁלְּפִלְיָטוֹס שֶׁנִּטַּל פִּיהָ, טְהוֹרָה, 
מִפְּנֵי שֶׁהִיא סוֹרַחַת אֶת הַיָּד.  
לָגִינִין גְּדוֹלִים שֶׁנִּטַּל פִּיהֶם, טְמֵאִין, 
מִפְּנֵי שֶׁהוּא מַתְקִינָן לִכְבָשִׁין.  
וְאַרְפַּקְס שֶׁלִּזְכוֹכִית, טְהוֹרָה.  

<small>ה</small>
אָמַר רְבִּי יוֹסֵה: 
"אַשְׁרֵיךְ כֵּלִים, 
שֶׁנִּכְנְסַת בְּטֻמְאָה, 
וְיָצָאת בְּטַהֲרָה!"
